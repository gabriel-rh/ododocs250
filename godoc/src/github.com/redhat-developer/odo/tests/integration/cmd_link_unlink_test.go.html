    19  		commonVar = helper.CommonBeforeEach()
    20  		// wait until timeout(sec) for odo to see all the operators installed by setup script in the namespace
    21  		odoArgs := []string{"catalog", "list", "services"}
    22  		operator := "redis-operator"
    23  		helper.WaitForCmdOut("odo", odoArgs, 5, true, func(output string) bool {
    24  			return strings.Contains(output, operator)
    25  		})
    26  	})

    32  	Context("Running the help command", func() {
    33  		It("should display the help", func() {
    34  			By("for the link command", func() {
    35  				appHelp := helper.Cmd("odo", "link", "-h").ShouldPass().Out()
    36  				helper.MatchAllInOutput(appHelp, []string{"Link current or provided component", "backed by an Operator", "or another component"})
    37  			})
    38  			By("for the unlink command", func() {
    39  				appHelp := helper.Cmd("odo", "unlink", "-h").ShouldPass().Out()
    40  				Expect(appHelp).To(ContainSubstring("Unlink component or service from a component"))
    41  			})
    42  		})
    43  	})

    45  	When("two components are deployed", func() {
    46  		var frontendContext, backendContext, frontendURL, frontendComp, backendComp string

    48  		// checkDescribe: checks that the linked component and related variables are present in the output of odo describe
    49  		var checkDescribe = func(contextDir string, compName string, pushed bool, bindAsFiles bool) {
    50  			stdOut := helper.Cmd("odo", "describe", "--context", contextDir).ShouldPass().Out()
    51  			Expect(stdOut).To(ContainSubstring("Linked Services"))
    52  			Expect(stdOut).To(ContainSubstring(compName))
    53  			if pushed {
    54  				if bindAsFiles {
    55  					Expect(stdOut).To(ContainSubstring("/bindings"))
    56  				} else {
    57  					Expect(stdOut).To(ContainSubstring("SERVICE_BACKEND_IP"))
    58  					Expect(stdOut).To(ContainSubstring("SERVICE_BACKEND_PORT"))
    59  				}
    60  			}
    61  		}
    62  
    63  		// createAndPush: creates component, a URL for it and deploys it
    64  		var createAndPush = func(compType string, compName string, contextDir string) {
    65  			helper.CopyExample(filepath.Join("source", compType), contextDir)
    66  			helper.Cmd("odo", "create", compName, "--context", contextDir, "--project", commonVar.Project, "--devfile", helper.GetExamplePath("source", "devfiles", compType, "devfile-registry.yaml")).ShouldPass()
    67  			helper.Cmd("odo", "url", "create", "--port", "8080", "--context", contextDir).ShouldPass()
    68  			helper.Cmd("odo", "push", "--context", contextDir).ShouldPass()
    69  		}
    70  
    71  		BeforeEach(func() {
    72  			frontendComp = fmt.Sprintf("frontend-%v", helper.RandString(3))
    73  			frontendContext = helper.CreateNewContext()
    74  			createAndPush("nodejs", frontendComp, frontendContext)
    75  			frontendURL = helper.DetermineRouteURL(frontendContext)
    76  
    77  			backendComp = fmt.Sprintf("backend-%v", helper.RandString(3))
    78  			backendContext = helper.CreateNewContext()
    79  			createAndPush("python", backendComp, backendContext)
    80  		})
    81  
    82  		AfterEach(func() {

    95  			})

    98  				helper.Cmd("odo", "unlink", backendComp, "--context", frontendContext).ShouldPass()
    99  			})
   100  
   101  			When("the link is pushed", func() {
   102  				BeforeEach(func() {
   103  					helper.Cmd("odo", "push", "--context", frontendContext).ShouldPass()
   104  				})
   105  				It("should ensure that the proper envFrom entry was created", func() {
   106  					envFromOutput := commonVar.CliRunner.GetEnvFromEntry(frontendComp, "app", commonVar.Project)
   107  					Expect(envFromOutput).To(ContainSubstring(backendComp))
   108  					helper.HttpWaitFor(frontendURL, "Hello world from node.js!", 20, 1)
   109  				})
   110  				It("should find the link and environment variables in odo describe", func() {
   111  					checkDescribe(frontendContext, backendComp, true, false)
   112  				})
   113  				It("should find the linked environment variable", func() {
   114  					stdOut := helper.Cmd("odo", "exec", "--context", frontendContext, "--", "sh", "-c", "echo $SERVICE_BACKEND_IP").ShouldPass().Out()
   115  					Expect(stdOut).To(Not(BeEmpty()))
   116  				})
   117  				It("should not allow re-linking", func() {

   124  					helper.Cmd("odo", "delete", "-f", "--context", backendContext).ShouldPass()
   125  					helper.Cmd("odo", "delete", "-f", "--context", frontendContext).ShouldPass()
   126  				})
