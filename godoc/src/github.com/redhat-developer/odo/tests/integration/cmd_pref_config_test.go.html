    54  	When("running odo config view", func() {
    55  		It("should pass if devfile is present in current dir", func() {

    57  			cmpName := helper.RandString(6)
    58  			helper.Cmd("odo", "create", cmpName, "--devfile", helper.GetExamplePath("source", "devfiles", "nodejs", "devfile.yaml")).ShouldPass()

    60  			helper.MatchAllInOutput(out, []string{"runtime", "Memory: 1024Mi"})
    61  		})

    63  		It("should fail if devfile not present in current dir", func() {

    65  			Expect(err).To(ContainSubstring("the current directory does not represent an odo component."))
    66  		})

    69  	Context("When viewing global config", func() {
    70  		var newContext string
    71  		// ConsentTelemetry is set to false in helper.CommonBeforeEach so that it does not prompt to set a value
    72  		// during the tests, but we want to check preference values as they would be in real time and hence
    73  		// we set the GLOBALODOCONFIG variable to a value in new context
    74  		var _ = JustBeforeEach(func() {
    75  			newContext = helper.CreateNewContext()
    76  			os.Setenv("GLOBALODOCONFIG", filepath.Join(newContext, "preference.yaml"))
    77  		})
    78  		var _ = JustAfterEach(func() {
    79  			helper.DeleteDir(newContext)
    80  		})
    81  		It("should get the default global config keys", func() {
    82  			configOutput := helper.Cmd("odo", "preference", "view").ShouldPass().Out()
    83  			preferences := []string{"UpdateNotification", "NamePrefix", "Timeout", "BuildTimeout", "PushTimeout", "Ephemeral", "ConsentTelemetry"}
    84  			helper.MatchAllInOutput(configOutput, preferences)
    85  			for _, key := range preferences {
    86  				value := helper.GetPreferenceValue(key)
    87  				Expect(value).To(BeEmpty())
    88  			}
    89  		})
    90  	})

    92  	Context("When configuring global config values", func() {
    93  		preferences := []struct {
    94  			name, value, updateValue, invalidValue string
    95  		}{
    96  			{"UpdateNotification", "false", "true", "foo"},
    97  			{"Timeout", "5", "6", "foo"},
    98  			{"NamePrefix", "foo", "bar", ""},
    99  			{"BuildTimeout", "5", "7", "foo"},
   100  			// !! Do not test ConsentTelemetry with true because it sends out the telemetry data and messes up the statistics !!
   101  			{"ConsentTelemetry", "false", "false", "foo"},
   102  			{"PushTimeout", "4", "6", "f00"},
   103  		}
   104  
   105  		It("should successfully updated", func() {
   106  			for _, pref := range preferences {
   107  				helper.Cmd("odo", "preference", "set", pref.name, pref.value).ShouldPass()
   108  				value := helper.GetPreferenceValue(pref.name)
   109  				Expect(value).To(ContainSubstring(pref.value))
   110  
   111  				helper.Cmd("odo", "preference", "set", "-f", pref.name, pref.updateValue).ShouldPass()
   112  				value = helper.GetPreferenceValue(pref.name)
   113  				Expect(value).To(ContainSubstring(pref.updateValue))
   114  
   115  				helper.Cmd("odo", "preference", "unset", "-f", pref.name).ShouldPass()
   116  				value = helper.GetPreferenceValue(pref.name)
   117  				Expect(value).To(BeEmpty())
   118  			}
   119  			globalConfPath := os.Getenv("HOME")
   120  			os.RemoveAll(filepath.Join(globalConfPath, ".odo"))
   121  		})
   122  
   123  		It("should unsuccessfully update", func() {
   124  			for _, pref := range preferences {
   125  				// TODO: Remove this once we decide something on checking NamePrefix
   126  				if pref.name != "NamePrefix" {
   127  					helper.Cmd("odo", "preference", "set", "-f", pref.name, pref.invalidValue).ShouldFail()

   130  		})

   132  		It("should show json output", func() {
   133  			prefJSONOutput, err := helper.Unindented(helper.Cmd("odo", "preference", "view", "-o", "json").ShouldPass().Out())
