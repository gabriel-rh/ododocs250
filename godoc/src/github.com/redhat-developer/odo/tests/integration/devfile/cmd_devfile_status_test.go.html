    42  	testCombo := func(ingress bool, secure bool, name string) {
    43  		defer GinkgoRecover()
    44  		It("Verify that odo component status detects the URL status: "+name, func() {
    45  			openshift := os.Getenv("KUBERNETES") != "true"
    46  			if !ingress && !openshift {
    47  				Skip("Route-based URLs is an OpenShift only scenario")
    48  			}
    49  			urlHost := helper.RandString(12) + ".com"
    50  
    51  			urlParams := []string{"url", "create", "my-url", "--port", "3000"}
    52  			if secure {
    53  				urlParams = append(urlParams, "--secure")
    54  			}
    55  
    56  			if ingress {
    57  				urlParams = append(urlParams, "--ingress")
    58  				urlParams = append(urlParams, "--host", urlHost)
    59  			}
    60  
    61  			helper.Cmd("odo", urlParams...).ShouldPass()
    62  
    63  			helper.Cmd("odo", "push", "-o", "json", "--project", namespace).ShouldPass()
    64  
    65  			session := helper.CmdRunner("odo", "component", "status", "-o", "json", "--project", namespace, "--follow")
    66  
    67  			helper.WaitForOutputToContain("urlReachable", 180, 10, session)
    68  
    69  			stdoutContents := string(session.Out.Contents())
    70  
    71  			entries, err := utils.ParseMachineEventJSONLines(stdoutContents)
    72  			Expect(err).NotTo(HaveOccurred())
    73  
    74  			// Verify url status is present and correct
    75  			urlReachableEntry := utils.GetMostRecentEventOfType(machineoutput.TypeURLReachable, entries, true).(*machineoutput.URLReachable)
    76  
    77  			expectedKind := "ingress"
    78  			if !ingress || openshift {
    79  				expectedKind = "route"
    80  			}
    81  
    82  			Expect(urlReachableEntry.Kind).To(Equal(expectedKind))
    83  			Expect(urlReachableEntry.Reachable).To(Equal(!ingress || openshift)) // On non-openshift, the ingress URL is using a random hostname, so should not be resolveable
    84  			Expect(urlReachableEntry.Port).To(Equal(3000))
    85  			Expect(urlReachableEntry.Secure).To(Equal(secure))
    86  
    87  			utils.TerminateSession(session)
    88  
    89  		})

   106  			It("Verify that odo component status correctly reports supervisord status", func() {
   107  
   108  				session := helper.CmdRunner("odo", "component", "status", "-o", "json", "--project", namespace, "--follow")
   109  
   110  				helper.WaitForOutputToContain("supervisordStatus", 180, 10, session)
   111  
   112  				stdoutContents := string(session.Out.Contents())
   113  
   114  				entries, err := utils.ParseMachineEventJSONLines(stdoutContents)
   115  				Expect(err).NotTo(HaveOccurred())
   116  
   117  				// Verify supervisord status is present and correct
   118  				supervisordEntry := utils.GetMostRecentEventOfType(machineoutput.TypeSupervisordStatus, entries, true).(*machineoutput.SupervisordStatus)
   119  				count := 0
   120  				for _, status := range supervisordEntry.ProgramStatus {
   121  					if status.Program == string(common.DefaultDevfileRunCommand) {
   122  						Expect(status.Status).To(Equal("RUNNING"))
   123  					} else if status.Program == string(common.DefaultDevfileDebugCommand) {
   124  						Expect(status.Status).To(Equal("STOPPED"))
   125  					} else {
   126  						Fail(fmt.Sprintf("Unexpected program: %v", status.Program))
   127  					}
   128  					count++
   129  				}
   130  				Expect(count).To(Equal(2))
   131  
   132  				// Kill the node processes within the container, returns true when complete
   133  				Eventually(func() bool {
   134  					podName := cliRunner.GetRunningPodNameByComponent(cmpName, namespace)
   135  
   136  					contents := helper.GetCliRunner().Exec(podName, namespace, "--", "ps", "-ef")
   137  
   138  					pids := []string{}
   139  					for _, str := range strings.Split(contents, "\n") {
   140  
   141  						if strings.Contains(str, "node") || strings.Contains(str, "npm") {
   142  
   143  							fields := strings.Fields(str)
   144  							if len(fields) >= 2 {
   145  								pids = append(pids, fields[1])
   146  							}
   147  						}
   148  					}
   149  
   150  					for _, pid := range pids {
   151  						helper.GetCliRunner().Exec(podName, namespace, "--", "kill", "-9", pid)
   152  					}
   153  
   154  					// We expect (at least) 2 node processes
   155  					return len(pids) >= 2
   156  
   157  				}, 180, 10).Should(Equal(true))
   158  
   159  				// Wait for 'odo component status' to report that the programs are no longer RUNNING (EXITED or STOPPED)
   160  				Eventually(func() bool {
   161  					stdoutContents := string(session.Out.Contents())
   162  					entries, err := utils.ParseMachineEventJSONLines(stdoutContents)
   163  					if err != nil {
   164  						return false
   165  					}
   166  
   167  					supervisordStatus := utils.GetMostRecentEventOfType(machineoutput.TypeSupervisordStatus, entries, false).(*machineoutput.SupervisordStatus)
   168  					if supervisordStatus == nil {
   169  						return false
   170  					}
   171  
   172  					// All programs should be stopped, because we killed the node processes
   173  					for _, programStatus := range supervisordStatus.ProgramStatus {
   174  						if programStatus.Status == "RUNNING" {
   175  							return false
   176  						}
   177  					}
   178  
   179  					return true
   180  				}, 180, 10).Should(Equal(true))
   181  
   182  				utils.TerminateSession(session)
