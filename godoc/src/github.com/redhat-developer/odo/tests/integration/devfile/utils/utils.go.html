    30  // Specify OdoV2Watch for odo version 2(devfile)
    31  // platform is kube
    32  func OdoWatch(odoV2Watch OdoV2Watch, project, context, flag string, runner interface{}, platform string) {
    33  
    34  	// After the watch command has started (indicated via channel), simulate file system changes
    35  	startSimulationCh := make(chan bool)
    36  	go func() {
    37  		startMsg := <-startSimulationCh
    38  		if startMsg {
    39  			err := os.MkdirAll(filepath.Join(context, ".abc"), 0750)
    40  			Expect(err).To(BeNil())
    41  
    42  			err = os.MkdirAll(filepath.Join(context, "abcd"), 0750)
    43  			Expect(err).To(BeNil())
    44  
    45  			_, err = os.Create(filepath.Join(context, "a.txt"))
    46  			Expect(err).To(BeNil())
    47  
    48  			if odoV2Watch.SrcType == "openjdk" {
    49  				helper.ReplaceString(filepath.Join(context, "src", "main", "java", "MessageProducer.java"), "Hello", "Hello odo")
    50  			} else {
    51  				helper.ReplaceString(filepath.Join(context, "server.js"), "Hello", "Hello odo")
    52  			}
    53  
    54  		}
    55  	}()
    56  
    57  	success, err := helper.WatchNonRetCmdStdOut(
    58  		("odo watch " + flag + " --context " + context),
    59  		time.Duration(5)*time.Minute,
    60  		func(output string) bool {
    61  			// the test hangs up on the CI when the delay is set to 0
    62  			// so we only check if the start message was displayed correctly or not
    63  			if strings.Contains(flag, "delay 0") {
    64  				return true
    65  			}
    66  			// Returns true if the test has succeeded, false if not yet
    67  
    68  			stringsMatched := true
    69  
    70  			for _, stringToBeMatched := range odoV2Watch.StringsToBeMatched {
    71  				if !strings.Contains(output, stringToBeMatched) {
    72  					fmt.Fprintln(GinkgoWriter, "Missing string: ", stringToBeMatched)
    73  					stringsMatched = false
    74  				}
    75  			}
    76  
    77  			if stringsMatched {
    78  
    79  				// first push is successful
    80  				// now delete a folder and check if the deletion is propagated properly
    81  				// and the file is removed from the cluster
    82  				index := suffixarray.New([]byte(output))
    83  				offsets := index.Lookup([]byte(filepath.Join(context, "abcd")+" changed"), -1)
    84  
    85  				// the first occurrence of '<target-dir> changed' means the creation of it was pushed to the cluster
    86  				// and the first push was successful
    87  				if len(offsets) == 1 {
    88  					helper.DeleteDir(filepath.Join(context, "abcd"))
    89  				} else if len(offsets) > 1 {
    90  					// the occurrence of 'target-directory' more than once indicates that the deletion was propagated too
    91  					// Verify directory deleted from component pod
    92  					err := validateContainerExecListDir(odoV2Watch, runner, platform, project)
    93  					Expect(err).To(BeNil())
    94  					return true
    95  				}
    96  			}
    97  
    98  			return false
    99  		},
   100  		startSimulationCh,
   101  		func(output string) bool {
   102  			// Returns true to indicate the test should begin file system file change simulation
   103  			return strings.Contains(output, "Waiting for something to change")
   104  		})
   105  
   106  	Expect(success).To(Equal(true))
   107  	Expect(err).To(BeNil())
   108  }

   112  // After a successful push with watch, it tries to start a debug session
   113  func OdoWatchWithDebug(odoV2Watch OdoV2Watch, context, flag string) {
   114  
   115  	startSimulationCh := make(chan bool)
   116  	go func() {
   117  		startMsg := <-startSimulationCh
   118  		if startMsg {
   119  			helper.ReplaceString(filepath.Join(context, "server.js"), "Hello", "Hello odo")

   121  		}
   122  	}()
   123  
   124  	success, err := helper.WatchNonRetCmdStdOut(
   125  		("odo watch " + flag + " --context " + context),
   126  		time.Duration(5)*time.Minute,
   127  		func(output string) bool {
   128  			stringsMatched := true
   129  
   130  			for _, stringToBeMatched := range odoV2Watch.StringsToBeMatched {
   131  				if !strings.Contains(output, stringToBeMatched) {
   132  					stringsMatched = false
   133  				}
   134  			}
   135  
   136  			if stringsMatched {
   137  				httpPort, err := util.HTTPGetFreePort()
   138  				Expect(err).NotTo(HaveOccurred())
   139  				freePort := strconv.Itoa(httpPort)
   140  
   141  				stopChannel := make(chan bool)
   142  				go func() {
   143  					helper.Cmd("odo", "debug", "port-forward", "--local-port", freePort).WithTerminate(60*time.Second, stopChannel).ShouldRun()
   144  				}()
   145  
   146  				// 400 response expected because the endpoint expects a websocket request and we are doing a HTTP GET
   147  				// We are just using this to validate if nodejs agent is listening on the other side
   148  				helper.HttpWaitForWithStatus("http://localhost:"+freePort, "WebSockets request was expected", 12, 5, 400)
   149  				stopChannel <- true
   150  				return true
   151  			}
   152  
   153  			return false
   154  		},
   155  		startSimulationCh,
   156  		func(output string) bool {
   157  			return strings.Contains(output, "Waiting for something to change")
   158  		})
   159  
   160  	Expect(success).To(Equal(true))
   161  	Expect(err).To(BeNil())
   162  }

   214  func validateContainerExecListDir(odoV2Watch OdoV2Watch, runner interface{}, platform, project string) error {
   215  	var folderToCheck, podName string
   216  	cliRunner := runner.(helper.CliRunner)
   217  	switch platform {
   218  	case "kube":
   219  		folderToCheck = "/projects"
   220  		if odoV2Watch.FolderToCheck != "" {
   221  			folderToCheck = odoV2Watch.FolderToCheck
   222  		}
   223  		podName = cliRunner.GetRunningPodNameByComponent(odoV2Watch.CmpName, project)
   224  
   225  	default:
   226  		return fmt.Errorf("Platform %s is not supported", platform)
   227  	}
   228  
   229  	// check if contains a.txt, .abc && abcd is deleted
   230  	cliRunner.WaitForRunnerCmdOut([]string{"exec", podName, "--namespace", project,
   231  		"--", "ls", "-lai", folderToCheck}, 5, true, func(output string) bool {
   232  		return !(strings.Contains(output, "abcd")) && (strings.Contains(output, "a.txt")) && (strings.Contains(output, ".abc"))
   233  	})
   234  
   235  	return nil
   236  }

   251  func VerifyCatalogListComponent(output string, cmpName []string) error {
   252  	var data map[string]interface{}
   253  	listItems := []string{"items"}
   254  
   255  	if err := json.Unmarshal([]byte(output), &data); err != nil {
   256  		return err
   257  	}
   258  
   259  	for _, items := range listItems {
   260  		outputBytes, err := json.Marshal(data[items])
   261  		if err != nil {
   262  			return err
   263  		}
   264  		output = string(outputBytes)
   265  		helper.MatchAllInOutput(output, cmpName)
   266  	}
   267  	return nil
   268  }

   270  // VerifyContainerSyncEnv verifies the sync env in the container
   271  func VerifyContainerSyncEnv(podName, containerName, namespace, projectSourceValue, projectsRootValue string, cliRunner helper.CliRunner) {
   272  	envProjectsRoot, envProjectSource := "PROJECTS_ROOT", "PROJECT_SOURCE"
   273  	projectSourceMatched, projectsRootMatched := false, false
   274  
   275  	envNamesAndValues := cliRunner.GetContainerEnv(podName, "runtime", namespace)
   276  	envNamesAndValuesArr := strings.Fields(envNamesAndValues)
   277  
   278  	for _, envNamesAndValues := range envNamesAndValuesArr {
   279  		envNameAndValueArr := strings.Split(envNamesAndValues, ":")
   280  
   281  		if envNameAndValueArr[0] == envProjectSource && strings.Contains(envNameAndValueArr[1], projectSourceValue) {
   282  			projectSourceMatched = true
   283  		}
   284  
   285  		if envNameAndValueArr[0] == envProjectsRoot && strings.Contains(envNameAndValueArr[1], projectsRootValue) {
   286  			projectsRootMatched = true
   287  		}
   288  	}
   289  
   290  	Expect(projectSourceMatched).To(Equal(true))
   291  	Expect(projectsRootMatched).To(Equal(true))
   292  }
