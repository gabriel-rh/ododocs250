   170  func createComponent(componentName, appName, project, context string) {
   171  	helper.Cmd("odo", "create", "--project", project, componentName, "--context", context, "--app", appName, "--devfile", helper.GetExamplePath("source", "devfiles", "nodejs", "devfile.yaml")).ShouldPass()
   172  	helper.CopyExample(filepath.Join("source", "devfiles", "nodejs", "project"), context)
   173  	helper.Cmd("odo", "push", "--context", context).ShouldPass()
   174  }
