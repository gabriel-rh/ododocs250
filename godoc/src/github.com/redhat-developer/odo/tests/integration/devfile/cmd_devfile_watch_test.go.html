   112  			defer session.Kill()

   115  
   116  			// Create a new file A
   117  			fileAPath, _ := helper.CreateSimpleFile(commonVar.Context, "my-file-", ".txt")
   118  
   119  			// Wait for the new file to exist in the index
   120  			Eventually(func() bool {
   121  
   122  				newIndexAfterPush, readErr := util.ReadFileIndex(filepath.Join(commonVar.Context, ".odo", "odo-file-index.json"))
   123  				if readErr != nil {
   124  					fmt.Fprintln(GinkgoWriter, "New index not found or could not be read", readErr)
   125  					return false
   126  				}

   129  				if !exists {
   130  					fmt.Fprintln(GinkgoWriter, "path", fileAPath, "not found.", readErr)
   131  				}

   133  
   134  			}, 180, 10).Should(Equal(true))

   136  			// Delete file A and verify that it disappears from the index

   139  			Eventually(func() bool {
   140  
   141  				newIndexAfterPush, err := util.ReadFileIndex(filepath.Join(commonVar.Context, ".odo", "odo-file-index.json"))
