    16  // AnalyzePushConsoleOutput analyzes the output of 'odo push -o json' for the machine readable event push test above.
    17  func AnalyzePushConsoleOutput(pushConsoleOutput string) {
    18  
    19  	entries, err := ParseMachineEventJSONLines(pushConsoleOutput)
    20  	Expect(err).NotTo(HaveOccurred())
    21  
    22  	// Ensure we pass a sanity test on the minimum expected entries
    23  	if len(entries) < 4 {
    24  		Fail("Expected at least 4 entries, corresponding to command/action execution.")
    25  	}
    26  
    27  	// Ensure that all logText entries are wrapped inside commandExecutionBegin and commandExecutionComplete entries (e.g. no floating logTexts)
    28  	insideCommandExecution := false
    29  	for _, entry := range entries {
    30  
    31  		if entry.GetType() == machineoutput.TypeDevFileCommandExecutionBegin {
    32  			insideCommandExecution = true
    33  		}
    34  
    35  		if entry.GetType() == machineoutput.TypeDevFileCommandExecutionComplete {
    36  			insideCommandExecution = false
    37  		}
    38  
    39  		if entry.GetType() == machineoutput.TypeLogText {
    40  			Expect(insideCommandExecution).To(Equal(true))
    41  		}
    42  
    43  	}
    44  
    45  	// Ensure that the log output has the given structure:
    46  	// - look for the expected devbuild events, then look for the expected devrun events.
    47  	expectedEventOrder := []struct {
    48  		entryType   machineoutput.MachineEventLogEntryType
    49  		commandName string
    50  	}{
    51  		// first the devbuild command (and its action) should run
    52  		{
    53  			machineoutput.TypeDevFileCommandExecutionBegin,
    54  			"devbuild",
    55  		},
    56  		{
    57  			// at least one logged line of text
    58  			machineoutput.TypeLogText,
    59  			"",
    60  		},
    61  		{
    62  			machineoutput.TypeDevFileCommandExecutionComplete,
    63  			"devbuild",
    64  		},
    65  		// next the devbuild command (and its action) should run
    66  		{
    67  			machineoutput.TypeDevFileCommandExecutionBegin,
    68  			"devrun",
    69  		},
    70  		{
    71  			// at least one logged line of text
    72  			machineoutput.TypeLogText,
    73  			"",
    74  		},
    75  		{
    76  			machineoutput.TypeDevFileCommandExecutionComplete,
    77  			"devrun",
    78  		},
    79  	}
    80  	currIndex := -1
    81  	for _, nextEventOrder := range expectedEventOrder {
    82  		entry, newIndex := findNextEntryByType(currIndex, nextEventOrder.entryType, entries)
    83  		Expect(entry).NotTo(BeNil())
    84  		Expect(newIndex).To(BeNumerically(">=", 0))
    85  		Expect(newIndex).To(BeNumerically(">", currIndex)) // monotonically increasing index

    89  		Expect(commandName).To(Equal(nextEventOrder.commandName))

    91  		currIndex = newIndex
    92  	}

    96  // ParseMachineEventJSONLines parses console output into machine event log entries
    97  func ParseMachineEventJSONLines(consoleOutput string) ([]machineoutput.MachineEventLogEntry, error) {
    98  
    99  	lines := strings.Split(strings.Replace(consoleOutput, "\r\n", "\n", -1), "\n")
   100  
   101  	entries := []machineoutput.MachineEventLogEntry{}
   102  
   103  	// Ensure that all lines can be correctly parsed into their expected JSON structure
   104  	for _, line := range lines {
   105  
   106  		if !strings.HasPrefix(line, "{") {
   107  			continue
   108  		}
   109  
   110  		lineWrapper := machineoutput.MachineEventWrapper{}
   111  
   112  		err := json.Unmarshal([]byte(line), &lineWrapper)
   113  		if err != nil {
   114  			return nil, err
   115  		}
   116  
   117  		entry, err := lineWrapper.GetEntry()
   118  		if err != nil {
   119  			return nil, err
   120  		}
   121  
   122  		entries = append(entries, entry)
   123  
   124  	}
   125  
   126  	return entries, nil
   127  }

   129  // findNextEntryByType locates the next entry of a given type within a slice. Currently used for test purposes only.
   130  func findNextEntryByType(initialIndex int, typeToFind machineoutput.MachineEventLogEntryType, entries []machineoutput.MachineEventLogEntry) (machineoutput.MachineEventLogEntry, int) {
   131  
   132  	for index, entry := range entries {
   133  		if index < initialIndex {
   134  			continue
   135  		}
   136  
   137  		if entry.GetType() == typeToFind {
   138  			return entry, index
   139  		}
   140  	}
   141  
   142  	return nil, -1
