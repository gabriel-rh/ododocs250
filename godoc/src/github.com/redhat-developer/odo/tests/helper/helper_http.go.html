    16  // HttpWaitForWithStatus periodically (every interval) calls GET to given url
    17  // ends when result response contains match string and status code, or after the maxRetry
    18  func HttpWaitForWithStatus(url string, match string, maxRetry int, interval int, expectedCode int) {
    19  	fmt.Fprintf(GinkgoWriter, "Checking %s, for %s\n", url, match)
    20  
    21  	var body []byte
    22  
    23  	for i := 0; i < maxRetry; i++ {
    24  		fmt.Fprintf(GinkgoWriter, "try %d of %d\n", i, maxRetry)
    25  
    26  		// #nosec
    27  		// gosec:G107, G402 -> This is safe since it's just used for testing.
    28  		transporter := &http.Transport{
    29  			TLSClientConfig: &tls.Config{InsecureSkipVerify: true},
    30  		}
    31  		client := &http.Client{Transport: transporter}
    32  		resp, err := client.Get(url)
    33  		if err != nil {
    34  			// we log the error and sleep again because this could mean the component is not up yet
    35  			fmt.Fprintln(GinkgoWriter, "error while requesting:", err.Error())
    36  			time.Sleep(time.Duration(interval) * time.Second)
    37  			continue
    38  		}
    39  		defer resp.Body.Close()
    40  		if resp.StatusCode == expectedCode {
    41  			body, _ = ioutil.ReadAll(resp.Body)
    42  			if strings.Contains(string(body), match) {
    43  				return
    44  			}
    45  
    46  		}
    47  		time.Sleep(time.Duration(interval) * time.Second)
    48  	}
    49  	fmt.Fprintf(GinkgoWriter, "Last output from %s: %s\n", url, string(body))
    50  	Fail(fmt.Sprintf("Failed after %d retries. Content in %s doesn't include '%s'.", maxRetry, url, match))
    51  }
