    85  func StatFileInPodContainer(runner CliRunner, cmpName, containerName, appName, project, filepath string) string {
    86  	podName := runner.GetRunningPodNameByComponent(cmpName, project)
    87  	var result string
    88  	runner.CheckCmdOpInRemoteDevfilePod(
    89  		podName,
    90  		containerName,
    91  		project,
    92  		[]string{"stat", filepath},
    93  		func(cmdOp string, err error) bool {
    94  			//strip out access info as
    95  			// 1. Touching a file (such as running it in a script) modifies access times. This gives wrong value on mounts without noatime
    96  			// 2. We are not interested in Access info anyway.
    97  			re := regexp.MustCompile("(?m)[\r\n]+^.*Access.*$")
    98  			result = re.ReplaceAllString(cmdOp, "")
    99  			return true
   100  		},
   101  	)
   102  	return result
   103  }

   138  // CheckCmdOpInRemoteCmpPod runs the provided command on remote component pod and returns the return value of command output handler function passed to it
   139  func (oc OcRunner) CheckCmdOpInRemoteCmpPod(cmpName string, appName string, prjName string, cmd []string, checkOp func(cmdOp string, err error) bool) bool {
   140  	cmpDCName := fmt.Sprintf("%s-%s", cmpName, appName)
   141  	outPodName := Cmd(oc.path, "get", "pods", "--namespace", prjName,
   142  		"--selector=deploymentconfig="+cmpDCName,
   143  		"-o", "jsonpath='{.items[0].metadata.name}'").ShouldPass().Out()
   144  	podName := strings.Replace(outPodName, "'", "", -1)
   145  	session := CmdRunner(oc.path, append([]string{"exec", podName, "--namespace", prjName,
   146  		"-c", cmpDCName, "--"}, cmd...)...)
   147  	stdOut := string(session.Wait().Out.Contents())
   148  	stdErr := string(session.Wait().Err.Contents())
   149  	if stdErr != "" && session.ExitCode() != 0 {
   150  		return checkOp(stdOut, fmt.Errorf("cmd %s failed with error %s on pod %s", cmd, stdErr, podName))
   151  	}
   152  	return checkOp(stdOut, nil)
   153  }

   167  		return checkOp(stdOut, fmt.Errorf("cmd %s failed with error %s on pod %s", cmd, stdErr, podName))
   168  	}
   169  	return checkOp(stdOut, nil)

   445  // WaitForRunnerCmdOut runs "oc" command until it gets
   446  // the expected output.
   447  // It accepts 4 arguments
   448  // args (arguments to the program)
   449  // timeout (the time to wait for the output)
   450  // errOnFail (flag to set if test should fail if command fails)
   451  // check (function with output check logic)
   452  // It times out if the command doesn't fetch the
   453  // expected output  within the timeout period.
   454  func (oc OcRunner) WaitForRunnerCmdOut(args []string, timeout int, errOnFail bool, check func(output string) bool, includeStdErr ...bool) bool {
   455  	pingTimeout := time.After(time.Duration(timeout) * time.Minute)
   456  	// this is a test package so time.Tick() is acceptable
   457  	// nolint
   458  	tick := time.Tick(time.Second)
   459  	for {
   460  		select {
   461  		case <-pingTimeout:
   462  			Fail(fmt.Sprintf("Timeout after %v minutes", timeout))
   463  
   464  		case <-tick:
   465  			session := CmdRunner(oc.path, args...)
   466  			if errOnFail {
   467  				Eventually(session).Should(gexec.Exit(0), runningCmd(session.Command))
   468  			} else {
   469  				Eventually(session).Should(gexec.Exit(), runningCmd(session.Command))
   470  			}
   471  			session.Wait()
   472  			output := string(session.Out.Contents())
   473  
   474  			if len(includeStdErr) > 0 && includeStdErr[0] {
   475  				output += "\n"
   476  				output += string(session.Err.Contents())
   477  			}
   478  			if check(strings.TrimSpace(string(output))) {
   479  				return true
   480  			}
   481  		}
   482  	}
   483  }
