    30  func Cmd(program string, args ...string) *CmdWrapper {
    31  	prefix := fmt.Sprintf("[%s] ", filepath.Base(program))
    32  	prefixWriter := gexec.NewPrefixedWriter(prefix, GinkgoWriter)
    33  	command := exec.Command(program, args...)
    34  	return &CmdWrapper{
    35  		Cmd:     command,
    36  		program: program,
    37  		args:    args,
    38  		writer:  prefixWriter,
    39  	}

    42  func (cw *CmdWrapper) Runner() *CmdWrapper {
    43  	fmt.Fprintln(GinkgoWriter, runningCmd(cw.Cmd))
    44  	cw.session, cw.err = gexec.Start(cw.Cmd, cw.writer, cw.writer)
    45  	timeout := time.After(cw.timeout)
    46  	if cw.timeout > 0 {
    47  		select {
    48  		case <-cw.stopChan:
    49  			if cw.session != nil {
    50  				if runtime.GOOS == "windows" {
    51  					cw.session.Kill()
    52  				} else {
    53  					cw.session.Terminate()
    54  				}
    55  			}
    56  		case <-timeout:
    57  			if cw.session != nil {
    58  				if runtime.GOOS == "windows" {
    59  					cw.session.Kill()
    60  				} else {
    61  					cw.session.Terminate()
    62  				}
    63  			}
    64  		}
    65  	} else if cw.maxRetry > 0 {
    66  		cw.session.Wait()
    67  		// cw.pass is to check if it is used with ShouldPass Or ShouldFail
    68  		if !cw.pass {
    69  			// we retry on success because the user has set “ShouldFail” as true
    70  			// if exit code is 0 which means the program succeeded and hence we retry
    71  			if cw.session.ExitCode() == 0 {
    72  				time.Sleep(time.Duration(cw.intervalSeconds) * time.Second)
    73  				cw.maxRetry = cw.maxRetry - 1
    74  				cw.Runner()
    75  			}
    76  			return cw
    77  		} else {
    78  			// if exit code is not 0 which means the program Failed and hence we retry
    79  			if cw.session.ExitCode() != 0 {
    80  				time.Sleep(time.Duration(cw.intervalSeconds) * time.Second)
    81  				cw.maxRetry = cw.maxRetry - 1
    82  				cw.Runner()
    83  			}
    84  			return cw
    85  		}
    86  	}
    87  	return cw
    88  }
