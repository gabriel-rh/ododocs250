    25  // DeleteDir deletes the specified path; due to Windows behaviour (for example https://github.com/redhat-developer/odo/issues/3371)
    26  // where Windows temporarily holds a lock on files and folders, we keep trying to delete until the operation passes (or it expires)
    27  func DeleteDir(dir string) {
    28  	attempts := 0
    29  
    30  	errorReportedAtLeastOnce := false
    31  
    32  	err := RunWithExponentialBackoff(func() error {
    33  		attempts++
    34  
    35  		fmt.Fprintf(GinkgoWriter, "Deleting dir: %s\n", dir)
    36  		err := os.RemoveAll(dir)
    37  		if err == nil {
    38  			return nil
    39  		}
    40  
    41  		errorReportedAtLeastOnce = true
    42  		fmt.Fprintf(GinkgoWriter, "Unable to delete %s on attempt #%d, trying again...\n", dir, attempts)
    43  
    44  		return err
    45  	}, 16, time.Duration(2)*time.Minute)
    46  	Expect(err).NotTo(HaveOccurred())
    47  
    48  	if errorReportedAtLeastOnce {
    49  		fmt.Fprintf(GinkgoWriter, "Successfully deleted %s after #%d attempts\n", dir, attempts)
    50  	}
    51  }

    53  // RunWithExponentialBackoff keeps trying to run 'fxn' until it no longer returns an error; if the function never succeeded,
    54  // then the most recent error is returned.
    55  func RunWithExponentialBackoff(fxn func() error, maxDelayInSeconds int, expireDuration time.Duration) error {
    56  	expireTime := time.Now().Add(expireDuration)
    57  	delayInSeconds := 1
    58  
    59  	var err error
    60  
    61  	for {
    62  
    63  		err = fxn()
    64  
    65  		if err == nil || time.Now().After(expireTime) {
    66  			break
    67  		}
    68  
    69  		delayInSeconds *= 2 // exponential backoff
    70  		if delayInSeconds > maxDelayInSeconds {
    71  			delayInSeconds = maxDelayInSeconds
    72  		}
    73  		time.Sleep(time.Duration(delayInSeconds) * time.Second)
    74  
    75  	}
    76  	return err
    77  
    78  }

   202  // copyDir copy one directory to the other
   203  // this function is called recursively info should start as os.Stat(src)
   204  func copyDir(src string, dst string, info os.FileInfo) error {
   205  
   206  	if info.IsDir() {
   207  		files, err := ioutil.ReadDir(src)
   208  		if err != nil {
   209  			return err
   210  		}
   211  
   212  		for _, file := range files {
   213  			dsrt := filepath.Join(src, file.Name())
   214  			ddst := filepath.Join(dst, file.Name())
   215  			if err := copyDir(dsrt, ddst, file); err != nil {
   216  				return err
   217  			}
   218  		}
   219  		return nil
   220  	}
   221  
   222  	if err := os.MkdirAll(filepath.Dir(dst), os.ModePerm); err != nil {
   223  		return err
   224  	}
   225  
   226  	return util.CopyFile(src, dst, info)
   227  }

   288  // ReplaceDevfileField replaces the value of a given field in a specified
   289  // devfile.
   290  // Currently only the first match of the field is replaced. i.e if the
   291  // field is 'type' and there are two types throughout the devfile, only one
   292  // is replaced with the newValue
   293  func ReplaceDevfileField(devfileLocation, field, newValue string) error {
   294  	file, err := ioutil.ReadFile(devfileLocation)
   295  	if err != nil {
   296  		return err
   297  	}
   298  	lines := strings.Split(string(file), "\n")
   299  	for i, line := range lines {
   300  		if strings.Contains(line, field) {
   301  			lineSplit := strings.SplitN(lines[i], ":", 2)
   302  			lineSplit[1] = newValue
   303  			lines[i] = strings.Join(lineSplit, ": ")
   304  			break
   305  		}
   306  	}
   307  	output := strings.Join(lines, "\n")
   308  	err = ioutil.WriteFile(devfileLocation, []byte(output), 0600)
   309  	if err != nil {
   310  		return err
   311  	}
   312  	return nil
   313  }
