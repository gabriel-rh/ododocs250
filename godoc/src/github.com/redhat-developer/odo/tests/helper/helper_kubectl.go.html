    62  // CheckCmdOpInRemoteDevfilePod runs the provided command on remote component pod and returns the return value of command output handler function passed to it
    63  func (kubectl KubectlRunner) CheckCmdOpInRemoteDevfilePod(podName string, containerName string, prjName string, cmd []string, checkOp func(cmdOp string, err error) bool) bool {
    64  	var execOptions []string
    65  	execOptions = []string{"exec", podName, "--namespace", prjName, "--"}
    66  	if containerName != "" {
    67  		execOptions = []string{"exec", podName, "-c", containerName, "--namespace", prjName, "--"}
    68  	}
    69  	args := append(execOptions, cmd...)
    70  	session := CmdRunner(kubectl.path, args...)
    71  	stdOut := string(session.Wait().Out.Contents())
    72  	stdErr := string(session.Wait().Err.Contents())
    73  	if stdErr != "" && session.ExitCode() != 0 {
    74  		return checkOp(stdOut, fmt.Errorf("cmd %s failed with error %s on pod %s", cmd, stdErr, podName))
    75  	}
    76  	return checkOp(stdOut, nil)
    77  }

   190  func (kubectl KubectlRunner) GetEnvsDevFileDeployment(componentName, appName, projectName string) map[string]string {
   191  	var mapOutput = make(map[string]string)
   192  	selector := fmt.Sprintf("--selector=%s=%s,%s=%s", labels.ComponentLabel, componentName, applabels.ApplicationLabel, appName)
   193  	output := Cmd(kubectl.path, "get", "deployment", selector, "--namespace", projectName,
   194  		"-o", "jsonpath='{range .items[0].spec.template.spec.containers[0].env[*]}{.name}:{.value}{\"\\n\"}{end}'").ShouldPass().Out()
   195  
   196  	for _, line := range strings.Split(output, "\n") {
   197  		line = strings.TrimPrefix(line, "'")
   198  		splits := strings.Split(line, ":")
   199  		name := splits[0]
   200  		value := strings.Join(splits[1:], ":")
   201  		mapOutput[name] = value
   202  	}
   203  	return mapOutput
   204  }

   235  // VerifyResourceToBeDeleted verifies if a resource if deleted, or if not, if it is marked for deletion
   236  func (kubectl KubectlRunner) VerifyResourceToBeDeleted(ri ResourceInfo) {
   237  	deletedOrMarkedToDelete := func() bool {
   238  		session := CmdRunner(kubectl.path, "get", ri.ResourceType, ri.ResourceName, "--namespace", ri.Namespace, "-o", "jsonpath='{.metadata.deletionTimestamp}'")
   239  		exit := session.Wait().ExitCode()

   241  			// resources does not exist

   279  // WaitForCmdOut runs "kubectl" command until it gets
   280  // the expected output.
   281  // It accepts 4 arguments
   282  // args (arguments to the program)
   283  // timeout (the time to wait for the output)
   284  // errOnFail (flag to set if test should fail if command fails)
   285  // check (function with output check logic)
   286  // It times out if the command doesn't fetch the
   287  // expected output  within the timeout period.
   288  func (kubectl KubectlRunner) WaitForRunnerCmdOut(args []string, timeout int, errOnFail bool, check func(output string) bool, includeStdErr ...bool) bool {
   289  	pingTimeout := time.After(time.Duration(timeout) * time.Minute)
   290  	// this is a test package so time.Tick() is acceptable
   291  	// nolint
   292  	tick := time.Tick(time.Second)
   293  	for {
   294  		select {
   295  		case <-pingTimeout:
   296  			Fail(fmt.Sprintf("Timeout after %v minutes", timeout))
   297  
   298  		case <-tick:
   299  			session := CmdRunner(kubectl.path, args...)
   300  			if errOnFail {
   301  				Eventually(session).Should(gexec.Exit(0), runningCmd(session.Command))
   302  			} else {
   303  				Eventually(session).Should(gexec.Exit(), runningCmd(session.Command))
   304  			}
   305  			session.Wait()
   306  			output := string(session.Out.Contents())
   307  
   308  			if len(includeStdErr) > 0 && includeStdErr[0] {
   309  				output += "\n"
   310  				output += string(session.Err.Contents())
   311  			}
   312  			if check(strings.TrimSpace(string(output))) {
   313  				return true
   314  			}
   315  		}
   316  	}
   317  }
