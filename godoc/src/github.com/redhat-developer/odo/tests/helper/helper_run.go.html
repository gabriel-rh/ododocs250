    22  func CmdRunner(program string, args ...string) *gexec.Session {
    23  	//prefix ginkgo verbose output with program name
    24  	prefix := fmt.Sprintf("[%s] ", filepath.Base(program))
    25  	prefixWriter := gexec.NewPrefixedWriter(prefix, GinkgoWriter)
    26  	command := exec.Command(program, args...)
    27  	fmt.Fprintln(GinkgoWriter, runningCmd(command))
    28  	session, err := gexec.Start(command, prefixWriter, prefixWriter)
    29  	Expect(err).NotTo(HaveOccurred())
    30  	return session

    43  // WaitAndCheckForTerminatingState waits for the given interval
    44  // and checks if the given resource type has been deleted on the cluster or is in the terminating state
    45  // path is the path to the program's binary
    46  func WaitAndCheckForTerminatingState(path, resourceType, namespace string, timeoutMinutes int) bool {
    47  	pingTimeout := time.After(time.Duration(timeoutMinutes) * time.Minute)
    48  	// this is a test package so time.Tick() is acceptable
    49  	// nolint
    50  	tick := time.Tick(time.Second)
    51  	for {
    52  		select {
    53  		case <-pingTimeout:
    54  			Fail(fmt.Sprintf("Timeout after %d minutes", timeoutMinutes))
    55  
    56  		case <-tick:
    57  			session := CmdRunner(path, "get", resourceType, "--namespace", namespace)
    58  			Eventually(session).Should(gexec.Exit(0))
    59  			// https://github.com/kubernetes/kubectl/issues/847
    60  			outputStdErr := string(session.Wait().Err.Contents())
    61  			outputStdOut := string(session.Wait().Out.Contents())
    62  
    63  			// if the resource gets deleted before the check, we won't get the `terminating` state output
    64  			// thus we also check and exit when the resource has been deleted on the cluster.
    65  			if strings.Contains(strings.ToLower(outputStdErr), "no resources found") || strings.Contains(strings.ToLower(outputStdOut), "terminating") {
    66  				return true
    67  			}
    68  		}
    69  	}
    70  }

    72  // GetAnnotationsDeployment gets the annotations from the deployment
    73  // belonging to the given component, app and project
    74  func GetAnnotationsDeployment(path, componentName, appName, projectName string) map[string]string {
    75  	var mapOutput = make(map[string]string)
    76  
    77  	selector := fmt.Sprintf("--selector=%s=%s,%s=%s", labels.ComponentLabel, componentName, applabels.ApplicationLabel, appName)
    78  	output := Cmd(path, "get", "deployment", selector, "--namespace", projectName,
    79  		"-o", "go-template='{{ range $k, $v := (index .items 0).metadata.annotations}}{{$k}}:{{$v}}{{\"\\n\"}}{{end}}'").ShouldPass().Out()
    80  
    81  	for _, line := range strings.Split(output, "\n") {
    82  		line = strings.TrimPrefix(line, "'")
    83  		splits := strings.Split(line, ":")
    84  		if len(splits) < 2 {
    85  			continue
    86  		}
    87  		name := splits[0]
    88  		value := strings.Join(splits[1:], ":")
    89  		mapOutput[name] = value
    90  	}
    91  	return mapOutput
    92  }
