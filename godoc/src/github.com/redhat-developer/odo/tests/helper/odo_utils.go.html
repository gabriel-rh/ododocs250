    21  // GetConfigValueWithContext returns a local config value of given key and contextdir or
    22  // returns an empty string if value is not set
    23  func GetConfigValueWithContext(key string, context string) string {
    24  	var stdOut string
    25  	if context != "" {
    26  		stdOut = Cmd("odo", "config", "view", "--context", context).ShouldPass().Out()
    27  	} else {
    28  		stdOut = Cmd("odo", "config", "view").ShouldPass().Out()
    29  	}
    30  	re := regexp.MustCompile(key + `.+`)
    31  	odoConfigKeyValue := re.FindString(stdOut)
    32  	if odoConfigKeyValue == "" {
    33  		return fmt.Sprintf("%s not found", key)
    34  	}
    35  	trimKeyValue := strings.TrimSpace(odoConfigKeyValue)
    36  	if strings.Compare(key, trimKeyValue) != 0 {
    37  		return strings.TrimSpace(strings.SplitN(trimKeyValue, " ", 2)[1])
    38  	}
    39  	return ""
    40  }

    92  // DetermineRouteURLs takes context path as argument and returns the URLs
    93  // where the current component exposes it's service, these URLs can
    94  // then be used in order to interact with the deployed service running in Openshift
    95  func DetermineRouteURLs(context string) []string {
    96  	var stdOut string
    97  	if context != "" {
    98  		stdOut = Cmd("odo", "url", "list", "--context", context).ShouldPass().Out()
    99  	} else {
   100  		stdOut = Cmd("odo", "url", "list").ShouldPass().Out()
   101  	}
   102  	reURL := regexp.MustCompile(`\s+http(s?)://.\S+`)
   103  	odoURLs := reURL.FindAllString(stdOut, -1)
   104  	for i := range odoURLs {
   105  		odoURLs[i] = strings.TrimSpace(odoURLs[i])
   106  	}
   107  	return odoURLs
   108  }
