    32  // WaitForCmdOut runs a command until it gets
    33  // the expected output.
    34  // It accepts 5 arguments, program (program to be run)
    35  // args (arguments to the program)
    36  // timeout (the time to wait for the output)
    37  // errOnFail (flag to set if test should fail if command fails)
    38  // check (function with output check logic)
    39  // It times out if the command doesn't fetch the
    40  // expected output  within the timeout period.
    41  func WaitForCmdOut(program string, args []string, timeout int, errOnFail bool, check func(output string) bool, includeStdErr ...bool) bool {
    42  	pingTimeout := time.After(time.Duration(timeout) * time.Minute)
    43  	// this is a test package so time.Tick() is acceptable
    44  	// nolint
    45  	tick := time.Tick(time.Second)
    46  	for {
    47  		select {
    48  		case <-pingTimeout:
    49  			Fail(fmt.Sprintf("Timeout after %v minutes", timeout))
    50  
    51  		case <-tick:
    52  			session := CmdRunner(program, args...)
    53  			if errOnFail {
    54  				Eventually(session).Should(gexec.Exit(0), runningCmd(session.Command))
    55  			} else {
    56  				Eventually(session).Should(gexec.Exit(), runningCmd(session.Command))
    57  			}
    58  			session.Wait()
    59  			output := string(session.Out.Contents())
    60  
    61  			if len(includeStdErr) > 0 && includeStdErr[0] {
    62  				output += "\n"
    63  				output += string(session.Err.Contents())
    64  			}
    65  			if check(strings.TrimSpace(string(output))) {
    66  				return true
    67  			}
    68  		}
    69  	}
    70  }

   113  // WatchNonRetCmdStdOut runs an 'odo watch' command and stores the process' stdout output into buffer.
   114  // - startIndicatorFunc should check stdout output and return true when simulation is ready to begin (for example, buffer contains "Waiting for something to change")
   115  // - startSimulationCh will be sent a 'true' when startIndicationFunc first returns true, at which point files/directories should be created by associated goroutine
   116  // - success function is passed stdout buffer, and should return if the test conditions have passes
   117  func WatchNonRetCmdStdOut(cmdStr string, timeout time.Duration, success func(output string) bool, startSimulationCh chan bool, startIndicatorFunc func(output string) bool) (bool, error) {
   118  	var cmd *exec.Cmd
   119  	var buf bytes.Buffer
   120  	var errBuf bytes.Buffer
   121  
   122  	cmdStrParts := strings.Fields(cmdStr)
   123  
   124  	fmt.Fprintln(GinkgoWriter, "Running command: ", cmdStrParts)
   125  
   126  	cmd = exec.Command(cmdStrParts[0], cmdStrParts[1:]...)
   127  
   128  	cmd.Stdout = &buf
   129  	cmd.Stderr = &errBuf
   130  
   131  	ticker := time.NewTicker(10 * time.Second)
   132  	defer ticker.Stop()
   133  
   134  	timeoutCh := make(chan bool)
   135  	go func() {
   136  		time.Sleep(timeout)
   137  		timeoutCh <- true
   138  	}()
   139  
   140  	if err := cmd.Start(); err != nil {
   141  		return false, err
   142  	}
   143  	startedFileModification := false
   144  	for {
   145  		select {
   146  		case <-timeoutCh:
   147  			if buf.String() != "" {
   148  				_, err := fmt.Fprintln(GinkgoWriter, "Output from stdout ["+cmdStr+"]:")
   149  				Expect(err).To(BeNil())
   150  				_, err = fmt.Fprintln(GinkgoWriter, buf.String())
   151  				Expect(err).To(BeNil())
   152  			}
   153  			errBufStr := errBuf.String()
   154  			if errBufStr != "" {
   155  				_, err := fmt.Fprintln(GinkgoWriter, "Output from stderr ["+cmdStr+"]:")
   156  				Expect(err).To(BeNil())
   157  				_, err = fmt.Fprintln(GinkgoWriter, errBufStr)
   158  				Expect(err).To(BeNil())
   159  			}
   160  			Fail(fmt.Sprintf("Timeout after %.2f minutes", timeout.Minutes()))
   161  		case <-ticker.C: // Every 10 seconds...
   162  
   163  			// If we have not yet begun file modification, query the parameter function to see if we should, do so if true
   164  			if !startedFileModification && startIndicatorFunc(buf.String()) {
   165  				startedFileModification = true
   166  				startSimulationCh <- true
   167  			}
   168  			// Call success(...) to determine if stdout contains expected text, exit if true
   169  			if success(buf.String()) {
   170  				if err := cmd.Process.Kill(); err != nil {
   171  					return true, err
   172  				}
   173  				return true, nil
   174  			}
   175  		}
   176  	}
   177  }

   179  // RunCmdWithMatchOutputFromBuffer starts the command, and command stdout is attached to buffer.
   180  // we read data from buffer line by line, and if expected string is matched it returns true
   181  // It is different from WaitforCmdOut which gives stdout in one go using session.Out.Contents()
   182  // for commands like odo log -f which streams continuous data and does not terminate by their own
   183  // we need to read the stream data from buffer.
   184  func RunCmdWithMatchOutputFromBuffer(timeoutAfter time.Duration, matchString, program string, args ...string) (bool, error) {
   185  	var buf, errBuf bytes.Buffer
   186  
   187  	command := exec.Command(program, args...)
   188  	command.Stdout = &buf
   189  	command.Stderr = &errBuf
   190  
   191  	timeoutCh := time.After(timeoutAfter)
   192  	matchOutputCh := make(chan bool)
   193  	errorCh := make(chan error)
   194  
   195  	_, err := fmt.Fprintln(GinkgoWriter, runningCmd(command))
   196  	if err != nil {
   197  		return false, err
   198  	}
   199  
   200  	err = command.Start()
   201  	if err != nil {
   202  		return false, err
   203  	}
   204  
   205  	// go routine which is reading data from buffer until expected string matched
   206  	go func() {
   207  		for {
   208  			line, err := buf.ReadString('\n')
   209  			if err != nil && err != io.EOF {
   210  				errorCh <- err
   211  			}
   212  			if len(line) > 0 {
   213  				_, err = fmt.Fprintln(GinkgoWriter, line)
   214  				if err != nil {
   215  					errorCh <- err
   216  				}
   217  				if strings.Contains(line, matchString) {
   218  					matchOutputCh <- true
   219  				}
   220  			}
   221  		}
   222  	}()
   223  
   224  	for {
   225  		select {
   226  		case <-timeoutCh:
   227  			fmt.Fprintln(GinkgoWriter, errBuf.String())
   228  			return false, errors.New("Timeout waiting for the condition")
   229  		case <-matchOutputCh:
   230  			return true, nil
   231  		case <-errorCh:
   232  			fmt.Fprintln(GinkgoWriter, errBuf.String())
   233  			return false, <-errorCh
   234  		}
   235  	}
   236  
   237  }

   287  // CommonBeforeEach is common function runs before every test Spec (It)
   288  // returns CommonVar values that are used within the test script
   289  func CommonBeforeEach() CommonVar {
   290  	SetDefaultEventuallyTimeout(10 * time.Minute)

   292  
   293  	commonVar := CommonVar{}

   297  	LocalKubeconfigSet(commonVar.Context)

   299  	commonVar.OriginalWorkingDirectory = Getwd()
   300  	os.Setenv("GLOBALODOCONFIG", filepath.Join(commonVar.Context, "preference.yaml"))
   301  	// Set ConsentTelemetry to false so that it does not prompt to set a preference value

   303  	err := cfg.SetConfiguration(preference.ConsentTelemetrySetting, "false")

   305  	SetDefaultDevfileRegistryAsStaging()

   309  // CommonAfterEach is common function that cleans up after every test Spec (It)
   310  func CommonAfterEach(commonVar CommonVar) {
   311  	// delete the random project/namespace created in CommonBeforeEach
   312  	commonVar.CliRunner.DeleteNamespaceProject(commonVar.Project)
   313

   325  // GjsonMatcher validates if []results from gjson.GetMany match the expected values for each json path requested
   326  // Values is an array of results returned by the gjson.GetMany function
   327  // Expected is an array of strings, defines the expected values for each of the paths requested in the gjson.GetMany function
   328  // For documentation about gjson see https://github.com/tidwall/gjson#get-multiple-values-at-once
   329  func GjsonMatcher(values []gjson.Result, expected []string) bool {
   330  	matched := 0
   331  	for i, v := range values {
   332  		if strings.Contains(v.String(), expected[i]) {
   333  			matched++
   334  		}
   335  	}
   336  	numVars := len(expected)
   337  	return matched == numVars
   338  }
