    14  // following command will run tests in Describe section below in parallel (in 2 nodes)
    15  // ginkgo -nodes=2 -focus="Example of a clean test" slowSpecThreshold=120 -randomizeAllSpecs  tests/e2e/
    16  var _ = Describe("Example of a clean test", func() {
    17  	// new clean project and context for each test
    18  	var project string
    19  	var context string
    20  
    21  	// current directory and component name (before any test runs) so that it can be restored  after all testing is done
    22  	var originalDir string
    23  	var cmpName string
    24  
    25  	BeforeEach(func() {
    26  		// Set default timeout for Eventually assertions
    27  		// commands like odo push, might take a long time
    28  		SetDefaultEventuallyTimeout(10 * time.Minute)
    29  		context = helper.CreateNewContext()
    30  		os.Setenv("GLOBALODOCONFIG", filepath.Join(context, "preference.yaml"))
    31  		project = helper.CreateRandProject()
    32  
    33  		// we will be testing components that are created from the current directory
    34  		// switch to the clean context dir before each test
    35  		originalDir = helper.Getwd()
    36  		helper.Chdir(context)
    37  
    38  	})
    39  
    40  	AfterEach(func() {
    41  		helper.DeleteProject(project)
    42  		// go back to original directory after each test
    43  		helper.Chdir(originalDir)
    44  		helper.DeleteDir(context)
    45  		os.Unsetenv("GLOBALODOCONFIG")
    46  	})
    47  
    48  	Context("when --project flag is used", func() {
    49  		JustBeforeEach(func() {
    50  			cmpName = "nodejs"
    51  		})
    52  
    53  		It("create local nodejs component and push code", func() {
    54  			helper.CopyExample(filepath.Join("source", "nodejs"), context)
    55  			helper.Cmd("odo", "component", "create", "nodejs", cmpName, "--project", project).ShouldPass()
    56  			// verify that config was properly created
    57  			info := helper.LocalEnvInfo(context)
    58  			Expect(info.GetApplication(), "app")
    59  			Expect(info.GetName(), cmpName)
    60  
    61  			output := helper.Cmd("odo", "push").ShouldPass().Out()
    62  			Expect(output).To(ContainSubstring("Changes successfully pushed to component"))
    63  		})
    64  
    65  		It("create, push and list local nodejs component", func() {
    66  			appName := "testing"
    67  			helper.CopyExample(filepath.Join("source", "nodejs"), context)
    68  			helper.Cmd("odo", "component", "create", "nodejs", cmpName, "--app", appName, "--project", project, "--context", context).ShouldPass()
    69  
    70  			// verify that config was properly created
    71  			info := helper.LocalEnvInfo(context)
    72  			Expect(info.GetApplication(), appName)
    73  			Expect(info.GetName(), cmpName)
    74  			helper.Cmd("odo", "push").ShouldPass()
    75  
    76  			// list the component name
    77  			cmpListOutput := helper.Cmd("odo", "list", "--app", appName, "--project", project).ShouldPass().Out()
    78  			Expect(cmpListOutput).To(ContainSubstring(cmpName))
    79  		})
    80  
    81  	})
