    23  // StartURLHttpRequestStatusWatchForK8S begins testing URLs for responses, outputting the result to console
    24  func StartURLHttpRequestStatusWatchForK8S(client kclient.ClientInterface, localConfigProvider *localConfigProvider.LocalConfigProvider, loggingClient machineoutput.MachineEventLoggingClient) {
    25  
    26  	// This is a non-blocking function so that other status watchers may start as needed
    27  	go func() {
    28  
    29  		var urlList []statusURL
    30  
    31  		for {
    32  			var err error
    33  			urlList, err = getURLsForKubernetes(client, *localConfigProvider, true)
    34  
    35  			if err == nil {
    36  				// Success!
    37  				break
    38  			} else {
    39  				// Try again in a few seconds...
    40  				klog.V(4).Infof("Unable to get URLs: %v", err)
    41  				time.Sleep(URLFailureWaitTime)
    42  			}
    43  		}
    44  
    45  		if len(urlList) == 0 {
    46  			return
    47  		}
    48  
    49  		singleEntry := [][]statusURL{urlList}
    50  
    51  		startURLTester(singleEntry, loggingClient)
    52  
    53  	}()
    54  }

    64  func getURLsForKubernetes(client kclient.ClientInterface, lcProvider localConfigProvider.LocalConfigProvider, ignoreUnpushed bool) ([]statusURL, error) {
    65  	var err error
    66  	componentName := lcProvider.GetName()
    67  
    68  	routesSupported := false
    69  
    70  	if routesSupported, err = client.IsRouteSupported(); err != nil {
    71  		// Fallback to Kubernetes client on error
    72  		routesSupported = false
    73  	}
    74  
    75  	urlClient := NewClient(ClientOptions{
    76  		LocalConfigProvider: lcProvider,
    77  		Client:              client,
    78  		IsRouteSupported:    routesSupported,
    79  	})
    80  	urls, err := urlClient.List()
    81  
    82  	if err != nil {
    83  		return nil, err
    84  	}
    85  	urlList := []statusURL{}
    86  
    87  	for _, u := range urls.Items {
    88  
    89  		// Ignore unpushed URLs, they necessarily are unreachable
    90  		if u.Status.State != StateTypePushed && ignoreUnpushed {
    91  			continue
    92  		}
    93  
    94  		var properURL, protocol string
    95  
    96  		if u.Spec.Kind != localConfigProvider.ROUTE {
    97  			protocol = GetProtocol(routev1.Route{}, ConvertExtensionV1IngressURLToIngress(u, componentName))
    98  			properURL = GetURLString(protocol, "", u.Spec.Host)
    99  		} else {
   100  			protocol = u.Spec.Protocol
   101  			properURL = GetURLString(protocol, u.Spec.Host, "")
   102  		}
   103  
   104  		statusURLVal := statusURL{
   105  			name:   u.Name,
   106  			url:    properURL,
   107  			kind:   string(u.Spec.Kind),
   108  			port:   u.Spec.Port,
   109  			secure: protocol == "https",
   110  		}
   111  
   112  		urlList = append(urlList, statusURLVal)
   113  
   114  	}
   115  
   116  	return urlList, nil
   117  }

   127  // startURLTestGoRoutine tests one or more urls ('urls' param); if at least one of them is successful, a success is reported.
   128  // If a success was previously reported, additional successes will not be reported (until at least one failure occurs)
   129  // Likewise if a failure was previously reported.
   130  func startURLTestGoRoutine(urls []statusURL, delayBetweenRequests time.Duration, loggingClient machineoutput.MachineEventLoggingClient) {
   131  
   132  	go func() {
   133  
   134  		var previousResult *bool = nil
   135  
   136  		for {
   137  
   138  			successfulMatch := (*statusURL)(nil)
   139  			atLeastOneSuccess := false
   140  
   141  			for _, currURL := range urls {
   142  				anyResponseReceived, err := testURL(currURL.url)
   143  
   144  				if err != nil {
   145  					klog.V(4).Infof("Error on connecting to URL '%s' %v", currURL.url, err)
   146  				}
   147  
   148  				if anyResponseReceived {
   149  					match := currURL
   150  					successfulMatch = &match
   151  					atLeastOneSuccess = true
   152  					break
   153  				}
   154  			}
   155  
   156  			// If this is the first time we have seen a result for this URL, OR the result has changed from last time
   157  			if previousResult == nil || *previousResult != atLeastOneSuccess {
   158  
   159  				if successfulMatch != nil {
   160  					// At least one of the URLs was reachable, so report success for it
   161  					loggingClient.URLReachable((*successfulMatch).name, (*successfulMatch).url, (*successfulMatch).port, (*successfulMatch).secure, (*successfulMatch).kind, true, machineoutput.TimestampNow())
   162  				} else {
   163  					// Otherwise report failure for all URLs
   164  					for _, currURL := range urls {
   165  						loggingClient.URLReachable(currURL.name, currURL.url, currURL.port, currURL.secure, currURL.kind, atLeastOneSuccess, machineoutput.TimestampNow())
   166  					}
   167  				}
   168  			}
   169  
   170  			previousResult = &atLeastOneSuccess
   171  
   172  			time.Sleep(delayBetweenRequests)
   173  		}
   174  	}()
   175  }

   177  // testURL tests a single URL, returning true if ANY response was received, or false otherwise (plus an error if applicable)
   178  func testURL(url string) (bool, error) {
   179  
   180  	// Suppress 'G402 (CWE-295): TLS InsecureSkipVerify set true': since we are not using the contents of the HTTP response,
   181  	// the fact that it can be MITM-ed is irrelevant.
   182  	/* #nosec */
   183  	tr := &http.Transport{
   184  		TLSClientConfig: &tls.Config{InsecureSkipVerify: true},
   185  	}
   186  
   187  	client := &http.Client{Transport: tr}
   188  
   189  	resp, err := client.Get(url)
   190  	if err != nil || resp == nil {
   191  		errMsg := "Get request failed for " + url + " , with no response code."
   192  		if err != nil {
   193  			return false, errors.Wrapf(err, errMsg)
   194  		}
   195  		return false, errors.New(errMsg)
   196  	}
   197  
   198  	defer resp.Body.Close()
   199  
   200  	// Any response code (including 5XX, 4XX) is considered a success; we only use this to determine if the process
   201  	// is responding to requests.
   202  	klog.V(4).Infof("Get request succeeded for '%s', with response code %d", url, resp.StatusCode)
   203  
   204  	return true, nil
   205  
   206  }
