    69  // Push creates and deletes the required URLs
    70  func Push(parameters PushParameters) error {
    71  	urlLOCAL := make(map[string]URL)
    72  
    73  	localConfigProviderURLs, err := parameters.LocalConfigProvider.ListURLs()
    74  	if err != nil {
    75  		return err
    76  	}
    77  
    78  	// get the local URLs
    79  	for _, url := range localConfigProviderURLs {
    80  		if !parameters.IsRouteSupported && url.Kind == localConfigProvider.ROUTE {
    81  			// display warning since Host info is missing
    82  			log.Warningf("Unable to create ingress, missing host information for Endpoint %v, please check instructions on URL creation (refer `odo url create --help`)\n", url.Name)
    83  			continue
    84  		}
    85  
    86  		urlLOCAL[url.Name] = NewURLFromLocalURL(url)
    87  	}
    88  
    89  	log.Info("\nApplying URL changes")
    90  
    91  	urlCLUSTER := make(map[string]URL)
    92  
    93  	// get the URLs on the cluster
    94  	urlList, err := parameters.URLClient.ListFromCluster()
    95  	if err != nil {
    96  		return err
    97  	}
    98  
    99  	for _, url := range urlList.Items {
   100  		urlCLUSTER[url.Name] = url
   101  	}
   102  
   103  	urlChange := false
   104  
   105  	// find URLs to delete
   106  	for urlName, urlSpec := range urlCLUSTER {
   107  		val, ok := urlLOCAL[urlName]
   108  		configMismatch := false
   109  		if ok {
   110  			// since the host stored in an ingress
   111  			// is the combination of name and host of the url
   112  			if val.Spec.Kind == localConfigProvider.INGRESS {
   113  				// in case of a secure ingress type URL with no user given tls secret
   114  				// the default secret name is used during creation
   115  				// thus setting it to the local URLs to avoid config mismatch
   116  				if val.Spec.Secure && val.Spec.TLSSecret == "" {
   117  					val.Spec.TLSSecret = getDefaultTLSSecretName(urlName, parameters.LocalConfigProvider.GetName(), parameters.LocalConfigProvider.GetApplication())
   118  				}
   119  				val.Spec.Host = fmt.Sprintf("%v.%v", urlName, val.Spec.Host)
   120  			} else if val.Spec.Kind == localConfigProvider.ROUTE {
   121  				// we don't allow the host input for route based URLs
   122  				// removing it for the urls from the cluster to avoid config mismatch
   123  				urlSpec.Spec.Host = ""
   124  			}
   125  
   126  			if val.Spec.Protocol == "" {
   127  				if val.Spec.Secure {
   128  					val.Spec.Protocol = "https"
   129  				} else {
   130  					val.Spec.Protocol = "http"
   131  				}
   132  			}
   133  
   134  			if !reflect.DeepEqual(val.Spec, urlSpec.Spec) {
   135  				configMismatch = true
   136  				klog.V(4).Infof("config and cluster mismatch for url %s", urlName)
   137  			}
   138  		}
   139  
   140  		if !ok || configMismatch {
   141  			// delete the url
   142  			err := parameters.URLClient.Delete(urlName, urlSpec.Spec.Kind)
   143  			if err != nil {
   144  				return err
   145  			}
   146  			log.Successf("URL %s successfully deleted", urlName)
   147  			urlChange = true
   148  			delete(urlCLUSTER, urlName)
   149  			continue
   150  		}
   151  	}
   152  
   153  	// find URLs to create
   154  	for urlName, urlInfo := range urlLOCAL {
   155  		_, ok := urlCLUSTER[urlName]
   156  		if !ok {
   157  			host, err := parameters.URLClient.Create(urlInfo)
   158  			if err != nil {
   159  				return err
   160  			}
   161  			log.Successf("URL %s: %s%s created", urlName, host, urlInfo.Spec.Path)
   162  			urlChange = true
   163  		}
   164  	}
   165  
   166  	if !urlChange {
   167  		log.Success("URLs are synced with the cluster, no changes are required.")
   168  	}
   169  
   170  	return nil
   171  }
