    38  // ListFromCluster lists both route and ingress based URLs from the cluster
    39  func (k kubernetesClient) ListFromCluster() (URLList, error) {
    40  	if k.componentName == "" || k.appName == "" {
    41  		return URLList{}, fmt.Errorf("the component name, the app name or both are empty")
    42  	}
    43  	labelSelector := componentlabels.GetSelector(k.componentName, k.appName)
    44  	klog.V(4).Infof("Listing ingresses with label selector: %v", labelSelector)
    45  	ingresses, err := k.client.ListIngresses(labelSelector)
    46  	if err != nil {
    47  		return URLList{}, errors.Wrap(err, "unable to list ingress")
    48  	}
    49  
    50  	var routes []routev1.Route
    51  	if k.isRouteSupported {
    52  		routes, err = k.client.ListRoutes(labelSelector)
    53  		if err != nil {
    54  			return URLList{}, errors.Wrap(err, "unable to list routes")
    55  		}
    56  	}
    57  
    58  	var clusterURLs []URL
    59  	clusterURLs = append(clusterURLs, NewURLsFromKubernetesIngressList(ingresses)...)
    60  	for _, r := range routes {
    61  		// ignore the routes created by ingresses
    62  		if r.OwnerReferences != nil && r.OwnerReferences[0].Kind == "Ingress" {
    63  			continue
    64  		}
    65  		clusterURL := NewURL(r)
    66  		clusterURLs = append(clusterURLs, clusterURL)
    67  	}
    68  
    69  	return NewURLList(clusterURLs), nil
    70  }

    72  // List lists both route/ingress based URLs and local URLs with respective states
    73  func (k kubernetesClient) List() (URLList, error) {
    74  	// get the URLs present on the cluster
    75  	clusterURLMap := make(map[string]URL)
    76  	var clusterURLs URLList
    77  	var err error
    78  	if k.client != nil {
    79  		clusterURLs, err = k.ListFromCluster()
    80  		if err != nil {
    81  			return URLList{}, errors.Wrap(err, "unable to list routes")
    82  		}
    83  	}
    84  
    85  	for _, url := range clusterURLs.Items {
    86  		clusterURLMap[url.Name] = url
    87  	}
    88  
    89  	localMap := make(map[string]URL)
    90  	if k.localConfigProvider != nil {
    91  		// get the URLs present on the localConfigProvider
    92  		localURLS, err := k.localConfigProvider.ListURLs()
    93  		if err != nil {
    94  			return URLList{}, err
    95  		}
    96  		for _, url := range localURLS {
    97  			if !k.isRouteSupported && url.Kind == localConfigProvider.ROUTE {
    98  				continue
    99  			}
   100  			localURL := NewURLFromEnvinfoURL(url, k.componentName)
   101  			if localURL.Spec.Protocol == "" {
   102  				if localURL.Spec.Secure {
   103  					localURL.Spec.Protocol = "https"
   104  				} else {
   105  					localURL.Spec.Protocol = "http"
   106  				}
   107  			}
   108  			localMap[url.Name] = localURL
   109  		}
   110  	}
   111  
   112  	// find the URLs which are present on the cluster but not on the localConfigProvider
   113  	// if not found on the localConfigProvider, mark them as 'StateTypeLocallyDeleted'
   114  	// else mark them as 'StateTypePushed'
   115  	var urls sortableURLs
   116  	for URLName, clusterURL := range clusterURLMap {
   117  		_, found := localMap[URLName]
   118  		if found {
   119  			// URL is in both local env file and cluster
   120  			clusterURL.Status.State = StateTypePushed
   121  			urls = append(urls, clusterURL)
   122  		} else {
   123  			// URL is on the cluster but not in local env file
   124  			clusterURL.Status.State = StateTypeLocallyDeleted
   125  			urls = append(urls, clusterURL)
   126  		}
   127  	}
   128  
   129  	// find the URLs which are present on the localConfigProvider but not on the cluster
   130  	// if not found on the cluster, mark them as 'StateTypeNotPushed'
   131  	for localName, localURL := range localMap {
   132  		_, remoteURLFound := clusterURLMap[localName]
   133  		if !remoteURLFound {
   134  			// URL is in the local env file but not pushed to cluster
   135  			localURL.Status.State = StateTypeNotPushed
   136  			urls = append(urls, localURL)
   137  		}
   138  	}
   139  
   140  	// sort urls by name to get consistent output
   141  	sort.Sort(urls)
   142  	urlList := NewURLList(urls)
   143  	return urlList, nil
   144  }

   146  // Delete deletes the URL with the given name and kind
   147  func (k kubernetesClient) Delete(name string, kind localConfigProvider.URLKind) error {
   148  	if k.componentName == "" || k.appName == "" {
   149  		return fmt.Errorf("the component name, the app name or both are empty")
   150  	}
   151  	selector := util.ConvertLabelsToSelector(urlLabels.GetLabels(name, k.componentName, k.appName, false))
   152  
   153  	switch kind {
   154  	case localConfigProvider.INGRESS:
   155  		ingress, err := k.client.GetOneIngressFromSelector(selector)
   156  		if err != nil {
   157  			return err
   158  		}
   159  		return k.client.DeleteIngress(ingress.GetName())
   160  	case localConfigProvider.ROUTE:
   161  		route, err := k.client.GetOneRouteFromSelector(selector)
   162  		if err != nil {
   163  			return err
   164  		}
   165  		return k.client.DeleteRoute(route.Name)
   166  	default:
   167  		return fmt.Errorf("url type is not supported")
   168  	}
   169  }

   171  // Create creates a route or ingress based on the given URL
   172  func (k kubernetesClient) Create(url URL) (string, error) {
   173  	if k.componentName == "" || k.appName == "" {
   174  		return "", fmt.Errorf("the component name, the app name or both are empty")
   175  	}
   176  
   177  	if url.Spec.Kind != localConfigProvider.INGRESS && url.Spec.Kind != localConfigProvider.ROUTE {
   178  		return "", fmt.Errorf("urlKind %s is not supported for URL creation", url.Spec.Kind)
   179  	}
   180  
   181  	if !url.Spec.Secure && url.Spec.TLSSecret != "" {
   182  		return "", fmt.Errorf("secret name can only be used for secure URLs")
   183  	}
   184  
   185  	labels := urlLabels.GetLabels(url.Name, k.componentName, k.appName, true)
   186  
   187  	if url.Spec.Kind == localConfigProvider.INGRESS {
   188  		return k.createIngress(url, labels)
   189  	} else {
   190  		if !k.isRouteSupported {
   191  			return "", errors.Errorf("routes are not available on non OpenShift clusters")
   192  		}
   193  
   194  		return k.createRoute(url, labels)
   195  	}
   196

   199  // createIngress creates a ingress for the given URL with the given labels
   200  func (k kubernetesClient) createIngress(url URL, labels map[string]string) (string, error) {
   201  	if url.Spec.Host == "" {
   202  		return "", errors.Errorf("the host cannot be empty")
   203  	}
   204  
   205  	service, err := k.client.GetOneService(k.componentName, k.appName)
   206  	if err != nil {
   207  		return "", err
   208  	}
   209  
   210  	ingressDomain := fmt.Sprintf("%v.%v", url.Name, url.Spec.Host)
   211  
   212  	// generate the owner reference
   213  	if k.deployment == nil {
   214  		k.deployment, err = k.client.GetOneDeployment(k.componentName, k.appName)
   215  		if err != nil {
   216  			return "", err
   217  		}
   218  	}
   219  	ownerReference := generator.GetOwnerReference(k.deployment)
   220  
   221  	if url.Spec.Secure {
   222  		if len(url.Spec.TLSSecret) != 0 {
   223  			// get the user given secret
   224  			_, err = k.client.GetSecret(url.Spec.TLSSecret, k.client.GetCurrentNamespace())
   225  			if err != nil {
   226  				return "", errors.Wrap(err, "unable to get the provided secret: "+url.Spec.TLSSecret)
   227  			}
   228  		} else {
   229  			// get the default secret
   230  			defaultTLSSecretName := getDefaultTLSSecretName(url.Name, k.componentName, k.appName)
   231  			_, err = k.client.GetSecret(defaultTLSSecretName, k.client.GetCurrentNamespace())
   232  
   233  			// create tls secret if it does not exist
   234  			if kerrors.IsNotFound(err) {
   235  				selfSignedCert, e := kclient.GenerateSelfSignedCertificate(url.Spec.Host)
   236  				if e != nil {
   237  					return "", errors.Wrap(e, "unable to generate self-signed certificate for clutser: "+url.Spec.Host)
   238  				}
   239  				// create tls secret
   240  				secretLabels := componentlabels.GetLabels(k.componentName, k.appName, true)
   241  				objectMeta := metav1.ObjectMeta{
   242  					Name:   defaultTLSSecretName,
   243  					Labels: secretLabels,
   244  					OwnerReferences: []v1.OwnerReference{
   245  						ownerReference,
   246  					},
   247  				}
   248  				secret, e := k.client.CreateTLSSecret(selfSignedCert.CertPem, selfSignedCert.KeyPem, objectMeta)
   249  				if e != nil {
   250  					return "", errors.Wrap(e, "unable to create tls secret")
   251  				}
   252  				url.Spec.TLSSecret = secret.Name
   253  			} else if err != nil {
   254  				return "", err
   255  			} else {
   256  				// tls secret found for this component
   257  				url.Spec.TLSSecret = defaultTLSSecretName
   258  			}
   259  
   260  		}
   261  
   262  	}
   263  
   264  	suffix := util.GetAdler32Value(url.Name + k.appName + k.componentName)
   265  	ingressName, err := util.NamespaceOpenShiftObject(url.Name, suffix)
   266  	if err != nil {
   267  		return "", err
   268  	}
   269  	objectMeta := generator.GetObjectMeta(k.componentName, k.client.GetCurrentNamespace(), labels, nil)
   270  	// to avoid error due to duplicate ingress name defined in different devfile components
   271  	objectMeta.Name = ingressName
   272  	objectMeta.OwnerReferences = append(objectMeta.OwnerReferences, ownerReference)
   273  
   274  	ingressParam := generator.IngressParams{
   275  		ObjectMeta: objectMeta,
   276  		IngressSpecParams: generator.IngressSpecParams{
   277  			ServiceName:   service.Name,
   278  			IngressDomain: ingressDomain,
   279  			PortNumber:    intstr.FromInt(url.Spec.Port),
   280  			TLSSecretName: url.Spec.TLSSecret,
   281  			Path:          url.Spec.Path,
   282  		},
   283  	}
   284  	ingress := unions.NewKubernetesIngressFromParams(ingressParam)
   285  	// Pass in the namespace name, link to the service (componentName) and labels to create a ingress
   286  	i, err := k.client.CreateIngress(*ingress)
   287  	if err != nil {
   288  		return "", fmt.Errorf("unable to create ingress %w", err)
   289  	}
   290  	return i.GetURLString(), nil
   291  }

   293  // createRoute creates a route for the given URL with the given labels
   294  func (k kubernetesClient) createRoute(url URL, labels map[string]string) (string, error) {
   295  	// to avoid error due to duplicate ingress name defined in different devfile components
   296  	// we avoid using the getResourceName() and use the previous method from s2i
   297  	// as the host name, which is automatically created on openshift,
   298  	// can become more than 63 chars, which is invalid
   299  	suffix := util.GetAdler32Value(url.Name + k.appName + k.componentName)
   300  	routeName, err := util.NamespaceOpenShiftObject(url.Name, suffix)
   301  	if err != nil {
   302  		return "", errors.Wrapf(err, "unable to create namespaced name")
   303  	}
   304  
   305  	if k.deployment == nil {
   306  		k.deployment, err = k.client.GetOneDeployment(k.componentName, k.appName)
   307  		if err != nil {
   308  			return "", err
   309  		}
   310  	}
   311  	ownerReference := generator.GetOwnerReference(k.deployment)
   312  
   313  	service, err := k.client.GetOneService(k.componentName, k.appName)
   314  	if err != nil {
   315  		return "", err
   316  	}
   317  
   318  	// Pass in the namespace name, link to the service (componentName) and labels to create a route
   319  	route, err := k.client.CreateRoute(routeName, service.Name, intstr.FromInt(url.Spec.Port), labels, url.Spec.Secure, url.Spec.Path, ownerReference)
   320  	if err != nil {
   321  		if kerrors.IsAlreadyExists(err) {
   322  			return "", fmt.Errorf("url named %q already exists in the same app named %q", url.Name, k.appName)
   323  		}
   324  		return "", errors.Wrap(err, "unable to create route")
   325  	}
   326  	return GetURLString(GetProtocol(*route, iextensionsv1.Ingress{}), route.Spec.Host, ""), nil
   327  }
