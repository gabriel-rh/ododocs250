   114  // NewURLFromEnvinfoURL creates a URL from a EnvinfoURL
   115  func NewURLFromEnvinfoURL(envinfoURL localConfigProvider.LocalURL, serviceName string) URL {
   116  	hostString := fmt.Sprintf("%s.%s", envinfoURL.Name, envinfoURL.Host)
   117  	// default to route kind if none is provided
   118  	kind := envinfoURL.Kind
   119  	if kind == "" {
   120  		kind = localConfigProvider.ROUTE
   121  	}
   122  	url := URL{
   123  		TypeMeta: metav1.TypeMeta{
   124  			Kind:       URLKind,
   125  			APIVersion: machineoutput.APIVersion,
   126  		},
   127  		ObjectMeta: metav1.ObjectMeta{
   128  			Name: envinfoURL.Name,
   129  		},
   130  		Spec: URLSpec{
   131  			Host:      envinfoURL.Host,
   132  			Protocol:  envinfoURL.Protocol,
   133  			Port:      envinfoURL.Port,
   134  			Secure:    envinfoURL.Secure,
   135  			Kind:      kind,
   136  			TLSSecret: envinfoURL.TLSSecret,
   137  			Path:      envinfoURL.Path,
   138  		},
   139  	}
   140  	if kind == localConfigProvider.INGRESS {
   141  		url.Spec.Host = hostString
   142  		if envinfoURL.Secure && len(envinfoURL.TLSSecret) > 0 {
   143  			url.Spec.TLSSecret = envinfoURL.TLSSecret
   144  		} else if envinfoURL.Secure {
   145  			url.Spec.TLSSecret = fmt.Sprintf("%s-tlssecret", serviceName)
   146  		}
   147  	}
   148  	return url
   149  }

   184  func NewURLFromKubernetesIngress(ki *unions.KubernetesIngress, skipIfGenerated bool) URL {
   185  	if skipIfGenerated && ki.IsGenerated() {
   186  		return URL{}
   187  	}
   188  	u := URL{
   189  		TypeMeta: metav1.TypeMeta{
   190  			Kind:       URLKind,
   191  			APIVersion: machineoutput.APIVersion,
   192  		},
   193  	}
   194  	if ki.NetworkingV1Ingress != nil {
   195  		u.ObjectMeta = metav1.ObjectMeta{Name: ki.NetworkingV1Ingress.Labels[urlLabels.URLLabel]}
   196  		u.Spec = URLSpec{
   197  			Host:   ki.NetworkingV1Ingress.Spec.Rules[0].Host,
   198  			Port:   int(ki.NetworkingV1Ingress.Spec.Rules[0].HTTP.Paths[0].Backend.Service.Port.Number),
   199  			Secure: ki.NetworkingV1Ingress.Spec.TLS != nil,
   200  			Path:   ki.NetworkingV1Ingress.Spec.Rules[0].HTTP.Paths[0].Path,
   201  			Kind:   localConfigProvider.INGRESS,
   202  		}
   203  		if len(ki.NetworkingV1Ingress.Spec.TLS) > 0 {
   204  			u.Spec.TLSSecret = ki.NetworkingV1Ingress.Spec.TLS[0].SecretName
   205  		}
   206  		if u.Spec.Secure {
   207  			u.Spec.Protocol = "https"
   208  		} else {
   209  			u.Spec.Protocol = "http"
   210  		}
   211  	} else if ki.ExtensionV1Beta1Ingress != nil {
   212  		u.ObjectMeta = metav1.ObjectMeta{Name: ki.ExtensionV1Beta1Ingress.Labels[urlLabels.URLLabel]}
   213  		u.Spec = URLSpec{
   214  			Host:   ki.ExtensionV1Beta1Ingress.Spec.Rules[0].Host,
   215  			Port:   int(ki.ExtensionV1Beta1Ingress.Spec.Rules[0].HTTP.Paths[0].Backend.ServicePort.IntVal),
   216  			Secure: ki.ExtensionV1Beta1Ingress.Spec.TLS != nil,
   217  			Path:   ki.ExtensionV1Beta1Ingress.Spec.Rules[0].HTTP.Paths[0].Path,
   218  			Kind:   localConfigProvider.INGRESS,
   219  		}
   220  		if len(ki.ExtensionV1Beta1Ingress.Spec.TLS) > 0 {
   221  			u.Spec.TLSSecret = ki.ExtensionV1Beta1Ingress.Spec.TLS[0].SecretName
   222  		}
   223  		if u.Spec.Secure {
   224  			u.Spec.Protocol = "https"
   225  		} else {
   226  			u.Spec.Protocol = "http"
   227  		}
   228  	}
   229  	return u
