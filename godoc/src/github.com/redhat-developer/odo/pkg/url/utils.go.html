    39  // ConvertEnvInfoURL converts EnvInfoURL to URL
    40  func ConvertEnvInfoURL(envInfoURL localConfigProvider.LocalURL, serviceName string) URL {
    41  	hostString := fmt.Sprintf("%s.%s", envInfoURL.Name, envInfoURL.Host)
    42  	// default to route kind if none is provided
    43  	kind := envInfoURL.Kind
    44  	if kind == "" {
    45  		kind = localConfigProvider.ROUTE
    46  	}
    47  	url := URL{
    48  		TypeMeta: metav1.TypeMeta{
    49  			Kind:       "url",
    50  			APIVersion: apiVersion,
    51  		},
    52  		ObjectMeta: metav1.ObjectMeta{
    53  			Name: envInfoURL.Name,
    54  		},
    55  		Spec: URLSpec{
    56  			Host:      envInfoURL.Host,
    57  			Protocol:  envInfoURL.Protocol,
    58  			Port:      envInfoURL.Port,
    59  			Secure:    envInfoURL.Secure,
    60  			Kind:      kind,
    61  			TLSSecret: envInfoURL.TLSSecret,
    62  			Path:      envInfoURL.Path,
    63  		},
    64  	}
    65  	if kind == localConfigProvider.INGRESS {
    66  		url.Spec.Host = hostString
    67  		if envInfoURL.Secure && len(envInfoURL.TLSSecret) > 0 {
    68  			url.Spec.TLSSecret = envInfoURL.TLSSecret
    69  		} else if envInfoURL.Secure {
    70  			url.Spec.TLSSecret = fmt.Sprintf("%s-tlssecret", serviceName)
    71  		}
    72  	}
    73  	return url
    74  }
