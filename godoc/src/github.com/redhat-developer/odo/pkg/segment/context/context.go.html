    53  // SetClusterType sets clusterType property for telemetry data when a component is pushed or a project is created/set
    54  func SetClusterType(ctx context.Context, client kclient.ClientInterface) {
    55  	var value string
    56  	if client == nil {
    57  		value = NOTFOUND
    58  	} else {
    59  		// We are not checking ServerVersion to decide the cluster type because it does not always return the version,
    60  		// it sometimes fails to retrieve the data if user is using minishift or plain oc cluster
    61  		isOC, err := client.IsProjectSupported()
    62  		if err != nil {
    63  			klog.V(3).Info(errors.Wrap(err, "unable to detect project support"))
    64  			value = NOTFOUND
    65  		} else {
    66  			if isOC {
    67  				isOC4, err := client.IsCSVSupported()
    68  				// TODO: Add a unit test for this case
    69  				if err != nil {
    70  					value = "openshift"
    71  				} else {
    72  					if isOC4 {
    73  						value = "openshift4"
    74  					} else {
    75  						value = "openshift3"
    76  					}
    77  				}
    78  			} else {
    79  				value = "kubernetes"
    80  			}
    81  		}
    82  	}
    83  	setContextProperty(ctx, ClusterType, value)
    84  }

   116  // set safely sets value for a key in storage
   117  func (p *properties) set(name string, value interface{}) {
   118  	p.lock.Lock()
   119  	defer p.lock.Unlock()
   120  	p.storage[name] = value
   121  }
   122  
   123  // values safely retrieves a deep copy of the storage
   124  func (p *properties) values() map[string]interface{} {
   125  	p.lock.Lock()
   126  	defer p.lock.Unlock()
   127  	ret := make(map[string]interface{})
   128  	for k, v := range p.storage {
   129  		ret[k] = v
   130  	}
   131  	return ret
   132  }

   134  // propertiesFromContext retrieves the properties instance from the context
   135  func propertiesFromContext(ctx context.Context) *properties {
   136  	value := ctx.Value(key)
   137  	if cast, ok := value.(*properties); ok {
   138  		return cast
   139  	}
   140  	return nil
   141  }
