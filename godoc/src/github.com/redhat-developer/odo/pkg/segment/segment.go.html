    71  func newCustomClient(preference preference.Client, telemetryFilePath string, segmentEndpoint string) (*Client, error) {
    72  	// get the locale information
    73  	tag, err := locale.Detect()

    83  			Timezone: getTimeZoneRelativeToUTC(),
    84  			OS: analytics.OSInfo{
    85  				Name: runtime.GOOS,
    86  			},
    87  			Locale: tag.String(),
    88  		},
    89  	})
    90  	if err != nil {
    91  		return nil, err
    92  	}
    93  	return &Client{
    94  		SegmentClient:     client,
    95  		Preference:        preference,
    96  		TelemetryFilePath: telemetryFilePath,

   100  // getTimeZoneRelativeToUTC returns time zone relative to UTC (UTC +0530, UTC 0000, etc.)
   101  func getTimeZoneRelativeToUTC() string {
   102  	// t is a string array of RHC8222Z representation of current time
   103  	// example - [13 Sep 21 16:37 +0530]
   104  	var t = strings.Split(time.Now().Format(time.RFC822Z), " ")
   105  	return fmt.Sprintf("UTC %s", t[len(t)-1])
   106  }

   113  // Upload prepares the data to be sent to segment and send it once the client connection closes
   114  func (c *Client) Upload(data TelemetryData) error {
   115  	// if the user has not consented for telemetry, return
   116  	if !IsTelemetryEnabled(c.Preference) {
   117  		return nil
   118  	}
   119  
   120  	// obtain the user ID
   121  	userId, uerr := getUserIdentity(c.TelemetryFilePath)
   122  	if uerr != nil {
   123  		return uerr
   124  	}
   125  
   126  	// add information to the data
   127  	properties := analytics.NewProperties()
   128  	for k, v := range data.Properties.CmdProperties {
   129  		if k != scontext.TelemetryStatus {
   130  			properties = properties.Set(k, v)
   131  		}
   132  	}
   133  
   134  	properties = properties.Set("version", data.Properties.Version).
   135  		Set("success", data.Properties.Success).
   136  		Set("duration(ms)", data.Properties.Duration).
   137  		Set("tty", data.Properties.Tty)
   138  	// in case the command executed unsuccessfully, add information about the error in the data
   139  	if data.Properties.Error != "" {
   140  		properties = properties.Set("error", data.Properties.Error).Set("error-type", data.Properties.ErrorType)
   141  	}
   142  
   143  	// send the Identify message data that helps identify the user on segment
   144  	err := c.SegmentClient.Enqueue(analytics.Identify{
   145  		UserId: userId,
   146  		Traits: addConfigTraits(),
   147  	})
   148  	if err != nil {
   149  		klog.V(4).Infof("Cannot send Identify telemetry event: %q", err)
   150  		// This doesn't have to be a fatal error, as we can still try to track normal event
   151  		// There just might be some missing information about the user, but this will be only
   152  		// in case that this was the first time we tried to send identify event for give userId.
   153  	}
   154  
   155  	// queue the data that has telemetry information
   156  	return c.SegmentClient.Enqueue(analytics.Track{
   157  		UserId:     userId,
   158  		Event:      data.Event,
   159  		Properties: properties,
   160  	})
   161  }

   165  	traits := analytics.NewTraits().Set("os", runtime.GOOS)

   183  // getUserIdentity returns the anonymous ID if it exists, else creates a new one and sends the data to Segment
   184  func getUserIdentity(telemetryFilePath string) (string, error) {
   185  	var id []byte
   186  
   187  	// Get-or-Create the '$HOME/.redhat' directory
   188  	if err := os.MkdirAll(filepath.Dir(telemetryFilePath), os.ModePerm); err != nil {
   189  		return "", err
   190  	}
   191  
   192  	// Get-or-Create the anonymousID file that contains a UUID
   193  	if _, err := os.Stat(telemetryFilePath); !os.IsNotExist(err) {
   194  		id, err = ioutil.ReadFile(telemetryFilePath)
   195  		if err != nil {
   196  			return "", err
   197  		}
   198  	}
   199  
   200  	// check if the id is a valid uuid, if not, nil is returned
   201  	if uuid.Parse(strings.TrimSpace(string(id))) == nil {
   202  		id = []byte(uuid.NewRandom().String())
   203  		if err := ioutil.WriteFile(telemetryFilePath, id, 0600); err != nil {
   204  			return "", err
   205  		}
   206  	}
   207  	return strings.TrimSpace(string(id)), nil
   208  }

   210  // SetError sanitizes any PII(Personally Identifiable Information) from the error
   211  func SetError(err error) (errString string) {
   212  	if err == nil {
   213  		return ""
   214  	}
   215  	errString = err.Error()
   216  
   217  	// Sanitize user information
   218  	errString = sanitizeUserInfo(errString)
   219  
   220  	// Sanitize file path
   221  	errString = sanitizeFilePath(errString)
   222  
   223  	// Sanitize exec commands: For errors when a command exec fails in cases like odo exec or odo test, we do not want to know the command that the user executed, so we simply return
   224  	errString = sanitizeExec(errString)
   225  
   226  	// Sanitize URL
   227  	errString = sanitizeURL(errString)
   228  
   229  	return errString
   230  }

   249  // IsTelemetryEnabled returns true if user has consented to telemetry
   250  func IsTelemetryEnabled(cfg preference.Client) bool {
   251  	klog.V(4).Info("Checking telemetry enable status")
   252  	// The env variable gets precedence in this decision.
   253  	// In case a non-bool value was passed to the env var, we ignore it
   254  	disableTelemetry, _ := strconv.ParseBool(os.Getenv(DisableTelemetryEnv))
   255  	if disableTelemetry {
   256  		klog.V(4).Infof("Sending telemetry disabled by %s=%t\n", DisableTelemetryEnv, disableTelemetry)
   257  		return false
   258  	} else if cfg.GetConsentTelemetry() {
   259  		return true
   260  	}
   261  	return false
   262  }

   274  // sanitizeFilePath sanitizes file paths from error string
   275  func sanitizeFilePath(errString string) string {
   276  	for _, str := range strings.Split(errString, " ") {
   277  		if strings.Count(str, string(os.PathSeparator)) > 1 {
   278  			errString = strings.ReplaceAll(errString, str, Sanitizer)
   279  		}
   280  	}
   281  	return errString
   282  }

   284  // sanitizeURL sanitizes URLs from the error string
   285  func sanitizeURL(errString string) string {
   286  	// the following regex parses hostnames and ip addresses
   287  	// references - https://www.oreilly.com/library/view/regular-expressions-cookbook/9780596802837/ch07s16.html
   288  	// https://www.oreilly.com/library/view/regular-expressions-cookbook/9781449327453/ch08s15.html
   289  	urlPattern, err := regexp.Compile(`((https?|ftp|smtp)://)?((?:[0-9]{1,3}\.){3}[0-9]{1,3}(:([0-9]{1,5}))?|([a-z0-9]+(-[a-z0-9]+)*\.)+[a-z]{2,})`)
   290  	if err != nil {
   291  		return errString
   292  	}
   293  	errString = urlPattern.ReplaceAllString(errString, Sanitizer)
   294  	return errString
   295  }

   297  // sanitizeExec sanitizes commands from the error string that might have been executed by users while running commands like odo test or odo exec
   298  func sanitizeExec(errString string) string {
   299  	pattern, _ := regexp.Compile("exec command.*")
   300  	errString = pattern.ReplaceAllString(errString, fmt.Sprintf("exec command %s", Sanitizer))
   301  	return errString
   302  }
