    63  func verifyRegistryOptions(isSet bool, ro library.RegistryOptions) error {
    64  	if ro.SkipTLSVerify {
    65  		return errors.New("SkipTLSVerify should be set to false by default")
    66  	}
    67  
    68  	return verifyTelemetryData(isSet, ro.Telemetry)
    69  }

    71  func verifyTelemetryData(isSet bool, data library.TelemetryData) error {
    72  	if !isSet {
    73  		if data.Locale == "" && data.User == "" {
    74  			return nil
    75  		}
    76  
    77  		return fmt.Errorf("Locale %s and User %s should be unset when telemetry is not enabled ", data.Locale, data.User)
    78  
    79  	} else {
    80  		//we don't care what value locale and user have been set to.  We just want to make sure they are not empty
    81  		if data.Locale != "" && data.User != "" {
    82  			return nil
    83  		}
    84  
    85  		return fmt.Errorf("Locale %s and User %s should be set when telemetry is enabled ", data.Locale, data.User)
    86  	}
