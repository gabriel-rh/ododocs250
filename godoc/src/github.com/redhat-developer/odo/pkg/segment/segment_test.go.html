    95  func TestClientUploadWithConsent(t *testing.T) {
    96  	body, server := mockServer()
    97  	defer server.Close()
    98  	defer close(body)
    99  
   100  	tests := []struct {
   101  		cmd      string
   102  		testName string
   103  		err      error
   104  		success  bool
   105  		errType  string
   106  		version  string
   107  	}{
   108  		{
   109  			testName: "command ran successfully",
   110  			err:      nil,
   111  			success:  true,
   112  			errType:  "",
   113  			version:  version.VERSION,
   114  		},
   115  		{
   116  			testName: "command failed",
   117  			err:      errors.New("some error occurred"),
   118  			success:  false,
   119  			errType:  "*errors.errorString",
   120  			version:  version.VERSION,
   121  		},
   122  	}
   123  	for _, tt := range tests {
   124  		t.Log("Running test: ", tt.testName)
   125  		t.Run(tt.testName, func(t *testing.T) {
   126  			ctrl := gomock.NewController(t)
   127  			cfg := preference.NewMockClient(ctrl)
   128  			cfg.EXPECT().GetConsentTelemetry().Return(true)
   129  
   130  			c, err := newCustomClient(cfg, createConfigDir(t), server.URL)
   131  			if err != nil {
   132  				t.Error(err)
   133  			}
   134  			uploadData := fakeTelemetryData("odo create", tt.err, context.Background())
   135  			// upload the data to Segment
   136  			if err = c.Upload(uploadData); err != nil {
   137  				t.Error(err)
   138  			}
   139  			// segment.Client.SegmentClient uploads the data to server when a condition is met or when the connection is closed.
   140  			// This condition can be added by setting a BatchSize or Interval to the SegmentClient.
   141  			// BatchSize or Interval conditions have not been set for segment.Client.SegmentClient, so we will need to
   142  			// close the connection in order to upload the data to server.
   143  			// In case a condition is added, we can close the client in the teardown.
   144  			if err = c.Close(); err != nil {
   145  				t.Error(err)
   146  			}
   147  			// Note: This will need to be changed if segment.Client.SegmentClient has BatchSize or Interval set to something.
   148  			select {
   149  			case x := <-body:
   150  				s := segmentResponse{}
   151  				if err1 := json.Unmarshal(x, &s); err1 != nil {
   152  					t.Error(err1)
   153  				}
   154  				// Response returns 2 Batches in response -
   155  				// 1) identify - user's system information in case the server did not already have this information,
   156  				// and 2) track - information about the fired command
   157  				// This condition checks if both the responses were received
   158  				if s.Batch[0].Type != "identify" && s.Batch[1].Type != "track" {
   159  					t.Errorf("Missing Identify or Track information.\nIdentify: %v\nTrack:%v", s.Batch[0].Type, s.Batch[1].Type)
   160  				}
   161  				if s.Batch[0].Traits.OS != runtime.GOOS {
   162  					t.Error("OS does not match")
   163  				}
   164  				if !tt.success {
   165  					if s.Batch[1].Properties.Error != tt.err.Error() {
   166  						t.Error("Error does not match")
   167  					}
   168  				} else {
   169  					if s.Batch[1].Properties.Error != "" {
   170  						t.Error("Error does not match")
   171  					}
   172  				}
   173  				if s.Batch[1].Properties.Success != tt.success {
   174  					t.Error("Success does not match")
   175  				}
   176  				if s.Batch[1].Properties.ErrorType != tt.errType {
   177  					t.Error("Error Type does not match")
   178  				}
   179  				if !strings.Contains(s.Batch[1].Properties.Version, version.VERSION) {
   180  					t.Error("Odo version does not match")
   181  				}
   182  
   183  			default:
   184  				t.Error("Server should receive data")
   185  			}
   186  		})
   187  	}
   188  }
