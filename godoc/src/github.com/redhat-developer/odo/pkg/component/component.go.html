    46  	retVal := ""
    47  	if path != "" {
    48  		retVal = filepath.Base(path)
    49  	} else {
    50  		currDir, err := os.Getwd()
    51  		if err != nil {

    61  // Parameters: desired default component name(w/o prefix) and slice of existing component names
    62  // Returns: Unique component name and error if any
    63  func GetDefaultComponentName(cfg preference.Client, componentPath string, componentType string, existingComponentList ComponentList) (string, error) {
    64  	var prefix string
    65  	var err error
    66  
    67  	// Get component names from component list
    68  	var existingComponentNames []string
    69  	for _, component := range existingComponentList.Items {
    70  		existingComponentNames = append(existingComponentNames, component.Name)
    71  	}
    72  
    73  	// If there's no prefix in config file, or its value is empty string use safe default - the current directory along with component type
    74  	if cfg.NamePrefix() == nil || *cfg.NamePrefix() == "" {
    75  		prefix, err = GetComponentDir(componentPath)
    76  		if err != nil {
    77  			return "", errors.Wrap(err, "unable to generate random component name")
    78  		}
    79  		prefix = util.TruncateString(prefix, componentRandomNamePartsMaxLen)
    80  	} else {
    81  		// Set the required prefix into componentName
    82  		prefix = *cfg.NamePrefix()
    83  	}
    84  
    85  	// Generate unique name for the component using prefix and unique random suffix
    86  	componentName, err := util.GetRandomName(
    87  		fmt.Sprintf("%s-%s", componentType, prefix),
    88  		componentNameMaxLen,
    89  		existingComponentNames,
    90  		componentNameMaxRetries,
    91  	)
    92  	if err != nil {
    93  		return "", errors.Wrap(err, "unable to generate random component name")
    94  	}
    95  
    96  	return util.GetDNS1123Name(componentName), nil
    97  }

   200  			}
   201  		}
   202  
   203  		return component, devfile, nil
   204  	}
   205  	return Component{}, parser.DevfileObj{}, nil

   271  func ListDevfileComponentsInPath(client kclient.ClientInterface, paths []string) ([]Component, error) {
   272  	var components []Component
   273  	var err error
   274  	for _, path := range paths {
   275  		err = filepath.Walk(path, func(path string, f os.FileInfo, err error) error {
   276  			// we check for .odo/env/env.yaml folder first and then find devfile.yaml, this could be changed
   277  			// TODO: optimise this
   278  			if f != nil && strings.Contains(f.Name(), ".odo") {
   279  				// lets find if there is a devfile and an env.yaml
   280  				dir := filepath.Dir(path)
   281  				data, err := envinfo.NewEnvSpecificInfo(dir)
   282  				if err != nil {
   283  					return err
   284  				}
   285  
   286  				// if the .odo folder doesn't contain a proper env file
   287  				if data.GetName() == "" || data.GetApplication() == "" || data.GetNamespace() == "" {
   288  					return nil
   289  				}
   290  
   291  				// we just want to confirm if the devfile is correct
   292  				_, err = parser.ParseDevfile(parser.ParserArgs{
   293  					Path: location.DevfileLocation(dir),
   294  				})
   295  				if err != nil {
   296  					return err
   297  				}
   298  				con, _ := filepath.Abs(filepath.Dir(path))
   299  
   300  				comp := NewComponent(data.GetName())
   301  				comp.Status.State = StateTypeUnknown
   302  				comp.Spec.App = data.GetApplication()
   303  				comp.Namespace = data.GetNamespace()
   304  				comp.Status.Context = con
   305  
   306  				// since the config file maybe belong to a component of a different project
   307  				if client != nil {
   308  					client.SetNamespace(data.GetNamespace())
   309  					deployment, err := client.GetOneDeployment(comp.Name, comp.Spec.App)
   310  					if err != nil {
   311  						comp.Status.State = StateTypeNotPushed
   312  					} else if deployment != nil {
   313  						comp.Status.State = StateTypePushed
   314  					}
   315  				}
   316  
   317  				components = append(components, comp)
   318  			}
   319  
   320  			return nil
   321  		})
   322  
   323  	}
   324  	return components, err
   325  }

   360  // getRemoteComponentMetadata provides component metadata from the cluster
   361  func getRemoteComponentMetadata(client kclient.ClientInterface, componentName string, applicationName string, getUrls, getStorage bool) (Component, error) {
   362  	fromCluster, err := GetPushedComponent(client, componentName, applicationName)
   363  	if err != nil || fromCluster == nil {
   364  		return Component{}, errors.Wrapf(err, "unable to get remote metadata for %s component", componentName)
   365  	}
   366  
   367  	// Component Type
   368  	componentType, err := fromCluster.GetType()
   369  	if err != nil {
   370  		return Component{}, errors.Wrap(err, "unable to get source type")
   371  	}
   372  
   373  	// init component
   374  	component := newComponentWithType(componentName, componentType)
   375  
   376  	// URL
   377  	if getUrls {
   378  		urls, e := fromCluster.GetURLs()
   379  		if e != nil {
   380  			return Component{}, e
   381  		}
   382  		component.Spec.URLSpec = urls
   383  		urlsNb := len(urls)
   384  		if urlsNb > 0 {
   385  			res := make([]string, 0, urlsNb)
   386  			for _, url := range urls {
   387  				res = append(res, url.Name)
   388  			}
   389  			component.Spec.URL = res
   390  		}
   391  	}
   392  
   393  	// Storage
   394  	if getStorage {
   395  		appStore, e := fromCluster.GetStorage()
   396  		if e != nil {
   397  			return Component{}, errors.Wrap(e, "unable to get storage list")
   398  		}
   399  
   400  		component.Spec.StorageSpec = appStore
   401  		var storageList []string
   402  		for _, store := range appStore {
   403  			storageList = append(storageList, store.Name)
   404  		}
   405  		component.Spec.Storage = storageList
   406  	}
   407  
   408  	// Environment Variables
   409  	envVars := fromCluster.GetEnvVars()
   410  	var filteredEnv []corev1.EnvVar
   411  	for _, env := range envVars {
   412  		if !strings.Contains(env.Name, "ODO") {
   413  			filteredEnv = append(filteredEnv, env)
   414  		}
   415  	}
   416  
   417  	// Secrets
   418  	linkedSecrets := fromCluster.GetLinkedSecrets()
   419  	err = setLinksServiceNames(client, linkedSecrets, componentlabels.GetSelector(componentName, applicationName))
   420  	if err != nil {
   421  		return Component{}, fmt.Errorf("unable to get name of services: %w", err)
   422  	}
   423  	component.Status.LinkedServices = linkedSecrets
   424  
   425  	// Annotations
   426  	component.Annotations = fromCluster.GetAnnotations()
   427  
   428  	// Labels
   429  	component.Labels = fromCluster.GetLabels()
   430  
   431  	component.Namespace = client.GetCurrentNamespace()
   432  	component.Spec.App = applicationName
   433  	component.Spec.Env = filteredEnv
   434  	component.Status.State = StateTypePushed
   435  
   436  	return component, nil
   437  }

   439  // setLinksServiceNames sets the service name of the links from the info in ServiceBindingRequests present in the cluster
   440  func setLinksServiceNames(client kclient.ClientInterface, linkedSecrets []SecretMount, selector string) error {
   441  	ok, err := client.IsServiceBindingSupported()
   442  	if err != nil {
   443  		return fmt.Errorf("unable to check if service binding is supported: %w", err)
   444  	}
   445  
   446  	serviceBindings := map[string]string{}
   447  	if ok {
   448  		// service binding operator is installed on the cluster
   449  		list, err := client.ListDynamicResource(kclient.ServiceBindingGroup, kclient.ServiceBindingVersion, kclient.ServiceBindingResource)
   450  		if err != nil || list == nil {
   451  			return err
   452  		}
   453  
   454  		for _, u := range list.Items {
   455  			var sbr servicebinding.ServiceBinding
   456  			js, err := u.MarshalJSON()
   457  			if err != nil {
   458  				return err
   459  			}
   460  			err = json.Unmarshal(js, &sbr)
   461  			if err != nil {
   462  				return err
   463  			}
   464  			services := sbr.Spec.Services
   465  			if len(services) != 1 {
   466  				return errors.New("the ServiceBinding resource should define only one service")
   467  			}
   468  			service := services[0]
   469  			if service.Kind == "Service" {
   470  				serviceBindings[sbr.Status.Secret] = service.Name
   471  			} else {
   472  				serviceBindings[sbr.Status.Secret] = service.Kind + "/" + service.Name
   473  			}
   474  		}
   475  	} else {
   476  		// service binding operator is not installed
   477  		// get the secrets instead of the service binding objects to retrieve the link data
   478  		secrets, err := client.ListSecrets(selector)
   479  		if err != nil {
   480  			return err
   481  		}
   482  
   483  		// get the services to get their names against the component names
   484  		services, err := client.ListServices("")
   485  		if err != nil {
   486  			return err
   487  		}
   488  
   489  		serviceCompMap := make(map[string]string)
   490  		for _, gotService := range services {
   491  			serviceCompMap[gotService.Labels[componentlabels.ComponentLabel]] = gotService.Name
   492  		}
   493  
   494  		for _, secret := range secrets {
   495  			serviceName, serviceOK := secret.Labels[service.ServiceLabel]
   496  			_, linkOK := secret.Labels[service.LinkLabel]
   497  			serviceKind, serviceKindOK := secret.Labels[service.ServiceKind]
   498  			if serviceKindOK && serviceOK && linkOK {
   499  				if serviceKind == "Service" {
   500  					if _, ok := serviceBindings[secret.Name]; !ok {
   501  						serviceBindings[secret.Name] = serviceCompMap[serviceName]
   502  					}
   503  				} else {
   504  					// service name is stored as kind-name in the labels
   505  					parts := strings.SplitN(serviceName, "-", 2)
   506  					if len(parts) < 2 {
   507  						continue
   508  					}
   509  
   510  					serviceName = fmt.Sprintf("%v/%v", parts[0], parts[1])
   511  					if _, ok := serviceBindings[secret.Name]; !ok {
   512  						serviceBindings[secret.Name] = serviceName
   513  					}
   514  				}
   515  			}
   516  		}
   517  	}
   518  
   519  	for i, linkedSecret := range linkedSecrets {
   520  		linkedSecrets[i].ServiceName = serviceBindings[linkedSecret.SecretName]
   521  	}
   522  	return nil
   523  }
