   102  func (d devfileComponent) GetLinkedSecrets() (secretMounts []SecretMount) {
   103  	for _, container := range d.d.Spec.Template.Spec.Containers {
   104  		for _, env := range container.EnvFrom {
   105  			if env.SecretRef != nil {
   106  				secretMounts = append(secretMounts, SecretMount{
   107  					SecretName:  env.SecretRef.Name,
   108  					MountVolume: false,
   109  				})
   110  			}
   111  		}
   112  	}
   113  
   114  	for _, volume := range d.d.Spec.Template.Spec.Volumes {
   115  		if volume.Secret != nil {
   116  			mountPath := ""
   117  			for _, container := range d.d.Spec.Template.Spec.Containers {
   118  				for _, mount := range container.VolumeMounts {
   119  					if mount.Name == volume.Name {
   120  						mountPath = mount.MountPath
   121  						break
   122  					}
   123  				}
   124  			}
   125  			secretMounts = append(secretMounts, SecretMount{
   126  				SecretName:  volume.Secret.SecretName,
   127  				MountVolume: true,
   128  				MountPath:   mountPath,
   129  			})
   130  		}
   131  	}
   132  
   133  	return secretMounts
   134  }

   160  func getType(component provider) (string, error) {
   161  	if componentType, ok := component.GetAnnotations()[componentlabels.ComponentTypeAnnotation]; ok {
   162  		return componentType, nil
   163  	} else if _, ok = component.GetLabels()[componentlabels.ComponentTypeLabel]; ok {
   164  		klog.V(1).Info("No annotation assigned; retuning 'Not available' since labels are assigned. Annotations will be assigned when user pushes again.")
   165  		return NotAvailable, nil
   166  	}
   167  	return "", fmt.Errorf("%s component doesn't provide a type annotation; consider pushing the component again", component.GetName())
   168  }

   170  // GetPushedComponents retrieves a map of PushedComponents from the cluster, keyed by their name
   171  func GetPushedComponents(c kclient.ClientInterface, applicationName string) (map[string]PushedComponent, error) {
   172  	applicationSelector := fmt.Sprintf("%s=%s", applabels.ApplicationLabel, applicationName)
   173  
   174  	deploymentList, err := c.ListDeployments(applicationSelector)
   175  	if err != nil {
   176  		return nil, errors.Wrapf(err, "unable to list components")
   177  	}
   178  	res := make(map[string]PushedComponent, len(deploymentList.Items))
   179  	for _, d := range deploymentList.Items {
   180  		deployment := d
   181  		storageClient := storage.NewClient(storage.ClientOptions{
   182  			Client:     c,
   183  			Deployment: &deployment,
   184  		})
   185  
   186  		urlClient := url.NewClient(url.ClientOptions{
   187  			Client:     c,
   188  			Deployment: &deployment,
   189  		})
   190  		comp := newPushedComponent(applicationName, &devfileComponent{d: d}, c, storageClient, urlClient)
   191  		res[comp.GetName()] = comp
   192  	}
   193  
   194  	return res, nil
   195  }

   207  // GetPushedComponent returns an abstraction over the cluster representation of the component
   208  func GetPushedComponent(c kclient.ClientInterface, componentName, applicationName string) (PushedComponent, error) {
   209  	d, err := c.GetOneDeployment(componentName, applicationName)
   210  	if err != nil {
   211  		if isIgnorableError(err) {
   212  			return nil, nil
   213  		}
   214  		return nil, err
   215  	}
   216  	storageClient := storage.NewClient(storage.ClientOptions{
   217  		Client:     c,
   218  		Deployment: d,
   219  	})
   220  
   221  	urlClient := url.NewClient(url.ClientOptions{
   222  		Client:     c,
   223  		Deployment: d,
   224  	})
   225  	return newPushedComponent(applicationName, &devfileComponent{d: *d}, c, storageClient, urlClient), nil
   226  }

   228  func isIgnorableError(err error) bool {
   229  	e := errors.Cause(err)
   230  	if e != nil {
   231  		err = e
   232  	}
   233  	if _, ok := err.(*kclient.DeploymentNotFoundError); ok {
   234  		return true
   235  	}
   236  	return kerrors.IsNotFound(err) || kerrors.IsForbidden(err) || kerrors.IsUnauthorized(err)
