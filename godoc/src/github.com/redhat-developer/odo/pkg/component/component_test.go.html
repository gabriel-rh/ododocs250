   196  				if !ok {
   197  					return false, nil, fmt.Errorf("expected a ListAction, got %v", action)
   198  				}

   200  					return true, &tt.deploymentList, nil
   201  				}

   204  				var deploymentLabels1 map[string]string
   205  				if len(tt.deploymentList.Items) == 2 {

   209  				switch listAction.GetListRestrictions().Labels.String() {
   210  				case util.ConvertLabelsToSelector(deploymentLabels0):
   211  					return true, &tt.deploymentList.Items[0], nil
   212  				case util.ConvertLabelsToSelector(deploymentLabels1):
   213  					return true, &tt.deploymentList.Items[1], nil
   214  				default:
   215  					return true, &tt.deploymentList, nil
   216  				}
   217  			})

   219  			results, err := List(client, applabels.GetSelector("app"))
