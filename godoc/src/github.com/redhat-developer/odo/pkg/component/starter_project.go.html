    39  // GetStarterProject gets starter project value from flag --starter.
    40  func GetStarterProject(projects []devfilev1.StarterProject, projectPassed string) (project *devfilev1.StarterProject, err error) {
    41  
    42  	nOfProjects := len(projects)
    43  
    44  	if nOfProjects == 0 {
    45  		return nil, errors.Errorf("no starter project found in devfile.")
    46  	}
    47  
    48  	// Determine what project to be used
    49  	if nOfProjects == 1 && projectPassed == defaultStarterProjectName {
    50  		project = &projects[0]
    51  	} else if nOfProjects > 1 && projectPassed == defaultStarterProjectName {
    52  		project = &projects[0]
    53  		log.Warning("There are multiple projects in this devfile but none have been specified in --starter. Downloading the first: " + project.Name)
    54  	} else { //If the user has specified a project
    55  		var availableNames []string
    56  
    57  		projectFound := false
    58  		for indexOfProject, projectInfo := range projects {
    59  			availableNames = append(availableNames, projectInfo.Name)
    60  			if projectInfo.Name == projectPassed { //Get the index
    61  				project = &projects[indexOfProject]
    62  				projectFound = true
    63  			}
    64  		}
    65  
    66  		if !projectFound {
    67  			availableNamesString := strings.Join(availableNames, ",")
    68  			return nil, errors.Errorf("the project: %s specified in --starter does not exist, available projects: %s", projectPassed, availableNamesString)
    69  		}
    70  	}
    71  
    72  	return project, err
    73  
    74  }

    76  // DownloadStarterProject Downloads first starter project from list of starter projects in devfile
    77  func DownloadStarterProject(starterProject *devfilev1.StarterProject, decryptedToken string, contextDir string) error {
    78  	var path string
    79  	var err error
    80  	// Retrieve the working directory in order to clone correctly
    81  	if contextDir == "" {
    82  		path, err = os.Getwd()
    83  		if err != nil {
    84  			return errors.Wrapf(err, "Could not get the current working directory.")
    85  		}
    86  	} else {
    87  		path = contextDir
    88  	}
    89  
    90  	// We will check to see if the project has a valid directory
    91  	err = util.IsValidProjectDir(path, location.DevfileLocation(""))
    92  	if err != nil {
    93  		return err
    94  	}
    95  
    96  	log.Info("\nStarter Project")
    97  
    98  	if starterProject.Git != nil {
    99  		err := downloadGitProject(starterProject, decryptedToken, path)
   100  
   101  		if err != nil {
   102  			return err
   103  		}
   104  
   105  	} else if starterProject.Zip != nil {
   106  		url := starterProject.Zip.Location
   107  		sparseDir := starterProject.SubDir
   108  		downloadSpinner := log.Spinnerf("Downloading starter project %s from %s", starterProject.Name, url)
   109  		err := checkoutProject(sparseDir, url, path, decryptedToken)
   110  		if err != nil {
   111  			downloadSpinner.End(false)
   112  			return err
   113  		}
   114  		downloadSpinner.End(true)
   115  	} else {
   116  		return errors.Errorf("Project type not supported")
   117  	}
   118  
   119  	return nil
   120  }

   122  // downloadGitProject downloads the git starter projects from devfile.yaml
   123  func downloadGitProject(starterProject *devfilev1.StarterProject, starterToken, path string) error {
   124  	remoteName, remoteUrl, revision, err := parsercommon.GetDefaultSource(starterProject.Git.GitLikeProjectSource)
   125  	if err != nil {
   126  		return errors.Wrapf(err, "unable to get default project source for starter project %s", starterProject.Name)
   127  	}
   128  
   129  	// convert revision to referenceName type, ref name could be a branch or tag
   130  	// if revision is not specified it would be the default branch of the project
   131  	refName := plumbing.ReferenceName(revision)
   132  
   133  	if plumbing.IsHash(revision) {
   134  		// Specifying commit in the reference name is not supported by the go-git library
   135  		// while doing git.PlainClone()
   136  		log.Warning("Specifying commit in 'revision' is not yet supported in odo.")
   137  		// overriding revision to empty as we do not support this
   138  		revision = ""
   139  	}
   140  
   141  	if revision != "" {
   142  		// lets consider revision to be a branch name first
   143  		refName = plumbing.NewBranchReferenceName(revision)
   144  	}
   145  
   146  	downloadSpinner := log.Spinnerf("Downloading starter project %s from %s", starterProject.Name, remoteUrl)
   147  	defer downloadSpinner.End(false)
   148  
   149  	cloneOptions := &git.CloneOptions{
   150  		URL:           remoteUrl,
   151  		RemoteName:    remoteName,
   152  		ReferenceName: refName,
   153  		SingleBranch:  true,
   154  		// we don't need history for starter projects
   155  		Depth: 1,
   156  	}
   157  
   158  	if starterToken != "" {
   159  		cloneOptions.Auth = &http.BasicAuth{
   160  			Username: registryUtil.RegistryUser,
   161  			Password: starterToken,
   162  		}
   163  	}
   164  
   165  	originalPath := ""
   166  	if starterProject.SubDir != "" {
   167  		originalPath = path
   168  		path, err = ioutil.TempDir("", "")
   169  		if err != nil {
   170  			return err
   171  		}
   172  	}
   173  
   174  	_, err = git.PlainClone(path, false, cloneOptions)
   175  
   176  	if err != nil {
   177  
   178  		// it returns the following error if no matching ref found
   179  		// if we get this error, we are trying again considering revision as tag, only if revision is specified.
   180  		if _, ok := err.(git.NoMatchingRefSpecError); !ok || revision == "" {
   181  			return err
   182  		}
   183  
   184  		// try again to consider revision as tag name
   185  		cloneOptions.ReferenceName = plumbing.NewTagReferenceName(revision)
   186  		// remove if any .git folder downloaded in above try
   187  		_ = os.RemoveAll(filepath.Join(path, ".git"))
   188  		_, err = git.PlainClone(path, false, cloneOptions)
   189  		if err != nil {
   190  			return err
   191  		}
   192  	}
   193  
   194  	// we don't want to download project be a git repo
   195  	err = os.RemoveAll(filepath.Join(path, ".git"))
   196  	if err != nil {
   197  		// we don't need to return (fail) if this happens
   198  		log.Warning("Unable to delete .git from cloned starter project")
   199  	}
   200  
   201  	if starterProject.SubDir != "" {
   202  		err = util.GitSubDir(path, originalPath,
   203  			starterProject.SubDir)
   204  		if err != nil {
   205  			return err
   206  		}
   207  	}
   208  	downloadSpinner.End(true)
   209  
   210  	return nil
   211  
   212  }
