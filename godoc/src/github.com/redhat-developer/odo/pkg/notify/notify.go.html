    16  // getLatestReleaseTag polls odo's upstream GitHub repository to get the
    17  // tag of the latest release
    18  func getLatestReleaseTag() (string, error) {
    19  
    20  	request := util.HTTPRequestParams{
    21  		URL: VersionFetchURL,
    22  	}
    23  
    24  	// Make request and cache response for 60 minutes
    25  	body, err := util.HTTPGetRequest(request, 60)
    26  	if err != nil {
    27  		return "", errors.Wrap(err, "error getting latest release")
    28  	}
    29  
    30  	return strings.TrimSuffix(string(body), "\n"), nil

    33  // CheckLatestReleaseTag returns the latest release tag if a newer latest
    34  // release is available, else returns an empty string
    35  func CheckLatestReleaseTag(currentVersion string) (string, error) {
    36  	currentSemver, err := semver.Make(strings.TrimPrefix(currentVersion, "v"))
    37  	if err != nil {
    38  		return "", errors.Wrapf(err, "unable to make semver from the current version: %v", currentVersion)
    39  	}
    40  
    41  	latestTag, err := getLatestReleaseTag()
    42  	if err != nil {
    43  		return "", errors.Wrap(err, "unable to get latest release tag")
    44  	}
    45  
    46  	latestSemver, err := semver.Make(strings.TrimPrefix(latestTag, "v"))
    47  	if err != nil {
    48  		return "", errors.Wrapf(err, "unable to make semver from the latest release tag: %v", latestTag)
    49  	}
    50  
    51  	if currentSemver.LT(latestSemver) {
    52  		return latestTag, nil
    53  	}
    54  
    55  	return "", nil
    56  }
