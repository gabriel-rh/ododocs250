    89  // checkIfConfigurationNeeded checks against a set of commands that do *NOT* need configuration.
    90  func (o *Cobra) CheckIfConfigurationNeeded() (bool, error) {
    91  	// Here we will check for parent commands, if the match a certain criteria, we will skip
    92  	// using the configuration.
    93  	//
    94  	// For example, `odo create` should NOT check to see if there is actually a configuration yet.
    95  	if o.cmd.HasParent() {
    96  
    97  		// Gather necessary preliminary information
    98  		parentCommand := o.cmd.Parent()
    99  		rootCommand := o.cmd.Root()
   100  		flagValue := o.FlagValueIfSet(util.ApplicationFlagName)
   101  
   102  		// Find the first child of the command, as some groups are allowed even with non existent configuration
   103  		firstChildCommand := getFirstChildOfCommand(o.cmd)
   104  
   105  		// This should *never* happen, but added just to be safe
   106  		if firstChildCommand == nil {
   107  			return false, fmt.Errorf("Unable to get first child of command")
   108  		}
   109  		// Case 1 : if command is create operation just allow it
   110  		if o.cmd.Name() == "create" && (parentCommand.Name() == "component" || parentCommand.Name() == rootCommand.Name()) {
   111  			return true, nil
   112  		}
   113  		// Case 2 : if command is describe or delete and app flag is used just allow it
   114  		if (firstChildCommand.Name() == "describe" || firstChildCommand.Name() == "delete") && len(flagValue) > 0 {
   115  			return true, nil
   116  		}
   117  		// Case 3 : if command is list, just allow it
   118  		if firstChildCommand.Name() == "list" {
   119  			return true, nil
   120  		}
   121  		// Case 4 : Check if firstChildCommand is project. If so, skip validation of context
   122  		if firstChildCommand.Name() == "project" {
   123  			return true, nil
   124  		}
   125  		// Case 5 : Check if specific flags are set for specific first child commands
   126  		if firstChildCommand.Name() == "app" {
   127  			return true, nil
   128  		}
   129  		// Case 6 : Check if firstChildCommand is catalog and request is to list or search
   130  		if firstChildCommand.Name() == "catalog" {
   131  			return true, nil
   132  		}
   133  		// Case 7: Check if firstChildCommand is component and  request is list
   134  		if (firstChildCommand.Name() == "component" || firstChildCommand.Name() == "service") && o.cmd.Name() == "list" {
   135  			return true, nil
   136  		}
   137  		// Case 8 : Check if firstChildCommand is component and app flag is used
   138  		if firstChildCommand.Name() == "component" && len(flagValue) > 0 {
   139  			return true, nil
   140  		}
   141  		// Case 9 : Check if firstChildCommand is logout and app flag is used
   142  		if firstChildCommand.Name() == "logout" {
   143  			return true, nil
   144  		}
   145  		// Case 10: Check if firstChildCommand is service and command is create or delete. Allow it if that's the case
   146  		if firstChildCommand.Name() == "service" && (o.cmd.Name() == "create" || o.cmd.Name() == "delete") {
   147  			return true, nil
   148  		}
   149  		// Case 11 : if command is deploy
   150  		if o.cmd.Name() == "deploy" {
   151  			return true, nil
   152  		}
   153  
   154  	} else {
   155  		return true, nil
   156  	}
   157  
   158  	return false, nil
   159  }

   161  // getFirstChildOfCommand gets the first child command of the root command of command
   162  func getFirstChildOfCommand(command *cobra.Command) *cobra.Command {
   163  	// If command does not have a parent no point checking
   164  	if command.HasParent() {
   165  		// Get the root command and set current command and its parent
   166  		rootCommand := command.Root()
   167  		parentCommand := command.Parent()
   168  		mainCommand := command
   169  		for {
   170  			// if parent is root, then we have our first child in c
   171  			if parentCommand == rootCommand {
   172  				return mainCommand
   173  			}
   174  			// Traverse backwards making current command as the parent and parent as the grandparent
   175  			mainCommand = parentCommand
   176  			parentCommand = mainCommand.Parent()
   177  		}
   178  	}
   179  	return nil
   180  }
