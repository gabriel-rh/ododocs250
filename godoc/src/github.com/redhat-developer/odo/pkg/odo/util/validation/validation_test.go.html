   110  		},
   111  		{

   114  			expected:    []survey.Validator{testValidator},
   115  		},

   117  
   118  	for _, tt := range tests {
   119  		t.Run(tt.name, func(t *testing.T) {
   120  			validator, chain := internalGetValidatorFor(tt.validatable)
   121  
   122  			// if validator chain is empty, only possible result is NilValidator
   123  			if len(chain) == 0 {
