    46  // NewParsedArgs creates a parsed representation of the provided arguments for the specified command. Mostly exposed for tests.
    47  func NewParsedArgs(args complete.Args, cmd *cobra.Command) parsedArgs {
    48  	typed := getUserTypedCommands(args, cmd)
    49  	commands, flagValues := getCommandsAndFlags(typed, cmd)
    50  
    51  	complete.Log("Parsed commands values for full input: %v", commands)
    52  	complete.Log("Parsed flag values for full input: %v", flagValues)
    53  
    54  	// the parsing above won't work properly when we have previously supplied flags and we are trying to complete
    55  	// another flag
    56  	// For example when the user has typed "odo service create --plan prod -p "
    57  	// the flagValues var above would not contains the plan flag and value
    58  	// So in the following part of the code we strip the last thing the user added and parse again
    59  	// any new parsed flags will be added
    60  	// There is no need to do this extra parsing if the user has not already added a flag. So as a heuristic
    61  	// we only perform this parsing when at the user has added at least 2 tokens
    62  	if len(typed) > 2 {
    63  		typedWithoutLast := make([]string, len(typed))
    64  		copy(typedWithoutLast, typed)
    65  		typedWithoutLast = typedWithoutLast[:len(typedWithoutLast)-1]
    66  		_, flagValuesWithoutLast := getCommandsAndFlags(typedWithoutLast, cmd)
    67  
    68  		complete.Log("Parsed flag values for input without last: %v", flagValuesWithoutLast)
    69  
    70  		// now we merge the two maps
    71  		for k, v := range flagValuesWithoutLast {
    72  			flagValues[k] = v
    73  		}
    74  	}
    75  
    76  	parsed := parsedArgs{
    77  		original:   args,
    78  		typed:      typed,
    79  		commands:   commands,
    80  		flagValues: flagValues,
    81  	}
    82  	_ = cmd.ParseFlags(typed)
    83  
    84  	return parsed
    85  }

   139  // getCommandsAndFlags returns the commands and flags from the given input
   140  func getCommandsAndFlags(args []string, c *cobra.Command) (map[string]bool, map[string]string) {
   141  	strippedCommandsMap := make(map[string]bool)
   142  	setFlags := make(map[string]string)
   143  
   144  	if len(args) == 0 {
   145  		return strippedCommandsMap, setFlags
   146  	}
   147  	err := c.ParseFlags(args)
   148  	if err != nil {
   149  		return strippedCommandsMap, setFlags
   150  	}
   151  
   152  	flags := c.Flags()
   153  
   154  	cmds := flags.Args()
   155  	flags.Visit(func(i *flag.Flag) {
   156  		setFlags[i.Name] = i.Value.String()
   157  	})
   158  
   159  	// send a map of commands for faster searching
   160  	for _, strippedCommand := range cmds {
   161  		strippedCommandsMap[strippedCommand] = true
   162  	}
   163  
   164  	return strippedCommandsMap, setFlags
   165  }

   167  // getUserTypedCommands returns only the user typed entities by excluding the cobra predefined commands
   168  func getUserTypedCommands(args complete.Args, command *cobra.Command) []string {
   169  	var commands []string
   170  
   171  	// get only the user typed commands/flags and remove the cobra defined commands
   172  	found := false
   173  	for _, arg := range args.Completed {
   174  		if arg == command.Name() && !found {
   175  			found = true
   176  			continue
   177  		}
   178  		if found {
   179  			commands = append(commands, arg)
   180  		}
   181  	}
   182  
   183  	return commands
   184  }
