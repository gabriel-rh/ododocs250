    27  // If the context is provided, then that is printed, if not, then the cause is
    28  // detected using errors.Cause(err)

    30  func LogErrorAndExit(err error, context string, a ...interface{}) {
    31  
    32  	if err != nil {
    33  
    34  		// If it's JSON, we'll output  the error
    35  		if log.IsJSON() {
    36  
    37  			// Machine readble error output
    38  			machineOutput := machineoutput.GenericError{
    39  				TypeMeta: metav1.TypeMeta{
    40  					Kind:       machineoutput.ErrorKind,
    41  					APIVersion: machineoutput.APIVersion,
    42  				},
    43  				Message: err.Error(),
    44  			}
    45  
    46  			// Output the error
    47  			machineoutput.OutputError(machineOutput)
    48  
    49  		} else {
    50  			if context == "" {
    51  				log.Error(errors.Cause(err))
    52  			} else {
    53  				printstring := fmt.Sprintf("%s%s", strings.Title(context), "\nError: %v")
    54  				log.Errorf(printstring, err)
    55  			}
    56  		}
    57  
    58  		// Always exit 1 anyways
    59  		os.Exit(1)

    61  	}

    76  func PrintComponentInfo(client kclient.ClientInterface, currentComponentName string, componentDesc component.Component, applicationName string, project string) error {
    77  
    78  	log.Describef("Component Name: ", currentComponentName)
    79  	log.Describef("Type: ", componentDesc.Spec.Type)
    80  
    81  	// Source
    82  	if componentDesc.Spec.Source != "" {
    83  		log.Describef("Source: ", componentDesc.Spec.Source)
    84  	}
    85  
    86  	// Env
    87  	if componentDesc.Spec.Env != nil {
    88  
    89  		// Retrieve all the environment variables
    90  		var output string
    91  		for _, env := range componentDesc.Spec.Env {
    92  			output += fmt.Sprintf(" · %v=%v\n", env.Name, env.Value)
    93  		}
    94  
    95  		// Cut off the last newline and output
    96  		if len(output) > 0 {
    97  			output = output[:len(output)-1]
    98  			log.Describef("Environment Variables:\n", output)
    99  		}
   100  
   101  	}
   102  
   103  	// Storage
   104  	if len(componentDesc.Spec.Storage) > 0 {
   105  
   106  		var storages storage.StorageList
   107  
   108  		if componentDesc.Status.State == "Pushed" {
   109  			// Retrieve the storage list
   110  			storages = storage.StorageList{Items: componentDesc.Spec.StorageSpec}
   111  		} else {
   112  			envInfo, err := envinfo.New()
   113  			if err != nil {
   114  				return err
   115  			}
   116  			storageLocal, err := envInfo.ListStorage()
   117  			if err != nil {
   118  				return err
   119  			}
   120  			storages = storage.ConvertListLocalToMachine(storageLocal)
   121  
   122  		}
   123  
   124  		// Gather the output
   125  		var output string
   126  		for _, store := range storages.Items {
   127  			output += fmt.Sprintf(" · %v of size %v mounted to %v\n", store.Name, store.Spec.Size, store.Spec.Path)
   128  		}
   129  
   130  		// Cut off the last newline and output
   131  		if len(output) > 0 {
   132  			output = output[:len(output)-1]
   133  			log.Describef("Storage:\n", output)
   134  		}
   135  
   136  	}
   137  
   138  	// URL
   139  	if componentDesc.Spec.URL != nil {
   140  		var output string
   141  
   142  		// S2I Only (odo describe exits by default on Devfile by default anyways..)
   143  		// if the component is not pushed
   144  		if componentDesc.Status.State == component.StateTypeNotPushed {
   145  			// Gather the output
   146  			for i, componentURL := range componentDesc.Spec.URL {
   147  				output += fmt.Sprintf(" · URL named %s will be exposed via %v\n", componentURL, componentDesc.Spec.Ports[i])
   148  			}
   149  		} else {
   150  			// Retrieve the URLs
   151  			urls := urlPkg.URLList{Items: componentDesc.Spec.URLSpec}
   152  
   153  			// Gather the output
   154  			for _, componentURL := range componentDesc.Spec.URL {
   155  				url := urls.Get(componentURL)
   156  
   157  				var urlString string
   158  
   159  				switch url.Spec.Kind {
   160  				case localConfigProvider.ROUTE:
   161  					urlString = urlPkg.GetURLString(url.Spec.Protocol, url.Spec.Host, "")
   162  				case localConfigProvider.INGRESS:
   163  					urlString = urlPkg.GetURLString(url.Spec.Protocol, "", url.Spec.Host)
   164  				default:
   165  					continue
   166  				}
   167  
   168  				output += fmt.Sprintf(" · %v exposed via %v\n", urlString, url.Spec.Port)
   169  			}
   170  
   171  		}
   172  
   173  		// Cut off the last newline and output
   174  		if len(output) > 0 {
   175  			output = output[:len(output)-1]
   176  			log.Describef("URLs:\n", output)
   177  		}
   178  
   179  	}
   180  
   181  	// Linked services
   182  	if len(componentDesc.Status.LinkedServices) > 0 {
   183  
   184  		// Gather the output
   185  		var output string
   186  		for _, linkedService := range componentDesc.Status.LinkedServices {
   187  
   188  			// Let's also get the secrets / environment variables that are being passed in.. (if there are any)
   189  			secrets, err := client.GetSecret(linkedService.SecretName, project)
   190  			if err != nil {
   191  				return err
   192  			}
   193  
   194  			if len(secrets.Data) > 0 {
   195  				// Iterate through the secrets to throw in a string
   196  				var secretOutput string
   197  				for i := range secrets.Data {
   198  					secretOutput += fmt.Sprintf("    · %v\n", i)
   199  				}
   200  
   201  				if len(secretOutput) > 0 {
   202  					// Cut off the last newline
   203  					secretOutput = secretOutput[:len(secretOutput)-1]
   204  					output += fmt.Sprintf(" · %s\n   Environment Variables:\n%s\n", linkedService.SecretName, secretOutput)
   205  				}
   206  
   207  			} else {
   208  				output += fmt.Sprintf(" · %s\n", linkedService.SecretName)
   209  			}
   210  
   211  		}
   212  
   213  		if len(output) > 0 {
   214  			// Cut off the last newline and output
   215  			output = output[:len(output)-1]
   216  			log.Describef("Linked Services:\n", output)
   217  
   218  		}
   219  
   220  	}
   221  	return nil
   222  }

   238  // ModifyAdditionalFlags modifies the flags and updates the descriptions
   239  // as well as changes whether or not machine readable output
   240  // has been passed in..
   241  //
   242  // Return the flag usages for the help output
   243  func ModifyAdditionalFlags(cmd *cobra.Command) string {
   244  
   245  	// Hide the machine readable output if the command
   246  	// does not have the annotation.
   247  	machineOutput := cmd.Annotations["machineoutput"]
   248  	f := cmd.InheritedFlags()
   249  
   250  	f.VisitAll(func(f *pflag.Flag) {
   251  		// Remove json flag if machineoutput has not been passed in
   252  		if f.Name == "o" && machineOutput == "json" {
   253  			f.Hidden = false
   254  		}
   255  	})
   256  
   257  	return CapitalizeFlagDescriptions(f)
   258  }

   260  // CapitalizeFlagDescriptions adds capitalizations
   261  func CapitalizeFlagDescriptions(f *pflag.FlagSet) string {
   262  	f.VisitAll(func(f *pflag.Flag) {
   263  		cap := []rune(f.Usage)
   264  		cap[0] = unicode.ToUpper(cap[0])
   265  		f.Usage = string(cap)
   266  	})
   267  	return f.FlagUsages()
   268  }

   302  // MapFromParameters takes an array of strings which are of the format "key=value" and returns a map from these
   303  // It returns error if any of the string in the array doesn't adhere to "key=value" format
   304  func MapFromParameters(params []string) (map[string]string, error) {
   305  	paramsMap := make(map[string]string)
   306  
   307  	for _, kv := range params {
   308  		kvSlice := strings.SplitN(kv, "=", 2)
   309  		// key value not provided in format of key=value
   310  		if len(kvSlice) != 2 {
   311  			return nil, errors.New("parameters not provided in key=value format")
   312  		}
   313  		paramsMap[kvSlice[0]] = kvSlice[1]
   314  	}
   315  
   316  	return paramsMap, nil
   317  }
