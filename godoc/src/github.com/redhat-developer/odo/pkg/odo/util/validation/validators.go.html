    85  // KeyEqValFormatValidator ensures that all the parts of the input follow the key=value format
    86  // examples of valid input are:
    87  // NAME=JANE
    88  // PORT=8080,PATH=/health
    89  func KeyEqValFormatValidator(portsStr interface{}) error {
    90  	if s, ok := portsStr.(string); ok {
    91  		parts := util.GetSplitValuesFromStr(s)
    92  		for _, part := range parts {
    93  			kvParts := strings.Split(part, "=")
    94  			if len(kvParts) != 2 {
    95  				return fmt.Errorf("Part '%s' does not have the correct format", part)
    96  			}
    97  			if len(strings.TrimSpace(kvParts[0])) != len(kvParts[0]) ||
    98  				len(strings.TrimSpace(kvParts[1])) != len(kvParts[1]) {
    99  				return fmt.Errorf("Spaces are not allowed in '%s'", part)
   100  			}
   101  		}
   102  
   103  		return nil
   104  	}
   105  
   106  	return fmt.Errorf("can only validate strings, got %v", portsStr)
   107  }

   116  // internalGetValidatorFor exposed for testing purposes
   117  func internalGetValidatorFor(prop Validatable) (validator Validator, chain []survey.Validator) {
   118  	// make sure we don't run into issues when composing validators
   119  	validatorChain := make([]survey.Validator, 0, 5)
   120  
   121  	if prop.Required {
   122  		validatorChain = append(validatorChain, survey.Required)
   123  	}
   124  
   125  	switch prop.Type {
   126  	case "integer":
   127  		validatorChain = append(validatorChain, survey.Validator(IntegerValidator))
   128  	}
   129  
   130  	for i := range prop.AdditionalValidators {
   131  		validatorChain = append(validatorChain, survey.Validator(prop.AdditionalValidators[i]))
   132  	}
   133  
   134  	if len(validatorChain) > 0 {
   135  		return Validator(survey.ComposeValidators(validatorChain...)), validatorChain
   136  	}
   137  
   138  	return NilValidator, validatorChain
   139  }
