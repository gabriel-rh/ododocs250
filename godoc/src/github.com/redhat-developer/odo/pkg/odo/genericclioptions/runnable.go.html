    41  func GenericRun(o Runnable, cmd *cobra.Command, args []string) {
    42  	var err error
    43  	startTime := time.Now()
    44  	cfg, _ := preference.NewClient()
    45  	disableTelemetry, _ := strconv.ParseBool(os.Getenv(segment.DisableTelemetryEnv))
    46  
    47  	// Prompt the user to consent for telemetry if a value is not set already
    48  	// Skip prompting if the preference command is called
    49  	// This prompt has been placed here so that it does not prompt the user when they call --help
    50  
    51  	if !cfg.IsSet(preference.ConsentTelemetrySetting) && cmd.Parent().Name() != "preference" {
    52  		if !segment.RunningInTerminal() {
    53  			klog.V(4).Infof("Skipping telemetry question because there is no terminal (tty)\n")
    54  		} else if disableTelemetry {
    55  			klog.V(4).Infof("Skipping telemetry question due to %s=%t\n", segment.DisableTelemetryEnv, disableTelemetry)
    56  		} else {
    57  			var consentTelemetry bool
    58  			prompt := &survey.Confirm{Message: "Help odo improve by allowing it to collect usage data. Read about our privacy statement: https://developers.redhat.com/article/tool-data-collection. You can change your preference later by changing the ConsentTelemetry preference.", Default: true}
    59  			err = survey.AskOne(prompt, &consentTelemetry, nil)
    60  			ui.HandleError(err)
    61  			if err == nil {
    62  				if err1 := cfg.SetConfiguration(preference.ConsentTelemetrySetting, strconv.FormatBool(consentTelemetry)); err1 != nil {
    63  					klog.V(4).Info(err1.Error())
    64  				}
    65  			}
    66  		}
    67  	}
    68  	// set value for telemetry status in context so that we do not need to call IsTelemetryEnabled every time to check its status
    69  	scontext.SetTelemetryStatus(cmd.Context(), segment.IsTelemetryEnabled(cfg))
    70  
    71  	// Send data to telemetry in case of user interrupt
    72  	captureSignals := []os.Signal{syscall.SIGHUP, syscall.SIGTERM, syscall.SIGQUIT, os.Interrupt}
    73  	go commonutil.StartSignalWatcher(captureSignals, func(receivedSignal os.Signal) {
    74  		scontext.SetSignal(cmd.Context(), receivedSignal)
    75  		startTelemetry(cmd, errors.Wrapf(terminal.InterruptErr, "user interrupted the command execution"), startTime)
    76  	})
    77  
    78  	// CheckMachineReadableOutput
    79  	// fixes / checks all related machine readable output functions
    80  	util.LogErrorAndExit(CheckMachineReadableOutputCommand(cmd), "")
    81  
    82  	// LogErrorAndExit is used so that we get -o (jsonoutput) for cmds which have json output implemented
    83  	util.LogErrorAndExit(checkConflictingFlags(cmd, args), "")
    84  	// Run completion, validation and run.
    85  	// Only upload data to segment for completion and validation if a non-nil error is returned.
    86  	err = o.Complete(cmdline.NewCobra(cmd), args)
    87  	if err != nil {
    88  		startTelemetry(cmd, err, startTime)
    89  	}
    90  	util.LogErrorAndExit(err, "")
    91  
    92  	err = o.Validate()
    93  	if err != nil {
    94  		startTelemetry(cmd, err, startTime)
    95  	}
    96  	util.LogErrorAndExit(err, "")
    97  
    98  	err = o.Run()
    99  	startTelemetry(cmd, err, startTime)
   100  	util.LogErrorAndExit(err, "")
   101  }

   105  func startTelemetry(cmd *cobra.Command, err error, startTime time.Time) {
   106  	if scontext.GetTelemetryStatus(cmd.Context()) && !strings.Contains(cmd.CommandPath(), "telemetry") {
   107  		uploadData := &segment.TelemetryData{
   108  			Event: cmd.CommandPath(),
   109  			Properties: segment.TelemetryProperties{
   110  				Duration:      time.Since(startTime).Milliseconds(),
   111  				Success:       err == nil,
   112  				Tty:           segment.RunningInTerminal(),
   113  				Version:       fmt.Sprintf("odo %v (%v)", version.VERSION, version.GITCOMMIT),
   114  				CmdProperties: scontext.GetContextProperties(cmd.Context()),
   115  			},
   116  		}
   117  		if err != nil {
   118  			uploadData.Properties.Error = segment.SetError(err)
   119  			uploadData.Properties.ErrorType = segment.ErrorType(err)
   120  		}
   121  		data, err1 := json.Marshal(uploadData)
   122  		if err1 != nil {
   123  			klog.V(4).Infof("Failed to marshall telemetry data. %q", err1.Error())
   124  		}
   125  		command := exec.Command(os.Args[0], "telemetry", string(data))
   126  		if err1 = command.Start(); err1 != nil {
   127  			klog.V(4).Infof("Failed to start the telemetry process. Error: %q", err1.Error())
   128  			return
   129  		}
   130  		if err1 = command.Process.Release(); err1 != nil {
   131  			klog.V(4).Infof("Failed to release the process. %q", err1.Error())
   132  			return
   133  		}
   134  	}
   135  }

   137  // checkConflictingFlags checks for conflicting flags. Currently --context cannot be provided
   138  // with either --app, --project and --component as that information can be fetched from the local
   139  // config.
   140  func checkConflictingFlags(cmd *cobra.Command, args []string) error {
   141  
   142  	// we allow providing --context with --app and --project in case of `odo create` or `odo component create`
   143  	if cmd.Name() == "create" {
   144  		if cmd.HasParent() {
   145  			if cmd.Parent().Name() == "odo" || cmd.Parent().Name() == "component" {
   146  				return nil
   147  			}
   148  		}
   149  	}
   150  	app := stringFlagLookup(cmd, "app")
   151  	project := stringFlagLookup(cmd, "project")
   152  	context := stringFlagLookup(cmd, "context")
   153  	component := stringFlagLookup(cmd, "component")
   154  	all, _ := strconv.ParseBool(stringFlagLookup(cmd, "all"))
   155  	// TODO: Move this to a method under DeleteOptions, similar to CreateOptions.checkConflictingFlags
   156  	if cmd.Name() == "delete" {
   157  		if cmd.HasParent() {
   158  			if cmd.Parent().Name() == "odo" || cmd.Parent().Name() == "component" {
   159  				var componentName string
   160  				if len(args) > 0 {
   161  					componentName = args[0]
   162  				}
   163  				if (context != "") && (project != "" || componentName != "") {
   164  					return fmt.Errorf("cannot provide --project or component name when --context is provided")
   165  				}
   166  				if project != "" && componentName == "" && app == "" {
   167  					return fmt.Errorf("cannot provide --project without --app and component name")
   168  				}
   169  				if all && ((componentName != "" && app != "" && project != "") || (componentName != "")) {
   170  					return fmt.Errorf("cannot provide --all when component name, --app and --project are provided")
   171  				}
   172  			}
   173  		}
   174  	}
   175  	if (context != "") && (app != "" || project != "" || component != "") {
   176  		return fmt.Errorf("cannot provide --app, --project or --component flag when --context is provided")
   177  	}
   178  	return nil
   179  }

   190  // CheckMachineReadableOutputCommand performs machine-readable output functions required to
   191  // have it work correctly
   192  func CheckMachineReadableOutputCommand(cmd *cobra.Command) error {
   193  
   194  	// Get the needed values
   195  	outputFlag := pflag.Lookup("o")
   196  	hasFlagChanged := outputFlag != nil && outputFlag.Changed
   197  	machineOutput := cmd.Annotations["machineoutput"]
   198  
   199  	// Check the valid output
   200  	if hasFlagChanged && outputFlag.Value.String() != "json" {
   201  		return fmt.Errorf("Please input a valid output format for -o, available format: json")
   202  	}
   203  
   204  	// Check that if -o json has been passed, that the command actually USES json.. if not, error out.
   205  	if hasFlagChanged && outputFlag.Value.String() == "json" && machineOutput == "" {
   206  
   207  		// By default we "disable" logging, so activate it so that the below error can be shown.
   208  		_ = flag.Set("o", "")
   209  
   210  		// Return the error
   211  		return fmt.Errorf("Machine readable output is not yet implemented for this command")
   212  	}
   213  
   214  	// Before running anything, we will make sure that no verbose output is made
   215  	// This is a HACK to manually override `-v 4` to `-v 0` (in which we have no klog.V(0) in our code...
   216  	// in order to have NO verbose output when combining both `-o json` and `-v 4` so json output
   217  	// is not malformed / mixed in with normal logging
   218  	if log.IsJSON() {
   219  		_ = flag.Set("v", "0")
   220  	} else {
   221  		// Override the logging level by the value (if set) by the ODO_LOG_LEVEL env
   222  		// The "-v" flag set on command line will take precedence over ODO_LOG_LEVEL env
   223  		v := flag.CommandLine.Lookup("v").Value.String()
   224  		if level, ok := os.LookupEnv("ODO_LOG_LEVEL"); ok && v == "0" {
   225  			_ = flag.CommandLine.Set("v", level)
   226  		}
   227  	}
   228  	return nil
   229  }
