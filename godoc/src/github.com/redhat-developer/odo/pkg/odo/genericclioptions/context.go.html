    90  // New creates a context based on the given parameters
    91  func New(parameters CreateParameters) (*Context, error) {
    92  	ctx := internalCxt{}
    93  	var err error
    94  
    95  	ctx.EnvSpecificInfo, err = GetValidEnvInfo(parameters.cmdline)
    96  	if err != nil {
    97  		return nil, err
    98  	}
    99  	ctx.LocalConfigProvider = ctx.EnvSpecificInfo
   100  
   101  	ctx.project = resolveProject(parameters.cmdline, ctx.EnvSpecificInfo)
   102  
   103  	ctx.application = resolveApp(parameters.cmdline, ctx.EnvSpecificInfo, parameters.appIfNeeded)
   104  
   105  	ctx.component = resolveComponent(parameters.cmdline, ctx.EnvSpecificInfo)
   106  
   107  	ctx.componentContext = parameters.componentContext
   108  
   109  	ctx.outputFlag = parameters.cmdline.FlagValueIfSet(util.OutputFlagName)
   110  
   111  	if !parameters.offline {
   112  		ctx.KClient, err = parameters.cmdline.GetKubeClient()
   113  		if err != nil {
   114  			return nil, err
   115  		}
   116  		if e := ctx.resolveProjectAndNamespace(parameters.cmdline, ctx.EnvSpecificInfo); e != nil {
   117  			return nil, e
   118  		}
   119  
   120  		if parameters.cmdline.FlagValueIfSet(util.ComponentFlagName) != "" {
   121  			if err = ctx.checkComponentExistsOrFail(); err != nil {
   122  				return nil, err
   123  			}
   124  		}
   125  
   126  		if parameters.routeAvailability {
   127  			isRouteSupported, err := ctx.KClient.IsRouteSupported()
   128  			if err != nil {
   129  				return nil, err
   130  			}
   131  			ctx.EnvSpecificInfo.SetIsRouteSupported(isRouteSupported)
   132  		}
   133  	}
   134  
   135  	ctx.devfilePath = location.DevfileLocation(parameters.componentContext)
   136  	isDevfile := odoutil.CheckPathExists(ctx.devfilePath)
   137  	if parameters.devfile && isDevfile {
   138  		// Parse devfile and validate
   139  		devObj, err := devfile.ParseAndValidateFromFile(ctx.devfilePath)
   140  		if err != nil {
   141  			return nil, fmt.Errorf("failed to parse the devfile %s, with error: %s", ctx.devfilePath, err)
   142  		}
   143  		err = validate.ValidateDevfileData(devObj.Data)
   144  		if err != nil {
   145  			return nil, err
   146  		}
   147  		ctx.EnvSpecificInfo.SetDevfileObj(devObj)
   148  	}
   149  
   150  	return &Context{
   151  		internalCxt: ctx,
   152  	}, nil
   153  }

   172  // ComponentAllowingEmpty retrieves the optionally specified component or the current one if it is set, allowing empty
   173  // components (instead of exiting with an error) if so specified
   174  func (o *Context) ComponentAllowingEmpty(allowEmpty bool, optionalComponent ...string) (string, error) {
   175  	switch len(optionalComponent) {
   176  	case 0:
   177  		// if we're not specifying a component to resolve, get the current one (resolved in NewContext as cmp)
   178  		// so nothing to do here unless the calling context doesn't allow no component to be set in which case we return an error
   179  		if !allowEmpty && len(o.component) == 0 {
   180  			return "", fmt.Errorf("No component is set")
   181  		}
   182  	case 1:
   183  		cmp := optionalComponent[0]
   184  		o.component = cmp
   185  	default:
   186  		// safeguard: fail if more than one optional string is passed because it would be a programming error
   187  		return "", fmt.Errorf("ComponentAllowingEmpty function only accepts one optional argument, was given: %v", optionalComponent)
   188  	}
   189  
   190  	return o.component, nil
   191  }
