    23  // resolveProjectAndNamespace resolve project in Context and namespace in Kubernetes and OpenShift clients
    24  func (o *internalCxt) resolveProjectAndNamespace(cmdline cmdline.Cmdline, configProvider localConfigProvider.LocalConfigProvider) error {
    25  	var namespace string
    26  	projectFlag := cmdline.FlagValueIfSet(util.ProjectFlagName)
    27  	if len(projectFlag) > 0 {
    28  		// if namespace flag was set, check that the specified namespace exists and use it
    29  		_, err := o.KClient.GetNamespaceNormal(projectFlag)
    30  
    31  		// do not error out when its odo delete -a, so that we let users delete the local config on missing namespace
    32  		if err != nil {
    33  			if cmdline.GetParentName() != "project" && !(cmdline.GetName() == "delete" && cmdline.IsFlagSet("all")) {
    34  				return err
    35  			}
    36  		}
    37  		namespace = projectFlag
    38  	} else {
    39  		namespace = configProvider.GetNamespace()
    40  		if namespace == "" {
    41  			namespace = o.KClient.GetCurrentNamespace()
    42  			if len(namespace) <= 0 {
    43  				errFormat := "Could not get current namespace. Please create or set a namespace\n"
    44  				err := checkProjectCreateOrDeleteOnlyOnInvalidNamespace(cmdline, errFormat)
    45  				if err != nil {
    46  					return err
    47  				}
    48  			}
    49  		}
    50  
    51  		// check that the specified namespace exists
    52  		_, err := o.KClient.GetNamespaceNormal(namespace)
    53  		if err != nil {
    54  			var errFormat string
    55  			if kerrors.IsForbidden(err) {
    56  				errFormat = "You are currently not logged in into the cluster. Use `odo login` first to perform any operation on cluster"
    57  			} else {
    58  				errFormat = fmt.Sprintf("You don't have permission to create or set namespace %q or the namespace doesn't exist. Please create or set a different namespace\n\t", namespace)
    59  			}
    60  
    61  			// errFormat := fmt.Sprint(e1, "%s project create|set <project_name>")
    62  			err = checkProjectCreateOrDeleteOnlyOnInvalidNamespaceNoFmt(cmdline, errFormat)
    63  			if err != nil {
    64  				return err
    65  			}
    66  		}
    67  	}
    68  	o.KClient.SetNamespace(namespace)
    69  	o.project = namespace
    70  	return nil
    71  }
