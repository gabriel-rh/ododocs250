    47  	o.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline).NeedDevfile(o.contextFlag))

    69  	svcDefined, err := o.Backend.ServiceDefined(o.Context, o.serviceName)

    71  		return err

    74  	svcDeployed, err := svc.OperatorSvcExists(o.KClient, o.serviceName)
    75  	if err != nil {
    76  		return err

    78  
    79  	if !svcDefined && !svcDeployed {

    82  	return nil
