    32  // CompleteServiceCreate contains logic to complete the "odo service create" call for the case of Operator backend
    33  func (b *OperatorBackend) CompleteServiceCreate(o *CreateOptions, args []string) (err error) {
    34  	// since interactive mode is not supported for Operators yet, set it to false
    35  	o.interactive = false
    36  
    37  	// if user has just used "odo service create", simply return
    38  	if o.fromFileFlag == "" && len(args) == 0 {
    39  		return
    40  	}
    41  
    42  	// if user wants to create service from file and use a name given on CLI
    43  	if o.fromFileFlag != "" {
    44  		if len(args) == 1 {
    45  			o.ServiceName = args[0]
    46  		}
    47  		return
    48  	}
    49  
    50  	// split the name provided on CLI and populate servicetype & customresource
    51  	o.ServiceType, b.CustomResource, err = svc.SplitServiceKindName(args[0])
    52  	if err != nil {
    53  		return fmt.Errorf("invalid service name, use the format <operator-type>/<crd-name>")
    54  	}
    55  
    56  	// if two args are given, first is service type and second one is service name
    57  	if len(args) == 2 {
    58  		o.ServiceName = args[1]
    59  	}
    60  
    61  	return nil
    62  }

    64  func (b *OperatorBackend) ValidateServiceCreate(o *CreateOptions) error {
    65  	u := unstructured.Unstructured{}
    66  	// if the user wants to create service from a file, we check for
    67  	// existence of file and validate if the requested operator and CR
    68  	// exist on the cluster
    69  	if o.fromFileFlag != "" {
    70  		if _, err := os.Stat(o.fromFileFlag); err != nil {
    71  			return errors.Wrap(err, "unable to find specified file")
    72  		}
    73  
    74  		// Parse the file to find Operator and CR info
    75  		fileContents, err := ioutil.ReadFile(o.fromFileFlag)
    76  		if err != nil {
    77  			return err
    78  		}
    79  
    80  		err = yaml.Unmarshal(fileContents, &u.Object)
    81  		if err != nil {
    82  			return err
    83  		}
    84  
    85  		gvk := u.GroupVersionKind()
    86  		b.group, b.version, b.kind = gvk.Group, gvk.Version, gvk.Kind
    87  
    88  		if u.GetName() == "" {
    89  			return ErrNoMetadataName
    90  		}
    91  
    92  		if o.ServiceName != "" && !o.DryRunFlag {
    93  			// First check if service with provided name already exists
    94  			svcFullName := strings.Join([]string{b.kind, o.ServiceName}, "/")
    95  			exists, e := svc.OperatorSvcExists(o.KClient, svcFullName)
    96  			if e != nil {
    97  				return e
    98  			}
    99  			if exists {
   100  				return fmt.Errorf("service %q already exists; please provide a different name or delete the existing service first", svcFullName)
   101  			}
   102  			u.SetName(o.ServiceName)
   103  		} else {
   104  			o.ServiceName = u.GetName()
   105  		}
   106  
   107  		csvPtr, err := o.KClient.GetCSVWithCR(u.GetKind())
   108  		if err != nil {
   109  			// error only occurs when OperatorHub is not installed.
   110  			// k8s doesn't have it installed by default but OCP does
   111  			return err
   112  		}
   113  		csv := *csvPtr
   114  
   115  		// CRD is valid. We can use it further to create a service from it.
   116  		b.CustomResourceDefinition = u.Object
   117  
   118  		// Validate spec
   119  		hasCR, cr := o.KClient.CheckCustomResourceInCSV(b.kind, &csv)
   120  		if !hasCR {
   121  			return fmt.Errorf("the %q resource doesn't exist in specified %q operator", b.CustomResource, b.group)
   122  		}
   123  
   124  		crd, err := o.KClient.GetCRDSpec(cr, b.group, b.kind)
   125  		if err != nil {
   126  			return err
   127  		}
   128  
   129  		err = validate.AgainstSchema(crd, u.Object["spec"], strfmt.Default)
   130  		if err != nil {
   131  			return err
   132  		}
   133  
   134  	} else if b.CustomResource != "" {
   135  		// make sure that CSV of the specified ServiceType exists
   136  		csv, err := o.KClient.GetClusterServiceVersion(o.ServiceType)
   137  		if err != nil {
   138  			// error only occurs when OperatorHub is not installed.
   139  			// k8s doesn't have it installed by default but OCP does
   140  			return err
   141  		}
   142  		b.group, b.version, b.resource, err = svc.GetGVRFromOperator(csv, b.CustomResource)
   143  		if err != nil {
   144  			return err
   145  		}
   146  
   147  		// if the service name is blank then we set it to custom resource name
   148  		if o.ServiceName == "" {
   149  			o.ServiceName = strings.ToLower(b.CustomResource)
   150  		}
   151  
   152  		hasCR, cr := o.KClient.CheckCustomResourceInCSV(b.CustomResource, &csv)
   153  		if !hasCR {
   154  			return fmt.Errorf("the %q resource doesn't exist in specified %q operator", b.CustomResource, b.group)
   155  		}
   156  
   157  		crd, err := o.KClient.GetCRDSpec(cr, b.group, b.CustomResource)
   158  		if err != nil {
   159  			return err
   160  		}
   161  
   162  		if len(o.parametersFlag) != 0 {
   163  			builtCRD, e := svc.BuildCRDFromParams(o.ParametersMap, crd, b.group, b.version, b.CustomResource)
   164  			if e != nil {
   165  				return e
   166  			}
   167  
   168  			u.Object = builtCRD
   169  		} else {
   170  			almExample, e := svc.GetAlmExample(csv, b.CustomResource, o.ServiceType)
   171  			if e != nil {
   172  				return e
   173  			}
   174  
   175  			u.Object = almExample
   176  		}
   177  
   178  		if o.ServiceName != "" && !o.DryRunFlag {
   179  			// First check if service with provided name already exists
   180  			svcFullName := strings.Join([]string{b.CustomResource, o.ServiceName}, "/")
   181  			exists, e := svc.OperatorSvcExists(o.KClient, svcFullName)
   182  			if e != nil {
   183  				return e
   184  			}
   185  			if exists {
   186  				return fmt.Errorf("service %q already exists; please provide a different name or delete the existing service first", svcFullName)
   187  			}
   188  		}
   189  
   190  		u.SetName(o.ServiceName)
   191  		if u.GetName() == "" {
   192  			return ErrNoMetadataName
   193  		}
   194  
   195  		// CRD is valid. We can use it further to create a service from it.
   196  		b.CustomResourceDefinition = u.Object
   197  
   198  		if o.ServiceName == "" {
   199  			o.ServiceName = u.GetName()
   200  		}
   201  
   202  		// Validate spec
   203  		err = validate.AgainstSchema(crd, u.Object["spec"], strfmt.Default)
   204  		if err != nil {
   205  			return err
   206  		}
   207  
   208  	} else {
   209  		// This block is executed only when user has neither provided a
   210  		// file nor a valid `odo service create <operator-name>` to start
   211  		// the service from an Operator. So we raise an error because the
   212  		// correct way is to execute:
   213  		// `odo service create <operator-name>/<crd-name>`
   214  
   215  		return fmt.Errorf("please use a valid command to start an Operator backed service; desired format: %q", "odo service create <operator-name>/<crd-name>")
   216  	}
   217  
   218  	return nil
   219  }

   221  func (b *OperatorBackend) RunServiceCreate(o *CreateOptions) (err error) {
   222  	s := &log.Status{}
   223  
   224  	// if cluster has resources of type CSV and o.CustomResource is not
   225  	// empty, we're expected to create an Operator backed service
   226  	if o.DryRunFlag {
   227  		// if it's dry run, only print the alm-example (o.CustomResourceDefinition) and exit
   228  		jsonCR, err := json.MarshalIndent(b.CustomResourceDefinition, "", "  ")
   229  		if err != nil {
   230  			return err
   231  		}
   232  
   233  		// convert json to yaml
   234  		yamlCR, err := yaml.JSONToYAML(jsonCR)
   235  		if err != nil {
   236  			return err
   237  		}
   238  
   239  		log.Info(string(yamlCR))
   240  
   241  		return nil
   242  	} else {
   243  
   244  		if o.inlinedFlag {
   245  			crdYaml, err := yaml.Marshal(b.CustomResourceDefinition)
   246  			if err != nil {
   247  				return err
   248  			}
   249  
   250  			err = devfile.AddKubernetesComponentToDevfile(string(crdYaml), o.ServiceName, o.EnvSpecificInfo.GetDevfileObj())
   251  			if err != nil {
   252  				return err
   253  			}
   254  
   255  		} else {
   256  			crdYaml, err := yaml.Marshal(b.CustomResourceDefinition)
   257  			if err != nil {
   258  				return err
   259  			}
   260  
   261  			err = devfile.AddKubernetesComponent(string(crdYaml), o.ServiceName, o.contextFlag, o.EnvSpecificInfo.GetDevfileObj())
   262  			if err != nil {
   263  				return err
   264  			}
   265  		}
   266  
   267  		if log.IsJSON() {
   268  			svcFullName := strings.Join([]string{b.CustomResource, o.ServiceName}, "/")
   269  			svc := NewServiceItem(svcFullName)
   270  			svc.Manifest = b.CustomResourceDefinition
   271  			svc.InDevfile = true
   272  			machineoutput.OutputSuccess(svc)
   273  		}
   274  	}
   275  	s.End(true)
   276  	return
   277  }

   289  	// "name" is of the form CR-Name/Instance-Name so we split it
   290  	_, instanceName, err := svc.SplitServiceKindName(name)

   303  func (b *OperatorBackend) DescribeService(o *DescribeOptions, serviceName, app string) error {
   304  
   305  	clusterList, _, err := svc.ListOperatorServices(o.KClient)
   306  	if err != nil {
   307  		return err
   308  	}
   309  	var clusterFound *unstructured.Unstructured
   310  	for i, clusterInstance := range clusterList {
   311  		fullName := strings.Join([]string{clusterInstance.GetKind(), clusterInstance.GetName()}, "/")
   312  		if fullName == serviceName {
   313  			clusterFound = &clusterList[i]
   314  			break
   315  		}
   316  	}
   317  
   318  	devfileList, err := svc.ListDevfileServices(o.KClient, o.EnvSpecificInfo.GetDevfileObj(), o.contextFlag)
   319  	if err != nil {
   320  		return err
   321  	}
   322  	devfileService, inDevfile := devfileList[serviceName]
   323  
   324  	item := NewServiceItem(serviceName)
   325  	item.InDevfile = inDevfile
   326  	item.Deployed = clusterFound != nil
   327  	if item.Deployed {
   328  		item.Manifest = clusterFound.Object
   329  	} else if item.InDevfile {
   330  		item.Manifest = devfileService.Object
   331  	}
   332  
   333  	if log.IsJSON() {
   334  		machineoutput.OutputSuccess(item)
   335  		return nil
   336  	}
   337  
   338  	return PrintHumanReadableOutput(item)
   339  }

   341  // PrintHumanReadableOutput outputs the description of a service in a human readable format
   342  func PrintHumanReadableOutput(item *serviceItem) error {
   343  	log.Describef("Version: ", "%s", item.Manifest["apiVersion"])
   344  	log.Describef("Kind: ", "%s", item.Manifest["kind"])
   345  	metadata, ok := item.Manifest["metadata"].(map[string]interface{})
   346  	if !ok {
   347  		return errors.New("unable to get name from manifest")
   348  	}
   349  	log.Describef("Name: ", "%s", metadata["name"])
   350  	spec, ok := item.Manifest["spec"].(map[string]interface{})
   351  	if !ok {
   352  		return errors.New("unable to get specifications from manifest")
   353  	}
   354  
   355  	var tab bytes.Buffer
   356  
   357  	wr := tabwriter.NewWriter(&tab, 5, 2, 3, ' ', tabwriter.TabIndent)
   358  	fmt.Fprint(wr, "NAME", "\t", "VALUE", "\n")
   359  	displayParameters(wr, spec, "")
   360  	wr.Flush()
   361  
   362  	log.Describef("Parameters:\n", tab.String())
   363  
   364  	return nil
   365  }

   367  // displayParameters adds lines describing fields of a given map
   368  func displayParameters(wr *tabwriter.Writer, spec map[string]interface{}, prefix string) {
   369  	keys := make([]string, len(spec))
   370  	i := 0
   371  	for key := range spec {
   372  		keys[i] = key
   373  		i++
   374  	}
   375  
   376  	for _, k := range keys {
   377  		v := spec[k]
   378  		switch val := v.(type) {
   379  		case map[string]interface{}:
   380  			displayParameters(wr, val, prefix+k+".")
   381  		default:
   382  			fmt.Fprintf(wr, "%s%s\t%v\n", prefix, k, val)
   383  		}
   384  	}
   385  }
