    67  // Complete completes watch args
    68  func (wo *WatchOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    69  	wo.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline).NeedDevfile(wo.contextFlag))
    70  	if err != nil {
    71  		return err
    72  	}
    73  	// Set the source path to either the context or current working directory (if context not set)
    74  	wo.sourcePath, err = util.GetAbsPath(wo.contextFlag)
    75  	if err != nil {
    76  		return errors.Wrap(err, "unable to get source path")
    77  	}
    78  
    79  	// Apply ignore information
    80  	err = genericclioptions.ApplyIgnore(&wo.ignoreFlag, wo.sourcePath)
    81  	if err != nil {
    82  		return errors.Wrap(err, "unable to apply ignore information")
    83  	}
    84  
    85  	// The namespace was retrieved from the --project flag (or from the kube client if not set) and stored in kclient when initializing the context
    86  	platformContext := kubernetes.KubernetesContext{
    87  		Namespace: wo.KClient.GetCurrentNamespace(),
    88  	}
    89  
    90  	wo.initialDevfileHandler, err = adapters.NewComponentAdapter(wo.EnvSpecificInfo.GetName(), wo.contextFlag, wo.GetApplication(), wo.EnvSpecificInfo.GetDevfileObj(), platformContext)
    91  
    92  	return err
    93  }

    95  // Validate validates the watch parameters
    96  func (wo *WatchOptions) Validate() (err error) {
    97  
    98  	// Delay interval cannot be -ve
    99  	if wo.delayFlag < 0 {
   100  		return fmt.Errorf("Delay cannot be lesser than 0 and delay=0 means changes will be pushed as soon as they are detected which can cause performance issues")
   101  	}
   102  	// Print a debug message warning user if delay is set to 0
   103  	if wo.delayFlag == 0 {
   104  		klog.V(4).Infof("delay=0 means changes will be pushed as soon as they are detected which can cause performance issues")
   105  	}
   106  
   107  	if wo.debugCommandFlag != "" && wo.EnvSpecificInfo != nil && wo.EnvSpecificInfo.GetRunMode() != envinfo.Debug {
   108  		return fmt.Errorf("please start the component in debug mode using `odo push --debug` to use the --debug-command flag")
   109  	}
   110  	exists, err := wo.initialDevfileHandler.DoesComponentExist(wo.EnvSpecificInfo.GetName(), wo.GetApplication())
   111  	if err != nil {
   112  		return err
   113  	}
   114  	if !exists {
   115  		return fmt.Errorf("component does not exist. Please use `odo push` to create your component")
   116  	}
   117  	return nil
   118  }

   121  func (wo *WatchOptions) Run() (err error) {
   122  	err = watch.DevfileWatchAndPush(
   123  		os.Stdout,
   124  		watch.WatchParameters{
   125  			ComponentName:       wo.EnvSpecificInfo.GetName(),
   126  			ApplicationName:     wo.Context.GetApplication(),
   127  			Path:                wo.sourcePath,
   128  			FileIgnores:         util.GetAbsGlobExps(wo.sourcePath, wo.ignoreFlag),
   129  			PushDiffDelay:       wo.delayFlag,
   130  			StartChan:           nil,
   131  			ExtChan:             make(chan bool),
   132  			DevfileWatchHandler: wo.regenerateAdapterAndPush,
   133  			Show:                wo.showLogFlag,
   134  			DevfileBuildCmd:     strings.ToLower(wo.buildCommandFlag),
   135  			DevfileRunCmd:       strings.ToLower(wo.runCommandFlag),
   136  			DevfileDebugCmd:     strings.ToLower(wo.debugCommandFlag),
   137  			EnvSpecificInfo:     wo.EnvSpecificInfo,
   138  		},
   139  	)
   140  	if err != nil {
   141  		return errors.Wrapf(err, "Error while trying to watch %s", wo.sourcePath)
   142  	}
   143  	return err
   144  }

   189  // regenerateAdapterAndPush is used as a DevfileWatchHandler in WatchParameters; it is a wrapper around adapter.Push()
   190  // that first regenerates the component adapter before calling push. This ensures that it has picked up the latest
   191  // devfile.yaml changes
   192  func (wo *WatchOptions) regenerateAdapterAndPush(pushParams common.PushParameters, watchParams watch.WatchParameters) error {
   193  	var adapter common.ComponentAdapter
   194  
   195  	adapter, err := wo.regenerateComponentAdapterFromWatchParams(watchParams)
   196  	if err != nil {
   197  		return errors.Wrapf(err, "unable to generate component from watch parameters")
   198  	}
   199  
   200  	err = adapter.Push(pushParams)
   201  	if err != nil {
   202  		return errors.Wrapf(err, "watch command was unable to push component")
   203  	}
   204  
   205  	return err
   206  }

   211  	// Parse devfile and validate
