   102  // Run contains the logic for the command
   103  func (o *UnsetOptions) Run() error {
   104  	if o.envArrayFlag != nil {
   105  
   106  		if err := o.EnvSpecificInfo.GetDevfileObj().RemoveEnvVars(o.envArrayFlag); err != nil {
   107  			return err
   108  		}
   109  		log.Success("Environment variables were successfully updated")
   110  		if o.nowFlag {
   111  			return o.DevfilePush()
   112  		}
   113  		log.Italic("\nRun `odo push` command to apply changes to the cluster")
   114  		return nil
   115  	}
   116  	if isSet := config.IsSetInDevfile(o.EnvSpecificInfo.GetDevfileObj(), o.paramName); isSet {
   117  		if !o.forceFlag && !ui.Proceed(fmt.Sprintf("Do you want to unset %s in the devfile", o.paramName)) {
   118  			fmt.Println("Aborted by the user.")
   119  			return nil
   120  		}
   121  		err := config.DeleteDevfileConfiguration(o.EnvSpecificInfo.GetDevfileObj(), strings.ToLower(o.paramName))
   122  		log.Success("Devfile was successfully updated.")
   123  		if o.nowFlag {
   124  			return o.DevfilePush()
   125  		}
   126  		return err
   127  	}
   128  	return fmt.Errorf("config already unset, cannot unset a configuration which is not set")
   129  }
