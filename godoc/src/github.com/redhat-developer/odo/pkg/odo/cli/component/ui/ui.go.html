    13  // SelectStarterProject allows user to select starter project in the prompt
    14  func SelectStarterProject(projects []devfilev1.StarterProject) string {
    15  
    16  	if len(projects) == 0 {
    17  		return ""
    18  	}
    19  
    20  	projectNames := getProjectNames(projects)
    21  
    22  	var download = false
    23  	var selectedProject string
    24  	prompt := &survey.Confirm{Message: "Do you want to download a starter project"}
    25  	err := survey.AskOne(prompt, &download, nil)
    26  	ui.HandleError(err)
    27  
    28  	if !download {
    29  		return ""
    30  	}
    31  
    32  	// select the only starter project in devfile
    33  	if len(projectNames) == 1 {
    34  		return projectNames[0]
    35  	}
    36  
    37  	// If multiple projects present give options to select
    38  	promptSelect := &survey.Select{
    39  		Message: "Which starter project do you want to download",
    40  		Options: projectNames,
    41  	}
    42  
    43  	err = survey.AskOne(promptSelect, &selectedProject, survey.Required)
    44  	ui.HandleError(err)
    45  	return selectedProject
