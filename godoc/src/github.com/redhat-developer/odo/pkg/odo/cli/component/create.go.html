   125  func (co *CreateOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
   126  	// GETTERS
   127  	// Get context
   128  	co.Context, err = getContext(co.nowFlag, cmdline)
   129  	if err != nil {
   130  		return err
   131  	}
   132  	// Get the app name
   133  	co.appFlag = genericclioptions.ResolveAppFlag(cmdline)
   134  	// Get the project name
   135  	co.devfileMetadata.componentNamespace = co.Context.GetProject()
   136  	// Get DevfilePath
   137  	co.DevfilePath = location.DevfileLocation(co.contextFlag)
   138  	//Check whether the directory already contains a devfile, this check should happen early
   139  	co.devfileMetadata.userCreatedDevfile = util.CheckPathExists(co.DevfilePath)
   140  	// EnvFilePath is the path of env file for devfile component
   141  	envFilePath := getEnvFilePath(co.contextFlag)
   142  	// This is required so that .odo is created in the correct context
   143  	co.PushOptions.componentContext = co.contextFlag
   144  	// Use Interactive mode if: 1) no args are passed || 2) the devfile exists || 3) --devfile is used
   145  	if len(args) == 0 && !util.CheckPathExists(co.DevfilePath) && co.devfileMetadata.devfilePath.value == "" {
   146  		co.interactive = true
   147  	}
   148  	// CONFLICT CHECK
   149  	// Check if a component exists
   150  	if util.CheckPathExists(envFilePath) && util.CheckPathExists(co.DevfilePath) {
   151  		return errors.New("this directory already contains a component")
   152  	}
   153  	// Check if there is a dangling env file; delete the env file if found
   154  	if util.CheckPathExists(envFilePath) && !util.CheckPathExists(co.DevfilePath) {
   155  		log.Warningf("Found a dangling env file without a devfile, overwriting it")
   156  		// Note: if the IF condition seems to have a side-effect, it is better to do the condition check separately, like below
   157  		err = util.DeletePath(envFilePath)
   158  		if err != nil {
   159  			return err
   160  		}
   161  	}
   162  	//Check if the directory already contains a devfile when --devfile flag is passed
   163  	if util.CheckPathExists(co.DevfilePath) && co.devfileMetadata.devfilePath.value != "" && !util.PathEqual(co.DevfilePath, co.devfileMetadata.devfilePath.value) {
   164  		return errors.New("this directory already contains a devfile, you can't specify devfile via --devfile")
   165  	}
   166  
   167  	// Initialize envinfo
   168  	err = co.InitEnvInfoFromContext()
   169  	if err != nil {
   170  		return err
   171  	}
   172  
   173  	// Fetch the necessary devfile and create the component
   174  	log.Info("Devfile Object Creation")
   175  	switch {
   176  	case co.devfileMetadata.userCreatedDevfile:
   177  		co.createMethod = UserCreatedDevfileMethod{}
   178  	case co.devfileMetadata.devfilePath.value != "":
   179  		//co.devfileName = "" for user provided devfile
   180  		fileErr := util.ValidateFile(co.devfileMetadata.devfilePath.value)
   181  		urlErr := util.ValidateURL(co.devfileMetadata.devfilePath.value)
   182  		if fileErr != nil && urlErr != nil {
   183  			return errors.Errorf("the devfile path you specify is invalid with either file error %q or url error %q", fileErr, urlErr)
   184  		} else if fileErr == nil {
   185  			co.createMethod = FileCreateMethod{}
   186  		} else if urlErr == nil {
   187  			co.createMethod = HTTPCreateMethod{}
   188  		}
   189  	case co.interactive:
   190  		co.createMethod = InteractiveCreateMethod{}
   191  	default:
   192  		co.createMethod = DirectCreateMethod{}
   193  	}
   194  	err = co.createMethod.CheckConflicts(co, args)
   195  	if err != nil {
   196  		return err
   197  	}
   198  	err = co.createMethod.FetchDevfileAndCreateComponent(co, cmdline, args)
   199  	if err != nil {
   200  		co.createMethod.Rollback(co.DevfilePath, co.contextFlag)
   201  		return err
   202  	}
   203  
   204  	// From this point forward, rollback should be triggered if an error is encountered; rollback should delete all the files that were created by odo
   205  	defer func() {
   206  		if err != nil {
   207  			co.createMethod.Rollback(co.DevfilePath, co.contextFlag)
   208  		}
   209  	}()
   210  	// Set the starter project token if required
   211  	if co.devfileMetadata.starter != "" {
   212  		secure := registryUtil.IsSecure(co.prefClient, co.devfileMetadata.devfileRegistry.Name)
   213  		if co.devfileMetadata.starterToken == "" && secure {
   214  			var token string
   215  			token, err = keyring.Get(fmt.Sprintf("%s%s", util.CredentialPrefix, co.devfileMetadata.devfileRegistry.Name), registryUtil.RegistryUser)
   216  			if err != nil {
   217  				return errors.Wrap(err, "unable to get secure registry credential from keyring")
   218  			}
   219  			co.devfileMetadata.starterToken = token
   220  		}
   221  	}
   222  
   223  	scontext.SetDevfileName(cmdline.Context(), co.devfileName)
   224  	// Adding component type to telemetry data
   225  	scontext.SetComponentType(cmdline.Context(), co.devfileMetadata.componentType)
   226  
   227  	return nil
   228  }

   230  func (co *CreateOptions) Validate() (err error) {
   231  	defer func() {
   232  		if err != nil {
   233  			co.createMethod.Rollback(co.DevfilePath, co.contextFlag)
   234  		}
   235  	}()
   236  	log.Info("Validation")
   237  	// Validate if the devfile component name that user wants to create adheres to the k8s naming convention
   238  	spinner := log.Spinner("Validating if devfile name is correct")
   239  	defer spinner.End(false)
   240  
   241  	err = util.ValidateK8sResourceName("component name", co.devfileMetadata.componentName)
   242  	if err != nil {
   243  		return err
   244  	}
   245  	spinner.End(true)
   246  
   247  	// Validate if the devfile is compatible with odo; this checks the resolved/flattened version of devfile
   248  	spinner = log.Spinner("Validating the devfile for odo")
   249  	defer spinner.End(false)
   250  
   251  	_, err = odoDevfile.ParseAndValidateFromFile(co.DevfilePath)
   252  	if err != nil {
   253  		return err
   254  	}
   255  	spinner.End(true)
   256  
   257  	return nil
   258  }

   260  func (co *CreateOptions) Run() (err error) {
   261  	defer func() {
   262  		if err != nil {
   263  			co.createMethod.Rollback(co.DevfilePath, co.contextFlag)
   264  		}
   265  	}()
   266  
   267  	devObj, err := devfileParseFromFile(co.DevfilePath, false)
   268  	if err != nil {
   269  		return errors.New("Failed to parse the devfile")
   270  	}
   271  
   272  	devfileData, err := ioutil.ReadFile(co.DevfilePath)
   273  	if err != nil {
   274  		return err
   275  	}
   276  	// WARN: Starter Project uses go-git that overrides the directory content, there by deleting the existing devfile.
   277  	err = decideAndDownloadStarterProject(devObj, co.devfileMetadata.starter, co.devfileMetadata.starterToken, co.interactive, co.contextFlag)
   278  	if err != nil {
   279  		return errors.Wrap(err, "failed to download project for devfile component")
   280  	}
   281  
   282  	// TODO: We should not have to rewrite to the file. Fix the starter project.
   283  	err = ioutil.WriteFile(co.DevfilePath, devfileData, 0644) // #nosec G306
   284  	if err != nil {
   285  		return err
   286  	}
   287  
   288  	// If user provided a custom name, re-write the devfile
   289  	// ENSURE: co.devfileMetadata.componentName != ""
   290  	if co.devfileMetadata.componentName != devObj.GetMetadataName() {
   291  		spinner := log.Spinnerf("Updating the devfile with component name %q", co.devfileMetadata.componentName)
   292  		defer spinner.End(false)
   293  
   294  		// WARN: SetMetadataName will rewrite to the devfile
   295  		err = devObj.SetMetadataName(co.devfileMetadata.componentName)
   296  		if err != nil {
   297  			return errors.New("Failed to update the devfile")
   298  		}
   299  		spinner.End(true)
   300  	}
   301  
   302  	// Generate env file
   303  	err = co.EnvSpecificInfo.SetComponentSettings(envinfo.ComponentSettings{
   304  		Name:               co.devfileMetadata.componentName,
   305  		Project:            co.devfileMetadata.componentNamespace,
   306  		AppName:            co.appFlag,
   307  		UserCreatedDevfile: co.devfileMetadata.userCreatedDevfile,
   308  	})
   309  	if err != nil {
   310  		return errors.Wrap(err, "failed to create env file for devfile component")
   311  	}
   312  
   313  	// Prepare .gitignore file
   314  	sourcePath, err := util.GetAbsPath(co.contextFlag)
   315  	if err != nil {
   316  		return errors.Wrap(err, "unable to get source path")
   317  	}
   318  
   319  	ignoreFile, err := util.TouchGitIgnoreFile(sourcePath)
   320  	if err != nil {
   321  		return err
   322  	}
   323  
   324  	err = util.AddFileToIgnoreFile(ignoreFile, filepath.Join(co.contextFlag, EnvDirectory))
   325  	if err != nil {
   326  		return err
   327  	}
   328  
   329  	if co.nowFlag {
   330  		err = co.DevfilePush()
   331  		if err != nil {
   332  			return fmt.Errorf("failed to push changes: %w", err)
   333  		}
   334  	} else {
   335  		log.Italic("\nPlease use `odo push` command to create the component with source deployed")
   336  	}
   337  
   338  	if log.IsJSON() {
   339  		return co.DevfileJSON()
   340  	}
   341  
   342  	return nil
   343  }
