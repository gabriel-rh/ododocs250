    41  // Complete DeployOptions after they've been created
    42  func (o *DeployOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    43  	o.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline).NeedDevfile(o.contextFlag))
    44  	if err != nil {
    45  		return err
    46  	}
    47  
    48  	envFileInfo, err := envinfo.NewEnvSpecificInfo(o.contextFlag)
    49  	if err != nil {
    50  		return errors.Wrap(err, "unable to retrieve configuration information")
    51  	}
    52  	if !envFileInfo.Exists() {
    53  		var cmpName string
    54  		cmpName, err = component.GatherName(o.EnvSpecificInfo.GetDevfileObj(), o.GetDevfilePath())
    55  		if err != nil {
    56  			return errors.Wrap(err, "unable to retrieve component name")
    57  		}
    58  		err = envFileInfo.SetComponentSettings(envinfo.ComponentSettings{Name: cmpName, Project: o.GetProject(), AppName: "app"})
    59  		if err != nil {
    60  			return errors.Wrap(err, "failed to write new env.yaml file")
    61  		}
    62  
    63  	} else if envFileInfo.GetComponentSettings().Project != o.GetProject() {
    64  		err = envFileInfo.SetConfiguration("project", o.GetProject())
    65  		if err != nil {
    66  			return errors.Wrap(err, "failed to update project in env.yaml file")
    67  		}
    68  	}
    69  	return
    70  }
