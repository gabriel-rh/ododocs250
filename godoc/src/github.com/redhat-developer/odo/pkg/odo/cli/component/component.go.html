    23  // Complete completes component options
    24  func (co *ComponentOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    25  	co.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline))
    26  	if err != nil {
    27  		co.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline).IsOffline())
    28  		if err != nil {
    29  			return err
    30  		}
    31  	}
    32  
    33  	co.componentName, err = co.Context.Component(args...)
    34  	if err != nil {
    35  		return nil
    36  	}
    37  	return
    38  }
