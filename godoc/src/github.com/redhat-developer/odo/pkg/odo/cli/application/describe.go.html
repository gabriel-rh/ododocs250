    72  // Run contains the logic for the odo command
    73  func (o *DescribeOptions) Run() (err error) {
    74  	if o.IsJSON() {
    75  		appDef := o.appClient.GetMachineReadableFormat(o.appName, o.GetProject())
    76  		machineoutput.OutputSuccess(appDef)
    77  		return nil
    78  	}
    79  
    80  	componentList, err := o.appClient.ComponentList(o.appName)
    81  	if err != nil {
    82  		return err
    83  	}
    84  
    85  	if len(componentList) == 0 {
    86  		fmt.Printf("Application %s has no components or services deployed.", o.appName)
    87  		return
    88  	}
    89  
    90  	fmt.Printf("Application Name: %s has %v component(s):\n--------------------------------------\n",
    91  		o.appName, len(componentList))
    92  	for _, currentComponent := range componentList {
    93  		err := util.PrintComponentInfo(o.KClient, currentComponent.Name, currentComponent, o.appName, o.GetProject())
    94  		if err != nil {
    95  			return err
    96  		}
    97  		fmt.Println("--------------------------------------")
    98  	}
    99  
   100  	return nil
   101  }
