    75  	} else {
    76  		// here we use information from user's kubeconfig

    78  		if util.CheckKubeConfigExist() {
    79  			klog.V(4).Infof("New Context")

    92  func (lo *ListOptions) Validate() (err error) {
    93  
    94  	if len(lo.GetApplication()) != 0 && lo.allAppsFlag {
    95  		klog.V(4).Infof("either --app and --all-apps both provided or provided --all-apps in a folder has app, use --all-apps anyway")
    96  	}
    97  
    98  	if util.CheckPathExists(lo.devfilePath) {
    99  		if lo.GetApplication() == "" && lo.KClient.GetCurrentNamespace() == "" {
   100  			return odoutil.ThrowContextError()
   101  		}
   102  		return nil
   103  	}
   104  	var project, app string
   105  
   106  	if !util.CheckKubeConfigExist() {
   107  		project = lo.EnvSpecificInfo.GetNamespace()
   108  		app = lo.EnvSpecificInfo.GetApplication()
   109  	} else {
   110  		project = lo.Context.GetProject()
   111  		app = lo.GetApplication()
   112  	}
   113  	if !lo.allAppsFlag && lo.pathFlag == "" && (project == "" || app == "") {
   114  		return odoutil.ThrowContextError()
   115  	}
   116  	return nil

   120  // Run has the logic to perform the required actions as part of command
   121  func (lo *ListOptions) Run() error {
   122  	var otherComps []component.Component
   123  	// --path workflow
   124  
   125  	if len(lo.pathFlag) != 0 {
   126  
   127  		devfileComps, err := component.ListDevfileComponentsInPath(lo.KClient, filepath.SplitList(lo.pathFlag))
   128  		if err != nil {
   129  			return err
   130  		}
   131  
   132  		combinedComponents := component.NewCombinedComponentList(devfileComps, otherComps)
   133  
   134  		if log.IsJSON() {
   135  			machineoutput.OutputSuccess(combinedComponents)
   136  		} else {
   137  			HumanReadableOutputInPath(os.Stdout, combinedComponents)
   138  		}
   139  		return nil
   140  	}
   141  
   142  	// non --path workflow below
   143  	// read the code like
   144  	// -> experimental
   145  	//	or	|-> --all-apps
   146  	//		|-> the current app
   147  	// -> non-experimental
   148  	//	or	|-> --all-apps
   149  	//		|-> the current app
   150  
   151  	// experimental workflow
   152  
   153  	devfileComponents := []component.Component{}
   154  	var selector string
   155  	// TODO: wrap this into a component list for docker support
   156  	if lo.allAppsFlag {
   157  		selector = project.GetSelector()
   158  	} else {
   159  		selector = applabels.GetSelector(lo.GetApplication())
   160  	}
   161  
   162  	currentComponentState := component.StateTypeNotPushed
   163  
   164  	if lo.KClient != nil {
   165  		devfileComponentsOut, err := component.ListDevfileComponents(lo.KClient, selector)
   166  		if err != nil {
   167  			return err
   168  		}
   169  
   170  		devfileComponents = devfileComponentsOut.Items
   171  		for _, comp := range devfileComponents {
   172  			if lo.EnvSpecificInfo != nil {
   173  				// if we can find a component from the listing from server then the local state is pushed
   174  				if lo.EnvSpecificInfo.EnvInfo.MatchComponent(comp.Name, comp.Spec.App, comp.Namespace) {
   175  					currentComponentState = component.StateTypePushed
   176  				}
   177  			}
   178  		}
   179  	}
   180  
   181  	// 1st condition - only if we are using the same application or all-apps are provided should we show the current component
   182  	// 2nd condition - if the currentComponentState is unpushed that means it didn't show up in the list above
   183  	if lo.EnvSpecificInfo != nil {
   184  		envinfo := lo.EnvSpecificInfo.EnvInfo
   185  		if (envinfo.GetApplication() == lo.GetApplication() || lo.allAppsFlag) && currentComponentState == component.StateTypeNotPushed {
   186  			comp := component.NewComponent(envinfo.GetName())
   187  			comp.Status.State = component.StateTypeNotPushed
   188  			comp.Namespace = envinfo.GetNamespace()
   189  			comp.Spec.App = envinfo.GetApplication()
   190  			comp.Spec.Type = lo.componentType
   191  			devfileComponents = append(devfileComponents, comp)
   192  		}
   193  	}
   194  
   195  	// list components managed by other sources/tools
   196  	if lo.allAppsFlag {
   197  		selector = project.GetNonOdoSelector()
   198  	} else {
   199  		selector = applabels.GetNonOdoSelector(lo.GetApplication())
   200  	}
   201  
   202  	otherComponents, err := component.List(lo.KClient, selector)
   203  	if err != nil {
   204  		return fmt.Errorf("failed to fetch components not managed by odo: %w", err)
   205  	}
   206  	otherComps = otherComponents.Items
   207  
   208  	combinedComponents := component.NewCombinedComponentList(devfileComponents, otherComps)
   209  	if log.IsJSON() {
   210  		machineoutput.OutputSuccess(combinedComponents)
   211  	} else {
   212  		HumanReadableOutput(os.Stdout, combinedComponents)
   213  	}
   214  
   215  	return nil
   216  }

   225  		Long:        "List all components in the current application.",
   226  		Example:     fmt.Sprintf(listExample, fullName),

   232  	}
   233  	odoutil.AddContextFlag(componentListCmd, &o.contextFlag)
   234  	componentListCmd.Flags().StringVar(&o.pathFlag, "path", "", "path of the directory to scan for odo component directories")

   237  
   238  	//Adding `--project` flag
   239  	projectCmd.AddProjectFlag(componentListCmd)

   242  
   243  	completion.RegisterCommandFlagHandler(componentListCmd, "path", completion.FileCompletionHandler)

   248  func HumanReadableOutputInPath(wr io.Writer, o component.CombinedComponentList) {
   249  	w := tabwriter.NewWriter(wr, 5, 2, 3, ' ', tabwriter.TabIndent)
   250  	defer w.Flush()
   251  
   252  	// if we dont have any components then
   253  	if len(o.DevfileComponents) == 0 {
   254  		fmt.Fprintln(w, "No components found")
   255  		return
   256  	}
   257  
   258  	if len(o.DevfileComponents) != 0 {
   259  		fmt.Fprintln(w, "Devfile Components: ")
   260  		fmt.Fprintln(w, "APP", "\t", "NAME", "\t", "PROJECT", "\t", "STATE", "\t", "CONTEXT")
   261  		for _, comp := range o.DevfileComponents {
   262  			fmt.Fprintln(w, comp.Spec.App, "\t", comp.Name, "\t", comp.Namespace, "\t", comp.Status.State, "\t", comp.Status.Context)
   263  		}
   264  		fmt.Fprintln(w)
   265  	}
   266  }

   268  func HumanReadableOutput(wr io.Writer, o component.CombinedComponentList) {
   269  	w := tabwriter.NewWriter(wr, 5, 2, 3, ' ', tabwriter.TabIndent)
   270  	defer w.Flush()
   271  
   272  	if len(o.DevfileComponents) == 0 && len(o.OtherComponents) == 0 {
   273  		log.Info("There are no components deployed.")
   274  		return
   275  	}
   276  
   277  	if len(o.DevfileComponents) != 0 || len(o.OtherComponents) != 0 {
   278  		fmt.Fprintln(w, "APP", "\t", "NAME", "\t", "PROJECT", "\t", "TYPE", "\t", "STATE", "\t", "MANAGED BY ODO")
   279  		for _, comp := range o.DevfileComponents {
   280  			fmt.Fprintln(w, comp.Spec.App, "\t", comp.Name, "\t", comp.Namespace, "\t", comp.Spec.Type, "\t", comp.Status.State, "\t", "Yes")
   281  		}
   282  		for _, comp := range o.OtherComponents {
   283  			fmt.Fprintln(w, comp.Spec.App, "\t", comp.Name, "\t", comp.Namespace, "\t", comp.Spec.Type, "\t", component.StateTypePushed, "\t", "No")
   284  		}
   285  		fmt.Fprintln(w)
   286  	}
