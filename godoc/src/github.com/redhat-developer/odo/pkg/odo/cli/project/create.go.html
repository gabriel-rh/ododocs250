    74  // Run runs the project create command
    75  func (pco *ProjectCreateOptions) Run() (err error) {
    76  	// Create the "spinner"
    77  	s := &log.Status{}
    78  
    79  	// If the --wait parameter has been passed, we add a spinner..
    80  	if pco.waitFlag {
    81  		s = log.Spinner("Waiting for project to come up")
    82  		defer s.End(false)
    83  	}
    84  
    85  	// Create the project & end the spinner (if there is any..)
    86  	err = pco.prjClient.Create(pco.projectName, pco.waitFlag)
    87  	if err != nil {
    88  		return err
    89  	}
    90  	s.End(true)
    91  
    92  	successMessage := fmt.Sprintf(`Project %q is ready for use`, pco.projectName)
    93  	log.Successf(successMessage)
    94  
    95  	// Set the current project when created
    96  	err = pco.prjClient.SetCurrent(pco.projectName)
    97  	if err != nil {
    98  		return err
    99  	}
   100  
   101  	log.Successf("New project created and now using project: %v", pco.projectName)
   102  
   103  	// If -o json has been passed, let's output the appropriate json output.
   104  	if log.IsJSON() {
   105  		prj := project.NewProject(pco.projectName, true)
   106  		machineoutput.OutputSuccess(prj)
   107  	}
   108  
   109  	return nil
   110  }
