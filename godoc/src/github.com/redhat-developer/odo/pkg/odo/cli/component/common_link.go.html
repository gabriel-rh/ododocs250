    65  func (o *commonLinkOptions) complete(cmdline cmdline.Cmdline, args []string, context string) (err error) {
    66  	o.operationName = cmdline.GetName()
    67  	o.suppliedName = args[0]
    68  
    69  	o.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline).NeedDevfile(context))
    70  	if err != nil {
    71  		return err
    72  	}
    73  
    74  	o.csvSupport, _ = o.KClient.IsCSVSupported()
    75  
    76  	o.serviceType, o.serviceName, err = svc.IsOperatorServiceNameValid(o.suppliedName)
    77  	if err != nil {
    78  		// error indicates the service name provided by user doesn't adhere to <crd-name>/<instance-name>
    79  		// so it's another odo component that they want to link/unlink to/from
    80  		o.serviceName = o.suppliedName
    81  		o.isTargetAService = false
    82  		o.serviceType = "Service" // Kubernetes Service
    83  
    84  		// TODO find the service using an app name to link components in other apps
    85  		// requires modification of the app flag or finding some other way
    86  		var s *v1.Service
    87  		s, err = o.Context.KClient.GetOneService(o.suppliedName, o.EnvSpecificInfo.GetApplication())
    88  		if kerrors.IsNotFound(err) {
    89  			return fmt.Errorf("couldn't find component named %q. Refer %q to see list of running components", o.suppliedName, "odo list")
    90  		}
    91  		if err != nil {
    92  			return err
    93  		}
    94  		o.serviceName = s.Name
    95  	} else {
    96  		o.isTargetAService = true
    97  	}
    98  
    99  	if o.operationName == unlink {
   100  		// rest of the code is specific to link operation
   101  		return nil
   102  	}
   103  
   104  	componentName := o.EnvSpecificInfo.GetName()
   105  
   106  	deployment, err := o.KClient.GetOneDeployment(componentName, o.EnvSpecificInfo.GetApplication())
   107  	if err != nil {
   108  		return err
   109  	}
   110  
   111  	deploymentGVR, err := o.KClient.GetDeploymentAPIVersion()
   112  	if err != nil {
   113  		return err
   114  	}
   115  
   116  	paramsMap, err := util.MapFromParameters(o.mappings)
   117  	if err != nil {
   118  		return err
   119  	}
   120  
   121  	// MappingsMap is a map of mappings to be used in the ServiceBinding we create for an "odo link"
   122  	var mappingsMap []servicebinding.Mapping
   123  	for kv := range paramsMap {
   124  		mapping := servicebinding.Mapping{
   125  			Name:  kv,
   126  			Value: paramsMap[kv],
   127  		}
   128  		mappingsMap = append(mappingsMap, mapping)
   129  	}
   130  
   131  	var service servicebinding.Service
   132  	if o.isTargetAService {
   133  		// since the service exists, let's get more info to populate service binding request
   134  		// first get the CR itself
   135  		cr, err := o.KClient.GetCustomResource(o.serviceType)
   136  		if err != nil {
   137  			return err
   138  		}
   139  
   140  		// now get the group, version, kind information from CR
   141  		group, version, kind, err := svc.GetGVKFromCR(cr)
   142  		if err != nil {
   143  			return err
   144  		}
   145  
   146  		service = servicebinding.Service{
   147  			Id: &o.serviceName, // Id field is helpful if user wants to inject mappings (custom binding data)
   148  			NamespacedRef: servicebinding.NamespacedRef{
   149  				Ref: servicebinding.Ref{
   150  					Group:   group,
   151  					Version: version,
   152  					Kind:    kind,
   153  					Name:    o.serviceName,
   154  				},
   155  			},
   156  		}
   157  	} else {
   158  		service = servicebinding.Service{
   159  			Id: &o.serviceName, // Id field is helpful if user wants to inject mappings (custom binding data)
   160  			NamespacedRef: servicebinding.NamespacedRef{
   161  				Ref: servicebinding.Ref{
   162  					Version: "v1",
   163  					Kind:    "Service",
   164  					Name:    o.serviceName,
   165  				},
   166  			},
   167  		}
   168  	}
   169  
   170  	o.serviceBinding = &servicebinding.ServiceBinding{
   171  		TypeMeta: metav1.TypeMeta{
   172  			APIVersion: strings.Join([]string{kclient.ServiceBindingGroup, kclient.ServiceBindingVersion}, "/"),
   173  			Kind:       kclient.ServiceBindingKind,
   174  		},
   175  		ObjectMeta: metav1.ObjectMeta{
   176  			Name: o.getServiceBindingName(componentName),
   177  		},
   178  		Spec: servicebinding.ServiceBindingSpec{
   179  			DetectBindingResources: true,
   180  			BindAsFiles:            o.bindAsFiles,
   181  			Application: servicebinding.Application{
   182  				Ref: servicebinding.Ref{
   183  					Name:     deployment.Name,
   184  					Group:    deploymentGVR.Group,
   185  					Version:  deploymentGVR.Version,
   186  					Resource: deploymentGVR.Resource,
   187  				},
   188  			},
   189  			Mappings: mappingsMap,
   190  			Services: []servicebinding.Service{service},
   191  		},
   192  	}
   193  
   194  	return nil
   195  }

   197  func (o *commonLinkOptions) validate() (err error) {
   198  	if o.EnvSpecificInfo == nil {
   199  		return fmt.Errorf("failed to find environment info to validate")
   200  	}
   201  
   202  	var svcFullName string
   203  
   204  	if o.isTargetAService {
   205  		// let's validate if the service exists
   206  		svcFullName = strings.Join([]string{o.serviceType, o.serviceName}, "/")
   207  		svcExists, err := svc.OperatorSvcExists(o.KClient, svcFullName)
   208  		if err != nil {
   209  			return err
   210  		}
   211  		if !svcExists {
   212  			return fmt.Errorf("couldn't find service named %q. Refer %q to see list of running services", svcFullName, "odo service list")
   213  		}
   214  	} else {
   215  		svcFullName = o.serviceName
   216  		if o.suppliedName == o.EnvSpecificInfo.GetName() {
   217  			if o.operationName == unlink {
   218  				return fmt.Errorf("the component %q cannot be unlinked from itself", o.suppliedName)
   219  			} else {
   220  				return fmt.Errorf("the component %q cannot be linked with itself", o.suppliedName)
   221  			}
   222  		}
   223  	}
   224  
   225  	if o.operationName == unlink {
   226  		_, found, err := svc.FindDevfileServiceBinding(o.EnvSpecificInfo.GetDevfileObj(), o.serviceType, o.serviceName, o.GetComponentContext())
   227  		if err != nil {
   228  			return err
   229  		}
   230  		if !found {
   231  			if o.getLinkType() == "service" {
   232  				return fmt.Errorf("failed to unlink the %s %q since no link was found in the configuration referring this %s", o.getLinkType(), svcFullName, o.getLinkType())
   233  			}
   234  			return fmt.Errorf("failed to unlink the %s %q since no link was found in the configuration referring this %s", o.getLinkType(), o.suppliedName, o.getLinkType())
   235  		}
   236  		return nil
   237  	}
   238  
   239  	return nil
   240  }

   242  func (o *commonLinkOptions) run() (err error) {
   243  	if o.Context.EnvSpecificInfo != nil {
   244  		if o.operationName == unlink {
   245  			return o.unlinkOperator()
   246  		}
   247  		return o.linkOperator()
   248  	}
   249  
   250  	var component string
   251  	if o.Context.EnvSpecificInfo != nil {
   252  		component = o.EnvSpecificInfo.GetName()
   253  		err = o.operation(o.secretName, component, o.GetApplication())
   254  	} else {
   255  		component, err = o.Component()
   256  		if err != nil {
   257  			return err
   258  		}
   259  		err = o.operation(o.secretName, component, o.GetApplication())
   260  	}
   261  
   262  	if err != nil {
   263  		return err
   264  	}
   265  
   266  	switch o.operationName {
   267  	case "link":
   268  		log.Successf("The %s %s has been successfully linked to the component %s\n", o.getLinkType(), o.suppliedName, component)
   269  	case "unlink":
   270  		log.Successf("The %s %s has been successfully unlinked from the component %s\n", o.getLinkType(), o.suppliedName, component)
   271  	default:
   272  		return fmt.Errorf("unknown operation %s", o.operationName)
   273  	}
   274  
   275  	secret, err := o.KClient.GetSecret(o.secretName, o.GetProject())
   276  	if err != nil {
   277  		return err
   278  	}
   279  
   280  	if len(secret.Data) == 0 {
   281  		log.Infof("There are no secret environment variables to expose within the %s service", o.suppliedName)
   282  	} else {
   283  		if o.operationName == "link" {
   284  			log.Infof("The below secret environment variables were added to the '%s' component:\n", component)
   285  		} else {
   286  			log.Infof("The below secret environment variables were removed from the '%s' component:\n", component)
   287  		}
   288  
   289  		// Output the environment variables
   290  		for i := range secret.Data {
   291  			fmt.Printf("· %v\n", i)
   292  		}
   293  
   294  		// Retrieve the first variable to use as an example.
   295  		// Have to use a range to access the map
   296  		var exampleEnv string
   297  		for i := range secret.Data {
   298  			exampleEnv = i
   299  			break
   300  		}
   301  
   302  		// Output what to do next if first linking...
   303  		if o.operationName == "link" {
   304  			log.Italicf(`
   305  You can now access the environment variables from within the component pod, for example:
   306  $%s is now available as a variable within component %s`, exampleEnv, component)
   307  		}
   308  	}
   309  	return
   310  }

   312  // getServiceBindingName creates a name to be used for creation/deletion of SBR during link/unlink operations
   313  func (o *commonLinkOptions) getServiceBindingName(componentName string) string {
   314  	if len(o.name) > 0 {
   315  		return o.name
   316  	}
   317  	if !o.isTargetAService {
   318  		return strings.Join([]string{componentName, o.serviceName}, "-")
   319  	}
   320  	return strings.Join([]string{componentName, strings.ToLower(o.serviceType), o.serviceName}, "-")
   321  }

   323  // linkOperator creates a service binding resource and links
   324  // the current component with the given odo service or
   325  // the current component with the given component's service
   326  // and stores the link info in the env
   327  func (o *commonLinkOptions) linkOperator() (err error) {
   328  	// Convert ServiceBinding -> JSON -> Map -> YAML
   329  	// JSON conversion step is necessary to inline TypeMeta
   330  
   331  	intermediate, err := json.Marshal(o.serviceBinding)
   332  	if err != nil {
   333  		return err
   334  	}
   335  
   336  	serviceBindingMap := make(map[string]interface{})
   337  	err = json.Unmarshal(intermediate, &serviceBindingMap)
   338  	if err != nil {
   339  		return err
   340  	}
   341  
   342  	yamlDesc, err := yaml.Marshal(serviceBindingMap)
   343  	if err != nil {
   344  		return err
   345  	}
   346  
   347  	// check if the component is already linked to the requested component/service
   348  	_, found, err := svc.FindDevfileServiceBinding(o.EnvSpecificInfo.GetDevfileObj(), o.serviceType, o.serviceName, o.GetComponentContext())
   349  	if err != nil {
   350  		return err
   351  	}
   352  	if found {
   353  		return fmt.Errorf("component %q is already linked with the %s %q", o.Context.EnvSpecificInfo.GetName(), o.getLinkType(), o.suppliedName)
   354  	}
   355  
   356  	if o.inlined {
   357  		err = devfile.AddKubernetesComponentToDevfile(string(yamlDesc), o.serviceBinding.Name, o.EnvSpecificInfo.GetDevfileObj())
   358  		if err != nil {
   359  			return err
   360  		}
   361  	} else {
   362  		err = devfile.AddKubernetesComponent(string(yamlDesc), o.serviceBinding.Name, o.GetComponentContext(), o.EnvSpecificInfo.GetDevfileObj())
   363  		if err != nil {
   364  			return err
   365  		}
   366  	}
   367  
   368  	log.Successf("Successfully created link between component %q and %s %q\n", o.Context.EnvSpecificInfo.GetName(), o.getLinkType(), o.suppliedName)
   369  	log.Italic("To apply the link, please use `odo push`")
   370  	return err
   371  }

   373  // unlinkOperator deletes the service binding resource from the devfile
   374  func (o *commonLinkOptions) unlinkOperator() (err error) {
   375  
   376  	// We already tested `found` in `validateForOperator`
   377  	name, _, err := svc.FindDevfileServiceBinding(o.EnvSpecificInfo.GetDevfileObj(), o.serviceType, o.serviceName, o.GetComponentContext())
   378  	if err != nil {
   379  		return err
   380  	}
   381  
   382  	err = devfile.DeleteKubernetesComponentFromDevfile(name, o.EnvSpecificInfo.GetDevfileObj(), o.GetComponentContext())
   383  	if err != nil {
   384  		return err
   385  	}
   386  
   387  	log.Successf("Successfully unlinked component %q from %s %q\n", o.Context.EnvSpecificInfo.GetName(), o.getLinkType(), o.suppliedName)
   388  	log.Italic("To apply the changes, please use `odo push`")
   389  	return nil
   390  }
