    78  // Run the project delete command
    79  func (pdo *ProjectDeleteOptions) Run() (err error) {
    80  
    81  	// Create the "spinner"
    82  	s := &log.Status{}
    83  
    84  	// This to set the project in the file and runtime
    85  	err = pdo.prjClient.SetCurrent(pdo.projectName)
    86  	if err != nil {
    87  		return err
    88  	}
    89  
    90  	// Prints out what will be deleted
    91  	// This function doesn't support devfile components.
    92  	// TODO: fix this once we have proper abstraction layer on top of devfile components
    93  	//err = printDeleteProjectInfo(pdo.Context, pdo.projectName)
    94  	//if err != nil {
    95  	//	return err
    96  	//}
    97  
    98  	if log.IsJSON() || pdo.forceFlag || ui.Proceed(fmt.Sprintf("Are you sure you want to delete project %v", pdo.projectName)) {
    99  
   100  		// If the --wait parameter has been passed, we add a spinner..
   101  		if pdo.waitFlag {
   102  			s = log.Spinner("Waiting for project to be deleted")
   103  			defer s.End(false)
   104  		}
   105  
   106  		err := pdo.prjClient.Delete(pdo.projectName, pdo.waitFlag)
   107  		if err != nil {
   108  			return err
   109  		}
   110  		s.End(true)
   111  
   112  		successMessage := fmt.Sprintf("Deleted project : %v", pdo.projectName)
   113  		log.Success(successMessage)
   114  		log.Warning("Warning! Projects are asynchronously deleted from the cluster. odo does its best to delete the project. Due to multi-tenant clusters, the project may still exist on a different node.")
   115  
   116  		if log.IsJSON() {
   117  			machineoutput.SuccessStatus(project.ProjectKind, pdo.projectName, successMessage)
   118  		}
   119  		return nil
   120  	}
   121  
   122  	return fmt.Errorf("aborting deletion of project: %v", pdo.projectName)
   123  }
