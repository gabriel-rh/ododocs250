    54  // Complete completes VersionOptions after they have been created
    55  func (o *VersionOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    56  	if !o.clientFlag {
    57  		// Let's fetch the info about the server, ignoring errors
    58  		client, err := kclient.New()
    59  
    60  		if err == nil {
    61  			// checking the value of timeout in preference
    62  			var timeout time.Duration
    63  			if o.prefClient != nil {
    64  				timeout = time.Duration(o.prefClient.GetTimeout()) * time.Second
    65  			} else {
    66  				// the default timeout will be used
    67  				// when the value is not readable from preference
    68  				timeout = preference.DefaultTimeout * time.Second
    69  			}
    70  			o.serverInfo, _ = client.GetServerVersion(timeout)
    71  		}
    72  	}
    73  	return nil
    74  }

    81  // Run contains the logic for the odo service create command
    82  func (o *VersionOptions) Run() (err error) {
    83  	// If verbose mode is enabled, dump all KUBECLT_* env variables
    84  	// this is usefull for debuging oc plugin integration
    85  	for _, v := range os.Environ() {
    86  		if strings.HasPrefix(v, "KUBECTL_") {
    87  			klog.V(4).Info(v)
    88  		}
    89  	}
    90  
    91  	fmt.Println("odo " + odoversion.VERSION + " (" + odoversion.GITCOMMIT + ")")
    92  
    93  	if !o.clientFlag && o.serverInfo != nil {
    94  		// make sure we only include OpenShift info if we actually have it
    95  		openshiftStr := ""
    96  		if len(o.serverInfo.OpenShiftVersion) > 0 {
    97  			openshiftStr = fmt.Sprintf("OpenShift: %v\n", o.serverInfo.OpenShiftVersion)
    98  		}
    99  		fmt.Printf("\n"+
   100  			"Server: %v\n"+
   101  			"%v"+
   102  			"Kubernetes: %v\n",
   103  			o.serverInfo.Address,
   104  			openshiftStr,
   105  			o.serverInfo.KubernetesVersion)
   106  	}
   107  
   108  	return nil
   109  }

   111  // NewCmdVersion implements the version odo command
   112  func NewCmdVersion(name, fullName string) *cobra.Command {
   113  	prefClient, err := preference.NewClient()
   114  	if err != nil {
   115  		klog.V(3).Info(errors.Wrap(err, "unable to read preference file"))
   116  	}
   117  	o := NewVersionOptions(prefClient)
   118  	// versionCmd represents the version command
   119  	var versionCmd = &cobra.Command{
   120  		Use:     name,
   121  		Short:   versionLongDesc,
   122  		Long:    versionLongDesc,
   123  		Example: fmt.Sprintf(versionExample, fullName),
   124  		Run: func(cmd *cobra.Command, args []string) {
   125  			genericclioptions.GenericRun(o, cmd, args)
   126  		},
   127  	}
   128  
   129  	// Add a defined annotation in order to appear in the help menu
   130  	versionCmd.Annotations = map[string]string{"command": "utility"}
   131  	versionCmd.SetUsageTemplate(util.CmdUsageTemplate)
   132  	versionCmd.Flags().BoolVar(&o.clientFlag, "client", false, "Client version only (no server required).")
   133  
   134  	return versionCmd

   138  func GetLatestReleaseInfo(info chan<- string) {
   139  	newTag, err := notify.CheckLatestReleaseTag(odoversion.VERSION)
   140  	if err != nil {
   141  		// The error is intentionally not being handled because we don't want
   142  		// to stop the execution of the program because of this failure

   144  	}
   145  	if len(newTag) > 0 {
   146  		info <- fmt.Sprintf(`

   149  visit %s to update.
   150  If you wish to disable this notification, run:
   151  odo preference set UpdateNotification false
   152  ---`, fmt.Sprint(newTag), OdoReleasesPage)

   154  	}
