    74  func (o *CreateOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    75  	o.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline).NeedDevfile(o.contextFlag))
    76  	if err != nil {
    77  		return err
    78  	}
    79  	// we convert the param list provided in the format of key=value list
    80  	// to a map
    81  	o.ParametersMap, err = util.MapFromParameters(o.parametersFlag)
    82  	if err != nil {
    83  		return err
    84  	}
    85  
    86  	err = validDevfileDirectory(o.contextFlag)
    87  	if err != nil {
    88  		return err
    89  	}
    90  	//if no args are provided and if request is not from file, user wants interactive mode
    91  	if o.fromFileFlag == "" && len(args) == 0 {
    92  		return fmt.Errorf("odo doesn't support interactive mode for creating Operator backed service")
    93  	}
    94  	o.Backend = NewOperatorBackend()
    95  	o.interactive = false
    96  	return o.Backend.CompleteServiceCreate(o, args)
    97  }

   105  func (o *CreateOptions) Run() (err error) {
   106  	err = o.Backend.RunServiceCreate(o)
   107  	if err != nil {
   108  		return fmt.Errorf("service %q already exists in configuration", o.ServiceName)
   109  	}
   110  
   111  	// Information on what to do next; don't do this if "--dry-run" was requested as it gets appended to the file
   112  	if !o.DryRunFlag {
   113  		log.Info("Successfully added service to the configuration; do 'odo push' to create service on the cluster")
   114  	}
   115  
   116  	return nil
