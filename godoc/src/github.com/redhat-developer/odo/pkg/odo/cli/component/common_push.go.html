    71  // ResolveProject completes the push options as needed
    72  func (cpo *CommonPushOptions) ResolveProject(prjName string) (err error) {
    73  
    74  	// check if project exist
    75  	isPrjExists, err := cpo.prjClient.Exists(prjName)
    76  	if err != nil {
    77  		return errors.Wrapf(err, "failed to check if project with name %s exists", prjName)
    78  	}
    79  	if !isPrjExists {
    80  		err = cpo.prjClient.Create(prjName, true)
    81  		if err != nil {
    82  			return errors.Wrapf(
    83  				err,
    84  				"project %s does not exist. Failed creating it. Please try after creating project using `odo project create <project_name>`",
    85  				prjName,
    86  			)
    87  		}
    88  	}
    89  	cpo.Context.KClient.SetNamespace(prjName)
    90  	return
    91  }

   106  func retrieveCmdNamespace(cmdline cmdline.Cmdline) (componentNamespace string, err error) {

   123  // GatherName parses the Devfile and retrieves an appropriate name in two ways.
   124  // 1. If metadata.name exists, we use it
   125  // 2. If metadata.name does NOT exist, we use the folder name where the devfile.yaml is located
   126  func GatherName(devObj parser.DevfileObj, devfilePath string) (string, error) {
   127  
   128  	metadata := devObj.Data.GetMetadata()
   129  
   130  	klog.V(4).Infof("metadata.Name: %s", metadata.Name)
   131  
   132  	// 1. Use metadata.name if it exists
   133  	if metadata.Name != "" {
   134  
   135  		// Remove any suffix's that end with `-`. This is because many Devfile's use the original v1 Devfile pattern of
   136  		// having names such as "foo-bar-" in order to prepend container names such as "foo-bar-container1"
   137  		return strings.TrimSuffix(metadata.Name, "-"), nil
   138  	}
   139  
   140  	// 2. Use the folder name as a last resort if nothing else exists
   141  	sourcePath, err := util.GetAbsPath(devfilePath)
   142  	if err != nil {
   143  		return "", errors.Wrap(err, "unable to get source path")
   144  	}
   145  	klog.V(4).Infof("Source path: %s", sourcePath)
   146  	klog.V(4).Infof("devfile dir: %s", filepath.Dir(sourcePath))
   147  
   148  	return filepath.Base(filepath.Dir(sourcePath)), nil
   149  }
