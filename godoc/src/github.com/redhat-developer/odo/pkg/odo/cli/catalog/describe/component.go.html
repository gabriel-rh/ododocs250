    96  // Run contains the logic for the command associated with DescribeComponentOptions
    97  func (o *DescribeComponentOptions) Run() (err error) {
    98  	w := tabwriter.NewWriter(os.Stdout, 5, 2, 3, ' ', tabwriter.TabIndent)
    99  	if log.IsJSON() {
   100  		if len(o.devfileComponents) > 0 {
   101  			out := []DevfileComponentDescription{}
   102  
   103  			for _, devfileComponent := range o.devfileComponents {
   104  				devObj, err := GetDevfile(devfileComponent)
   105  				if err != nil {
   106  					return err
   107  				}
   108  				out = append(out, DevfileComponentDescription{RegistryName: devfileComponent.Registry.Name, Devfile: devObj.Data})
   109  			}
   110  			machineoutput.OutputSuccess(out)
   111  		}
   112  	} else {
   113  		if len(o.devfileComponents) > 1 {
   114  			log.Warningf("There are multiple components named \"%s\" in different multiple devfile registries.\n", o.componentName)
   115  		}
   116  		if len(o.devfileComponents) > 0 {
   117  			fmt.Fprintln(w, "Devfile Component(s):")
   118  
   119  			for _, devfileComponent := range o.devfileComponents {
   120  				fmt.Fprintln(w, "\n* Registry: "+devfileComponent.Registry.Name)
   121  
   122  				devObj, err := GetDevfile(devfileComponent)
   123  				if err != nil {
   124  					return err
   125  				}
   126  
   127  				projects, err := devObj.Data.GetStarterProjects(parsercommon.DevfileOptions{})
   128  				if err != nil {
   129  					return err
   130  				}
   131  				// only print project info if there is at least one project in the devfile
   132  				err = o.PrintDevfileStarterProjects(w, projects, devObj)
   133  				if err != nil {
   134  					return err
   135  				}
   136  			}
   137  		} else {
   138  			fmt.Fprintln(w, "There are no Odo devfile components with the name \""+o.componentName+"\"")
   139  		}
   140  	}
   141  
   142  	return nil
   143  }

   163  // listDevfileComponentsByName lists all the devfile components that have the same name as the specified component
   164  func listDevfileComponentsByName(catalogDevfileList catalog.DevfileComponentTypeList, componentName string) []catalog.DevfileComponentType {
   165  	var components []catalog.DevfileComponentType

   174  // GetDevfile downloads the devfile in memory and return the devfile object
   175  func GetDevfile(devfileComponent catalog.DevfileComponentType) (parser.DevfileObj, error) {
   176  	devObj, err := getDevFileNoValidation(devfileComponent)
   177  	if err != nil {
   178  		return devObj, err
   179  	}
   180  
   181  	err = validate.ValidateDevfileData(devObj.Data)
   182  	if err != nil {
   183  		return devObj, err
   184  	}
   185  	return devObj, nil
   186  }

   188  func getDevFileNoValidation(devfileComponent catalog.DevfileComponentType) (parser.DevfileObj, error) {
   189  	if strings.Contains(devfileComponent.Registry.URL, "github") {
   190  		devObj, err := devfile.ParseAndValidateFromURL(devfileComponent.Registry.URL + devfileComponent.Link)
   191  		return devObj, errors.Wrapf(err, "Failed to download devfile.yaml from Github-based registry for devfile component: %s", devfileComponent.Name)
   192  	}
   193  	registryURL, err := url.Parse(devfileComponent.Registry.URL)
   194  	if err != nil {
   195  		return parser.DevfileObj{}, errors.Wrapf(err, "Failed to parse registry URL for devfile component: %s", devfileComponent.Name)
   196  	}
   197  	registryURL.Path = path.Join(registryURL.Path, "devfiles", devfileComponent.Name)
   198  	devObj, err := devfile.ParseAndValidateFromURL(registryURL.String())
   199  	return devObj, errors.Wrapf(err, "Failed to download devfile.yaml from OCI-based registry for devfile component: %s", devfileComponent.Name)
   200  }

   202  // PrintDevfileStarterProjects prints all the starter projects in a devfile
   203  // If no starter projects exists in the devfile, it prints the whole devfile
   204  func (o *DescribeComponentOptions) PrintDevfileStarterProjects(w *tabwriter.Writer, projects []devfilev1.StarterProject, devObj parser.DevfileObj) error {
   205  	if len(projects) > 0 {
   206  		fmt.Fprintln(w, "\nStarter Projects:")
   207  		for _, project := range projects {
   208  			yamlData, err := yaml.Marshal(project)
   209  			if err != nil {
   210  				return errors.Wrapf(err, "Failed to marshal devfile object into yaml")
   211  			}
   212  			fmt.Printf("---\n%s", string(yamlData))
   213  		}
   214  	} else {
   215  		fmt.Fprintln(w, "The Odo devfile component \""+o.componentName+"\" has no starter projects.")
   216  		yamlData, err := yaml.Marshal(devObj)
   217  		if err != nil {
   218  			return errors.Wrapf(err, "Failed to marshal devfile object into yaml")
   219  		}
   220  		fmt.Printf("---\n%s", string(yamlData))
   221  	}
   222  	return nil
   223  }
