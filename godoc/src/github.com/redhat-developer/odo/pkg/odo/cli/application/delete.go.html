    77  // Run contains the logic for the odo command
    78  func (o *DeleteOptions) Run() (err error) {
    79  	if o.IsJSON() {
    80  		return o.appClient.Delete(o.appName)
    81  	}
    82  
    83  	// Print App Information which will be deleted
    84  	err = printAppInfo(o.appClient, o.appName, o.GetProject())
    85  	if err != nil {
    86  		return err
    87  	}
    88  
    89  	if o.forceFlag || ui.Proceed(fmt.Sprintf("Are you sure you want to delete the application: %v from project: %v", o.appName, o.GetProject())) {
    90  		err = o.appClient.Delete(o.appName)
    91  		if err != nil {
    92  			return err
    93  		}
    94  		log.Infof("Deleted application: %s from project: %v", o.appName, o.GetProject())
    95  	} else {
    96  		log.Infof("Aborting deletion of application: %v", o.appName)
    97  	}
    98  	return nil
    99  }

   101  // printAppInfo will print information about the app requested for deletion
   102  func printAppInfo(appClient application.Client, appName string, projectName string) error {
   103  	components, err := appClient.ComponentList(appName)
   104  	if err != nil {
   105  		return err
   106  	}
   107  	if len(components) != 0 {
   108  		log.Info("This application has following components that will be deleted")
   109  		for _, currentComponent := range components {
   110  			log.Info("component named", currentComponent.Name)
   111  
   112  			if len(currentComponent.Spec.URL) != 0 {
   113  				log.Info("This component has following urls that will be deleted with component")
   114  				for _, u := range currentComponent.Spec.URLSpec {
   115  					log.Info("URL named", u.GetName(), "with host", u.Spec.Host, "having protocol", u.Spec.Protocol, "at port", u.Spec.Port)
   116  				}
   117  			}
   118  
   119  			if len(currentComponent.Spec.Storage) != 0 {
   120  				log.Info("The component has following storages which will be deleted with the component")
   121  				for _, storage := range currentComponent.Spec.StorageSpec {
   122  					store := storage
   123  					log.Info("Storage named", store.GetName(), "of size", store.Spec.Size)
   124  				}
   125  			}
   126  		}
   127  	}
   128  	return nil
   129  }
