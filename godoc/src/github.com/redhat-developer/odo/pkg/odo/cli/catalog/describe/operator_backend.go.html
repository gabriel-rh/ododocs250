    37  func (ohb *operatorBackend) CompleteDescribeService(dso *DescribeServiceOptions, args []string) error {
    38  	ohb.Name = args[0]
    39  	oprType, CR, err := service.SplitServiceKindName(ohb.Name)
    40  	if err != nil {
    41  		klog.V(2).Infof("could not determine csv, falling back to describing all of them")
    42  		oprType = args[0]
    43  		CR = ""
    44  	}
    45  	// we check if the cluster supports ClusterServiceVersion or not.
    46  	isCSVSupported, err := dso.KClient.IsCSVSupported()
    47  	if err != nil {
    48  		// if there is an error checking it, we return the error.
    49  		return err
    50  	}
    51  	// if its not supported then we return an error
    52  	if !isCSVSupported {
    53  		return errors.New("it seems the cluster doesn't support Operators. Please install OLM and try again")
    54  	}
    55  	ohb.OperatorType = oprType
    56  	ohb.CustomResource = CR
    57  	return nil
    58  }

    60  func (ohb *operatorBackend) ValidateDescribeService(dso *DescribeServiceOptions) error {
    61  	var err error
    62  	if ohb.OperatorType == "" {
    63  		return errors.New("invalid service name provided. should either be <operator-type> or <operator-type>/<crd-name>")
    64  	}
    65  	// make sure that CSV of the specified OperatorType exists
    66  	ohb.CSV, err = dso.KClient.GetClusterServiceVersion(ohb.OperatorType)
    67  	if err != nil {
    68  		// error only occurs when OperatorHub is not installed.
    69  		// k8s does't have it installed by default but OCP does
    70  		return err
    71  	}
    72  
    73  	//if both operator type and cr are known, validate that it exists
    74  	if ohb.OperatorType != "" && ohb.CustomResource != "" {
    75  		var hasCR bool
    76  		hasCR, ohb.CR = dso.KClient.CheckCustomResourceInCSV(ohb.CustomResource, &ohb.CSV)
    77  		if !hasCR {
    78  			return fmt.Errorf("the %q resource doesn't exist in specified %q operator", ohb.CustomResource, ohb.OperatorType)
    79  		}
    80  
    81  		ohb.CRDSpec, err = dso.KClient.GetCRDSpec(ohb.CR, ohb.OperatorType, ohb.CustomResource)
    82  		if err != nil {
    83  			return err
    84  		}
    85  	}
    86  	return nil

    90  func (ohb *operatorBackend) RunDescribeService(dso *DescribeServiceOptions) error {
    91  	if ohb.OperatorType != "" && ohb.CustomResource == "" {
    92  		//we don't have cr so list all possible crds
    93  		ohb.CRDList = service.NewOperatorBackedCRList(ohb.OperatorType, ohb.CSV.Spec.DisplayName, ohb.CSV.Spec.Description)

    97  				Kind:        custRes.Kind,
    98  				Description: custRes.Description,
    99  			})
   100  		}
   101  		if log.IsJSON() {
   102  			machineoutput.OutputSuccess(ohb.CRDList)
   103  		} else {
   104  			HumanReadableCRListOutput(os.Stdout, ohb.CRDList)
   105  		}
   106  		return nil
   107  	}
   108  	if dso.exampleFlag {
   109  		almExample, err := service.GetAlmExample(ohb.CSV, ohb.CustomResource, ohb.OperatorType)
   110  		if err != nil {
   111  			return err
   112  		}
   113  		if log.IsJSON() {
   114  			jsonExample := service.NewOperatorExample(almExample)
   115  			jsonCR, err := json.MarshalIndent(jsonExample, "", "  ")
   116  			if err != nil {
   117  				return err
   118  			}
   119  
   120  			fmt.Println(string(jsonCR))
   121  
   122  		} else {
   123  			yamlCR, err := yaml.Marshal(almExample)
   124  			if err != nil {
   125  				return err
   126  			}
   127  
   128  			log.Info(string(yamlCR))
   129  		}
   130  	}
   131  	svc := service.NewOperatorBackedService(ohb.Name, ohb.CR.Kind, ohb.CR.Version, ohb.CR.Description, ohb.CR.DisplayName, ohb.CRDSpec)
   132  	if log.IsJSON() {
   133  		machineoutput.OutputSuccess(svc)
   134  	} else {

   167  // displayProperties displays the properties of an OpenAPI schema in a human readable form
   168  // required fields are displayed first
   169  func displayProperties(w io.Writer, schema *spec.Schema, prefix string) {
   170  	required := schema.Required
   171  	requiredMap := map[string]bool{}
   172  	for _, req := range required {
   173  		requiredMap[req] = true
   174  	}
   175  
   176  	reqKeys := []string{}
   177  	for key := range schema.Properties {
   178  		if requiredMap[key] {
   179  			reqKeys = append(reqKeys, key)
   180  		}
   181  	}
   182  	sort.Strings(reqKeys)
   183  
   184  	nonReqKeys := []string{}
   185  	for key := range schema.Properties {
   186  		if !requiredMap[key] {
   187  			nonReqKeys = append(nonReqKeys, key)
   188  		}
   189  	}
   190  	sort.Strings(nonReqKeys)
   191  	keys := append(reqKeys, nonReqKeys...)
   192  
   193  	for _, key := range keys {
   194  		property := schema.Properties[key]
   195  		requiredInfo := ""
   196  		if requiredMap[key] {
   197  			requiredInfo = "-required-"
   198  		}
   199  		fmt.Fprintf(w, "%s%s (%s)   %s\n", strings.Repeat(" ", 3+2*strings.Count(prefix, ".")), prefix+key, getTypeString(property), requiredInfo)
   200  		nl := false
   201  		if len(property.Title) > 0 {
   202  			fmt.Fprintf(w, "%s\n", indentText(property.Title, 5+2*strings.Count(prefix, ".")))
   203  			nl = true
   204  		}
   205  		if len(property.Description) > 0 {
   206  			fmt.Fprintf(w, "%s\n", indentText(property.Description, 5+2*strings.Count(prefix, ".")))
   207  			nl = true
   208  		}
   209  		if !nl {
   210  			fmt.Fprintln(w)
   211  		}
   212  		if property.Type.Contains("object") {
   213  			displayProperties(w, &property, prefix+key+".")
   214  		} else if property.Type.Contains("array") && property.Items.Schema.Type.Contains("object") {
   215  			displayProperties(w, property.Items.Schema, prefix+key+".*.")
   216  		}
   217  	}
   218  }

   220  func getTypeString(property spec.Schema) string {
   221  	if len(property.Type) != 1 {
   222  		// should not happen
   223  		return strings.Join(property.Type, ", ")
   224  	}
   225  	tpe := property.Type[0]
   226  	if tpe == "array" {
   227  		tpe = "[]" + getTypeString(*property.Items.Schema)
   228  	}
   229  	return tpe

   232  func indentText(t string, indent int) string {

   236  		res += strings.Repeat(" ", indent) + line + "\n"

   260  // Add adds the word to the line, returns true if we could, false if we
   261  // didn't have enough room. It's always possible to add to an empty line.
   262  func (l *line) Add(word string) bool {
   263  	newLine := line{
   264  		wrap:  l.wrap,
   265  		words: append(l.words, word),
   266  	}
   267  	if newLine.Len() <= l.wrap || len(l.words) == 0 {
   268  		l.words = newLine.words
   269  		return true
   270  	}
   271  	return false
   272  }

   274  func wrapString(str string, wrap int) []string {
   275  	wrapped := []string{}
   276  	l := line{wrap: wrap}
   277  	// track the last word added to the current line
   278  	lastWord := ""
   279  	flush := func() {
   280  		if !l.Empty() {
   281  			lastWord = ""
   282  			wrapped = append(wrapped, l.String())
   283  			l = line{wrap: wrap}
   284  		}
   285  	}
   286  
   287  	// iterate over the lines in the original description
   288  	for _, str := range strings.Split(str, "\n") {
   289  		// preserve code blocks and blockquotes as-is
   290  		if strings.HasPrefix(str, "    ") {
   291  			flush()
   292  			wrapped = append(wrapped, str)
   293  			continue
   294  		}
   295  
   296  		// preserve empty lines after the first line, since they can separate logical sections
   297  		if len(wrapped) > 0 && len(strings.TrimSpace(str)) == 0 {
   298  			flush()
   299  			wrapped = append(wrapped, "")
   300  			continue
   301  		}
   302  
   303  		// flush if we should start a new line
   304  		if shouldStartNewLine(lastWord, str) {
   305  			flush()
   306  		}
   307  		words := strings.Fields(str)
   308  		for _, word := range words {
   309  			lastWord = word
   310  			if !l.Add(word) {
   311  				flush()
   312  				if !l.Add(word) {
   313  					panic("Couldn't add to empty line.")
   314  				}
   315  			}
   316  		}
   317  	}
   318  	flush()
   319  	return wrapped
   320  }

   322  var bullet = regexp.MustCompile(`^(\d+\.?|-|\*)\s`)

   324  func shouldStartNewLine(lastWord, str string) bool {
   325  	// preserve line breaks ending in :
   326  	if strings.HasSuffix(lastWord, ":") {
   327  		return true
   328  	}
   329  
   330  	// preserve code blocks
   331  	if strings.HasPrefix(str, "    ") {
   332  		return true
   333  	}
   334  	str = strings.TrimSpace(str)
   335  	// preserve empty lines
   336  	if len(str) == 0 {
   337  		return true
   338  	}
   339  	// preserve lines that look like they're starting lists
   340  	if bullet.MatchString(str) {
   341  		return true
   342  	}
   343  	// otherwise combine
   344  	return false
   345  }
