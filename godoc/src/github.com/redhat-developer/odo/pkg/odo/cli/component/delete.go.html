    82  func (do *DeleteOptions) Run() (err error) {
    83  	klog.V(4).Infof("component delete called")
    84  	klog.V(4).Infof("args: %#v", do)
    85  
    86  	// odo delete --deploy || odo delete --all
    87  	if do.deployFlag || do.allFlag {
    88  		if do.forceFlag || ui.Proceed("Are you sure you want to undeploy?") {
    89  			err = do.DevfileUnDeploy()
    90  			if err != nil {
    91  				log.Errorf("error occurred while undeploying, cause: %v", err)
    92  			}
    93  		} else {
    94  			log.Error("Aborting the un-deployment")
    95  		}
    96  	}
    97  
    98  	// odo delete || odo delete --all
    99  	if !do.deployFlag || do.allFlag {
   100  		if do.forceFlag || ui.Proceed(fmt.Sprintf("Are you sure you want to delete the devfile component: %s?", do.EnvSpecificInfo.GetName())) {
   101  			err = do.DevfileComponentDelete()
   102  			if err != nil {
   103  				log.Errorf("error occurred while deleting component, cause: %v", err)
   104  			}
   105  		} else {
   106  			log.Error("Aborting deletion of component")
   107  		}
   108  	}
   109  
   110  	// Delete the configuration files
   111  	if do.allFlag {
   112  		log.Info("\nDeleting local config")
   113  		// Prompt and delete env folder
   114  		if do.forceFlag || ui.Proceed("Are you sure you want to delete env folder?") {
   115  			if !do.EnvSpecificInfo.Exists() {
   116  				return fmt.Errorf("env folder doesn't exist for the component")
   117  			}
   118  			if err = util.DeleteIndexFile(filepath.Dir(do.GetDevfilePath())); err != nil {
   119  				return err
   120  			}
   121  
   122  			err = do.EnvSpecificInfo.DeleteEnvInfoFile()
   123  			if err != nil {
   124  				return err
   125  			}
   126  			err = do.EnvSpecificInfo.DeleteEnvDirIfEmpty()
   127  			if err != nil {
   128  				return err
   129  			}
   130  			err = util.DeletePath(filepath.Join(do.contextFlag, util.DotOdoDirectory))
   131  			if err != nil {
   132  				return err
   133  			}
   134  			log.Successf("Successfully deleted env file")
   135  		} else {
   136  			log.Error("Aborting deletion of env folder")
   137  		}
   138  
   139  		if do.forceFlag {
   140  			if !util.CheckPathExists(do.GetDevfilePath()) {
   141  				return fmt.Errorf("devfile.yaml does not exist in the current directory")
   142  			}
   143  			if !do.EnvSpecificInfo.IsUserCreatedDevfile() {
   144  
   145  				// first remove the uri based files mentioned in the devfile
   146  				devfileObj, err := devfile.ParseAndValidateFromFile(do.GetDevfilePath())
   147  				if err != nil {
   148  					return err
   149  				}
   150  
   151  				err = common.RemoveDevfileURIContents(devfileObj, do.contextFlag)
   152  				if err != nil {
   153  					return err
   154  				}
   155  
   156  				empty, err := util.IsEmpty(filepath.Join(do.contextFlag, consts.UriFolder))
   157  				if err != nil && !os.IsNotExist(err) {
   158  					return err
   159  				}
   160  
   161  				if !os.IsNotExist(err) && empty {
   162  					err = os.RemoveAll(filepath.Join(do.contextFlag, consts.UriFolder))
   163  					if err != nil {
   164  						return err
   165  					}
   166  				}
   167  
   168  				err = util.DeletePath(do.GetDevfilePath())
   169  				if err != nil {
   170  					return err
   171  				}
   172  
   173  				log.Successf("Successfully deleted devfile.yaml file")
   174  
   175  			} else {
   176  				log.Info("Didn't delete the devfile as it was user provided")
   177  			}
   178  
   179  		} else if ui.Proceed("Are you sure you want to delete devfile.yaml?") {
   180  			if !util.CheckPathExists(do.GetDevfilePath()) {
   181  				return fmt.Errorf("devfile.yaml does not exist in the current directory")
   182  			}
   183  
   184  			// first remove the uri based files mentioned in the devfile
   185  			devfileObj, err := devfile.ParseAndValidateFromFile(do.GetDevfilePath())
   186  			if err != nil {
   187  				return err
   188  			}
   189  
   190  			err = common.RemoveDevfileURIContents(devfileObj, do.contextFlag)
   191  			if err != nil {
   192  				return err
   193  			}
   194  
   195  			empty, err := util.IsEmpty(filepath.Join(do.contextFlag, consts.UriFolder))
   196  			if err != nil && !os.IsNotExist(err) {
   197  				return err
   198  			}
   199  
   200  			if !os.IsNotExist(err) && empty {
   201  				err = os.RemoveAll(filepath.Join(do.contextFlag, consts.UriFolder))
   202  				if err != nil {
   203  					return err
   204  				}
   205  			}
   206  
   207  			err = util.DeletePath(do.GetDevfilePath())
   208  			if err != nil {
   209  				return err
   210  			}
   211  
   212  			log.Successf("Successfully deleted devfile.yaml file")
   213  		} else {
   214  			log.Error("Aborting deletion of devfile.yaml file")
   215  		}
   216  	}
   217  	return nil
   218  }
