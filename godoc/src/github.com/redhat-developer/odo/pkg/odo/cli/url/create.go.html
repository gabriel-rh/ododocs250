    86  func (o *CreateOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    87  	params := genericclioptions.NewCreateParameters(cmdline).NeedDevfile(o.GetComponentContext()).RequireRouteAvailability()
    88  	if o.nowFlag {
    89  		params.CreateAppIfNeeded()
    90  	}
    91  	o.Context, err = genericclioptions.New(params)
    92  	if err != nil {
    93  		return err
    94  	}
    95  
    96  	var urlType localConfigProvider.URLKind
    97  	if o.ingressFlag {
    98  		urlType = localConfigProvider.INGRESS
    99  	}
   100  
   101  	// get the name
   102  	if len(args) != 0 {
   103  		o.urlName = args[0]
   104  	}
   105  
   106  	// create the localURL
   107  	o.url = localConfigProvider.LocalURL{
   108  		Name:      o.urlName,
   109  		Port:      o.portFlag,
   110  		Secure:    o.secureFlag,
   111  		Host:      o.hostFlag,
   112  		TLSSecret: o.tlsSecretFlag,
   113  		Kind:      urlType,
   114  		Container: o.containerFlag,
   115  		Protocol:  o.protocolFlag,
   116  		Path:      o.pathFlag,
   117  	}
   118  
   119  	// complete the URL
   120  	err = o.Context.LocalConfigProvider.CompleteURL(&o.url)
   121  	if err != nil {
   122  		return err
   123  	}
   124  
   125  	if o.nowFlag {
   126  		prjName := o.Context.LocalConfigProvider.GetNamespace()
   127  		o.ResolveSrcAndConfigFlags()
   128  		err = o.ResolveProject(prjName)
   129  		if err != nil {
   130  			return err
   131  		}
   132  	}
   133  
   134  	return nil
   135  }

   137  // Validate validates the CreateOptions based on completed values
   138  func (o *CreateOptions) Validate() (err error) {
   139  	if !util.CheckOutputFlag(o.GetOutputFlag()) {
   140  		return fmt.Errorf("given output format %s is not supported", o.GetOutputFlag())
   141  	}
   142  
   143  	errorList := make([]string, 0)
   144  	// Check if url name is more than 63 characters long
   145  	if len(o.urlName) > 63 {
   146  		errorList = append(errorList, "URL name must be shorter than 63 characters")
   147  	}
   148  
   149  	// validate the URL
   150  	err = o.LocalConfigProvider.ValidateURL(o.url)
   151  	if err != nil {
   152  		errorList = append(errorList, err.Error())
   153  	}
   154  
   155  	if len(errorList) > 0 {
   156  		for i := range errorList {
   157  			errorList[i] = fmt.Sprintf("\t- %s", errorList[i])
   158  		}
   159  		return fmt.Errorf("URL creation failed:\n%s", strings.Join(errorList, "\n"))
   160  	}
   161  	return nil
   162  }

   164  // Run contains the logic for the odo url create command
   165  func (o *CreateOptions) Run() (err error) {
   166  
   167  	// create the URL and write it to the local config
   168  	err = o.Context.LocalConfigProvider.CreateURL(o.url)
   169  	if err != nil {
   170  		return err
   171  	}
   172  	log.Successf("URL %s created for component: %v", o.url.Name, o.LocalConfigProvider.GetName())
   173  
   174  	if o.nowFlag {
   175  		// if the now flag is specified, push the changes
   176  		o.CompleteDevfilePath()
   177  		o.EnvSpecificInfo = o.Context.EnvSpecificInfo
   178  		err = o.DevfilePush()
   179  		if err != nil {
   180  			return errors.Wrap(err, "failed to push changes")
   181  		}
   182  	} else {
   183  		log.Italic("\nTo apply the URL configuration changes, please use `odo push`")
   184  	}
   185  
   186  	if log.IsJSON() {
   187  		u := url.NewURLFromLocalURL(o.url)
   188  		u.Status.State = url.StateTypeNotPushed
   189  		if o.nowFlag {
   190  			u.Status.State = url.StateTypePushed
   191  		}
   192  		machineoutput.OutputSuccess(u)
   193  	}
   194  	return nil
   195  }
