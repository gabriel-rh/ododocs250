    75  // Complete completes all the required options for port-forward cmd.
    76  func (o *PortForwardOptions) Complete(cmdline cmdline.Cmdline, args []string) (err error) {
    77  	o.Context, err = genericclioptions.New(genericclioptions.NewCreateParameters(cmdline))
    78  	if err != nil {
    79  		return err
    80  	}
    81  
    82  	remotePort := o.Context.EnvSpecificInfo.GetDebugPort()
    83  
    84  	// try to listen on the given local port and check if the port is free or not
    85  	addressLook := "localhost:" + strconv.Itoa(o.localPortFlag)
    86  	listener, err := net.Listen("tcp", addressLook)
    87  	if err != nil {
    88  		// if the local-port flag is set by the user, return the error and stop execution
    89  		if cmdline.IsFlagSet("local-port") {
    90  			return err
    91  		}
    92  		// else display a error message and auto select a new free port
    93  		log.Errorf("the local debug port %v is not free, cause: %v", o.localPortFlag, err)
    94  		o.localPortFlag, err = util.HTTPGetFreePort()
    95  		if err != nil {
    96  			return err
    97  		}
    98  		log.Infof("The local port %v is auto selected", o.localPortFlag)
    99  	} else {
   100  		err = listener.Close()
   101  		if err != nil {
   102  			return err
   103  		}
   104  	}
   105  
   106  	o.PortPair = fmt.Sprintf("%d:%d", o.localPortFlag, remotePort)
   107  
   108  	// Using Discard streams because nothing important is logged
   109  	o.PortForwarder = debug.NewDefaultPortForwarder(o.Context.EnvSpecificInfo.GetName(), o.Context.GetApplication(), o.Context.EnvSpecificInfo.GetNamespace(), o.KClient, k8sgenclioptions.NewTestIOStreamsDiscard())
   110  
   111  	o.StopChannel = make(chan struct{}, 1)
   112  	o.ReadyChannel = make(chan struct{})
   113  	return nil
   114  }

   124  // Run implements all the necessary functionality for port-forward cmd.
   125  func (o PortForwardOptions) Run() error {
   126  
   127  	signals := make(chan os.Signal, 1)
   128  	signal.Notify(signals, os.Interrupt,
   129  		syscall.SIGHUP,
   130  		syscall.SIGINT,
   131  		syscall.SIGTERM,
   132  		syscall.SIGQUIT)
   133  	defer signal.Stop(signals)
   134  	defer os.RemoveAll(debug.GetDebugInfoFilePath(o.Context.EnvSpecificInfo.GetName(), o.Context.GetApplication(), o.Context.EnvSpecificInfo.GetNamespace()))
   135  
   136  	go func() {
   137  		<-signals
   138  		if o.StopChannel != nil {
   139  			close(o.StopChannel)
   140  		}
   141  	}()
   142  
   143  	err := debug.CreateDebugInfoFile(o.PortForwarder, o.PortPair)
   144  	if err != nil {
   145  		return err
   146  	}
   147  
   148  	devfilePath := location.DevfileLocation(o.contextFlag)
   149  	return o.PortForwarder.ForwardPorts(o.PortPair, o.StopChannel, o.ReadyChannel, util.CheckPathExists(devfilePath))
   150  }
