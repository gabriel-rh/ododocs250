    45  func (icm InteractiveCreateMethod) FetchDevfileAndCreateComponent(co *CreateOptions, cmdline cmdline.Cmdline, args []string) error {
    46  	catalogDevfileList, err := validateAndFetchRegistry(co.devfileMetadata.devfileRegistry.Name)
    47  	if err != nil {
    48  		return err
    49  	}
    50  
    51  	//SET METADATA
    52  	// Component type: We provide devfile component list to let user choose
    53  	componentType := ui.SelectDevfileComponentType(catalogDevfileList.Items)
    54  
    55  	// Component name: User needs to specify the component name, by default it is component type that user chooses
    56  	componentName := ui.EnterDevfileComponentName(componentType)
    57  
    58  	// Component namespace: User needs to specify component namespace, by default it is the current active namespace
    59  	var componentNamespace string
    60  	if cmdline.IsFlagSet("project") {
    61  		componentNamespace, err = cmdline.FlagValue("project")
    62  		if err != nil {
    63  			return err
    64  		}
    65  	} else {
    66  		var currentNamespace string
    67  		if co.KClient != nil {
    68  			// Get the current project if no --project is passed, but --now is passed
    69  			currentNamespace = co.KClient.GetCurrentProjectName()
    70  		} else {
    71  			var client kclient.ClientInterface
    72  			client, err = kclient.New()
    73  			// if err != nil, currentNamespace will be an empty string
    74  			if err == nil && client != nil {
    75  				// Get the current project if no --project is passed and using offline context
    76  				currentNamespace = client.GetCurrentProjectName()
    77  			}
    78  		}
    79  		componentNamespace = ui.EnterDevfileComponentProject(currentNamespace)
    80  	}
    81  
    82  	co.devfileMetadata.componentType = componentType
    83  	co.devfileName = componentType
    84  	co.devfileMetadata.componentName = componentName
    85  	co.devfileMetadata.componentNamespace = componentNamespace
    86  
    87  	co.devfileMetadata.devfileLink, co.devfileMetadata.devfileRegistry, err = findDevfileFromRegistry(catalogDevfileList, co.devfileMetadata.devfileRegistry.Name, co.devfileMetadata.componentType)
    88  	if err != nil {
    89  		return err
    90  	}
    91  	return fetchDevfileFromRegistry(co.devfileMetadata.devfileRegistry, co.devfileMetadata.devfileLink, co.DevfilePath, co.devfileMetadata.componentType, co.contextFlag, co.prefClient)
    92  }

   106  func (dcm DirectCreateMethod) FetchDevfileAndCreateComponent(co *CreateOptions, cmdline cmdline.Cmdline, args []string) error {
   107  	catalogDevfileList, err := validateAndFetchRegistry(co.devfileMetadata.devfileRegistry.Name)
   108  	if err != nil {
   109  		return err
   110  	}
   111  	// SET METADATA
   112  	// The first argument passed will always be considered as component type
   113  	co.devfileMetadata.componentType = args[0]
   114  	co.devfileName = args[0]
   115  
   116  	var componentName string
   117  	if len(args) == 2 {
   118  		// If more than one argument is passed, then the second one will be considered as component name
   119  		componentName = args[1]
   120  	} else {
   121  		// If only component type is passed, then the component name will be created by odo
   122  		componentName, err = createDefaultComponentName(
   123  			co.devfileMetadata.componentType,
   124  			co.contextFlag,
   125  			co.prefClient,
   126  		)
   127  		if err != nil {
   128  			return err
   129  		}
   130  	}
   131  	co.devfileMetadata.componentName = componentName
   132  	co.devfileMetadata.devfileLink, co.devfileMetadata.devfileRegistry, err = findDevfileFromRegistry(catalogDevfileList, co.devfileMetadata.devfileRegistry.Name, co.devfileMetadata.componentType)
   133  	if err != nil {
   134  		return err
   135  	}
   136  	return fetchDevfileFromRegistry(co.devfileMetadata.devfileRegistry, co.devfileMetadata.devfileLink, co.DevfilePath, co.devfileMetadata.componentType, co.contextFlag, co.prefClient)

   248  func conflictCheckForDevfileFlag(args []string, registryName string) error {
   249  	// More than one arguments should not be allowed when --devfile is used
   250  	if len(args) > 1 {
   251  		return &DevfileExistsExtraArgsError{len(args)}
   252  	}
   253  	// Check if both --devfile and --registry flag are used, in which case raise an error
   254  	if registryName != "" {
   255  		return &DevfileFlagWithRegistryFlagError{}
   256  	}
   257  	return nil

   260  // validateAndFetchRegistry validates if the provided registryName exists and returns the devfile listed in the registy;
   261  // if the registryName is "", then it returns devfiles of all the available registries
   262  func validateAndFetchRegistry(registryName string) (catalog.DevfileComponentTypeList, error) {
   263  	// Validate if the component type is available
   264  	if registryName != "" {
   265  		registryExistSpinner := log.Spinnerf("Checking if the registry %q exists", registryName)
   266  		defer registryExistSpinner.End(false)
   267  		registryList, e := catalog.GetDevfileRegistries(registryName)
   268  		if e != nil {
   269  			return catalog.DevfileComponentTypeList{}, errors.Wrap(e, "failed to get registry")
   270  		}
   271  		if len(registryList) == 0 {
   272  			return catalog.DevfileComponentTypeList{}, errors.Errorf("registry %s doesn't exist, please specify a valid registry via --registry", registryName)
   273  		}
   274  		registryExistSpinner.End(true)
   275  	}
   276  
   277  	klog.V(4).Infof("Fetching the available devfile components")
   278  	// Get available devfile components for checking devfile compatibility
   279  	catalogDevfileList, err := catalog.ListDevfileComponents(registryName)
   280  	if err != nil {
   281  		return catalog.DevfileComponentTypeList{}, err
   282  	}
   283  
   284  	if registryName != "" && catalogDevfileList.Items == nil {
   285  		return catalog.DevfileComponentTypeList{}, errors.Errorf("can't create devfile component from registry %s", registryName)
   286  	}
   287  
   288  	if len(catalogDevfileList.DevfileRegistries) == 0 {
   289  		return catalog.DevfileComponentTypeList{}, errors.New("Registry is empty, please run `odo registry add <registry name> <registry URL>` to add a registry\n")
   290  	}
   291  	return catalogDevfileList, nil
   292  }

   294  // findDevfileFromRegistry finds the devfile and returns necessary information related to it
   295  func findDevfileFromRegistry(catalogDevfileList catalog.DevfileComponentTypeList, registryName, componentType string) (devfileLink string, devfileRegistry catalog.Registry, err error) {
   296  	devfileExistSpinner := log.Spinnerf("Checking if the devfile for %q exists on available registries", componentType)
   297  	defer devfileExistSpinner.End(false)
   298  	if registryName != "" {
   299  		devfileExistSpinner = log.Spinnerf("Checking if the devfile for %q exists on registry %q", componentType, registryName)
   300  	}
   301  
   302  	// Find the request devfile from the registry
   303  	for _, devfileComponent := range catalogDevfileList.Items {
   304  		if componentType == devfileComponent.Name {
   305  			devfileExistSpinner.End(true)
   306  			return devfileComponent.Link, devfileComponent.Registry, nil
   307  		}
   308  	}
   309  	return "", catalog.Registry{}, fmt.Errorf("devfile component type %q is not supported, please run `odo catalog list components` for a list of supported devfile component types", componentType)
   310  }

   312  // fetchDevfileFromRegistry fetches the required devfile from the list catalogDevfileList
   313  func fetchDevfileFromRegistry(registry catalog.Registry, devfileLink, devfilePath, componentType, componentContext string, prefClient preference.Client) (err error) {
   314  	// Download devfile from registry
   315  	registrySpinner := log.Spinnerf("Creating a devfile component from registry %q", registry.Name)
   316  	defer registrySpinner.End(false)
   317  
   318  	// For GitHub based registries
   319  	if registryUtil.IsGitBasedRegistry(registry.URL) {
   320  		registryUtil.PrintGitRegistryDeprecationWarning()
   321  
   322  		params := util.HTTPRequestParams{
   323  			URL: registry.URL + devfileLink,
   324  		}
   325  
   326  		secure := registryUtil.IsSecure(prefClient, registry.Name)
   327  		if secure {
   328  			var token string
   329  			token, err = keyring.Get(fmt.Sprintf("%s%s", util.CredentialPrefix, registry.Name), registryUtil.RegistryUser)
   330  			if err != nil {
   331  				return errors.Wrap(err, "unable to get secure registry credential from keyring")
   332  			}
   333  			params.Token = token
   334  		}
   335  
   336  		devfileData, err := util.DownloadFileInMemoryWithCache(params, prefClient.GetRegistryCacheTime())
   337  		if err != nil {
   338  			return err
   339  		}
   340  
   341  		err = ioutil.WriteFile(devfilePath, devfileData, 0644) // #nosec G306
   342  		if err != nil {
   343  			return err
   344  		}
   345  	} else {
   346  		err := registryLibrary.PullStackFromRegistry(registry.URL, componentType, componentContext, segment.GetRegistryOptions())
   347  		if err != nil {
   348  			return err
   349  		}
   350  	}
   351  	registrySpinner.End(true)
   352  	return nil
   353  }

   355  // getMetadataForExistingDevfile sets metadata for a user provided devfile; UserCreatedDevfileCreateMethod, HTTPCreateMethod, and FileCreateMethod
   356  func getMetadataForExistingDevfile(co *CreateOptions, args []string) (componentName, componentType string, err error) {
   357  	devObj, err := devfileParseFromFile(co.DevfilePath, false)
   358  	if err != nil {
   359  		return "", "", err
   360  	}
   361  	componentType = component.GetComponentTypeFromDevfileMetadata(devObj.Data.GetMetadata())
   362  
   363  	// Set component name
   364  	if len(args) > 0 {
   365  		// user provided name: `odo create mynode`
   366  		componentName = args[0]
   367  	} else {
   368  		if devObj.GetMetadataName() != "" {
   369  			// devfile provided name: `.metadata.name`
   370  			componentName = devObj.GetMetadataName()
   371  		} else {
   372  			// default name
   373  			componentName, err = createDefaultComponentName(co.devfileMetadata.componentType, co.contextFlag, co.prefClient)
   374  			if err != nil {
   375  				return "", "", err
   376  			}
   377  		}
   378  	}
   379  
   380  	return
   381  }
