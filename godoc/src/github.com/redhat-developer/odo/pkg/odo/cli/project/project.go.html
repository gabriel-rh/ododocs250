    16  func NewCmdProject(name, fullName string) *cobra.Command {
    17  
    18  	projectCreateCmd := NewCmdProjectCreate(createRecommendedCommandName, odoutil.GetFullName(fullName, createRecommendedCommandName))
    19  	projectSetCmd := NewCmdProjectSet(setRecommendedCommandName, odoutil.GetFullName(fullName, setRecommendedCommandName))
    20  	projectListCmd := NewCmdProjectList(listRecommendedCommandName, odoutil.GetFullName(fullName, listRecommendedCommandName))
    21  	projectDeleteCmd := NewCmdProjectDelete(deleteRecommendedCommandName, odoutil.GetFullName(fullName, deleteRecommendedCommandName))
    22  	projectGetCmd := NewCmdProjectGet(getRecommendedCommandName, odoutil.GetFullName(fullName, getRecommendedCommandName))
    23  
    24  	projectCmd := &cobra.Command{
    25  		Use:   name + " [options]",
    26  		Short: "Perform project operations",
    27  		Long:  "Perform project operations",
    28  		Example: fmt.Sprintf("%s\n\n%s\n\n%s\n\n%s\n\n%s",
    29  			projectSetCmd.Example,
    30  			projectCreateCmd.Example,
    31  			projectListCmd.Example,
    32  			projectDeleteCmd.Example,
    33  			projectGetCmd.Example),
    34  		// 'odo project' is the same as 'odo project get'
    35  		// 'odo project <project_name>' is the same as 'odo project set <project_name>'
    36  		Run: func(cmd *cobra.Command, args []string) {
    37  			if len(args) > 0 && args[0] != getRecommendedCommandName && args[0] != setRecommendedCommandName {
    38  				projectSetCmd.Run(cmd, args)
    39  			} else {
    40  				projectGetCmd.Run(cmd, args)
    41  			}
    42  		},
    43  	}
    44  
    45  	projectCmd.AddCommand(projectGetCmd)
    46  	projectCmd.AddCommand(projectSetCmd)
    47  	projectCmd.AddCommand(projectCreateCmd)
    48  	projectCmd.AddCommand(projectDeleteCmd)
    49  	projectCmd.AddCommand(projectListCmd)
    50  
    51  	// Add a defined annotation in order to appear in the help menu
    52  	projectCmd.Annotations = map[string]string{"command": "main"}
    53  	projectCmd.SetUsageTemplate(odoutil.CmdUsageTemplate)
    54  
    55  	completion.RegisterCommandHandler(projectSetCmd, completion.ProjectNameCompletionHandler)
    56  	completion.RegisterCommandHandler(projectDeleteCmd, completion.ProjectNameCompletionHandler)
    57  
    58  	return projectCmd
    59  }
