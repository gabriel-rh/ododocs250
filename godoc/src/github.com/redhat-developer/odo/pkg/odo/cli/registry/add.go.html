    82  func (o *AddOptions) Run() (err error) {
    83  	isSecure := false

    87  
    88  	err = o.prefClient.RegistryHandler(o.operation, o.registryName, o.registryURL, false, isSecure)
    89  	if err != nil {

    92  
    93  	if o.tokenFlag != "" {
    94  		err = keyring.Set(util.CredentialPrefix+o.registryName, o.user, o.tokenFlag)
    95  		if err != nil {
    96  			return errors.Wrap(err, "unable to store registry credential to keyring")
    97  		}
    98  	}
    99  
   100  	log.Info("New registry successfully added")
