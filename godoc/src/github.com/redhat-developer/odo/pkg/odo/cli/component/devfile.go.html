    24  // DevfilePush has the logic to perform the required actions for a given devfile
    25  func (po *PushOptions) DevfilePush() error {
    26  
    27  	// Wrap the push so that we can capture the error in JSON-only mode
    28  	err := po.devfilePushInner()
    29  
    30  	if err != nil && log.IsJSON() {
    31  		eventLoggingClient := machineoutput.NewConsoleMachineEventLoggingClient()
    32  		eventLoggingClient.ReportError(err, machineoutput.TimestampNow())
    33  
    34  		// Suppress the error to prevent it from being output by the generic machine-readable handler (which will produce invalid JSON for our purposes)
    35  		err = nil
    36  
    37  		// os.Exit(1) since we are suppressing the generic machine-readable handler's exit code logic
    38  		os.Exit(1)
    39  	}
    40  
    41  	if err != nil {
    42  		return err
    43  	}
    44  
    45  	// push is successful, save the runMode used
    46  	runMode := envinfo.Run
    47  	if po.debugFlag {
    48  		runMode = envinfo.Debug
    49  	}
    50  
    51  	return po.EnvSpecificInfo.SetRunMode(runMode)
    52  }

    54  func (po *PushOptions) devfilePushInner() (err error) {
    55  	devObj, err := devfile.ParseAndValidateFromFile(po.DevfilePath)
    56  	if err != nil {
    57  		return err
    58  	}
    59  	componentName := po.EnvSpecificInfo.GetName()
    60  
    61  	// Set the source path to either the context or current working directory (if context not set)
    62  	po.sourcePath, err = util.GetAbsPath(po.componentContext)
    63  	if err != nil {
    64  		return errors.Wrap(err, "unable to get source path")
    65  	}
    66  
    67  	// Apply ignore information
    68  	err = genericclioptions.ApplyIgnore(&po.ignoreFlag, po.sourcePath)
    69  	if err != nil {
    70  		return errors.Wrap(err, "unable to apply ignore information")
    71  	}
    72  
    73  	var platformContext interface{}
    74  	kc := kubernetes.KubernetesContext{
    75  		Namespace: po.KClient.GetCurrentNamespace(),
    76  	}
    77  	platformContext = kc
    78  
    79  	devfileHandler, err := adapters.NewComponentAdapter(componentName, po.sourcePath, po.GetApplication(), devObj, platformContext)
    80  	if err != nil {
    81  		return err
    82  	}
    83  
    84  	pushParams := common.PushParameters{
    85  		Path:            po.sourcePath,
    86  		IgnoredFiles:    po.ignoreFlag,
    87  		ForceBuild:      po.forceBuildFlag,
    88  		Show:            po.showFlag,
    89  		EnvSpecificInfo: *po.EnvSpecificInfo,
    90  		DevfileBuildCmd: strings.ToLower(po.buildCommandFlag),
    91  		DevfileRunCmd:   strings.ToLower(po.runCommandflag),
    92  		DevfileDebugCmd: strings.ToLower(po.debugCommandFlag),
    93  		Debug:           po.debugFlag,
    94  		DebugPort:       po.EnvSpecificInfo.GetDebugPort(),
    95  	}
    96  
    97  	_, err = po.EnvSpecificInfo.ListURLs()
    98  	if err != nil {
    99  		return err
   100  	}
   101  
   102  	// Start or update the component
   103  	err = devfileHandler.Push(pushParams)
   104  	if err != nil {
   105  		err = errors.Errorf("Failed to start component with name %q. Error: %v",
   106  			componentName,
   107  			err,
   108  		)
   109  	} else {
   110  		log.Infof("\nPushing devfile component %q", componentName)
   111  		log.Success("Changes successfully pushed to component")
   112  	}
   113  
   114  	return
   115  }

   118  func (lo LogOptions) DevfileComponentLog() error {
   119  	devObj, err := devfile.ParseAndValidateFromFile(lo.GetDevfilePath())
   120  	if err != nil {
   121  		return err
   122  	}
   123  
   124  	componentName := lo.Context.EnvSpecificInfo.GetName()
   125  
   126  	var platformContext interface{}
   127  	kc := kubernetes.KubernetesContext{
   128  		Namespace: lo.KClient.GetCurrentNamespace(),
   129  	}
   130  	platformContext = kc
   131  
   132  	devfileHandler, err := adapters.NewComponentAdapter(componentName, lo.contextFlag, lo.GetApplication(), devObj, platformContext)
   133  
   134  	if err != nil {
   135  		return err
   136  	}
   137  
   138  	var command devfilev1.Command
   139  	if lo.debugFlag {
   140  		command, err = common.GetDebugCommand(devObj.Data, "")
   141  		if err != nil {
   142  			return err
   143  		}
   144  		if reflect.DeepEqual(devfilev1.Command{}, command) {
   145  			return errors.Errorf("no debug command found in devfile, please run \"odo log\" for run command logs")
   146  		}
   147  
   148  	} else {
   149  		command, err = common.GetRunCommand(devObj.Data, "")
   150  		if err != nil {
   151  			return err
   152  		}
   153  	}
   154  
   155  	// Start or update the component
   156  	rd, err := devfileHandler.Log(lo.followFlag, command)
   157  	if err != nil {
   158  		log.Errorf(
   159  			"Failed to log component with name %s.\nError: %v",
   160  			componentName,
   161  			err,
   162  		)
   163  		return err
   164  	}
   165  
   166  	return util.DisplayLog(lo.followFlag, rd, os.Stdout, componentName, -1)
   167  }
