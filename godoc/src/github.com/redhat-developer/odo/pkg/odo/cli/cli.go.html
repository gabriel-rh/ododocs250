   115  func NewCmdOdo(name, fullName string) *cobra.Command {
   116  	rootCmd := odoRootCmd(name, fullName)
   117  
   118  	if len(os.Args) > 1 {
   119  		cmdPathPieces := os.Args[1:]
   120  		// only look for suitable extension executables if
   121  		// the specified command does not already exist
   122  		cmd, _, err := rootCmd.Find(cmdPathPieces)
   123  		if err == nil && cmd != rootCmd {
   124  			return rootCmd
   125  		}
   126  		handleErr := plugins.HandleCommand(plugins.NewExecHandler(pluginPrefix), cmdPathPieces)
   127  		if handleErr != nil {
   128  			return rootCmd
   129  		}
   130  	}
   131  	return rootCmd
   132  }

   222  // reconfigureCmdWithSubcmd reconfigures each root command with a list of all subcommands and lists them
   223  // beside the help output
   224  // Adapted from: https://github.com/cppforlife/knctl/blob/612840d3c9729b1c57b20ca0450acab0d6eceeeb/pkg/knctl/cmd/knctl.go#L224
   225  func reconfigureCmdWithSubcmd(cmd *cobra.Command) {
   226  	if len(cmd.Commands()) == 0 {
   227  		return
   228  	}
   229  
   230  	if cmd.Args == nil {
   231  		cmd.Args = cobra.ArbitraryArgs
   232  	}
   233  	if cmd.RunE == nil {
   234  		cmd.RunE = ShowSubcommands
   235  	}
   236  
   237  	var strs []string
   238  	for _, subcmd := range cmd.Commands() {
   239  		if !subcmd.Hidden {
   240  			strs = append(strs, strings.Split(subcmd.Use, " ")[0])
   241  		}
   242  	}
   243  
   244  	cmd.Short += " (" + strings.Join(strs, ", ") + ")"
   245  }

   247  // ShowSubcommands shows all available subcommands.
   248  // Adapted from: https://github.com/cppforlife/knctl/blob/612840d3c9729b1c57b20ca0450acab0d6eceeeb/pkg/knctl/cmd/knctl.go#L224
   249  func ShowSubcommands(cmd *cobra.Command, args []string) error {
   250  	var strs []string
   251  	for _, subcmd := range cmd.Commands() {
   252  		if !subcmd.Hidden {
   253  			strs = append(strs, subcmd.Name())
   254  		}
   255  	}
   256  	return fmt.Errorf("Subcommand not found, use one of the available commands: %s", strings.Join(strs, ", "))
   257  }

   259  // ShowHelp will show the help correctly (and whether or not the command is invalid...)
   260  // Taken from: https://github.com/cppforlife/knctl/blob/612840d3c9729b1c57b20ca0450acab0d6eceeeb/pkg/knctl/cmd/knctl.go#L71
   261  func ShowHelp(cmd *cobra.Command, args []string) error {
   262  
   263  	if len(args) == 0 {
   264  
   265  		// We will show a custom help when typing JUST `odo`, directing the user to use `odo --help` for a full help.
   266  		// Thus we will set cmd.SilenceUsage and cmd.SilenceErrors both to true so we do not output the usage or error out.
   267  		cmd.SilenceUsage = true
   268  		cmd.SilenceErrors = true
   269  
   270  		// Print out the default "help" usage
   271  		fmt.Println(rootDefaultHelp)
   272  		return nil
   273  	}
   274  
   275  	return fmt.Errorf("Invalid command - see available commands/subcommands above")
   276  }
