    60  // Run contains the logic for "odo registry list" command
    61  func (o *ListOptions) Run() (err error) {
    62  	registryList := o.prefClient.RegistryList()
    63  	if registryList == nil || len(*registryList) == 0 {
    64  		return fmt.Errorf("No devfile registries added to the configuration. Refer `odo registry add -h` to add one")
    65  	}
    66  
    67  	if log.IsJSON() {
    68  		machineoutput.OutputSuccess(machineoutput.NewRegistryListOutput(registryList))
    69  		return nil
    70  	}
    71  
    72  	w := tabwriter.NewWriter(os.Stdout, 5, 2, 3, ' ', tabwriter.TabIndent)
    73  	fmt.Fprintln(w, "NAME", "\t", "URL", "\t", "SECURE")
    74  	o.printRegistryList(w, registryList)
    75  	w.Flush()
    76  	if o.printGitRegistryDeprecationWarning {
    77  		util.PrintGitRegistryDeprecationWarning()
    78  	}
    79  	return nil
    80  }

    82  func (o *ListOptions) printRegistryList(w io.Writer, registryList *[]preference.Registry) {
    83  	if registryList == nil {
    84  		return
    85  	}
    86  
    87  	regList := *registryList
    88  	// Loop backwards here to ensure the registry display order is correct (display latest newly added registry firstly)
    89  	for i := len(regList) - 1; i >= 0; i-- {
    90  		registry := regList[i]
    91  		secure := "No"
    92  		if registry.Secure {
    93  			secure = "Yes"
    94  		}
    95  		fmt.Fprintln(w, registry.Name, "\t", registry.URL, "\t", secure)
    96  		if util.IsGitBasedRegistry(registry.URL) {
    97  			o.printGitRegistryDeprecationWarning = true
    98  		}
    99  	}
   100  }
