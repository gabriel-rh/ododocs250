    78  // Run contains the logic for "odo registry update" command
    79  func (o *UpdateOptions) Run() (err error) {
    80  	secureBeforeUpdate := false
    81  	secureAfterUpdate := false
    82  
    83  	secure := registryUtil.IsSecure(o.prefClient, o.registryName)
    84  	if secure {
    85  		secureBeforeUpdate = true
    86  	}
    87  
    88  	if o.tokenFlag != "" {
    89  		secureAfterUpdate = true
    90  	}
    91  
    92  	err = o.prefClient.RegistryHandler(o.operation, o.registryName, o.registryURL, o.forceFlag, secureAfterUpdate)
    93  	if err != nil {
    94  		return err
    95  	}
    96  
    97  	if secureAfterUpdate {
    98  		err = keyring.Set(util.CredentialPrefix+o.registryName, o.user, o.tokenFlag)
    99  		if err != nil {
   100  			return errors.Wrap(err, "unable to store registry credential to keyring")
   101  		}
   102  	} else if secureBeforeUpdate && !secureAfterUpdate {
   103  		err = keyring.Delete(util.CredentialPrefix+o.registryName, o.user)
   104  		if err != nil {
   105  			return errors.Wrap(err, "unable to delete registry credential from keyring")
   106  		}
   107  	}
   108  
   109  	return nil
   110  }
