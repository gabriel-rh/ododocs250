    71  func showBlankIfNil(intf interface{}) interface{} {
    72  	imm := reflect.ValueOf(intf)
    73  
    74  	// if the value is nil then we should return a blank string
    75  	if imm.IsNil() {
    76  		return ""
    77  	}
    78  
    79  	// if its a pointer then we should de-ref it because we cant de-ref an interface{}
    80  	if imm.Kind() == reflect.Ptr {
    81  		return imm.Elem().Interface()
    82  	}
    83  
    84  	return intf
    85  }
