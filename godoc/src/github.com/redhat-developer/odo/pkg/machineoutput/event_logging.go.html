    53  func (c *NoOpMachineEventLoggingClient) CreateContainerOutputWriter() (*io.PipeWriter, chan interface{}, *io.PipeWriter, chan interface{}) {
    54  
    55  	channels := []chan interface{}{make(chan interface{}), make(chan interface{})}
    56  
    57  	// Ensure there is always a result waiting on each of the channels
    58  	for _, channelPtr := range channels {
    59  		channelVal := channelPtr
    60  
    61  		go func(channel chan interface{}) {
    62  			for {
    63  				channel <- nil
    64  			}
    65  		}(channelVal)
    66  	}
    67  
    68  	return nil, channels[0], nil, channels[1]
    69  }

   341  // createWriterAndChannel is similar to the exec.CreateConsoleOutputWriterAndChannel(); see that function's comment for details.
   342  func createWriterAndChannel(stderr bool) (*io.PipeWriter, chan interface{}) {
   343  	reader, writer := io.Pipe()
   344  
   345  	closeChannel := make(chan interface{})
   346  
   347  	stream := "stdout"
   348  	if stderr {
   349  		stream = "stderr"
   350  	}
   351  
   352  	go func() {
   353  
   354  		bufReader := bufio.NewReader(reader)
   355  		for {
   356  			line, _, err := bufReader.ReadLine()
   357  			if err != nil {
   358  				if err != io.EOF {
   359  					klog.V(4).Infof("Unexpected error on reading container output reader: %v", err)
   360  				}
   361  				break
   362  			}
   363  
   364  			// Output log text event for each line we receive
   365  			json := MachineEventWrapper{
   366  				LogText: &LogText{
   367  					AbstractLogEvent: AbstractLogEvent{Timestamp: TimestampNow()},
   368  					Text:             string(line),
   369  					Stream:           stream,
   370  				},
   371  			}
   372  			OutputSuccessUnindented(json)
   373  		}
   374  
   375  		// Output a single nil event on the channel to inform that the last line of text has been
   376  		// received from the writer.
   377  		closeChannel <- nil
   378  	}()
   379  
   380  	return writer, closeChannel
   381  }
