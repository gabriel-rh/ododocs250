    68  // getEnvInfoFile first checks for the ENVINFO variable
    69  // then we check for directory and eventually the file (which we return as a string)
    70  func getEnvInfoFile(envDir string) (string, string, error) {
    71  	if env, ok := os.LookupEnv(envInfoEnvName); ok {
    72  		return env, filepath.Join(env, "..", "..", "..", "devfile.yaml"), nil
    73  	}
    74  
    75  	if envDir == "" {
    76  		var err error
    77  		envDir, err = os.Getwd()
    78  		if err != nil {
    79  			return "", "", err
    80  		}
    81  	}
    82  
    83  	return filepath.Join(envDir, ".odo", "env", envInfoFileName), location.DevfileLocation(envDir), nil
    84  }

    96  // newEnvSpecificInfo retrieves the env.yaml file, if it does not exist, we return a *BLANK* environment file.
    97  func newEnvSpecificInfo(envDir string, fs filesystem.Filesystem) (*EnvSpecificInfo, error) {
    98  	// Get the path of the environment file
    99  	envInfoFile, devfilePath, err := getEnvInfoFile(envDir)
   100  	if err != nil {
   101  		return nil, errors.Wrap(err, "failed to get the path of the environment file")
   102  	}
   103  
   104  	// Organize that information into a struct
   105  	e := EnvSpecificInfo{
   106  		EnvInfo:           NewEnvInfo(),
   107  		devfilePath:       devfilePath,
   108  		Filename:          envInfoFile,
   109  		envinfoFileExists: true,
   110  		fs:                fs,
   111  	}
   112  
   113  	// If the env.yaml file does not exist then we simply return and set e.envinfoFileExists as false
   114  	if _, err = e.fs.Stat(envInfoFile); os.IsNotExist(err) {
   115  		e.envinfoFileExists = false
   116  		return &e, nil
   117  	}
   118  
   119  	// Retrieve the environment file
   120  	err = getFromFile(&e.EnvInfo, e.Filename)
   121  	if err != nil {
   122  		return nil, err
   123  	}
   124  
   125  	return &e, nil
   126  }

   149  // SetConfiguration sets the environment specific info such as the Cluster Host, Name, etc.
   150  // we then **write** this data to the environment yaml file (see envInfoFileName const)
   151  func (esi *EnvSpecificInfo) SetConfiguration(parameter string, value interface{}) (err error) {
   152  	if p, ok := asLocallySupportedParameter(parameter); ok {
   153  		switch p {
   154  		case "name":
   155  			val := value.(string)
   156  			esi.componentSettings.Name = val
   157  		case "project":
   158  			val := value.(string)
   159  			esi.componentSettings.Project = val
   160  		case "debugport":
   161  			val, err := strconv.Atoi(value.(string))
   162  			if err != nil {
   163  				return errors.Wrap(err, "failed to set debug port")
   164  			}
   165  			esi.componentSettings.DebugPort = &val
   166  		case "url":
   167  			urlValue := value.(localConfigProvider.LocalURL)
   168  			if esi.componentSettings.URL != nil {
   169  				*esi.componentSettings.URL = append(*esi.componentSettings.URL, urlValue)
   170  			} else {
   171  				esi.componentSettings.URL = &[]localConfigProvider.LocalURL{urlValue}
   172  			}
   173  		}
   174  
   175  		return esi.writeToFile()
   176  	}
   177  	return errors.Errorf("unknown parameter: %q is not a parameter in the odo environment file, please refer `odo env set --help` to see valid parameters", parameter)
   178  
   179  }

   181  // DeleteEnvDirIfEmpty Deletes the env directory if its empty
   182  func (esi *EnvSpecificInfo) DeleteEnvDirIfEmpty() error {
   183  	envDir := filepath.Dir(esi.Filename)
   184  	_, err := esi.fs.Stat(envDir)
   185  	if os.IsNotExist(err) {
   186  		// If the Env dir doesn't exist then we dont mind
   187  		return nil
   188  	} else if err != nil {
   189  		// Possible to not have permission to the dir
   190  		return err
   191  	}
   192  	f, err := esi.fs.Open(envDir)
   193  	if err != nil {
   194  		return err
   195  	}
   196  	defer f.Close()
   197  	_, err = f.Readdir(1)
   198  
   199  	// If directory is empty we can remove it
   200  	if err == io.EOF {
   201  		klog.V(4).Info("Deleting the env directory as well because its empty")
   202  
   203  		return esi.fs.Remove(envDir)
   204  	}
   205  	return err
   206  }

   232  // DeleteConfiguration is used to delete environment specific info from local odo envinfo
   233  func (esi *EnvSpecificInfo) DeleteConfiguration(parameter string) error {
   234  	for _, manParam := range manParams {
   235  		if parameter == manParam {
   236  			return errors.Errorf("failed to unset %q: %q is mandatory parameter", parameter, parameter)
   237  		}
   238  	}
   239  
   240  	if p, ok := asLocallySupportedParameter(parameter); ok {
   241  		if err := util.DeleteConfiguration(&esi.componentSettings, p); err != nil {
   242  			return err
   243  		}
   244  		return esi.writeToFile()
   245  	}
   246  	return errors.Errorf("unknown parameter: %q is not a parameter in the odo environment file, please refer `odo env unset --help` to unset a valid parameter", parameter)
   247  
   248  }

   281  // GetContainers returns the Container components from the devfile
   282  // returns empty list if nil
   283  func (ei *EnvInfo) GetContainers() ([]localConfigProvider.LocalContainer, error) {
   284  	var localContainers []localConfigProvider.LocalContainer
   285  
   286  	devfileComponents, err := ei.devfileObj.Data.GetComponents(common.DevfileOptions{})
   287  	if err != nil {
   288  		return localContainers, err
   289  	}
   290  	for _, component := range devfileComponents {
   291  		if component.Container == nil {
   292  			continue
   293  		}
   294  		localContainers = append(localContainers, localConfigProvider.LocalContainer{
   295  			Name: component.Name,
   296  		})
   297  	}
   298  	return localContainers, nil
   299  }
