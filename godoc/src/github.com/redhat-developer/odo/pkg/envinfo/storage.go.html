    31  func (ei *EnvInfo) ValidateStorage(storage localConfigProvider.LocalStorage) error {

    33  	if storage.Ephemeral != nil && *storage.Ephemeral {

    42  	}
    43  
    44  	storageList, err := ei.ListStorage()
    45  	if err != nil {
    46  		return err
    47  	}
    48  
    49  	for _, store := range storageList {
    50  		if store.Name == storage.Name {
    51  			return fmt.Errorf("storage with name %s already exists", storage.Name)
    52  		}
    53  	}
    54  
    55  	if storage.Container == "" {
    56  		return nil
    57  	}
    58  
    59  	//check if specified container exists or not (if specified)
    60  	containerExists := false
    61  	containers, err := ei.GetContainers()
    62  	if err != nil {
    63  		return err
    64  	}
    65  
    66  	for _, c := range containers {
    67  		if c.Name == storage.Container {
    68  			containerExists = true
    69  		}
    70  	}

    92  // CreateStorage sets the storage related information in the local configuration
    93  func (ei *EnvInfo) CreateStorage(storage localConfigProvider.LocalStorage) error {
    94  	//initialize volume mount and volume container
    95  	vm := []devfilev1.VolumeMount{
    96  		{
    97  			Name: storage.Name,
    98  			Path: storage.Path,
    99  		},
   100  	}
   101  	vc := []devfilev1.Component{{
   102  		Name: storage.Name,
   103  		ComponentUnion: devfilev1.ComponentUnion{
   104  			Volume: &devfilev1.VolumeComponent{
   105  				Volume: devfilev1.Volume{
   106  					Size:      storage.Size,
   107  					Ephemeral: storage.Ephemeral,
   108  				},
   109  			},
   110  		},
   111  	}}
   112  	volumeExists := false
   113  	// Get all the containers in the devfile
   114  	containers, err := ei.GetContainers()
   115  	if err != nil {
   116  		return err
   117  	}
   118  
   119  	// Add volumeMount to all containers if no container is specified else to specified container(s) in the devfile
   120  	for _, c := range containers {
   121  		if storage.Container == "" || (storage.Container != "" && c.Name == storage.Container) {
   122  			if e := ei.devfileObj.Data.AddVolumeMounts(c.Name, vm); e != nil {
   123  				return e
   124  			}
   125  		}
   126  	}
   127  
   128  	// Get all the components to check if volume component exists
   129  	components, err := ei.devfileObj.Data.GetComponents(common.DevfileOptions{})
   130  	if err != nil {
   131  		return err
   132  	}
   133  
   134  	// check if volume component already exists
   135  	for _, component := range components {
   136  		if component.Volume != nil && component.Name == storage.Name {
   137  			volumeExists = true
   138  		}
   139  	}
   140  
   141  	// Add volume component to devfile. Think along the lines of a k8s pod spec's volumeMount and volume.
   142  	// Add if volume does not exist, otherwise update
   143  	if !volumeExists {
   144  		err = ei.devfileObj.Data.AddComponents(vc)
   145  		if err != nil {
   146  			return err
   147  		}
   148  	} else {
   149  		return fmt.Errorf("volume with name %s already exists", storage.Name)
   150  	}
   151  
   152  	err = ei.devfileObj.WriteYamlDevfile()
   153  	if err != nil {
   154  		return err
   155  	}
   156  
   157  	return nil
   158  }

   160  // ListStorage gets all the storage from the devfile.yaml
   161  func (ei *EnvInfo) ListStorage() ([]localConfigProvider.LocalStorage, error) {
   162  	var storageList []localConfigProvider.LocalStorage
   163  
   164  	volumeMap := make(map[string]devfilev1.Volume)
   165  	components, err := ei.devfileObj.Data.GetComponents(common.DevfileOptions{})
   166  	if err != nil {
   167  		return storageList, err
   168  	}
   169  
   170  	for _, component := range components {
   171  		if component.Volume == nil {
   172  			continue
   173  		}
   174  		if component.Volume.Size == "" {
   175  			component.Volume.Size = DefaultVolumeSize
   176  		}
   177  		volumeMap[component.Name] = component.Volume.Volume
   178  	}
   179  
   180  	for _, component := range components {
   181  		if component.Container == nil {
   182  			continue
   183  		}
   184  		for _, volumeMount := range component.Container.VolumeMounts {
   185  			vol, ok := volumeMap[volumeMount.Name]
   186  			if ok {
   187  				storageList = append(storageList, localConfigProvider.LocalStorage{
   188  					Name:      volumeMount.Name,
   189  					Size:      vol.Size,
   190  					Ephemeral: vol.Ephemeral,
   191  					Path:      GetVolumeMountPath(volumeMount),
   192  					Container: component.Name,
   193  				})
   194  			}
   195  		}
   196  	}
   197  
   198  	return storageList, nil
   199  }

   220  // GetStorageMountPath gets the mount path of the storage with the given storage name
   221  func (ei *EnvInfo) GetStorageMountPath(storageName string) (string, error) {
   222  	containers, err := ei.GetContainers()
   223  	if err != nil {
   224  		return "", err
   225  	}
   226  	if len(containers) == 0 {
   227  		return "", fmt.Errorf("invalid devfile: components.container: required value")
   228  	}
   229  
   230  	// go over all containers
   231  	for _, c := range containers {
   232  		// get all volume mount paths in current container
   233  		pt, err := ei.devfileObj.Data.GetVolumeMountPaths(storageName, c.Name)
   234  		if err != nil {
   235  			klog.V(2).Infof("Failed to get volume mount paths for storage %s in container %s: %s", storageName, c.Name, err.Error())
   236  		}
   237  		if len(pt) > 0 {
   238  			return pt[0], nil
   239  		}
   240  	}
   241  	// TODO: Below "if" storage needs to be mounted on multiple containers, then the return will have to be an array.
   242  	return "", nil
   243  }
