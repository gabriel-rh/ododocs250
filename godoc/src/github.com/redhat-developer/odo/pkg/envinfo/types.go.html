    62  func showBlankIfNil(intf interface{}) interface{} {
    63  	value := reflect.ValueOf(intf)
    64  
    65  	// if the value is nil then we should return a blank string

    67  		return ""
    68  	}

    70  	// if it's a pointer then we should de-ref it because we cant de-ref an interface{}

    73  	}
