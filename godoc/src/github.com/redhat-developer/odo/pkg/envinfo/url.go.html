    19  func (ei *EnvInfo) getPorts(container string) ([]string, error) {
    20  	var portList []string
    21  	containerComponents, err := ei.devfileObj.Data.GetDevfileContainerComponents(common.DevfileOptions{})
    22  	if err != nil {
    23  		return nil, err
    24  	}
    25  	containerExists := false
    26  	portMap := make(map[string]bool)
    27  	for _, component := range containerComponents {
    28  		if container == "" || container == component.Name {
    29  			containerExists = true
    30  			for _, endpoint := range component.Container.Endpoints {
    31  				portMap[strconv.FormatInt(int64(endpoint.TargetPort), 10)] = true
    32  			}
    33  		}
    34  	}
    35  	if !containerExists {
    36  		return portList, fmt.Errorf("the container specified: %s does not exist in devfile", container)
    37  	}
    38  	for port := range portMap {
    39  		portList = append(portList, port)
    40  	}
    41  	sort.Strings(portList)
    42  	return portList, nil
    43  }

    70  		}
    71  	}

    75  // CompleteURL completes the given URL with default values
    76  func (ei *EnvInfo) CompleteURL(url *localConfigProvider.LocalURL) error {
    77  	if url.Kind == "" {
    78  		if !ei.isRouteSupported {
    79  			url.Kind = localConfigProvider.INGRESS
    80  		} else {
    81  			url.Kind = localConfigProvider.ROUTE
    82  		}
    83  	}
    84  
    85  	if len(url.Path) > 0 && (strings.HasPrefix(url.Path, "/") || strings.HasPrefix(url.Path, "\\")) {
    86  		if len(url.Path) <= 1 {
    87  			url.Path = ""
    88  		} else {
    89  			// remove the leading / or \ from provided path
    90  			url.Path = string([]rune(url.Path)[1:])
    91  		}
    92  	}
    93  	// add leading / to path, if the path provided is empty, it will be set to / which is the default value of path
    94  	url.Path = "/" + url.Path
    95  
    96  	// get the port if not provided
    97  	if url.Container == "" {
    98  		ports, err := ei.GetComponentPorts()
    99  		if err != nil {
   100  			return err
   101  		}
   102  		err = ei.checkValidPort(url, fmt.Sprintf("component %s", ei.GetName()), ports)
   103  		if err != nil {
   104  			return err
   105  		}
   106  	} else {
   107  		ports, err := ei.GetContainerPorts(url.Container)
   108  		if err != nil {
   109  			return err
   110  		}
   111  		err = ei.checkValidPort(url, fmt.Sprintf("container %s", url.Container), ports)
   112  		if err != nil {
   113  			return err
   114  		}
   115  	}
   116  
   117  	// get the name for the URL if not provided
   118  	if len(url.Name) == 0 {
   119  		foundURL, err := findInvalidEndpoint(ei, url.Port)
   120  		if err != nil {
   121  			return err
   122  		}
   123  
   124  		if foundURL.Name != "" {
   125  			// found an URL that can be overridden or more info can be added to it
   126  			url.Name = foundURL.Name
   127  			ei.updateURL = true
   128  		} else {
   129  			url.Name = util.GetURLName(ei.GetName(), url.Port)
   130  		}
   131  	}
   132  
   133  	containerComponents, err := ei.devfileObj.Data.GetDevfileContainerComponents(common.DevfileOptions{})
   134  	if err != nil {
   135  		return err
   136  	}
   137  
   138  	if len(containerComponents) == 0 {
   139  		return fmt.Errorf("no valid components found in the devfile")
   140  	}
   141  
   142  	// if a container name is provided for the URL, return
   143  	if len(url.Container) > 0 {
   144  		return nil
   145  	}
   146  
   147  	containerPortMap := make(map[int]string)
   148  	portMap := make(map[string]bool)
   149  
   150  	// if a container name for the URL is not provided
   151  	// use a container which uses the given URL port in one of it's endpoints
   152  	for _, component := range containerComponents {
   153  		for _, endpoint := range component.Container.Endpoints {
   154  			containerPortMap[endpoint.TargetPort] = component.Name
   155  			portMap[strconv.FormatInt(int64(endpoint.TargetPort), 10)] = true
   156  		}
   157  	}
   158  	if containerName, exist := containerPortMap[url.Port]; exist {
   159  		url.Container = containerName
   160  	}
   161  
   162  	// container is not provided, or the specified port is not being used under any containers
   163  	// pick the first container to store the new endpoint
   164  	if len(url.Container) == 0 {
   165  		url.Container = containerComponents[0].Name
   166  	}
   167  
   168  	return nil
   169  }

   171  // ValidateURL validates the given URL
   172  func (ei *EnvInfo) ValidateURL(url localConfigProvider.LocalURL) error {
   173  
   174  	foundContainer := false
   175  	containerComponents, err := ei.devfileObj.Data.GetDevfileContainerComponents(common.DevfileOptions{})
   176  	if err != nil {
   177  		return err
   178  	}
   179  
   180  	if len(containerComponents) == 0 {
   181  		return fmt.Errorf("no valid components found in the devfile")
   182  	}
   183  
   184  	// map TargetPort with containerName
   185  	containerPortMap := make(map[int]string)
   186  	for _, component := range containerComponents {
   187  		if len(url.Container) > 0 && !foundContainer {
   188  			if component.Name == url.Container {
   189  				foundContainer = true
   190  			}
   191  		}
   192  		for _, endpoint := range component.Container.Endpoints {
   193  			if endpoint.Name == url.Name && !ei.updateURL {
   194  				return fmt.Errorf("url %v already exist in devfile endpoint entry under container %v", url.Name, component.Name)
   195  			}
   196  			containerPortMap[endpoint.TargetPort] = component.Name
   197  		}
   198  	}
   199  
   200  	if len(url.Container) > 0 && !foundContainer {
   201  		return fmt.Errorf("the container specified: %v does not exist in devfile", url.Container)
   202  	}
   203  	if containerName, exist := containerPortMap[url.Port]; exist {
   204  		if len(url.Container) > 0 && url.Container != containerName {
   205  			return fmt.Errorf("cannot set URL %v under container %v, TargetPort %v is being used under container %v", url.Name, url.Container, url.Port, containerName)
   206  		}
   207  	}
   208  
   209  	errorList := make([]string, 0)
   210  	if url.TLSSecret != "" && (url.Kind != localConfigProvider.INGRESS || !url.Secure) {
   211  		errorList = append(errorList, "TLS secret is only available for secure URLs of Ingress kind")
   212  	}
   213  
   214  	// check if a host is provided for route based URLs
   215  	if len(url.Host) > 0 {
   216  		if url.Kind == localConfigProvider.ROUTE {
   217  			errorList = append(errorList, "host is not supported for URLs of Route Kind")
   218  		}
   219  		if err := validation.ValidateHost(url.Host); err != nil {
   220  			errorList = append(errorList, err.Error())
   221  		}
   222  	} else if url.Kind == localConfigProvider.INGRESS {
   223  		errorList = append(errorList, "host must be provided in order to create URLS of Ingress Kind")
   224  	}
   225  
   226  	// check the protocol of the URL
   227  	if len(url.Protocol) > 0 {
   228  		switch strings.ToLower(url.Protocol) {
   229  		case string(devfilev1.HTTPEndpointProtocol):
   230  			break
   231  		case string(devfilev1.HTTPSEndpointProtocol):
   232  			break
   233  		case string(devfilev1.WSEndpointProtocol):
   234  			break
   235  		case string(devfilev1.WSSEndpointProtocol):
   236  			break
   237  		case string(devfilev1.TCPEndpointProtocol):
   238  			break
   239  		case string(devfilev1.UDPEndpointProtocol):
   240  			break
   241  		default:
   242  			errorList = append(errorList, fmt.Sprintf("endpoint protocol only supports %v|%v|%v|%v|%v|%v", devfilev1.HTTPEndpointProtocol, devfilev1.HTTPSEndpointProtocol, devfilev1.WSSEndpointProtocol, devfilev1.WSEndpointProtocol, devfilev1.TCPEndpointProtocol, devfilev1.UDPEndpointProtocol))
   243  		}
   244  	}
   245  
   246  	if !ei.updateURL {
   247  		urls, err := ei.ListURLs()
   248  		if err != nil {
   249  			return err
   250  		}
   251  		for _, localURL := range urls {
   252  			if url.Name == localURL.Name {
   253  				errorList = append(errorList, fmt.Sprintf("URL %s already exists", url.Name))
   254  			}
   255  		}
   256  	}
   257  
   258  	if len(errorList) > 0 {
   259  		return fmt.Errorf(strings.Join(errorList, "\n"))
   260  	}
   261  	return nil
   262  }

   278  // CreateURL write the given url to the env.yaml and devfile
   279  func (esi *EnvSpecificInfo) CreateURL(url localConfigProvider.LocalURL) error {
   280  
   281  	if !esi.updateURL {
   282  		newEndpointEntry := devfilev1.Endpoint{
   283  			Name:       url.Name,
   284  			Path:       url.Path,
   285  			Secure:     util.GetBoolPtr(url.Secure),
   286  			Exposure:   devfilev1.PublicEndpointExposure,
   287  			TargetPort: url.Port,
   288  			Protocol:   devfilev1.EndpointProtocol(strings.ToLower(url.Protocol)),
   289  		}
   290  
   291  		err := addEndpointInDevfile(esi.devfileObj, newEndpointEntry, url.Container)
   292  		if err != nil {
   293  			return errors.Wrapf(err, "failed to write endpoints information into devfile")
   294  		}
   295  	} else {
   296  		err := updateEndpointInDevfile(esi.devfileObj, url)
   297  		if err != nil {
   298  			return err
   299  		}
   300  	}
   301  
   302  	err := esi.SetConfiguration("url", localConfigProvider.LocalURL{Name: url.Name, Host: url.Host, TLSSecret: url.TLSSecret, Kind: url.Kind})
   303  	if err != nil {
   304  		return errors.Wrapf(err, "failed to persist the component settings to env file")
   305  	}
   306  	return nil
   307  }

   309  // ListURLs returns the urls from the env and devfile, returns default if nil
   310  func (ei *EnvInfo) ListURLs() ([]localConfigProvider.LocalURL, error) {
   311  
   312  	envMap := make(map[string]localConfigProvider.LocalURL)
   313  	if ei.componentSettings.URL != nil {
   314  		for _, url := range *ei.componentSettings.URL {
   315  			envMap[url.Name] = url
   316  		}
   317  	}
   318  
   319  	var urls []localConfigProvider.LocalURL
   320  
   321  	if ei.devfileObj.Data == nil {
   322  		return urls, nil
   323  	}
   324  
   325  	devfileComponents, err := ei.devfileObj.Data.GetDevfileContainerComponents(common.DevfileOptions{})
   326  	if err != nil {
   327  		return urls, err
   328  	}
   329  	for _, comp := range devfileComponents {
   330  		for _, localEndpoint := range comp.Container.Endpoints {
   331  			// only exposed endpoint will be shown as a URL in `odo url list`
   332  			if localEndpoint.Exposure == devfilev1.NoneEndpointExposure || localEndpoint.Exposure == devfilev1.InternalEndpointExposure {
   333  				continue
   334  			}
   335  
   336  			path := "/"
   337  			if localEndpoint.Path != "" {
   338  				path = localEndpoint.Path
   339  			}
   340  
   341  			secure := false
   342  			if util.SafeGetBool(localEndpoint.Secure) || localEndpoint.Protocol == "https" || localEndpoint.Protocol == "wss" {
   343  				secure = true
   344  			}
   345  
   346  			url := localConfigProvider.LocalURL{
   347  				Name:      localEndpoint.Name,
   348  				Port:      localEndpoint.TargetPort,
   349  				Secure:    secure,
   350  				Path:      path,
   351  				Container: comp.Name,
   352  			}
   353  
   354  			if envInfoURL, exist := envMap[localEndpoint.Name]; exist {
   355  				url.Host = envInfoURL.Host
   356  				url.TLSSecret = envInfoURL.TLSSecret
   357  				url.Kind = envInfoURL.Kind
   358  			} else {
   359  				url.Kind = localConfigProvider.ROUTE
   360  			}
   361  
   362  			urls = append(urls, url)
   363  		}
   364  	}
   365  
   366  	return urls, nil
   367  }

   409  func removeEndpointInDevfile(devObj parser.DevfileObj, urlName string) error {

   428  		if found {
   429  			break
   430  		}

   438  // updateEndpointInDevfile updates the endpoint of the given URL in the devfile
   439  func updateEndpointInDevfile(devObj parser.DevfileObj, url localConfigProvider.LocalURL) error {
   440  	components, err := devObj.Data.GetComponents(common.DevfileOptions{})
   441  	if err != nil {
   442  		return err
   443  	}
   444  	for _, component := range components {
   445  		if component.Container != nil && component.Name == url.Container {
   446  			for j := range component.ComponentUnion.Container.Endpoints {
   447  				endpoint := component.ComponentUnion.Container.Endpoints[j]
   448  
   449  				if endpoint.Name == url.Name {
   450  					// fill the default values
   451  					if endpoint.Exposure == "" {
   452  						endpoint.Exposure = devfilev1.PublicEndpointExposure
   453  					}
   454  					if endpoint.Path == "" {
   455  						endpoint.Path = "/"
   456  					}
   457  					if endpoint.Protocol == "" {
   458  						endpoint.Protocol = devfilev1.HTTPEndpointProtocol
   459  					}
   460  
   461  					// prevent write unless required
   462  					if endpoint.Exposure != devfilev1.PublicEndpointExposure || url.Secure != util.SafeGetBool(endpoint.Secure) ||
   463  						url.Path != endpoint.Path || url.Protocol != string(endpoint.Protocol) {
   464  						endpoint = devfilev1.Endpoint{
   465  							Name:       url.Name,
   466  							Path:       url.Path,
   467  							Secure:     &url.Secure,
   468  							Exposure:   devfilev1.PublicEndpointExposure,
   469  							TargetPort: url.Port,
   470  							Protocol:   devfilev1.EndpointProtocol(strings.ToLower(url.Protocol)),
   471  						}
   472  						component.ComponentUnion.Container.Endpoints[j] = endpoint
   473  						err := devObj.Data.UpdateComponent(component)
   474  						if err != nil {
   475  							return err
   476  						}
   477  						return devObj.WriteYamlDevfile()
   478  					}
   479  					return nil
   480  				}
   481  			}
   482  		}
   483  	}
   484  	return fmt.Errorf("url %s not found for updating", url.Name)
   485  }

   489  // urls with no host information on a vanilla k8s based cluster
   490  func findInvalidEndpoint(ei *EnvInfo, port int) (localConfigProvider.LocalURL, error) {
   491  	urls, err := ei.ListURLs()
   492  	if err != nil {
   493  		return localConfigProvider.LocalURL{}, err
   494  	}
   495  	for _, url := range urls {
   496  		if url.Kind == localConfigProvider.ROUTE && url.Port == port && !ei.isRouteSupported {
   497  			return url, nil
   498  		}
   499  	}
   500  	return localConfigProvider.LocalURL{}, nil
   501  }
