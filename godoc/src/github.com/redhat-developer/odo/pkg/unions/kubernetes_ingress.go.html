    60  //GetProtocol returns `https` if tls is configured on either networking v1 or extensions v1 ingress, else `http`
    61  func (ki *KubernetesIngress) GetProtocol() string {
    62  	if (ki.NetworkingV1Ingress != nil && len(ki.NetworkingV1Ingress.Spec.TLS) > 0) || (ki.ExtensionV1Beta1Ingress != nil && len(ki.ExtensionV1Beta1Ingress.Spec.TLS) > 0) {
