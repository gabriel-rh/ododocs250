    49  func AsDevfileSupportedParameter(param string) (string, bool) {

    54  // SetDevfileConfiguration allows setting all the parameters that are configurable in a devfile
    55  func SetDevfileConfiguration(d parser.DevfileObj, parameter string, value interface{}) error {
    56  
    57  	// we are ignoring this error becase a developer is usually aware of the type of value that is
    58  	// being passed. So consider this a shortcut, if you know its a string value use this strValue
    59  	// else parse it inside the switch case.
    60  	strValue, _ := value.(string)
    61  	if param, ok := AsDevfileSupportedParameter(parameter); ok {
    62  		switch param {
    63  		case "name":
    64  			return d.SetMetadataName(strValue)
    65  		case "ports":
    66  			arrValue := strings.Split(strValue, ",")
    67  			return d.SetPorts(arrValue...)
    68  		case "memory":
    69  			return d.SetMemory(strValue)
    70  		}
    71  
    72  	}
    73  	return errors.Errorf("unknown parameter :'%s' is not a configurable parameter in the devfile", parameter)
    74  
    75  }
