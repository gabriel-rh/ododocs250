    26  // List all applications names in current project by looking at `app` labels in deployments
    27  func (o kubernetesClient) List() ([]string, error) {
    28  	if o.client == nil {
    29  		return nil, nil
    30  	}
    31  
    32  	// Get all Deployments with the "app" label
    33  	deploymentAppNames, err := o.client.GetDeploymentLabelValues(applabels.ApplicationLabel, applabels.ApplicationLabel)
    34  	if err != nil {
    35  		return nil, errors.Wrap(err, "unable to list applications from deployments")
    36  	}
    37  
    38  	// Filter out any names, as there could be multiple components but within the same application
    39  	return util.RemoveDuplicates(deploymentAppNames), nil
    40  }
