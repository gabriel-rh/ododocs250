   103  // newPreferenceInfo gets the PreferenceInfo from preference file
   104  // or returns default PreferenceInfo if preference file does not exist
   105  func newPreferenceInfo() (*preferenceInfo, error) {
   106  	preferenceFile, err := getPreferenceFile()
   107  	klog.V(4).Infof("The path for preference file is %+v", preferenceFile)
   108  	if err != nil {
   109  		return nil, err
   110  	}
   111  
   112  	c := preferenceInfo{
   113  		Preference: newPreference(),
   114  		Filename:   preferenceFile,
   115  	}
   116  
   117  	// Default devfile registry
   118  	defaultRegistryList := []Registry{
   119  		{
   120  			Name:   DefaultDevfileRegistryName,
   121  			URL:    DefaultDevfileRegistryURL,
   122  			Secure: false,
   123  		},
   124  	}
   125  
   126  	// If the preference file doesn't exist then we return with default preference
   127  	if _, err = os.Stat(preferenceFile); os.IsNotExist(err) {
   128  		c.OdoSettings.RegistryList = &defaultRegistryList
   129  		return &c, nil
   130  	}
   131  
   132  	err = util.GetFromFile(&c.Preference, c.Filename)
   133  	if err != nil {
   134  		return nil, err
   135  	}
   136  
   137  	// Handle user has preference file but doesn't use dynamic registry before
   138  	if c.OdoSettings.RegistryList == nil {
   139  		c.OdoSettings.RegistryList = &defaultRegistryList
   140  	}
   141  
   142  	// Handle OCI-based default registry migration
   143  	if c.OdoSettings.RegistryList != nil {
   144  		for index, registry := range *c.OdoSettings.RegistryList {
   145  			if registry.Name == DefaultDevfileRegistryName && registry.URL == OldDefaultDevfileRegistryURL {
   146  				registryList := *c.OdoSettings.RegistryList
   147  				registryList[index].URL = DefaultDevfileRegistryURL
   148  				break
   149  			}
   150  		}
   151  	}
   152  
   153  	return &c, nil
   154  }

   156  // RegistryHandler handles registry add, update and delete operations
   157  func (c *preferenceInfo) RegistryHandler(operation string, registryName string, registryURL string, forceFlag bool, isSecure bool) error {
   158  	var registryList []Registry
   159  	var err error
   160  	var registryExist bool
   161  
   162  	// Registry list is empty
   163  	if c.OdoSettings.RegistryList == nil {
   164  		registryList, err = handleWithoutRegistryExist(registryList, operation, registryName, registryURL, isSecure)
   165  		if err != nil {
   166  			return err
   167  		}
   168  	} else {
   169  		// The target registry exists in the registry list
   170  		registryList = *c.OdoSettings.RegistryList
   171  		for index, registry := range registryList {
   172  			if registry.Name == registryName {
   173  				registryExist = true
   174  				registryList, err = handleWithRegistryExist(index, registryList, operation, registryName, registryURL, forceFlag, isSecure)
   175  				if err != nil {
   176  					return err
   177  				}
   178  			}
   179  		}
   180  
   181  		// The target registry doesn't exist in the registry list
   182  		if !registryExist {
   183  			registryList, err = handleWithoutRegistryExist(registryList, operation, registryName, registryURL, isSecure)
   184  			if err != nil {
   185  				return err
   186  			}
   187  		}
   188  	}
   189  
   190  	c.OdoSettings.RegistryList = &registryList
   191  	err = util.WriteToFile(&c.Preference, c.Filename)
   192  	if err != nil {
   193  		return errors.Errorf("unable to write the configuration of %q operation to preference file", operation)
   194  	}
   195  
   196  	return nil
   197  }

   199  func handleWithoutRegistryExist(registryList []Registry, operation string, registryName string, registryURL string, isSecure bool) ([]Registry, error) {
   200  	switch operation {
   201  
   202  	case "add":
   203  		registry := Registry{
   204  			Name:   registryName,
   205  			URL:    registryURL,
   206  			Secure: isSecure,
   207  		}
   208  		registryList = append(registryList, registry)
   209  
   210  	case "update", "delete":
   211  		return nil, errors.Errorf("failed to %v registry: registry %q doesn't exist", operation, registryName)
   212  	}
   213  
   214  	return registryList, nil
   215  }

   217  func handleWithRegistryExist(index int, registryList []Registry, operation string, registryName string, registryURL string, forceFlag bool, isSecure bool) ([]Registry, error) {
   218  	switch operation {
   219  
   220  	case "add":
   221  		return nil, errors.Errorf("failed to add registry: registry %q already exists", registryName)
   222  
   223  	case "update":
   224  		if !forceFlag {
   225  			if !ui.Proceed(fmt.Sprintf("Are you sure you want to update registry %q", registryName)) {
   226  				log.Info("Aborted by the user")
   227  				return registryList, nil
   228  			}
   229  		}
   230  
   231  		registryList[index].URL = registryURL
   232  		registryList[index].Secure = isSecure
   233  		log.Info("Successfully updated registry")
   234  
   235  	case "delete":
   236  		if !forceFlag {
   237  			if !ui.Proceed(fmt.Sprintf("Are you sure you want to delete registry %q", registryName)) {
   238  				log.Info("Aborted by the user")
   239  				return registryList, nil
   240  			}
   241  		}
   242  
   243  		copy(registryList[index:], registryList[index+1:])
   244  		registryList[len(registryList)-1] = Registry{}
   245  		registryList = registryList[:len(registryList)-1]
   246  		log.Info("Successfully deleted registry")
   247  	}
   248  
   249  	return registryList, nil
   250  }

   254  func (c *preferenceInfo) SetConfiguration(parameter string, value string) error {
   255  	if p, ok := asSupportedParameter(parameter); ok {
   256  		// processing values according to the parameter names
   257  		switch p {
   258  
   259  		case "timeout":
   260  			typedval, err := strconv.Atoi(value)
   261  			if err != nil {
   262  				return errors.Errorf("unable to set %q to %q", parameter, value)
   263  			}
   264  			if typedval < 0 {
   265  				return errors.Errorf("cannot set timeout to less than 0")
   266  			}
   267  			c.OdoSettings.Timeout = &typedval
   268  
   269  		case "buildtimeout":
   270  			typedval, err := strconv.Atoi(value)
   271  			if err != nil {
   272  				return errors.Errorf("unable to set %q to %q, value must be an integer", parameter, value)
   273  			}
   274  			if typedval < 0 {
   275  				return errors.Errorf("cannot set timeout to less than 0")
   276  			}
   277  			c.OdoSettings.BuildTimeout = &typedval
   278  
   279  		case "pushtimeout":
   280  			typedval, err := strconv.Atoi(value)
   281  			if err != nil {
   282  				return errors.Errorf("unable to set %q to %q, value must be an integer", parameter, value)
   283  			}
   284  			if typedval < 0 {
   285  				return errors.Errorf("cannot set timeout to less than 0")
   286  			}
   287  			c.OdoSettings.PushTimeout = &typedval
   288  
   289  		case "registrycachetime":
   290  			typedval, err := strconv.Atoi(value)
   291  			if err != nil {
   292  				return errors.Errorf("unable to set %q to %q, value must be an integer", parameter, value)
   293  			}
   294  			if typedval < 0 {
   295  				return errors.Errorf("cannot set timeout to less than 0")
   296  			}
   297  			c.OdoSettings.RegistryCacheTime = &typedval
   298  
   299  		case "updatenotification":
   300  			val, err := strconv.ParseBool(strings.ToLower(value))
   301  			if err != nil {
   302  				return errors.Errorf("unable to set %q to %q, value must be a boolean", parameter, value)
   303  			}
   304  			c.OdoSettings.UpdateNotification = &val
   305  
   306  		//	TODO: should we add a validator here? What is the use of nameprefix?
   307  		case "nameprefix":
   308  			c.OdoSettings.NamePrefix = &value
   309  
   310  		case "ephemeral":
   311  			val, err := strconv.ParseBool(strings.ToLower(value))
   312  			if err != nil {
   313  				return errors.Errorf("unable to set %q to %q, value must be a boolean", parameter, value)
   314  			}
   315  			c.OdoSettings.Ephemeral = &val
   316  
   317  		case "consenttelemetry":
   318  			val, err := strconv.ParseBool(strings.ToLower(value))
   319  			if err != nil {
   320  				return errors.Errorf("unable to set %q to %q, value must be a boolean", parameter, value)
   321  			}
   322  			c.OdoSettings.ConsentTelemetry = &val
   323  		}
   324  	} else {
   325  		return errors.Errorf("unknown parameter : %q is not a parameter in odo preference, run `odo preference -h` to see list of available parameters", parameter)
   326  	}
   327  
   328  	err := util.WriteToFile(&c.Preference, c.Filename)
   329  	if err != nil {
   330  		return errors.Errorf("unable to set %q, something is wrong with odo, kindly raise an issue at https://github.com/redhat-developer/odo/issues/new?template=Bug.md", parameter)
   331  	}
   332  	return nil
   333  }
