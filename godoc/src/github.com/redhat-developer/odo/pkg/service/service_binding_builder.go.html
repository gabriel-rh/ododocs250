    53  func ProvisionedService(ctx pipeline.Context) {
    54  	services, _ := ctx.Services()
    55  
    56  	for _, service := range services {
    57  		res := service.Resource()
    58  		secretName, found, err := unstructured.NestedString(res.Object, "status", "binding", "name")
    59  		if err != nil {
    60  			requestRetry(ctx, collect.ErrorReadingBindingReason, err)
    61  			return
    62  		}
    63  		if found {
    64  			if secretName != "" {
    65  				secret, err := ctx.ReadSecret(res.GetNamespace(), secretName)
    66  				if err != nil {
    67  					requestRetry(ctx, collect.ErrorReadingSecret, err)
    68  					return
    69  				}
    70  				ctx.AddBindings(&pipeline.SecretBackedBindings{Service: service, Secret: secret})
    71  			}
    72  		} else {
    73  			crd, err := service.CustomResourceDefinition()
    74  			if err != nil {
    75  				// If the user does not have permission to access CRD, ignore
    76  				if !kerrors.IsForbidden(err) {
    77  					requestRetry(ctx, collect.ErrorReadingCRD, err)
    78  					return
    79  				} else {
    80  					log.Warning("Skipping the check for CRD, user does not have access")
    81  				}
    82  			}
    83  			if crd == nil {
    84  				continue
    85  			}
    86  			v, ok := crd.Resource().GetAnnotations()[collect.ProvisionedServiceAnnotationKey]
    87  			if ok && v == "true" {
    88  				requestRetry(ctx, collect.ErrorReadingBindingReason, fmt.Errorf("CRD of service %v/%v indicates provisioned service, but no secret name provided under .status.binding.name", res.GetNamespace(), res.GetName()))
    89  				return
    90  			}
    91  		}
    92  	}

    95  func BindingDefinitions(ctx pipeline.Context) {
    96  	services, _ := ctx.Services()
    97  
    98  	for _, service := range services {
    99  		anns := make(map[string]string)
   100  		crd, err := service.CustomResourceDefinition()
   101  		if err != nil {
   102  			// If the user does not have permission to access CRD, ignore
   103  			if !kerrors.IsForbidden(err) {
   104  				requestRetry(ctx, collect.ErrorReadingCRD, err)
   105  				return
   106  			} else {
   107  				log.Warning("Skipping the check for CRD, user does not have access")
   108  			}
   109  		}
   110  		if crd != nil {
   111  			descr, err := crd.Descriptor()
   112  			if err != nil {
   113  				requestRetry(ctx, collect.ErrorReadingDescriptorReason, err)
   114  				return
   115  			}
   116  			if descr != nil {
   117  				util.MergeMaps(anns, bindingAnnotations(descr))
   118  			}
   119  			util.MergeMaps(anns, crd.Resource().GetAnnotations())
   120  		}
   121  
   122  		util.MergeMaps(anns, service.Resource().GetAnnotations())
   123  
   124  		for k, v := range anns {
   125  			definition, err := makeBindingDefinition(k, v, ctx)
   126  			if err != nil {
   127  				continue
   128  			}
   129  			service.AddBindingDef(definition)
   130  		}
   131  	}
   132  }

   154  func bindingAnnotations(crdDescription *olmv1alpha1.CRDDescription) map[string]string {
   155  	anns := make(map[string]string)
   156  	for _, sd := range crdDescription.StatusDescriptors {
   157  		objectType := getObjectType(sd.XDescriptors)
   158  		for _, xd := range sd.XDescriptors {
   159  			loadDescriptor(anns, sd.Path, xd, "status", objectType)
   160  		}
   161  	}
   162  
   163  	for _, sd := range crdDescription.SpecDescriptors {
   164  		objectType := getObjectType(sd.XDescriptors)
   165  		for _, xd := range sd.XDescriptors {
   166  			loadDescriptor(anns, sd.Path, xd, "spec", objectType)
   167  		}
   168  	}
   169  	return anns
   170  }

   172  func getObjectType(descriptors []string) string {
   173  	typeAnno := "urn:alm:descriptor:io.kubernetes:"
   174  	for _, desc := range descriptors {
   175  		if strings.HasPrefix(desc, typeAnno) {
   176  			return strings.TrimPrefix(desc, typeAnno)
   177  		}
   178  	}
   179  	return ""
   180  }

   182  func loadDescriptor(anns map[string]string, path string, descriptor string, root string, objectType string) {
   183  	if !strings.HasPrefix(descriptor, binding.AnnotationPrefix) {
   184  		return
   185  	}
   186  
   187  	keys := strings.Split(descriptor, ":")
   188  	key := binding.AnnotationPrefix
   189  	value := ""
   190  
   191  	if len(keys) > 1 {
   192  		key += "/" + keys[1]
   193  	} else {
   194  		key += "/" + path
   195  	}
   196  
   197  	p := []string{fmt.Sprintf("path={.%s.%s}", root, path)}
   198  	if len(keys) > 1 {
   199  		p = append(p, keys[2:]...)
   200  	}
   201  	if objectType != "" {
   202  		p = append(p, []string{fmt.Sprintf("objectType=%s", objectType)}...)
   203  	}
   204  
   205  	value += strings.Join(p, ",")
   206  	anns[key] = value
   207  }
