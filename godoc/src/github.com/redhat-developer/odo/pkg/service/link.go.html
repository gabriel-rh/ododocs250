    29  // PushLinks updates Link(s) from Kubernetes Inlined component in a devfile by creating new ones or removing old ones
    30  // returns true if the component needs to be restarted (when a link has been created or deleted)
    31  // if service binding operator is not present, it will call pushLinksWithoutOperator to create the links without it.
    32  func PushLinks(client kclient.ClientInterface, k8sComponents []devfile.Component, labels map[string]string, deployment *v1.Deployment, context string) (bool, error) {
    33  	serviceBindingSupport, err := client.IsServiceBindingSupported()
    34  	if err != nil {
    35  		return false, err
    36  	}
    37  
    38  	if !serviceBindingSupport {
    39  		return pushLinksWithoutOperator(client, k8sComponents, labels, deployment, context)
    40  	}
    41  
    42  	return pushLinksWithOperator(client, k8sComponents, labels, deployment, context)
    43  }

    45  // pushLinksWithOperator creates links or deletes links (if service binding operator is installed) between components and services
    46  // returns true if the component needs to be restarted (a secret was generated and added to the deployment)
    47  func pushLinksWithOperator(client kclient.ClientInterface, k8sComponents []devfile.Component, labels map[string]string, deployment *v1.Deployment, context string) (bool, error) {
    48  
    49  	ownerReference := generator.GetOwnerReference(deployment)
    50  	deployed, err := ListDeployedServices(client, labels)
    51  	if err != nil {
    52  		return false, err
    53  	}
    54  
    55  	for key, deployedResource := range deployed {
    56  		if !deployedResource.isLinkResource {
    57  			delete(deployed, key)
    58  		}
    59  	}
    60  
    61  	restartNeeded := false
    62  
    63  	// create an object on the kubernetes cluster for all the Kubernetes Inlined components
    64  	for _, c := range k8sComponents {
    65  		// get the string representation of the YAML definition of a CRD
    66  		strCRD := c.Kubernetes.Inlined
    67  		if c.Kubernetes.Uri != "" {
    68  			strCRD, err = getDataFromURI(c.Kubernetes.Uri, context, devfilefs.DefaultFs{})
    69  			if err != nil {
    70  				return false, err
    71  			}
    72  		}
    73  
    74  		// convert the YAML definition into map[string]interface{} since it's needed to create dynamic resource
    75  		u := unstructured.Unstructured{}
    76  		if e := yaml.Unmarshal([]byte(strCRD), &u.Object); e != nil {
    77  			return false, e
    78  		}
    79  
    80  		if !isLinkResource(u.GetKind()) {
    81  			// operator hub is not installed on the cluster
    82  			// or it's a service binding related resource
    83  			continue
    84  		}
    85  
    86  		crdName := u.GetName()
    87  		u.SetOwnerReferences([]metav1.OwnerReference{ownerReference})
    88  		u.SetLabels(labels)
    89  
    90  		err = createOperatorService(client, u)
    91  		delete(deployed, u.GetKind()+"/"+crdName)
    92  		if err != nil {
    93  			if strings.Contains(err.Error(), "already exists") {
    94  				// this could be the case when "odo push" was executed after making change to code but there was no change to the service itself
    95  				// TODO: better way to handle this might be introduced by https://github.com/redhat-developer/odo/issues/4553
    96  				continue // this ensures that services slice is not updated
    97  			} else {
    98  				return false, err
    99  			}
   100  		}
   101  
   102  		name := u.GetName()
   103  		log.Successf("Created link %q using Service Binding Operator on the cluster; component will be restarted", name)
   104  		restartNeeded = true
   105  	}
   106  
   107  	for key, val := range deployed {
   108  		if !isLinkResource(val.Kind) {
   109  			continue
   110  		}
   111  		err = DeleteOperatorService(client, key)
   112  		if err != nil {
   113  			return false, err
   114  
   115  		}
   116  
   117  		log.Successf("Deleted link %q using Service Binding Operator on the cluster; component will be restarted", key)
   118  		restartNeeded = true
   119  	}
   120  
   121  	if !restartNeeded {
   122  		log.Success("Links are in sync with the cluster, no changes are required")
   123  	}
   124  	return restartNeeded, nil
   125  }

   127  // pushLinksWithoutOperator creates links or deletes links (if service binding operator is not installed) between components and services
   128  // returns true if the component needs to be restarted (a secret was generated and added to the deployment)
   129  func pushLinksWithoutOperator(client kclient.ClientInterface, k8sComponents []devfile.Component, labels map[string]string, deployment *v1.Deployment, context string) (bool, error) {
   130  
   131  	// check csv support before proceeding
   132  	csvSupport, err := client.IsCSVSupported()
   133  	if err != nil {
   134  		return false, err
   135  	}
   136  
   137  	secrets, err := client.ListSecrets(componentlabels.GetSelector(labels[componentlabels.ComponentLabel], labels[applabels.ApplicationLabel]))
   138  	if err != nil {
   139  		return false, err
   140  	}
   141  
   142  	ownerReferences := generator.GetOwnerReference(deployment)
   143  
   144  	clusterLinksMap := make(map[string]string)
   145  	for _, secret := range secrets {
   146  		if value, ok := secret.GetLabels()[LinkLabel]; ok {
   147  			clusterLinksMap[value] = secret.Name
   148  		}
   149  	}
   150  
   151  	localLinksMap := make(map[string]string)
   152  	// create an object on the kubernetes cluster for all the Kubernetes Inlined components
   153  	for _, c := range k8sComponents {
   154  		// get the string representation of the YAML definition of a CRD
   155  		strCRD := c.Kubernetes.Inlined
   156  		if c.Kubernetes.Uri != "" {
   157  			strCRD, err = getDataFromURI(c.Kubernetes.Uri, context, devfilefs.DefaultFs{})
   158  			if err != nil {
   159  				return false, err
   160  			}
   161  		}
   162  
   163  		// convert the YAML definition into map[string]interface{} since it's needed to create dynamic resource
   164  		u := unstructured.Unstructured{}
   165  		if e := yaml.Unmarshal([]byte(strCRD), &u.Object); e != nil {
   166  			return false, e
   167  		}
   168  
   169  		if !isLinkResource(u.GetKind()) {
   170  			// not a service binding object, thus continue
   171  			continue
   172  		}
   173  		localLinksMap[c.Name] = strCRD
   174  	}
   175  
   176  	var processingPipeline pipeline.Pipeline
   177  
   178  	deploymentGVR, err := client.GetDeploymentAPIVersion()
   179  	if err != nil {
   180  		return false, err
   181  	}
   182  
   183  	var restartRequired bool
   184  
   185  	// delete the links not present on the devfile
   186  	for linkName, secretName := range clusterLinksMap {
   187  		if _, ok := localLinksMap[linkName]; !ok {
   188  
   189  			// recreate parts of the service binding request for deletion
   190  			var newServiceBinding servicebinding.ServiceBinding
   191  			newServiceBinding.Name = linkName
   192  			newServiceBinding.Namespace = client.GetCurrentNamespace()
   193  			newServiceBinding.Spec.Application = servicebinding.Application{
   194  				Ref: servicebinding.Ref{
   195  					Name:     deployment.Name,
   196  					Group:    deploymentGVR.Group,
   197  					Version:  deploymentGVR.Version,
   198  					Resource: deploymentGVR.Resource,
   199  				},
   200  			}
   201  			newServiceBinding.Status.Secret = secretName
   202  
   203  			// set the deletion time stamp to trigger deletion
   204  			timeNow := metav1.Now()
   205  			newServiceBinding.DeletionTimestamp = &timeNow
   206  
   207  			// if the pipeline was created before
   208  			// skip deletion
   209  			if processingPipeline == nil {
   210  				processingPipeline, err = getPipeline(client)
   211  				if err != nil {
   212  					return false, err
   213  				}
   214  			}
   215  			_, err = processingPipeline.Process(&newServiceBinding)
   216  			if err != nil {
   217  				return false, err
   218  			}
   219  
   220  			// since the library currently doesn't delete the secret after unbinding
   221  			// delete the secret manually
   222  			err = client.DeleteSecret(secretName, client.GetCurrentNamespace())
   223  			if err != nil {
   224  				return false, err
   225  			}
   226  			restartRequired = true
   227  			log.Successf("Deleted link %q on the cluster; component will be restarted", linkName)
   228  		}
   229  	}
   230  
   231  	var serviceCompMap map[string]string
   232  
   233  	// create the links
   234  	for linkName, strCRD := range localLinksMap {
   235  		if _, ok := clusterLinksMap[linkName]; !ok {
   236  			if serviceCompMap == nil {
   237  				// prevent listing of services unless required
   238  				services, e := client.ListServices("")
   239  				if e != nil {
   240  					return false, e
   241  				}
   242  
   243  				// get the services and get match them against the component
   244  				serviceCompMap = make(map[string]string)
   245  				for _, service := range services {
   246  					serviceCompMap[service.Name] = service.Labels[componentlabels.ComponentLabel]
   247  				}
   248  			}
   249  
   250  			// get the string representation of the YAML definition of a CRD
   251  			var serviceBinding servicebinding.ServiceBinding
   252  			err = yaml.Unmarshal([]byte(strCRD), &serviceBinding)
   253  			if err != nil {
   254  				return false, err
   255  			}
   256  
   257  			if len(serviceBinding.Spec.Services) != 1 {
   258  				continue
   259  			}
   260  
   261  			if !csvSupport && !isLinkResource(serviceBinding.Spec.Services[0].Kind) {
   262  				// ignore service binding objects linked to services if csv support is not present on the cluster
   263  				continue
   264  			}
   265  
   266  			// set the labels and namespace
   267  			serviceBinding.SetLabels(labels)
   268  			serviceBinding.Namespace = client.GetCurrentNamespace()
   269  			ns := client.GetCurrentNamespace()
   270  			serviceBinding.Spec.Services[0].Namespace = &ns
   271  
   272  			_, err = json.MarshalIndent(serviceBinding, " ", " ")
   273  			if err != nil {
   274  				return false, err
   275  			}
   276  
   277  			if processingPipeline == nil {
   278  				processingPipeline, err = getPipeline(client)
   279  				if err != nil {
   280  					return false, err
   281  				}
   282  			}
   283  
   284  			_, err = processingPipeline.Process(&serviceBinding)
   285  			if err != nil {
   286  				if kerrors.IsForbidden(err) {
   287  					// due to https://github.com/redhat-developer/service-binding-operator/issues/1003
   288  					return false, fmt.Errorf("please install the service binding operator")
   289  				}
   290  				return false, err
   291  			}
   292  
   293  			if len(serviceBinding.Status.Secret) == 0 {
   294  				return false, fmt.Errorf("no secret was provided by service binding's pipleine")
   295  			}
   296  
   297  			// get the generated secret and update it with the labels and owner reference
   298  			secret, err := client.GetSecret(serviceBinding.Status.Secret, client.GetCurrentNamespace())
   299  			if err != nil {
   300  				return false, err
   301  			}
   302  			secret.Labels = labels
   303  			secret.Labels[LinkLabel] = linkName
   304  			if _, ok := serviceCompMap[serviceBinding.Spec.Services[0].Name]; ok {
   305  				secret.Labels[ServiceLabel] = serviceCompMap[serviceBinding.Spec.Services[0].Name]
   306  			} else {
   307  				secret.Labels[ServiceLabel] = serviceBinding.Spec.Services[0].Name
   308  			}
   309  			secret.Labels[ServiceKind] = serviceBinding.Spec.Services[0].Kind
   310  			if serviceBinding.Spec.Services[0].Kind != "Service" {
   311  				// the service name is stored as kind-name as `/` is not a valid char for labels of kubernetes secrets
   312  				secret.Labels[ServiceLabel] = fmt.Sprintf("%v-%v", serviceBinding.Spec.Services[0].Kind, serviceBinding.Spec.Services[0].Name)
   313  			}
   314  			secret.SetOwnerReferences([]metav1.OwnerReference{ownerReferences})
   315  			_, err = client.UpdateSecret(secret, client.GetCurrentNamespace())
   316  			if err != nil {
   317  				return false, err
   318  			}
   319  			restartRequired = true
   320  			log.Successf("Created link %q on the cluster; component will be restarted", linkName)
   321  		}
   322  	}
   323  
   324  	if restartRequired {
   325  		return true, nil
   326  	} else {
   327  		log.Success("Links are in sync with the cluster, no changes are required")
   328  	}
   329  
   330  	return false, nil
   331  }

   333  // getPipeline gets the pipeline to process service binding requests
   334  func getPipeline(client kclient.ClientInterface) (pipeline.Pipeline, error) {
   335  	mgr, err := ctrl.NewManager(client.GetClientConfig(), ctrl.Options{
   336  		Scheme: runtime.NewScheme(),
   337  		// disable the health probes to prevent binding to them
   338  		HealthProbeBindAddress: "0",
   339  		// disable the prometheus metrics
   340  		MetricsBindAddress: "0",
   341  	})
   342  	if err != nil {
   343  		return nil, err
   344  	}
   345  	return OdoDefaultBuilder.WithContextProvider(context.Provider(client.GetDynamicClient(), context.ResourceLookup(mgr.GetRESTMapper()))).Build(), nil
   346  }
