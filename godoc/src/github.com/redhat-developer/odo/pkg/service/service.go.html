   335  func listDevfileServices(client kclient.ClientInterface, devfileObj parser.DevfileObj, componentContext string, fs devfilefs.Filesystem) (map[string]unstructured.Unstructured, error) {
   336  	if devfileObj.Data == nil {
   337  		return nil, nil
   338  	}
   339  	components, err := devfileObj.Data.GetComponents(common.DevfileOptions{
   340  		ComponentOptions: parsercommon.ComponentOptions{ComponentType: devfile.KubernetesComponentType},
   341  	})
   342  	if err != nil {
   343  		return nil, err
   344  	}
   345  
   346  	csvSupported, err := client.IsCSVSupported()
   347  	if err != nil {
   348  		return nil, err
   349  	}
   350  	var operatorGVRList []meta.RESTMapping
   351  	if csvSupported {
   352  		operatorGVRList, err = client.GetOperatorGVRList()
   353  		if err != nil {
   354  			return nil, err
   355  		}
   356  	}
   357  
   358  	services := map[string]unstructured.Unstructured{}
   359  	for _, c := range components {
   360  		u, err := GetK8sComponentAsUnstructured(c.Kubernetes, componentContext, fs)
   361  		if err != nil {
   362  			return nil, err
   363  		}
   364  		restMapping, err := client.GetRestMappingFromUnstructured(u)
   365  		if err != nil {
   366  			// getting a RestMapping would fail if there are no matches for the Kind field on the cluster
   367  			// this could be a case when an Operator backed service was added to devfile while working on a cluster
   368  			// that had the Operator installed but "odo service list" is run when that Operator is either no longer
   369  			// available or on a different cluster
   370  			services[strings.Join([]string{u.GetKind(), c.Name}, "/")] = u
   371  			continue
   372  		}
   373  		var match bool
   374  		for _, i := range operatorGVRList {
   375  			if i.Resource == restMapping.Resource {
   376  				// if it's an Operator backed service, it will match; if it's Pod, Deployment, etc. it won't
   377  				match = true
   378  				break
   379  			}
   380  		}
   381  		if match {
   382  			services[strings.Join([]string{u.GetKind(), c.Name}, "/")] = u
   383  		}
   384  	}
   385  	// final list of services includes Operator backed services both supported and unsupported by the underlying k8s cluster
   386  	// but it doesn't include things like Pod, Deployment, etc.
   387  	return services, nil
   388  }

   434  // PushKubernetesResources updates service(s) from Kubernetes Inlined component in a devfile by creating new ones or removing old ones
   435  func PushKubernetesResources(client kclient.ClientInterface, k8sComponents []devfile.Component, labels map[string]string, context string) error {
   436  	// check csv support before proceeding
   437  	csvSupported, err := client.IsCSVSupported()
   438  	if err != nil {
   439  		return err
   440  	}
   441  
   442  	var deployed map[string]DeployedInfo
   443  
   444  	if csvSupported {
   445  		deployed, err = ListDeployedServices(client, labels)
   446  		if err != nil {
   447  			return err
   448  		}
   449  
   450  		for key, deployedResource := range deployed {
   451  			if deployedResource.isLinkResource {
   452  				delete(deployed, key)
   453  			}
   454  		}
   455  	}
   456  
   457  	madeChange := false
   458  
   459  	// create an object on the kubernetes cluster for all the Kubernetes Inlined components
   460  	for _, c := range k8sComponents {
   461  		u, er := GetK8sComponentAsUnstructured(c.Kubernetes, context, devfilefs.DefaultFs{})
   462  		if er != nil {
   463  			return er
   464  		}
   465  
   466  		isOperatorBackedService, er := PushKubernetesResource(client, u, labels)
   467  		if er != nil {
   468  			return er
   469  		}
   470  		if csvSupported {
   471  			delete(deployed, u.GetKind()+"/"+u.GetName())
   472  		}
   473  		if isOperatorBackedService {
   474  			log.Successf("Created service %q on the cluster; refer %q to know how to link it to the component", strings.Join([]string{u.GetKind(), u.GetName()}, "/"), "odo link -h")
   475  		}
   476  		madeChange = true
   477  	}
   478  
   479  	if csvSupported {
   480  		for key, val := range deployed {
   481  			if isLinkResource(val.Kind) {
   482  				continue
   483  			}
   484  			err = DeleteOperatorService(client, key)
   485  			if err != nil {
   486  				return err
   487  
   488  			}
   489  
   490  			log.Successf("Deleted service %q from the cluster", key)
   491  			madeChange = true
   492  		}
   493  	}
   494  
   495  	if !madeChange {
   496  		log.Success("Services are in sync with the cluster, no changes are required")
   497  	}
   498  
   499  	return nil
   500  }

   502  // PushKubernetesResource pushes a Kubernetes resource (u) to the cluster using client
   503  // adding labels to the resource
   504  func PushKubernetesResource(client kclient.ClientInterface, u unstructured.Unstructured, labels map[string]string) (bool, error) {
   505  	if isLinkResource(u.GetKind()) {
   506  		// it's a service binding related resource
   507  		return false, nil
   508  	}
   509  
   510  	isOp, err := isOperatorBackedService(client, u)
   511  	if err != nil {
   512  		return false, err
   513  	}
   514  
   515  	// add labels to the CRD before creation
   516  	existingLabels := u.GetLabels()
   517  	if isOp {
   518  		u.SetLabels(mergeLabels(existingLabels, labels))
   519  	} else {
   520  		// Kubernetes built-in resource; only set managed-by label to it
   521  		u.SetLabels(mergeLabels(existingLabels, map[string]string{"app.kubernetes.io/managed-by": "odo"}))
   522  	}
   523  
   524  	err = createOperatorService(client, u)
   525  	return isOp, err
   526  }

   528  func isOperatorBackedService(client kclient.ClientInterface, u unstructured.Unstructured) (bool, error) {
   529  	restMapping, err := client.GetRestMappingFromUnstructured(u)
   530  	if err != nil {
   531  		return false, err
   532  	}
   533  	// check if the GVR of the CRD belongs to any of the CRs provided by any of the Operators
   534  	// if yes, it is an Operator backed service.
   535  	// if no, it is likely a Kubernetes built-in resource.
   536  	operatorGVRList, err := client.GetOperatorGVRList()
   537  	if err != nil {
   538  		return false, err
   539  	}
   540  
   541  	for _, i := range operatorGVRList {
   542  		if i.Resource == restMapping.Resource {
   543  			return true, nil
   544  		}
   545  	}
   546  	return false, nil
   547  }

   550  func GetK8sComponentAsUnstructured(component *devfile.KubernetesComponent, context string, fs devfilefs.Filesystem) (unstructured.Unstructured, error) {

   556  			return unstructured.Unstructured{}, err

   558  	}

   560  	// convert the YAML definition into map[string]interface{} since it's needed to create dynamic resource

   563  		return unstructured.Unstructured{}, err

   611  // UpdateServicesWithOwnerReferences adds an owner reference to an inlined Kubernetes resource (except service binding objects)
   612  // if not already present in the list of owner references
   613  func UpdateServicesWithOwnerReferences(client kclient.ClientInterface, k8sComponents []devfile.Component, ownerReference metav1.OwnerReference, context string) error {
   614  	for _, c := range k8sComponents {
   615  		// get the string representation of the YAML definition of a CRD
   616  		u, err := GetK8sComponentAsUnstructured(c.Kubernetes, context, devfilefs.DefaultFs{})
   617  		if err != nil {
   618  			return err
   619  		}
   620  
   621  		if isLinkResource(u.GetKind()) {
   622  			// ignore service binding resources
   623  			continue
   624  		}
   625  
   626  		restMapping, err := client.GetRestMappingFromUnstructured(u)
   627  		if err != nil {
   628  			return err
   629  		}
   630  
   631  		d, err := client.GetDynamicResource(restMapping.Resource.Group, restMapping.Resource.Version, restMapping.Resource.Resource, u.GetName())
   632  		if err != nil {
   633  			return err
   634  		}
   635  
   636  		found := false
   637  		for _, ownerRef := range d.GetOwnerReferences() {
   638  			if ownerRef.UID == ownerReference.UID {
   639  				found = true
   640  				break
   641  			}
   642  		}
   643  		if found {
   644  			continue
   645  		}
   646  		d.SetOwnerReferences(append(d.GetOwnerReferences(), ownerReference))
   647  
   648  		err = client.UpdateDynamicResource(restMapping.Resource.Group, restMapping.Resource.Version, restMapping.Resource.Resource, u.GetName(), d)
   649  		if err != nil {
   650  			return err
   651  		}
   652  	}
   653  	return nil
   654  }

   678  func getDataFromURI(uri, componentContext string, fs devfilefs.Filesystem) (string, error) {
   679  
   680  	parsedURL, err := url.Parse(uri)
   681  	if err != nil {
   682  		return "", err
   683  	}
   684  	if len(parsedURL.Host) != 0 && len(parsedURL.Scheme) != 0 {
   685  		params := util.HTTPRequestParams{
   686  			URL: uri,
   687  		}
   688  		dataBytes, err := util.DownloadFileInMemoryWithCache(params, 1)
   689  		if err != nil {
   690  			return "", err
   691  		}
   692  		return string(dataBytes), nil
   693  	} else {
   694  		dataBytes, err := fs.ReadFile(filepath.Join(componentContext, uri))
   695  		if err != nil {
   696  			return "", err
   697  		}
   698  		return string(dataBytes), nil
   699  	}

   702  // ValidateResourcesExist validates if the Kubernetes inlined components are installed on the cluster
   703  func ValidateResourcesExist(client kclient.ClientInterface, k8sComponents []devfile.Component, context string) error {
   704  	if len(k8sComponents) == 0 {
   705  		return nil
   706  	}
   707  
   708  	var unsupportedResources []string
   709  	for _, c := range k8sComponents {
   710  		kindErr, err := ValidateResourceExist(client, c, context)
   711  		if err != nil {
   712  			if kindErr != "" {
   713  				unsupportedResources = append(unsupportedResources, kindErr)
   714  			} else {
   715  				return err
   716  			}
   717  		}
   718  	}
   719  
   720  	if len(unsupportedResources) > 0 {
   721  		// tell the user about all the unsupported resources in one message
   722  		return fmt.Errorf("following resource(s) in the devfile are not supported by your cluster; please install corresponding Operator(s) before doing \"odo push\": %s", strings.Join(unsupportedResources, ", "))
   723  	}
   724  	return nil
   725  }

   727  func ValidateResourceExist(client kclient.ClientInterface, k8sComponent devfile.Component, context string) (kindErr string, err error) {
   728  	// get the string representation of the YAML definition of a CRD
   729  	u, err := GetK8sComponentAsUnstructured(k8sComponent.Kubernetes, context, devfilefs.DefaultFs{})
   730  	if err != nil {
   731  		return "", err
   732  	}
   733  
   734  	_, err = client.GetRestMappingFromUnstructured(u)
   735  	if err != nil && u.GetKind() != "ServiceBinding" {
   736  		// getting a RestMapping would fail if there are no matches for the Kind field on the cluster;
   737  		// but if it's a "ServiceBinding" resource, we don't add it to unsupported list because odo can create links
   738  		// without having SBO installed
   739  		return u.GetKind(), errors.New("resource not supported")
   740  	}
   741  	return "", nil
   742  }
