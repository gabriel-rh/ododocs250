    23  // CopyFile copies localPath directory or list of files in copyFiles list to the directory in running Pod.
    24  // copyFiles is list of changed files captured during `odo watch` as well as binary file path
    25  // During copying binary components, localPath represent base directory path to binary and copyFiles contains path of binary
    26  // During copying local source components, localPath represent base directory path whereas copyFiles is empty
    27  // During `odo watch`, localPath represent base directory path whereas copyFiles contains list of changed Files
    28  func CopyFile(client SyncClient, localPath string, compInfo common.ComponentInfo, targetPath string, copyFiles []string, globExps []string, ret util.IndexerRet) error {
    29  
    30  	// Destination is set to "ToSlash" as all containers being ran within OpenShift / S2I are all
    31  	// Linux based and thus: "\opt\app-root\src" would not work correctly.
    32  	dest := filepath.ToSlash(filepath.Join(targetPath, filepath.Base(localPath)))
    33  	targetPath = filepath.ToSlash(targetPath)
    34  
    35  	klog.V(4).Infof("CopyFile arguments: localPath %s, dest %s, targetPath %s, copyFiles %s, globalExps %s", localPath, dest, targetPath, copyFiles, globExps)
    36  	reader, writer := io.Pipe()
    37  	// inspired from https://github.com/kubernetes/kubernetes/blob/master/pkg/kubectl/cmd/cp.go#L235
    38  	go func() {
    39  		defer writer.Close()
    40

    42  		if err != nil {
    43  			log.Errorf("Error while creating tar: %#v", err)
    44  			os.Exit(1)
    45  		}
    46  
    47  	}()
    48  
    49  	err := client.ExtractProjectToComponent(compInfo, targetPath, reader)

    63  // makeTar function is copied from https://github.com/kubernetes/kubernetes/blob/master/pkg/kubectl/cmd/cp.go#L309
    64  // srcPath is ignored if files is set
    65  func makeTar(srcPath, destPath string, writer io.Writer, files []string, globExps []string, ret util.IndexerRet, fs filesystem.Filesystem) error {
    66  	// TODO: use compression here?
    67  	tarWriter := taro.NewWriter(writer)
    68  	defer tarWriter.Close()
    69  	srcPath = filepath.Clean(srcPath)
    70  
    71  	// "ToSlash" is used as all containers within OpenShift are Linux based
    72  	// and thus \opt\app-root\src would be an invalid path. Backward slashes
    73  	// are converted to forward.
    74  	destPath = filepath.ToSlash(filepath.Clean(destPath))
    75  	uniquePaths := make(map[string]bool)
    76  	klog.V(4).Infof("makeTar arguments: srcPath: %s, destPath: %s, files: %+v", srcPath, destPath, files)
    77  	if len(files) != 0 {
    78  		//watchTar
    79  		for _, fileName := range files {
    80  
    81  			if _, ok := uniquePaths[fileName]; ok {
    82  				continue
    83  			} else {
    84  				uniquePaths[fileName] = true
    85  			}
    86  
    87  			if checkFileExistWithFS(fileName, fs) {
    88  
    89  				matched, err := util.IsGlobExpMatch(fileName, globExps)
    90  				if err != nil {
    91  					return err
    92  				}
    93  				if matched {
    94  					continue
    95  				}
    96  
    97  				// Fetch path of source file relative to that of source base path so that it can be passed to recursiveTar
    98  				// which uses path relative to base path for taro header to correctly identify file location when untarred
    99  
   100  				// Yes, now that the file exists, now we need to get the absolute path.. if we don't, then when we pass in:
   101  				// 'odo push --context foobar' instead of 'odo push --context ~/foobar' it will NOT work..
   102  				fileAbsolutePath, err := util.GetAbsPath(fileName)
   103  				if err != nil {
   104  					return err
   105  				}
   106  				klog.V(4).Infof("Got abs path: %s", fileAbsolutePath)
   107  				klog.V(4).Infof("Making %s relative to %s", srcPath, fileAbsolutePath)
   108  
   109  				// We use "FromSlash" to make this OS-based (Windows uses \, Linux & macOS use /)
   110  				// we get the relative path by joining the two
   111  				destFile, err := filepath.Rel(filepath.FromSlash(srcPath), filepath.FromSlash(fileAbsolutePath))
   112  				if err != nil {
   113  					return err
   114  				}
   115  
   116  				// Now we get the source file and join it to the base directory.
   117  				srcFile := filepath.Join(filepath.Base(srcPath), destFile)
   118  
   119  				if value, ok := ret.NewFileMap[destFile]; ok && value.RemoteAttribute != "" {
   120  					destFile = value.RemoteAttribute
   121  				}
   122  
   123  				klog.V(4).Infof("makeTar srcFile: %s", srcFile)
   124  				klog.V(4).Infof("makeTar destFile: %s", destFile)
   125  
   126  				// The file could be a regular file or even a folder, so use recursiveTar which handles symlinks, regular files and folders
   127  				err = linearTar(filepath.Dir(srcPath), srcFile, filepath.Dir(destPath), destFile, tarWriter, fs)
   128  				if err != nil {
   129  					return err
   130  				}
   131  			}
   132  		}
   133  	}
   134  
   135  	return nil
   136  }

   138  // linearTar function is a modified version of https://github.com/kubernetes/kubernetes/blob/master/pkg/kubectl/cmd/cp.go#L319
   139  func linearTar(srcBase, srcFile, destBase, destFile string, tw *taro.Writer, fs filesystem.Filesystem) error {
   140  	if destFile == "" {
   141  		return fmt.Errorf("linear Tar error, destFile cannot be empty")
   142  	}
   143  
   144  	klog.V(4).Infof("recursiveTar arguments: srcBase: %s, srcFile: %s, destBase: %s, destFile: %s", srcBase, srcFile, destBase, destFile)
   145  
   146  	// The destination is a LINUX container and thus we *must* use ToSlash in order
   147  	// to get the copying over done correctly..
   148  	destBase = filepath.ToSlash(destBase)
   149  	destFile = filepath.ToSlash(destFile)
   150  	klog.V(4).Infof("Corrected destinations: base: %s file: %s", destBase, destFile)
   151  
   152  	joinedPath := filepath.Join(srcBase, srcFile)
   153  
   154  	stat, err := fs.Stat(joinedPath)
   155  	if err != nil {
   156  		return err
   157  	}
   158  
   159  	if stat.IsDir() {
   160  		files, err := fs.ReadDir(joinedPath)
   161  		if err != nil {
   162  			return err
   163  		}
   164  		if len(files) == 0 {
   165  			//case empty directory
   166  			hdr, _ := taro.FileInfoHeader(stat, joinedPath)
   167  			hdr.Name = destFile
   168  			if err := tw.WriteHeader(hdr); err != nil {
   169  				return err
   170  			}
   171  		}
   172  		return nil
   173  	} else if stat.Mode()&os.ModeSymlink != 0 {
   174  		//case soft link
   175  		hdr, _ := taro.FileInfoHeader(stat, joinedPath)
   176  		target, err := os.Readlink(joinedPath)
   177  		if err != nil {
   178  			return err
   179  		}
   180  
   181  		hdr.Linkname = target
   182  		hdr.Name = destFile
   183  		if err := tw.WriteHeader(hdr); err != nil {
   184  			return err
   185  		}
   186  	} else {
   187  		//case regular file or other file type like pipe
   188  		hdr, err := taro.FileInfoHeader(stat, joinedPath)
   189  		if err != nil {
   190  			return err
   191  		}
   192  		hdr.Name = destFile
   193  
   194  		err = tw.WriteHeader(hdr)
   195  		if err != nil {
   196  			return err
   197  		}
   198  
   199  		f, err := fs.Open(joinedPath)
   200  		if err != nil {
   201  			return err
   202  		}
   203  		defer f.Close() // #nosec G307
   204  
   205  		if _, err := io.Copy(tw, f); err != nil {
   206  			return err
   207  		}
   208  
   209  		return f.Close()
   210  	}
   211  
   212  	return nil
   213  }
