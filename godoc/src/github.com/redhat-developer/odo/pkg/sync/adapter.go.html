    32  // SyncFiles does a couple of things:
    33  // if files changed/deleted are passed in from watch, it syncs them to the component
    34  // otherwise, it checks which files have changed and syncs the delta
    35  // it returns a boolean execRequired and an error. execRequired tells us if files have
    36  // changed and devfile execution is required
    37  func (a Adapter) SyncFiles(syncParameters common.SyncParameters) (bool, error) {
    38  
    39  	// Whether to write the indexer content to the index file path (resolvePath)
    40  	forceWrite := false
    41  
    42  	// Ret from Indexer function
    43  	var ret util.IndexerRet
    44  
    45  	var deletedFiles []string
    46  	var changedFiles []string
    47  	pushParameters := syncParameters.PushParams
    48  	isForcePush := pushParameters.ForceBuild || !syncParameters.ComponentExists || syncParameters.PodChanged
    49  	isWatch := len(pushParameters.WatchFiles) > 0 || len(pushParameters.WatchDeletedFiles) > 0
    50  
    51  	// When this function is invoked by watch, the logic is:
    52  	// 1) If this is the first time that watch has called Push (in this OS process), then generate the file index
    53  	//    using the file indexer, and use that to sync files (eg don't use changed/deleted files list from watch at
    54  	//    this stage; these will be found by the indexer run).
    55  	//    - In the watch scenario, we need to first run the indexer for two reasons:
    56  	// 	    - In cases where the index doesn't initially exist, we need to create it (so we can ADD to it in
    57  	//        later calls to SyncFiles(...) )
    58  	// 	    - Even if it does initially exist, there is no guarantee that the remote pod is consistent with it; so
    59  	//        on our first invocation we need to compare the index with the remote pod (by regenerating the index
    60  	//        and using the changes files list from that to sync the results.)
    61  	//
    62  	// 2) For every other push/sync call after the first, don't run the file indexer, instead we use
    63  	//    the watch events to determine what changed, and ensure that the index is then updated based
    64  	//    on the watch events (to ensure future 'odo push' calls are correct)
    65  
    66  	// True if the index was updated based on the deleted/changed files values from the watch (and
    67  	// thus the indexer doesn't need to run), false otherwise
    68  	indexRegeneratedByWatch := false
    69  
    70  	// If watch files are specified _and_ this is not the first call (by this process) to SyncFiles by the watch command, then insert the
    71  	// changed files into the existing file index, and delete removed files from the index
    72  	if isWatch && !syncParameters.PushParams.DevfileScanIndexForWatch {
    73  
    74  		err := updateIndexWithWatchChanges(pushParameters)
    75  
    76  		if err != nil {
    77  			return false, err
    78  		}
    79  
    80  		changedFiles = pushParameters.WatchFiles
    81  		deletedFiles = pushParameters.WatchDeletedFiles
    82  		deletedFiles, err = util.RemoveRelativePathFromFiles(deletedFiles, pushParameters.Path)
    83  		if err != nil {
    84  			return false, errors.Wrap(err, "unable to remove relative path from list of changed/deleted files")
    85  		}
    86  		indexRegeneratedByWatch = true
    87  
    88  	}
    89  
    90  	if !indexRegeneratedByWatch {
    91  		// Calculate the files to sync
    92  		// Tries to sync the deltas unless it is a forced push
    93  		// if it is a forced push (isForcePush) reset the index to do a full sync
    94  		absIgnoreRules := util.GetAbsGlobExps(pushParameters.Path, pushParameters.IgnoredFiles)
    95  
    96  		var s *log.Status
    97  		if syncParameters.ComponentExists {
    98  			s = log.Spinner("Checking file changes for pushing")
    99  		} else {
   100  			// if the component doesn't exist, we don't check for changes in the files
   101  			// thus we show a different message
   102  			s = log.Spinner("Checking files for pushing")
   103  		}
   104  		defer s.End(false)
   105  
   106  		// Before running the indexer, make sure the .odo folder exists (or else the index file will not get created)
   107  		odoFolder := filepath.Join(pushParameters.Path, ".odo")
   108  		if _, err := os.Stat(odoFolder); os.IsNotExist(err) {
   109  			err = os.Mkdir(odoFolder, 0750)
   110  			if err != nil {
   111  				return false, errors.Wrap(err, "unable to create directory")
   112  			}
   113  		}
   114  
   115  		// If the pod changed, reset the index, which will cause the indexer to walk the directory
   116  		// tree and resync all local files.
   117  		// If it is a new component, reset index to make sure any previously existing file is cleaned up
   118  		if syncParameters.PodChanged || !syncParameters.ComponentExists {
   119  			err := util.DeleteIndexFile(pushParameters.Path)
   120  			if err != nil {
   121  				return false, errors.Wrap(err, "unable to reset the index file")
   122  			}
   123  		}
   124  
   125  		// Run the indexer and find the modified/added/deleted/renamed files
   126  		var err error
   127  		ret, err = util.RunIndexerWithRemote(pushParameters.Path, absIgnoreRules, syncParameters.Files)
   128  		s.End(true)
   129  
   130  		if err != nil {
   131  			return false, errors.Wrap(err, "unable to run indexer")
   132  		}
   133  
   134  		if len(ret.FilesChanged) > 0 || len(ret.FilesDeleted) > 0 {
   135  			forceWrite = true
   136  		}
   137  
   138  		// apply the glob rules from the .gitignore/.odoignore file
   139  		// and ignore the files on which the rules apply and filter them out
   140  		filesChangedFiltered, filesDeletedFiltered := util.FilterIgnores(ret.FilesChanged, ret.FilesDeleted, absIgnoreRules)
   141  
   142  		deletedFiles = append(filesDeletedFiltered, ret.RemoteDeleted...)
   143  		deletedFiles = append(deletedFiles, ret.RemoteDeleted...)
   144  		klog.V(4).Infof("List of files to be deleted: +%v", deletedFiles)
   145  		changedFiles = filesChangedFiltered
   146  		klog.V(4).Infof("List of files changed: +%v", changedFiles)
   147  
   148  		if len(filesChangedFiltered) == 0 && len(filesDeletedFiltered) == 0 && !isForcePush {
   149  			return false, nil
   150  		}
   151  
   152  		if isForcePush {
   153  			deletedFiles = append(deletedFiles, "*")
   154  		}
   155  	}
   156  
   157  	err := a.pushLocal(pushParameters.Path,
   158  		changedFiles,
   159  		deletedFiles,
   160  		isForcePush,
   161  		util.GetAbsGlobExps(pushParameters.Path, pushParameters.IgnoredFiles),
   162  		syncParameters.CompInfo,
   163  		ret,
   164  	)
   165  	if err != nil {
   166  		return false, errors.Wrapf(err, "failed to sync to component with name %s", a.ComponentName)
   167  	}
   168  	if forceWrite {
   169  		err = util.WriteFile(ret.NewFileMap, ret.ResolvedPath)
   170  		if err != nil {
   171  			return false, errors.Wrapf(err, "Failed to write file")
   172  		}
   173  	}
   174  
   175  	return true, nil
   176  }

   178  // pushLocal syncs source code from the user's disk to the component
   179  func (a Adapter) pushLocal(path string, files []string, delFiles []string, isForcePush bool, globExps []string, compInfo common.ComponentInfo, ret util.IndexerRet) error {
   180  	klog.V(4).Infof("Push: componentName: %s, path: %s, files: %s, delFiles: %s, isForcePush: %+v", a.ComponentName, path, files, delFiles, isForcePush)
   181  
   182  	// Edge case: check to see that the path is NOT empty.
   183  	emptyDir, err := util.IsEmpty(path)
   184  	if err != nil {
   185  		return errors.Wrapf(err, "unable to check directory: %s", path)
   186  	} else if emptyDir {
   187  		return errors.New(fmt.Sprintf("directory/file %s is empty", path))
   188  	}
   189  
   190  	// Sync the files to the pod
   191  	s := log.Spinner("Syncing files to the component")
   192  	defer s.End(false)
   193  
   194  	syncFolder := compInfo.SyncFolder
   195  
   196  	if syncFolder != generator.DevfileSourceVolumeMount {
   197  		// Need to make sure the folder already exists on the component or else sync will fail
   198  		klog.V(4).Infof("Creating %s on the remote container if it doesn't already exist", syncFolder)
   199  		cmdArr := getCmdToCreateSyncFolder(syncFolder)
   200  
   201  		err = common.ExecuteCommand(a.Client, compInfo, cmdArr, false, nil, nil)
   202  		if err != nil {
   203  			return err
   204  		}
   205  	}
   206  	// If there were any files deleted locally, delete them remotely too.
   207  	if len(delFiles) > 0 {
   208  		cmdArr := getCmdToDeleteFiles(delFiles, syncFolder)
   209  
   210  		err = common.ExecuteCommand(a.Client, compInfo, cmdArr, false, nil, nil)
   211  		if err != nil {
   212  			return err
   213  		}
   214  	}
   215  
   216  	if !isForcePush {
   217  		if len(files) == 0 && len(delFiles) == 0 {
   218  			// nothing to push
   219  			s.End(true)
   220  			return nil
   221  		}
   222  	}
   223  
   224  	if isForcePush || len(files) > 0 {
   225  		klog.V(4).Infof("Copying files %s to pod", strings.Join(files, " "))
   226  		err = CopyFile(a.Client, path, compInfo, syncFolder, files, globExps, ret)
   227  		if err != nil {
   228  			s.End(false)
   229  			return errors.Wrap(err, "unable push files to pod")
   230  		}
   231  	}
   232  	s.End(true)
   233  
   234  	return nil
   235  }

   237  // updateIndexWithWatchChanges uses the pushParameters.WatchDeletedFiles and pushParamters.WatchFiles to update
   238  // the existing index file; the index file is required to exist when this function is called.
   239  func updateIndexWithWatchChanges(pushParameters common.PushParameters) error {
   240  	indexFilePath, err := util.ResolveIndexFilePath(pushParameters.Path)
   241  
   242  	if err != nil {
   243  		return errors.Wrapf(err, "unable to resolve path: %s", pushParameters.Path)
   244  	}
   245  
   246  	// Check that the path exists
   247  	_, err = os.Stat(indexFilePath)
   248  	if err != nil {
   249  		// This shouldn't happen: in the watch case, SyncFiles should first be called with 'DevfileScanIndexForWatch' set to true, which
   250  		// will generate the index. Then, all subsequent invocations of SyncFiles will run with 'DevfileScanIndexForWatch' set to false,
   251  		// which will not regenerate the index (merely updating it based on changed files.)
   252  		//
   253  		// If you see this error it means somehow watch's SyncFiles was called without the index being first generated (likely because the
   254  		// above mentioned pushParam wasn't set). See SyncFiles(...) for details.
   255  		return errors.Wrapf(err, "resolved path doesn't exist: %s", indexFilePath)
   256  	}
   257  
   258  	// Parse the existing index
   259  	fileIndex, err := util.ReadFileIndex(indexFilePath)
   260  	if err != nil {
   261  		return errors.Wrapf(err, "Unable to read index from path: %s", indexFilePath)
   262  	}
   263  
   264  	rootDir := pushParameters.Path
   265  
   266  	// Remove deleted files from the existing index
   267  	for _, deletedFile := range pushParameters.WatchDeletedFiles {
   268  
   269  		relativePath, err := util.CalculateFileDataKeyFromPath(deletedFile, rootDir)
   270  
   271  		if err != nil {
   272  			klog.V(4).Infof("Error occurred for %s: %v", deletedFile, err)
   273  			continue
   274  		}
   275  		delete(fileIndex.Files, relativePath)
   276  		klog.V(4).Infof("Removing watch deleted file from index: %s", relativePath)
   277  	}
   278  
   279  	// Add changed files to the existing index
   280  	for _, addedOrModifiedFile := range pushParameters.WatchFiles {
   281  		relativePath, fileData, err := util.GenerateNewFileDataEntry(addedOrModifiedFile, rootDir)
   282  
   283  		if err != nil {
   284  			klog.V(4).Infof("Error occurred for %s: %v", addedOrModifiedFile, err)
   285  			continue
   286  		}
   287  		fileIndex.Files[relativePath] = *fileData
   288  		klog.V(4).Infof("Added/updated watched file in index: %s", relativePath)
   289  	}
   290  
   291  	// Write the result
   292  	return util.WriteFile(fileIndex.Files, indexFilePath)
   293  
   294  }

   296  // getCmdToCreateSyncFolder returns the command used to create the remote sync folder on the running container
   297  func getCmdToCreateSyncFolder(syncFolder string) []string {
   298  	return []string{"mkdir", "-p", syncFolder}

   301  // getCmdToDeleteFiles returns the command used to delete the remote files on the container that are marked for deletion
   302  func getCmdToDeleteFiles(delFiles []string, syncFolder string) []string {
   303  	rmPaths := util.GetRemoteFilesMarkedForDeletion(delFiles, syncFolder)
   304  	klog.V(4).Infof("remote files marked for deletion are %+v", rmPaths)
   305  	cmdArr := []string{"rm", "-rf"}
   306  
   307  	for _, remote := range rmPaths {
   308  		cmdArr = append(cmdArr, filepath.ToSlash(remote))
   309  	}
   310  	return cmdArr
   311  }
