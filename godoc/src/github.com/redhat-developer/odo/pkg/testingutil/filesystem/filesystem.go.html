    37  	Rename(oldpath, newpath string) error

    39  	Chtimes(name string, atime time.Time, mtime time.Time) error
    40  	RemoveAll(path string) error
    41  	Remove(name string) error
    42  	Chmod(name string, mode os.FileMode) error
    43  
    44  	// from "io/ioutil"
    45  	ReadFile(filename string) ([]byte, error)
    46  	WriteFile(filename string, data []byte, perm os.FileMode) error
    47  	TempDir(dir, prefix string) (string, error)
    48  	TempFile(dir, prefix string) (File, error)
    49  	ReadDir(dirname string) ([]os.FileInfo, error)
    50  	Walk(root string, walkFn filepath.WalkFunc) error
    51  }
