    14  // TempMkdir creates a temporary directory
    15  func TempMkdir(parentDir string, newDirPrefix string) (string, error) {
    16  	parentDir = filepath.FromSlash(parentDir)
    17  	dir, err := ioutil.TempDir(parentDir, newDirPrefix)
    18  	if err != nil {
    19  		return "", errors.Wrapf(err, "failed to create dir with prefix %s in directory %s", newDirPrefix, parentDir)
    20  	}
    21  	return dir, nil

    69  // SimulateFileModifications mock function to simulate requested file/folder operation
    70  // Parameters:
    71  //	basePath: The parent directory for file/folder involved in desired file operation
    72  //	fileModification: Meta-data of file/folder
    73  // Returns:
    74  //	path to file/folder involved in the operation
    75  //	error if any or nil
    76  func SimulateFileModifications(basePath string, fileModification FileProperties) (string, error) {
    77  	// Files/folders intended to be directly under basepath will be indicated by fileModification.FileParent set to empty string
    78  	if fileModification.FileParent != "" {
    79  		// If fileModification.FileParent is not empty, use it to generate file/folder absolute path
    80  		basePath = filepath.Join(basePath, fileModification.FileParent)
    81  	}
    82  
    83  	switch fileModification.ModificationType {
    84  	case CREATE:
    85  		if fileModification.FileType == Directory {
    86  			filePath, err := TempMkdir(basePath, fileModification.FilePath)
    87  			// t.Logf("In simulateFileModifications, Attempting to create folder %s in %s. Error : %v", fileModification.filePath, basePath, err)
    88  			return filePath, err
    89  		} else if fileModification.FileType == RegularFile {
    90  			folderPath, err := TempMkFile(basePath, fileModification.FilePath)
    91  			// t.Logf("In simulateFileModifications, Attempting to create file %s in %s", fileModification.filePath, basePath)
    92  			return folderPath, err
    93  		}
    94  	case DELETE:
    95  		if fileModification.FileType == Directory {
    96  			return filepath.Join(basePath, fileModification.FilePath), os.RemoveAll(filepath.Join(basePath, fileModification.FilePath))
    97  		} else if fileModification.FileType == RegularFile {
    98  			return filepath.Join(basePath, fileModification.FilePath), os.Remove(filepath.Join(basePath, fileModification.FilePath))
    99  		}
   100  	case UPDATE:
   101  		if fileModification.FileType == Directory {
   102  			return "", fmt.Errorf("Updating directory %s is not supported", fileModification.FilePath)
   103  		} else if fileModification.FileType == RegularFile {
   104  			f, err := os.Open(filepath.Join(basePath, fileModification.FilePath))
   105  			if err != nil {
   106  				return "", err
   107  			}
   108  			if _, err := f.WriteString("Hello from Odo"); err != nil {
   109  				return "", err
   110  			}
   111  			if err := f.Sync(); err != nil {
   112  				return "", err
   113  			}
   114  			if err := f.Close(); err != nil {
   115  				return "", err
   116  			}
   117  			return filepath.Join(basePath, fileModification.FilePath), nil
   118  		}
   119  	case APPEND:
   120  		if fileModification.FileType == RegularFile {
   121  			err := ioutil.WriteFile(filepath.Join(basePath, fileModification.FilePath), []byte("// Check watch command"), os.ModeAppend)
   122  			if err != nil {
   123  				return "", err
   124  			}
   125  			return filepath.Join(basePath, fileModification.FilePath), nil
   126  		}
   127  
   128  		return "", fmt.Errorf("Append not supported for file of type %v", fileModification.FileType)
   129  
   130  	default:
   131  		return "", fmt.Errorf("Unsupported file operation %s", fileModification.ModificationType)
   132  	}
   133  	return "", nil
   134  }
