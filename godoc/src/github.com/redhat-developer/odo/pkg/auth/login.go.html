    38  		RequestTimeout: 0,
    39  		IOStreams:      genericclioptions.IOStreams{Out: filteredWriter, In: os.Stdin, ErrOut: odolog.GetStderr()},
    40  	}
    41  
    42  	// initialize client-go client and read starting kubeconfig file
    43  
    44  	loadingRules := clientcmd.NewDefaultClientConfigLoadingRules()
    45  	configOverrides := &clientcmd.ConfigOverrides{}
    46  	kubeConfig := clientcmd.NewNonInteractiveDeferredLoadingClientConfig(loadingRules, configOverrides)
    47  
    48  	kubeconfig, _ := kubeConfig.RawConfig()
    49  
    50  	a.StartingKubeConfig = &kubeconfig
    51

    53  	if len(a.Server) == 0 {
    54  		if defaultContext, defaultContextExists := a.StartingKubeConfig.Contexts[a.StartingKubeConfig.CurrentContext]; defaultContextExists {
    55  			if cluster, exists := a.StartingKubeConfig.Clusters[defaultContext.Cluster]; exists {
    56  				a.Server = cluster.Server
    57  			}
    58  		}
    59  	}
    60

    62  	if len(a.DefaultNamespace) == 0 {
    63  		if defaultContext, defaultContextExists := a.StartingKubeConfig.Contexts[a.StartingKubeConfig.CurrentContext]; defaultContextExists {
    64  			if len(defaultContext.Namespace) > 0 {
    65  				a.DefaultNamespace = defaultContext.Namespace
    66  			}
    67  		}
    68  	}
    69  
    70  	// 1. Say we're connecting
    71  	odolog.Info("Connecting to the OpenShift cluster\n")
    72  
    73  	// 2. Handle the error messages here. This is copied over from:
    74  	// https://github.com/openshift/origin/blob/master/pkg/oc/cli/login/login.go#L60
    75  	// as unauthorized errors are handled MANUALLY by oc.
    76  	if err := a.GatherInfo(); err != nil {
    77  		if kapierrors.IsUnauthorized(err) {
    78  			fmt.Println("Login failed (401 Unauthorized)")
    79  			fmt.Println("Verify you have provided correct credentials.")

    86  				}
    87  			}
    88  		}

    96  	}

   102  	}

   107  // copyAndFilter captures the output, filters it and then spits it back out to stdout.
   108  // Kindly taken from https://stackoverflow.com/questions/54570268/filtering-the-output-of-a-terminal-output-using-golang
   109  func copyAndFilter(w io.Writer, r io.Reader) ([]byte, error) {
   110  	var out []byte
   111  	buf := make([]byte, 1024)
   112  	for {
   113  		n, err := r.Read(buf[:])
   114  		if n > 0 {
   115  			d := buf[:n]
   116  			out = append(out, d...)
   117  			if _, e := w.Write(filteredInformation(d)); e != nil {
   118  				return out, e
   119  			}
   120  		}
   121  		if err != nil {
   122  			// Read returns io.EOF at the end of file, which is not an error for us
   123  			if err == io.EOF {
   124  				err = nil
   125  			}
   126  			return out, err
   127  		}
   128  	}
   129  }

   131  // filteredInformation takes a list of strings ([]byte), replaces them and spits it back out
   132  // This is used since we utilize `oc login` with odo and require certain strings to be filtered / changed
   133  // to their odo equivalent
   134  func filteredInformation(s []byte) []byte {
   135  
   136  	// List of strings to correctly filter
   137  	s = bytes.Replace(s, []byte("oc new-project"), []byte("odo project create"), -1)
   138  	s = bytes.Replace(s, []byte("<projectname>"), []byte("<project-name>"), -1)
   139  	s = bytes.Replace(s, []byte("project <project-name>"), []byte("project set <project-name>"), -1)
   140  	s = bytes.Replace(s, []byte("odo projects"), []byte("odo project list"), -1)
   141  
   142  	return s
   143  }
