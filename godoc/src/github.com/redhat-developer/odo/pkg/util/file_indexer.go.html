    46  // ReadFileIndex tries to read the odo index file from the given location and returns the data from the file
    47  // if no such file is present, it means the folder hasn't been walked and thus returns a empty list
    48  func ReadFileIndex(filePath string) (*FileIndex, error) {
    49  	// Read operation
    50  	var fi FileIndex
    51  	if _, err := os.Stat(filePath); os.IsNotExist(err) {
    52  		return NewFileIndex(), nil
    53  	}
    54  
    55  	byteValue, err := ioutil.ReadFile(filePath)
    56  	if err != nil {
    57  		return nil, err
    58  	}
    59  	// Unmarshals the byte values and fill up the file read map
    60  	err = json.Unmarshal(byteValue, &fi)
    61  	if err != nil {
    62  		// This is added here for backward compatibility because
    63  		// if the marshalling fails then that means we are trying to
    64  		// read a very old version of the index and hence we can just
    65  		// ignore it and reset index
    66  		// TODO: we need to remove this later
    67  		return NewFileIndex(), nil
    68  	}
    69  	return &fi, nil
    70  }

    72  // ResolveIndexFilePath resolves the filepath of the odo index file in the .odo folder
    73  func ResolveIndexFilePath(directory string) (string, error) {
    74  	directoryFi, err := os.Stat(filepath.Join(directory))
    75  	if err != nil {
    76  		return "", err
    77  	}
    78  
    79  	switch mode := directoryFi.Mode(); {
    80  	case mode.IsDir():
    81  		// do directory stuff
    82  		return filepath.Join(directory, DotOdoDirectory, fileIndexName), nil
    83  	case mode.IsRegular():
    84  		// do file stuff
    85  		// for binary files
    86  		return filepath.Join(filepath.Dir(directory), DotOdoDirectory, fileIndexName), nil
    87  	}
    88  
    89  	return directory, nil
    90  }

   207  // RunIndexerWithRemote reads the existing index from the given directory and runs the indexer on it
   208  // with the given ignore rules
   209  // it also adds the file index to the .gitignore file and resolves the path
   210  func RunIndexerWithRemote(directory string, ignoreRules []string, remoteDirectories map[string]string) (ret IndexerRet, err error) {
   211  	directory = filepath.FromSlash(directory)
   212  	ret.ResolvedPath, err = ResolveIndexFilePath(directory)
   213  	if err != nil {
   214  		return ret, err
   215  	}
   216  
   217  	// check for .gitignore file and add odo-file-index.json to .gitignore
   218  	gitIgnoreFile, err := TouchGitIgnoreFile(directory)
   219  	if err != nil {
   220  		return ret, err
   221  	}
   222  
   223  	// add odo-file-index.json path to .gitignore
   224  	err = AddOdoFileIndex(gitIgnoreFile)
   225  	if err != nil {
   226  		return ret, err
   227  	}
   228  
   229  	// read the odo index file
   230  	existingFileIndex, err := ReadFileIndex(ret.ResolvedPath)
   231  	if err != nil {
   232  		return ret, err
   233  	}
   234  
   235  	returnedIndex, err := runIndexerWithExistingFileIndex(directory, ignoreRules, remoteDirectories, existingFileIndex)
   236  	if err != nil {
   237  		return IndexerRet{}, err
   238  	}
   239  	returnedIndex.ResolvedPath = ret.ResolvedPath
   240  	return returnedIndex, nil
   241  }

   243  // runIndexerWithExistingFileIndex visits the given directory and creates the new index data
   244  // it ignores the files and folders satisfying the ignoreRules
   245  func runIndexerWithExistingFileIndex(directory string, ignoreRules []string, remoteDirectories map[string]string, existingFileIndex *FileIndex) (ret IndexerRet, err error) {
   246  	destPath := ""
   247  	srcPath := directory
   248  
   249  	ret.NewFileMap = make(map[string]FileData)
   250  
   251  	fileChanged := make(map[string]bool)
   252  	filesDeleted := make(map[string]bool)
   253  	fileRemoteChanged := make(map[string]bool)
   254  
   255  	if len(remoteDirectories) == 0 {
   256  		// The file could be a regular file or even a folder, so use recursiveTar which handles symlinks, regular files and folders
   257  		pathOptions := recursiveCheckerPathOptions{directory, filepath.Dir(srcPath), filepath.Base(srcPath), filepath.Dir(destPath), filepath.Base(destPath)}
   258  		innerRet, err := recursiveChecker(pathOptions, ignoreRules, remoteDirectories, *existingFileIndex)
   259  
   260  		if err != nil {
   261  			return IndexerRet{}, err
   262  		}
   263  
   264  		for k, v := range innerRet.NewFileMap {
   265  			ret.NewFileMap[k] = v
   266  		}
   267  
   268  		for _, remote := range innerRet.FilesChanged {
   269  			fileChanged[remote] = true
   270  		}
   271  
   272  		for _, remote := range innerRet.RemoteDeleted {
   273  			fileRemoteChanged[remote] = true
   274  		}
   275  
   276  		for _, remote := range innerRet.FilesDeleted {
   277  			filesDeleted[remote] = true
   278  		}
   279  	}
   280  
   281  	for remoteAttribute := range remoteDirectories {
   282  		matches, err := filepath.Glob(filepath.Join(directory, remoteAttribute))
   283  		if err != nil {
   284  			return IndexerRet{}, err
   285  		}
   286  		if len(matches) == 0 {
   287  			return IndexerRet{}, fmt.Errorf("path %q doens't exist", remoteAttribute)
   288  		}
   289  		for _, fileName := range matches {
   290  			if checkFileExist(fileName) {
   291  				// Fetch path of source file relative to that of source base path so that it can be passed to recursiveTar
   292  				// which uses path relative to base path for taro header to correctly identify file location when untarred
   293  
   294  				// Yes, now that the file exists, now we need to get the absolute path.. if we don't, then when we pass in:
   295  				// 'odo push --context foobar' instead of 'odo push --context ~/foobar' it will NOT work..
   296  				fileAbsolutePath, err := util.GetAbsPath(fileName)
   297  				if err != nil {
   298  					return IndexerRet{}, err
   299  				}
   300  				klog.V(4).Infof("Got abs path: %s", fileAbsolutePath)
   301  				klog.V(4).Infof("Making %s relative to %s", srcPath, fileAbsolutePath)
   302  
   303  				// We use "FromSlash" to make this OS-based (Windows uses \, Linux & macOS use /)
   304  				// we get the relative path by joining the two
   305  				destFile, err := filepath.Rel(filepath.FromSlash(srcPath), filepath.FromSlash(fileAbsolutePath))
   306  				if err != nil {
   307  					return IndexerRet{}, err
   308  				}
   309  
   310  				// Now we get the source file and join it to the base directory.
   311  				srcFile := filepath.Join(filepath.Base(srcPath), destFile)
   312  
   313  				if value, ok := remoteDirectories[filepath.ToSlash(destFile)]; ok {
   314  					destFile = value
   315  				}
   316  
   317  				klog.V(4).Infof("makeTar srcFile: %s", srcFile)
   318  				klog.V(4).Infof("makeTar destFile: %s", destFile)
   319  
   320  				// The file could be a regular file or even a folder, so use recursiveTar which handles symlinks, regular files and folders
   321  				pathOptions := recursiveCheckerPathOptions{directory, filepath.Dir(srcPath), srcFile, filepath.Dir(destPath), destFile}
   322  				innerRet, err := recursiveChecker(pathOptions, ignoreRules, remoteDirectories, *existingFileIndex)
   323  				if err != nil {
   324  					return IndexerRet{}, err
   325  				}
   326  
   327  				for k, v := range innerRet.NewFileMap {
   328  					ret.NewFileMap[k] = v
   329  				}
   330  
   331  				for _, remote := range innerRet.FilesChanged {
   332  					fileChanged[remote] = true
   333  				}
   334  
   335  				for _, remote := range innerRet.RemoteDeleted {
   336  					fileRemoteChanged[remote] = true
   337  				}
   338  
   339  				for _, remote := range innerRet.FilesDeleted {
   340  					filesDeleted[remote] = true
   341  				}
   342  			} else {
   343  				return IndexerRet{}, fmt.Errorf("path %q doens't exist", fileName)
   344  			}
   345  		}
   346  	}
   347  
   348  	// find files which are deleted/renamed
   349  	for fileName, value := range existingFileIndex.Files {
   350  		if _, ok := ret.NewFileMap[fileName]; !ok {
   351  			klog.V(4).Infof("Deleting file: %s", fileName)
   352  
   353  			if value.RemoteAttribute != "" {
   354  				currentRemote := value.RemoteAttribute
   355  				for _, remote := range findRemoteFolderForDeletion(currentRemote, remoteDirectories) {
   356  					fileRemoteChanged[remote] = true
   357  				}
   358  			} else {
   359  				// check the *absolute* path to the file for glob rules
   360  				fileAbsolutePath, err := GetAbsPath(filepath.Join(directory, fileName))
   361  				if err != nil {
   362  					return ret, errors.Wrapf(err, "unable to retrieve absolute path of file %s", fileName)
   363  				}
   364  
   365  				matched, err := util.IsGlobExpMatch(fileAbsolutePath, ignoreRules)
   366  				if err != nil {
   367  					return IndexerRet{}, err
   368  				}
   369  				if matched {
   370  					continue
   371  				}
   372  				filesDeleted[fileName] = true
   373  			}
   374  		}
   375  	}
   376  
   377  	if len(fileRemoteChanged) > 0 {
   378  		ret.RemoteDeleted = []string{}
   379  	}
   380  	if len(fileChanged) > 0 {
   381  		ret.FilesChanged = []string{}
   382  	}
   383  	if len(filesDeleted) > 0 {
   384  		ret.FilesDeleted = []string{}
   385  	}
   386  	for remote := range fileRemoteChanged {
   387  		ret.RemoteDeleted = append(ret.RemoteDeleted, remote)
   388  	}
   389  	for remote := range fileChanged {
   390  		ret.FilesChanged = append(ret.FilesChanged, remote)
   391  	}
   392  	for remote := range filesDeleted {
   393  		ret.FilesDeleted = append(ret.FilesDeleted, remote)
   394  	}
   395  
   396  	return ret, nil
   397  }

   409  // recursiveChecker visits the current source and it's inner files and folders, if any
   410  // the destination values are used to record the appropriate remote location for file or folder
   411  // ignoreRules are used to ignore file and folders
   412  // remoteDirectories are used to find the remote destination of the file/folder and to delete files/folders left behind after the attributes are changed
   413  // existingFileIndex is used to check for file/folder changes
   414  func recursiveChecker(pathOptions recursiveCheckerPathOptions, ignoreRules []string, remoteDirectories map[string]string, existingFileIndex FileIndex) (IndexerRet, error) {
   415  	klog.V(4).Infof("recursiveTar arguments: srcBase: %s, srcFile: %s, destBase: %s, destFile: %s", pathOptions.srcBase, pathOptions.srcFile, pathOptions.destBase, pathOptions.destFile)
   416  
   417  	// The destination is a LINUX container and thus we *must* use ToSlash in order
   418  	// to get the copying over done correctly..
   419  	pathOptions.destBase = filepath.ToSlash(pathOptions.destBase)
   420  	pathOptions.destFile = filepath.ToSlash(pathOptions.destFile)
   421  	klog.V(4).Infof("Corrected destinations: base: %s file: %s", pathOptions.destBase, pathOptions.destFile)
   422  
   423  	joinedPath := filepath.Join(pathOptions.srcBase, pathOptions.srcFile)
   424  	matchedPathsDir, err := filepath.Glob(joinedPath)
   425  	if err != nil {
   426  		return IndexerRet{}, err
   427  	}
   428  
   429  	if len(matchedPathsDir) == 0 {
   430  		return IndexerRet{}, fmt.Errorf("path %q doens't exist", joinedPath)
   431  	}
   432  
   433  	joinedRelPath, err := filepath.Rel(pathOptions.directory, joinedPath)
   434  	if err != nil {
   435  		return IndexerRet{}, err
   436  	}
   437  
   438  	var ret IndexerRet
   439  	ret.NewFileMap = make(map[string]FileData)
   440  
   441  	fileChanged := make(map[string]bool)
   442  	fileRemoteChanged := make(map[string]bool)
   443  
   444  	for _, matchedPath := range matchedPathsDir {
   445  
   446  		// check if it matches a ignore rule
   447  		match, err := IsGlobExpMatch(matchedPath, ignoreRules)
   448  		if err != nil {
   449  			return IndexerRet{}, err
   450  		}
   451  		// the folder matches a glob rule and thus should be skipped
   452  		if match {
   453  			return IndexerRet{}, nil
   454  		}
   455  
   456  		stat, err := os.Stat(matchedPath)
   457  		if err != nil {
   458  			return IndexerRet{}, err
   459  		}
   460  
   461  		if joinedRelPath != "." {
   462  			// check for changes in the size and the modified date of the file or folder
   463  			// and if the file is newly added
   464  			if _, ok := existingFileIndex.Files[joinedRelPath]; !ok {
   465  				fileChanged[matchedPath] = true
   466  				klog.V(4).Infof("file added: %s", matchedPath)
   467  			} else if !stat.ModTime().Equal(existingFileIndex.Files[joinedRelPath].LastModifiedDate) {
   468  				fileChanged[matchedPath] = true
   469  				klog.V(4).Infof("last modified date changed: %s", matchedPath)
   470  			} else if stat.Size() != existingFileIndex.Files[joinedRelPath].Size {
   471  				fileChanged[matchedPath] = true
   472  				klog.V(4).Infof("size changed: %s", matchedPath)
   473  			}
   474  		}
   475  
   476  		if stat.IsDir() {
   477  
   478  			if stat.Name() == DotOdoDirectory || stat.Name() == ".git" {
   479  				return IndexerRet{}, nil
   480  			}
   481  
   482  			if joinedRelPath != "." {
   483  				folderData, folderChangedData, folderRemoteChangedData := handleRemoteDataFolder(pathOptions.destFile, matchedPath, joinedRelPath, remoteDirectories, existingFileIndex)
   484  				folderData.Size = stat.Size()
   485  				folderData.LastModifiedDate = stat.ModTime()
   486  				ret.NewFileMap[joinedRelPath] = folderData
   487  
   488  				for data, value := range folderChangedData {
   489  					fileChanged[data] = value
   490  				}
   491  
   492  				for data, value := range folderRemoteChangedData {
   493  					fileRemoteChanged[data] = value
   494  				}
   495  			}
   496  
   497  			// read the current folder and read inner files and folders
   498  			files, err := ioutil.ReadDir(matchedPath)
   499  			if err != nil {
   500  				return IndexerRet{}, err
   501  			}
   502  			if len(files) == 0 {
   503  				continue
   504  			}
   505  			for _, f := range files {
   506  				if _, ok := remoteDirectories[filepath.Join(joinedRelPath, f.Name())]; ok {
   507  					continue
   508  				}
   509  
   510  				opts := recursiveCheckerPathOptions{pathOptions.directory, pathOptions.srcBase, filepath.Join(pathOptions.srcFile, f.Name()), pathOptions.destBase, filepath.Join(pathOptions.destFile, f.Name())}
   511  				innerRet, err := recursiveChecker(opts, ignoreRules, remoteDirectories, existingFileIndex)
   512  				if err != nil {
   513  					return IndexerRet{}, err
   514  				}
   515  
   516  				for k, v := range innerRet.NewFileMap {
   517  					ret.NewFileMap[k] = v
   518  				}
   519  
   520  				for _, remote := range innerRet.FilesChanged {
   521  					fileChanged[remote] = true
   522  				}
   523  				for _, remote := range innerRet.RemoteDeleted {
   524  					fileRemoteChanged[remote] = true
   525  				}
   526  			}
   527  		} else {
   528  			fileData, fileChangedData, fileRemoteChangedData := handleRemoteDataFile(pathOptions.destFile, matchedPath, joinedRelPath, remoteDirectories, existingFileIndex)
   529  			fileData.Size = stat.Size()
   530  			fileData.LastModifiedDate = stat.ModTime()
   531  			ret.NewFileMap[joinedRelPath] = fileData
   532  
   533  			for data, value := range fileChangedData {
   534  				fileChanged[data] = value
   535  			}
   536  
   537  			for data, value := range fileRemoteChangedData {
   538  				fileRemoteChanged[data] = value
   539  			}
   540  		}
   541  	}
   542  
   543  	// remove duplicates in the records
   544  	if len(fileRemoteChanged) > 0 {
   545  		ret.RemoteDeleted = []string{}
   546  	}
   547  	if len(fileChanged) > 0 {
   548  		ret.FilesChanged = []string{}
   549  	}
   550  	for remote := range fileRemoteChanged {
   551  		ret.RemoteDeleted = append(ret.RemoteDeleted, remote)
   552  	}
   553  	for file := range fileChanged {
   554  		ret.FilesChanged = append(ret.FilesChanged, file)
   555  	}
   556  
   557  	return ret, nil
   558  }

   561  func handleRemoteDataFile(destFile, path, relPath string, remoteDirectories map[string]string, existingFileIndex FileIndex) (FileData, map[string]bool, map[string]bool) {
   562  	destFile = filepath.ToSlash(destFile)
   563  	fileChanged := make(map[string]bool)
   564  	fileRemoteChanged := make(map[string]bool)
   565  
   566  	remoteDeletionRequired := false
   567  
   568  	remoteAttribute := destFile
   569  	if len(remoteDirectories) == 0 {
   570  		// if no remote attributes specified currently
   571  		remoteAttribute = ""
   572  		if existingFileIndex.Files[relPath].RemoteAttribute != "" && existingFileIndex.Files[relPath].RemoteAttribute != destFile {
   573  			// remote attribute for the file exists in the index
   574  			// but the value doesn't match the current relative path
   575  			// we need to push the current file again and delete the previous location from the container
   576  
   577  			fileChanged[path] = true
   578  			if existingFileIndex.Files[relPath].RemoteAttribute != "" {
   579  				remoteDeletionRequired = true
   580  			}
   581  		}
   582  	} else {
   583  		if value, ok := remoteDirectories[relPath]; ok {
   584  			remoteAttribute = value
   585  		}
   586  
   587  		if existingFileData, ok := existingFileIndex.Files[relPath]; !ok {
   588  			// if the file data doesn't exist in the existing index, we mark the file for pushing
   589  			fileChanged[path] = true
   590  		} else {
   591  			// if the remote attribute is different in the file data from the existing index
   592  			// and the remote attribute is not same as the current relative path
   593  			// we mark the file for pushing and delete the remote paths
   594  			if existingFileData.RemoteAttribute != remoteAttribute && (remoteAttribute != relPath || existingFileData.RemoteAttribute != "") {
   595  				fileChanged[path] = true
   596  				remoteDeletionRequired = true
   597  			}
   598  		}
   599  	}
   600  
   601  	if remoteDeletionRequired {
   602  		// if remote deletion is required but the remote attribute is empty
   603  		// we use the relative path for deletion
   604  		currentRemote := existingFileIndex.Files[relPath].RemoteAttribute
   605  		if currentRemote == "" {
   606  			currentRemote = relPath
   607  		}
   608  		fileRemoteChanged[currentRemote] = true
   609  		for _, remote := range findRemoteFolderForDeletion(currentRemote, remoteDirectories) {
   610  			fileRemoteChanged[remote] = true
   611  		}
   612  	}
   613  
   614  	return FileData{
   615  		RemoteAttribute: filepath.ToSlash(remoteAttribute),
   616  	}, fileChanged, fileRemoteChanged
   617  }

   683  // findRemoteFolderForDeletion finds the remote directories which can be deleted by checking the remoteDirectories map
   684  func findRemoteFolderForDeletion(currentRemote string, remoteDirectories map[string]string) []string {
   685  	var remoteDelete []string
   686  	currentRemote = filepath.ToSlash(currentRemote)
   687  	for currentRemote != "" && currentRemote != "." && currentRemote != "/" {
   688  
   689  		found := false
   690  		for _, remote := range remoteDirectories {
   691  			if strings.HasPrefix(remote, currentRemote+"/") || remote == currentRemote {
   692  				found = true
   693  				break
   694  			}
   695  		}
   696  		if !found {
   697  			remoteDelete = append(remoteDelete, currentRemote)
   698  		}
   699  		currentRemote = filepath.ToSlash(filepath.Clean(filepath.Dir(currentRemote)))
   700  	}
   701  	return remoteDelete
   702  }
