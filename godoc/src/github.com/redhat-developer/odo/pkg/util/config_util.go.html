    18  	_, err := os.Stat(filepath.Dir(configFile))
    19  	if os.IsNotExist(err) {
    20  		err = os.MkdirAll(filepath.Dir(configFile), 0750)
    21  		if err != nil {
    22  			return errors.Wrap(err, "unable to create directory")
    23  		}
    24  	}
    25  	// Check whether config file is present or not
    26  	_, err = os.Stat(configFile)
    27  	if os.IsNotExist(err) {
    28  		file, err := os.Create(configFile)
    29  		if err != nil {
    30  			return errors.Wrap(err, "unable to create config file")
    31  		}
    32  		defer file.Close() // #nosec G307
    33  	}
    34

    71  // IsSet uses reflection to check if a parameter is set in a struct
    72  // using the name in a case insensitive manner
    73  // only supports flat structs
    74  // TODO: support deeper struct using recursion
    75  func IsSet(info interface{}, parameter string) bool {
    76  	imm := reflect.ValueOf(info)
    77  	if imm.Kind() == reflect.Ptr {
    78  		imm = imm.Elem()
    79  	}
    80  	val := imm.FieldByNameFunc(CaseInsensitive(parameter))
    81  
    82  	// If the value kind is string, only checks if the value is valid
    83  	// since by definition argument must be a chan, func, interface, map, pointer, or slice
    84  	// for IsNil() function of reflect package
    85  	if val.Kind() == reflect.String && val.IsValid() {
    86  		return true
    87  	}
    88  
    89  	if !val.IsValid() || val.IsNil() {
    90  		return false
    91  	}
    92  	if val.IsNil() {
    93  		return false
    94  	}
    95  	// if the value is a Ptr then we need to de-ref it
    96  	if val.Kind() == reflect.Ptr {
    97  		return true
    98  	}
    99  
   100  	return true
   101  }

   103  // CaseInsensitive returns a function which compares two words
   104  // caseinsensitively
   105  func CaseInsensitive(parameter string) func(word string) bool {
   106  	return func(word string) bool {
   107  		return strings.EqualFold(word, parameter)
   108  	}

   111  // DeleteConfiguration sets a parameter to null in a struct using reflection
   112  func DeleteConfiguration(info interface{}, parameter string) error {
   113  
   114  	imm := reflect.ValueOf(info)
   115  	if imm.Kind() == reflect.Ptr {
   116  		imm = imm.Elem()
   117  	}
   118  	val := imm.FieldByNameFunc(CaseInsensitive(parameter))
   119  	if !val.IsValid() {
   120  		return fmt.Errorf("unknown parameter :'%s' is not a parameter in odo config", parameter)
   121  	}
   122  
   123  	if val.CanSet() {
   124  		val.Set(reflect.Zero(val.Type()))
   125  		return nil
   126  	}
   127  	return fmt.Errorf("cannot set %s to nil", parameter)
   128  
   129  }
