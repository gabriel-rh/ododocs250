    86  // ConvertLabelsToSelector converts the given labels to selector
    87  // To pass operands such as !=, append a ! prefix to the value.
    88  // For E.g. map[string]string{"app.kubernetes.io/managed-by": "!odo"}
    89  // Using != operators also means that resource will be filtered even if it doesn't have the key.
    90  // So a resource not labelled with key "app.kubernetes.io/managed-by" will also be returned.
    91  func ConvertLabelsToSelector(labels map[string]string) string {
    92  	var selector string
    93  	isFirst := true
    94  	keys := make([]string, 0, len(labels))
    95  	for k := range labels {
    96  		keys = append(keys, k)
    97  	}
    98  	sort.Strings(keys)
    99  	for _, k := range keys {
   100  		v := labels[k]
   101  		if isFirst {
   102  			isFirst = false
   103  			if v == "" {
   104  				selector = selector + fmt.Sprintf("%v", k)
   105  			} else {
   106  				if strings.HasPrefix(v, "!") {
   107  					v = strings.Replace(v, "!", "", -1)
   108  					selector = fmt.Sprintf("%v!=%v", k, v)
   109  				} else {
   110  					selector = fmt.Sprintf("%v=%v", k, v)
   111  				}
   112  			}
   113  		} else {
   114  			if v == "" {
   115  				selector = selector + fmt.Sprintf(",%v", k)
   116  			} else {
   117  				if strings.HasPrefix(v, "!") {
   118  					v = strings.Replace(v, "!", "", -1)
   119  					selector = selector + fmt.Sprintf(",%v!=%v", k, v)
   120  				} else {
   121  					selector = selector + fmt.Sprintf(",%v=%v", k, v)
   122  				}
   123  			}
   124  		}
   125  	}
   126  	return selector

   335  // GetRandomName returns a randomly generated name which can be used for naming odo and/or openshift entities
   336  // prefix: Desired prefix part of the name
   337  // prefixMaxLen: Desired maximum length of prefix part of random name; if -1 is passed, no limit on length will be enforced
   338  // existList: List to verify that the returned name does not already exist
   339  // retries: number of retries to try generating a unique name
   340  // Returns:
   341  //		1. randomname: is prefix-suffix, where:
   342  //				prefix: string passed as prefix or fetched current directory of length same as the passed prefixMaxLen
   343  //				suffix: 4 char random string
   344  //      2. error: if requested number of retries also failed to generate unique name
   345  func GetRandomName(prefix string, prefixMaxLen int, existList []string, retries int) (string, error) {
   346  	prefix = TruncateString(GetDNS1123Name(strings.ToLower(prefix)), prefixMaxLen)
   347  	name := fmt.Sprintf("%s-%s", prefix, GenerateRandomString(4))
   348  
   349  	//Create a map of existing names for efficient iteration to find if the newly generated name is same as any of the already existing ones
   350  	existingNames := make(map[string]bool)
   351  	for _, existingName := range existList {
   352  		existingNames[existingName] = true
   353  	}
   354  
   355  	// check if generated name is already used in the existList
   356  	if _, ok := existingNames[name]; ok {
   357  		prevName := name
   358  		trial := 0
   359  		// keep generating names until generated name is not unique. So, loop terminates when name is unique and hence for condition is false
   360  		for ok {
   361  			trial = trial + 1
   362  			prevName = name
   363  			// Attempt unique name generation from prefix-suffix by concatenating prefix-suffix withrandom string of length 4
   364  			prevName = fmt.Sprintf("%s-%s", prevName, GenerateRandomString(4))
   365  			_, ok = existingNames[prevName]
   366  			if trial >= retries {
   367  				// Avoid infinite loops and fail after passed number of retries
   368  				return "", fmt.Errorf("failed to generate a unique name even after %d retrials", retries)
   369  			}
   370  		}
   371  		// If found to be unique, set name as generated name
   372  		name = prevName
   373  	}
   374  	// return name
   375  	return name, nil
   376  }

   379  func GetDNS1123Name(str string) string {
   380  	nonAllowedCharsRegex := regexp.MustCompile(`[^a-zA-Z0-9_-]+`)
   381  	withReplacedChars := strings.Replace(
   382  		nonAllowedCharsRegex.ReplaceAllString(str, "-"),

   387  		name = strings.ToLower(removeNonAlphaSuffix(removeNonAlphaPrefix("x" + withReplacedChars)))
   388  	}
   389  	return name

   652  func IsGlobExpMatch(strToMatch string, globExps []string) (bool, error) {

   670  		if matched {

   777  func HTTPGetRequest(request HTTPRequestParams, cacheFor int) ([]byte, error) {
   778  	// Build http request
   779  	req, err := http.NewRequest("GET", request.URL, nil)
   780  	if err != nil {
   781  		return nil, err
   782  	}

   784  		bearer := "Bearer " + request.Token

   789  		Transport: &http.Transport{
   790  			ResponseHeaderTimeout: ResponseHeaderTimeout,

   818  			klog.V(4).Info("Response won't be cached.")
   819  		}
   820  	}
   821

   824  		return nil, err
   825  	}
   826  	defer resp.Body.Close()

   829  		klog.V(4).Infof("Cached response used.")

   832  	// We have a non 1xx / 2xx status, return an error

   836  
   837  	// Process http response
   838  	bytes, err := ioutil.ReadAll(resp.Body)
   839  	if err != nil {
   840  		return nil, err
   841  	}
   842  
   843  	return bytes, err
   844  }

   953  // Unzip will decompress a zip archive, moving specified files and folders
   954  // within the zip file (parameter 1) to an output directory (parameter 2)
   955  // Source: https://golangcode.com/unzip-files-in-go/
   956  // pathToUnzip (parameter 3) is the path within the zip folder to extract
   957  func Unzip(src, dest, pathToUnzip string) ([]string, error) {
   958  	var filenames []string
   959  
   960  	r, err := zip.OpenReader(src)
   961  	if err != nil {
   962  		return filenames, err
   963  	}
   964  	defer r.Close()
   965  
   966  	// change path separator to correct character
   967  	pathToUnzip = filepath.FromSlash(pathToUnzip)
   968  
   969  	// removes first slash of pathToUnzip if present
   970  	pathToUnzip = strings.TrimPrefix(pathToUnzip, string(os.PathSeparator))
   971  
   972  	for _, f := range r.File {
   973  		// Store filename/path for returning and using later on
   974  		index := strings.Index(f.Name, "/")
   975  		filename := filepath.FromSlash(f.Name[index+1:])
   976  		if filename == "" {
   977  			continue
   978  		}
   979  
   980  		// if subDir has a pattern
   981  		match, err := filepath.Match(pathToUnzip, filename)
   982  		if err != nil {
   983  			return filenames, err
   984  		}
   985  
   986  		// destination filepath before trim
   987  		fpath := filepath.Join(dest, filename)
   988  
   989  		// used for pattern matching
   990  		fpathDir := filepath.Dir(fpath)
   991  
   992  		// check for prefix or match
   993  		if strings.HasPrefix(filename, pathToUnzip) {
   994  			filename = strings.TrimPrefix(filename, pathToUnzip)
   995  		} else if !strings.HasPrefix(filename, pathToUnzip) && !match && !sliceContainsString(fpathDir, filenames) {
   996  			continue
   997  		}
   998  		// adds trailing slash to destination if needed as filepath.Join removes it
   999  		if (len(filename) == 1 && os.IsPathSeparator(filename[0])) || filename == "" {
  1000  			fpath = dest + string(os.PathSeparator)
  1001  		} else {
  1002  			fpath = filepath.Join(dest, filename)
  1003  		}
  1004  		// Check for ZipSlip. More Info: http://bit.ly/2MsjAWE
  1005  		if !strings.HasPrefix(fpath, filepath.Clean(dest)+string(os.PathSeparator)) {
  1006  			return filenames, fmt.Errorf("%s: illegal file path", fpath)
  1007  		}
  1008  
  1009  		filenames = append(filenames, fpath)
  1010  
  1011  		if f.FileInfo().IsDir() {
  1012  			// Make Folder
  1013  			if err = os.MkdirAll(fpath, os.ModePerm); err != nil {
  1014  				return filenames, err
  1015  			}
  1016  			continue
  1017  		}
  1018  
  1019  		// Make File
  1020  		if err = os.MkdirAll(filepath.Dir(fpath), os.ModePerm); err != nil {
  1021  			return filenames, err
  1022  		}
  1023  
  1024  		outFile, err := os.OpenFile(fpath, os.O_WRONLY|os.O_CREATE|os.O_TRUNC, f.Mode())
  1025  		if err != nil {
  1026  			return filenames, err
  1027  		}
  1028  
  1029  		rc, err := f.Open()
  1030  		if err != nil {
  1031  			return filenames, err
  1032  		}
  1033  
  1034  		// limit the number of bytes copied from a file
  1035  		// This is set to the limit of file size in Github
  1036  		// which is 100MB
  1037  		limited := io.LimitReader(rc, 100*1024*1024)
  1038  
  1039  		_, err = io.Copy(outFile, limited)
  1040  
  1041  		// Close the file without defer to close before next iteration of loop
  1042  		outFile.Close()
  1043  		rc.Close()
  1044  
  1045  		if err != nil {
  1046  			return filenames, err
  1047  		}
  1048  	}
  1049  	return filenames, nil
  1050  }
