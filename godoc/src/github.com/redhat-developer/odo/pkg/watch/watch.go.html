    58  // addRecursiveWatch handles adding watches recursively for the path provided
    59  // and its subdirectories.  If a non-directory is specified, this call is a no-op.
    60  // Files matching glob pattern defined in ignores will be ignored.
    61  // Taken from https://github.com/openshift/origin/blob/85eb37b34f0657631592356d020cef5a58470f8e/pkg/util/fsnotification/fsnotification.go
    62  // path is the path of the file or the directory
    63  // ignores contains the glob rules for matching
    64  func addRecursiveWatch(watcher *fsnotify.Watcher, path string, ignores []string) error {
    65  
    66  	file, err := os.Stat(path)
    67  	if err != nil {
    68  		if os.IsNotExist(err) {
    69  			return nil
    70  		}
    71  		return fmt.Errorf("error introspecting path %s: %v", path, err)
    72  	}
    73  
    74  	mode := file.Mode()
    75  	if mode.IsRegular() {
    76  		matched, e := util.IsGlobExpMatch(path, ignores)
    77  		if e != nil {
    78  			return errors.Wrapf(e, "unable to watcher on %s", path)
    79  		}
    80  		if !matched {
    81  			klog.V(4).Infof("adding watch on path %s", path)
    82  
    83  			// checking if the file exits before adding the watcher to it
    84  			if !util.CheckPathExists(path) {
    85  				return nil
    86  			}
    87  
    88  			err = watcher.Add(path)
    89  			if err != nil {
    90  				klog.V(4).Infof("error adding watcher for path %s: %v", path, err)
    91  			}
    92  			return nil
    93  		}
    94  	}
    95  
    96  	folders := []string{}
    97  	err = filepath.Walk(path, func(newPath string, info os.FileInfo, err error) error {
    98  		if err != nil {
    99  			// Ignore the error if it's a 'path does not exist' error, no need to walk a non-existent path
   100  			if !util.CheckPathExists(newPath) {
   101  				klog.V(4).Infof("Walk func received an error for path %s, but the path doesn't exist so this is likely not an error. err: %v", path, err)
   102  				return nil
   103  			}
   104  			return errors.Wrapf(err, "unable to walk path: %s", newPath)
   105  		}
   106  
   107  		if info.IsDir() {
   108  			// If the current directory matches any of the ignore patterns, ignore them so that their contents are also not ignored
   109  			matched, err := util.IsGlobExpMatch(newPath, ignores)
   110  			if err != nil {
   111  				return errors.Wrapf(err, "unable to addRecursiveWatch on %s", newPath)
   112  			}
   113  			if matched {
   114  				klog.V(4).Infof("ignoring watch on path %s", newPath)
   115  				return filepath.SkipDir
   116  			}
   117  			// Append the folder we just walked on
   118  			folders = append(folders, newPath)
   119  		}
   120  		return nil
   121  	})
   122  	if err != nil {
   123  		return err
   124  	}
   125  	for _, folder := range folders {
   126  
   127  		if matched, _ := util.IsGlobExpMatch(folder, ignores); matched {
   128  			klog.V(4).Infof("ignoring watch for %s", folder)
   129  			continue
   130  		}
   131  
   132  		// checking if the file exits before adding the watcher to it
   133  		if !util.CheckPathExists(path) {
   134  			continue
   135  		}
   136  
   137  		klog.V(4).Infof("adding watch on path %s", folder)
   138  		err = watcher.Add(folder)
   139  		if err != nil {
   140  			// Linux "no space left on device" issues are usually resolved via
   141  			// $ sudo sysctl fs.inotify.max_user_watches=65536
   142  			// BSD / OSX: "too many open files" issues are ussualy resolved via
   143  			// $ sysctl variables "kern.maxfiles" and "kern.maxfilesperproc",
   144  			klog.V(4).Infof("error adding watcher for path %s: %v", folder, err)
   145  		}
   146  	}
   147  	return nil
   148  }

   153  // WatchAndPush watches path, if something changes in  that path it calls PushLocal
   154  // ignores .git/* by default
   155  // inspired by https://github.com/openshift/origin/blob/e785f76194c57bd0e1674c2f2776333e1e0e4e78/pkg/oc/cli/cmd/rsync/rsync.go#L257
   156  // Parameters:
   157  //	client: occlient instance
   158  //	out: io Writer instance
   159  // 	parameters: WatchParameters
   160  func WatchAndPush(client kclient.ClientInterface, out io.Writer, parameters WatchParameters) error {
   161  	// ToDo reduce number of parameters to this function by extracting them into a struct and passing the struct instance instead of passing each of them separately
   162  	// delayInterval int
   163  	klog.V(4).Infof("starting WatchAndPush, path: %s, component: %s, ignores %s", parameters.Path, parameters.ComponentName, parameters.FileIgnores)
   164  
   165  	// these variables must be accessed while holding the changeLock
   166  	// mutex as they are shared between goroutines to communicate
   167  	// sync state/events.
   168  	var (
   169  		changeLock   sync.Mutex
   170  		dirty        bool
   171  		lastChange   time.Time
   172  		watchError   error
   173  		deletedPaths []string
   174  		changedFiles []string
   175  	)
   176  
   177  	watcher, err := fsnotify.NewWatcher()
   178  	if err != nil {
   179  		return fmt.Errorf("error setting up filesystem watcher: %v", err)
   180  	}
   181  	defer watcher.Close()
   182  	defer close(parameters.ExtChan)
   183  
   184  	// This goroutine listens for either file change events from fsnotify, fs errors, or a terminate signal
   185  	// The results are stored in the variables defined in the var( ... ) block above
   186  	go func() {
   187  		for {
   188  			select {
   189  			case extMsg := <-parameters.ExtChan:
   190  				if extMsg {
   191  					changeLock.Lock()
   192  					watchError = ErrUserRequestedWatchExit
   193  					changeLock.Unlock()
   194  				}
   195  			case event := <-watcher.Events:
   196  				changeLock.Lock()
   197  				klog.V(4).Infof("filesystem watch event: %s", event)
   198  
   199  				isIgnoreEvent := shouldIgnoreEvent(event)
   200  
   201  				// add file name to changedFiles only once
   202  				alreadyInChangedFiles := false
   203  				for _, cfile := range changedFiles {
   204  					if cfile == event.Name {
   205  						alreadyInChangedFiles = true
   206  						break
   207  					}
   208  				}
   209  
   210  				// Filter out anything in ignores list from the list of changed files
   211  				// This is important in spite of not watching the
   212  				// ignores paths because, when a directory that is ignored, is deleted,
   213  				// because its parent is watched, the fsnotify automatically raises an event
   214  				// for it.
   215  				matched, globErr := util.IsGlobExpMatch(event.Name, parameters.FileIgnores)
   216  				klog.V(4).Infof("Matching %s with %s. Matched %v, err: %v", event.Name, parameters.FileIgnores, matched, globErr)
   217  				if globErr != nil {
   218  					watchError = errors.Wrap(globErr, "unable to watch changes")
   219  				}
   220  				if !alreadyInChangedFiles && !matched && !isIgnoreEvent {
   221  					// Append the new file change event to changedFiles if and only if the event is not a file remove event
   222  					if event.Op&fsnotify.Remove != fsnotify.Remove {
   223  						changedFiles = append(changedFiles, event.Name)
   224  					}
   225  				}
   226  
   227  				lastChange = time.Now()
   228  				dirty = true
   229  				// Rename operation triggers RENAME event on old path + CREATE event for renamed path so delete old path in case of rename
   230  				// Also weirdly, fsnotify raises a RENAME event for deletion of files/folders with space in their name so even that should be handled here
   231  				if event.Op&fsnotify.Remove == fsnotify.Remove || event.Op&fsnotify.Rename == fsnotify.Rename {
   232  					// On remove/rename, stop watching the resource
   233  					if e := watcher.Remove(event.Name); e != nil {
   234  						klog.V(4).Infof("error removing watch for %s: %v", event.Name, e)
   235  					}
   236  					// Append the file to list of deleted files
   237  					// When a file/folder is deleted, it raises 2 events:
   238  					//	a. RENAME with event.Name empty
   239  					//	b. REMOVE with event.Name as file name
   240  					if !alreadyInChangedFiles && !matched && event.Name != "" {
   241  						deletedPaths = append(deletedPaths, event.Name)
   242  					}
   243  				} else {
   244  					// On other ops, recursively watch the resource (if applicable)
   245  					if e := addRecursiveWatch(watcher, event.Name, parameters.FileIgnores); e != nil && watchError == nil {
   246  						klog.V(4).Infof("Error occurred in addRecursiveWatch, setting watchError to %v", e)
   247  						watchError = e
   248  					}
   249  				}
   250  				changeLock.Unlock()
   251  			case watchErr := <-watcher.Errors:
   252  				changeLock.Lock()
   253  				watchError = fmt.Errorf("error watching filesystem for changes: %v", watchErr)
   254  				changeLock.Unlock()
   255  			}
   256  		}
   257  	}()
   258  	// adding watch on the root folder and the sub folders recursively
   259  	// so directory and the path in addRecursiveWatch() are the same
   260  	err = addRecursiveWatch(watcher, parameters.Path, parameters.FileIgnores)
   261  	if err != nil {
   262  		return fmt.Errorf("error watching source path %s: %v", parameters.Path, err)
   263  	}
   264  
   265  	// Only signal start of watch if invoker is interested
   266  	if parameters.StartChan != nil {
   267  		parameters.StartChan <- true
   268  	}
   269  
   270  	var ticker *time.Ticker
   271  	delay := time.Duration(parameters.PushDiffDelay) * time.Second
   272  
   273  	// don't create a ticker if delay is 0 as it will trigger panic
   274  	if delay != 0 {
   275  		ticker = time.NewTicker(delay)
   276  		defer ticker.Stop()
   277  	}
   278  	showWaitingMessage := true
   279  
   280  	hasFirstSuccessfulPushOccurred := false
   281  
   282  	// This for{} loop waits for filesystem changes that are signaled by the above goroutine;
   283  	// - 'dirty' is used by the goroutine to indicate that at least one change has occurred
   284  	for {
   285  		changeLock.Lock()
   286  		if watchError != nil {
   287  			klog.V(4).Infof("Ending watch for {} loop with error %v\n", watchError)
   288  			return watchError
   289  		}
   290  		if showWaitingMessage {
   291  			if parameters.EnvSpecificInfo != nil && parameters.EnvSpecificInfo.GetRunMode() == envinfo.Debug {
   292  				fmt.Fprintf(out, "Component is running in debug mode\nPlease start port-forwarding in a different terminal\n")
   293  			}
   294  			fmt.Fprintf(out, "Waiting for something to change in %s\n", parameters.Path)
   295  			showWaitingMessage = false
   296  		}
   297  		// if a change happened more than 'delay' seconds ago, sync it now.
   298  		// if a change happened less than 'delay' seconds ago, sleep for 'delay' seconds
   299  		// and see if more changes happen, we don't want to sync when
   300  		// the filesystem is in the middle of changing due to a massive
   301  		// set of changes (such as a local build in progress).
   302  		if dirty && time.Now().After(lastChange.Add(delay)) {
   303  
   304  			deletedPaths = removeDuplicates(deletedPaths)
   305  
   306  			for _, file := range removeDuplicates(append(changedFiles, deletedPaths...)) {
   307  				fmt.Fprintf(out, "File %s changed\n", file)
   308  			}
   309  			if len(changedFiles) > 0 || len(deletedPaths) > 0 {
   310  				fmt.Fprintf(out, "Pushing files...\n")
   311  				fileInfo, err := os.Stat(parameters.Path)
   312  				if err != nil {
   313  					return errors.Wrapf(err, "%s: file doesn't exist", parameters.Path)
   314  				}
   315  				if fileInfo.IsDir() {
   316  					klog.V(4).Infof("Copying files %s to pod", changedFiles)
   317  
   318  					if parameters.DevfileWatchHandler != nil {
   319  						pushParams := common.PushParameters{
   320  							Path:                     parameters.Path,
   321  							WatchFiles:               changedFiles,
   322  							WatchDeletedFiles:        deletedPaths,
   323  							IgnoredFiles:             parameters.FileIgnores,
   324  							ForceBuild:               false,
   325  							DevfileBuildCmd:          parameters.DevfileBuildCmd,
   326  							DevfileRunCmd:            parameters.DevfileRunCmd,
   327  							DevfileDebugCmd:          parameters.DevfileDebugCmd,
   328  							DevfileScanIndexForWatch: !hasFirstSuccessfulPushOccurred,
   329  							EnvSpecificInfo:          *parameters.EnvSpecificInfo,
   330  							Debug:                    parameters.EnvSpecificInfo.GetRunMode() == envinfo.Debug,
   331  							DebugPort:                parameters.EnvSpecificInfo.GetDebugPort(),
   332  						}
   333  
   334  						err = parameters.DevfileWatchHandler(pushParams, parameters)
   335  
   336  					} else {
   337  						err = parameters.WatchHandler(client, parameters.ComponentName, parameters.ApplicationName, parameters.Path, out,
   338  							changedFiles, deletedPaths, false, parameters.FileIgnores, parameters.Show)
   339  					}
   340  
   341  				} else {
   342  					pathDir := filepath.Dir(parameters.Path)
   343  					klog.V(4).Infof("Copying file %s to pod", parameters.Path)
   344  
   345  					if parameters.DevfileWatchHandler != nil {
   346  						pushParams := common.PushParameters{
   347  							Path:                     pathDir,
   348  							WatchFiles:               changedFiles,
   349  							WatchDeletedFiles:        deletedPaths,
   350  							IgnoredFiles:             parameters.FileIgnores,
   351  							ForceBuild:               false,
   352  							DevfileBuildCmd:          parameters.DevfileBuildCmd,
   353  							DevfileRunCmd:            parameters.DevfileRunCmd,
   354  							DevfileDebugCmd:          parameters.DevfileDebugCmd,
   355  							DevfileScanIndexForWatch: !hasFirstSuccessfulPushOccurred,
   356  							EnvSpecificInfo:          *parameters.EnvSpecificInfo,
   357  							Debug:                    parameters.EnvSpecificInfo.GetRunMode() == envinfo.Debug,
   358  							DebugPort:                parameters.EnvSpecificInfo.GetDebugPort(),
   359  						}
   360  
   361  						err = parameters.DevfileWatchHandler(pushParams, parameters)
   362  					} else {
   363  						err = parameters.WatchHandler(client, parameters.ComponentName, parameters.ApplicationName, pathDir, out,
   364  							[]string{parameters.Path}, deletedPaths, false, parameters.FileIgnores, parameters.Show)
   365  					}
   366  
   367  				}
   368  				if err != nil {
   369  
   370  					// Log and output, but intentionally not exiting on error here.
   371  					// We don't want to break watch when push failed, it might be fixed with the next change.
   372  					klog.V(4).Infof("Error from Push: %v", err)
   373  					fmt.Fprintf(out, "%s - %s\n\n", PushErrorString, err.Error())
   374  				} else {
   375  					hasFirstSuccessfulPushOccurred = true
   376  				}
   377  				dirty = false
   378  				showWaitingMessage = true
   379  				// Reset changed files
   380  				changedFiles = []string{}
   381  				// Reset deleted Paths
   382  				deletedPaths = []string{}
   383  			}
   384  		}
   385  		changeLock.Unlock()
   386  		if ticker != nil {
   387  			<-ticker.C
   388  		}
   389  	}
   390  }

   392  func shouldIgnoreEvent(event fsnotify.Event) (ignoreEvent bool) {
   393  	if !(event.Op&fsnotify.Remove == fsnotify.Remove || event.Op&fsnotify.Rename == fsnotify.Rename) {
   394  		stat, err := os.Lstat(event.Name)
   395  		if err != nil {
   396  			// Some of the editors like vim and gedit, generate temporary buffer files during update to the file and deletes it soon after exiting from the editor
   397  			// So, its better to log the error rather than feeding it to error handler via `watchError = errors.Wrap(err, "unable to watch changes")`,
   398  			// which will terminate the watch
   399  			klog.V(4).Infof("Failed getting details of the changed file %s. Ignoring the change", event.Name)
   400  		}
   401  		// Some of the editors generate temporary buffer files during update to the file and deletes it soon after exiting from the editor
   402  		// So, its better to log the error rather than feeding it to error handler via `watchError = errors.Wrap(err, "unable to watch changes")`,
   403  		// which will terminate the watch
   404  		if stat == nil {
   405  			klog.V(4).Infof("Ignoring event for file %s as details about the file couldn't be fetched", event.Name)
   406  			ignoreEvent = true
   407  		}
   408  
   409  		// In windows, every new file created under a sub-directory of the watched directory, raises 2 events:
   410  		// 1. Write event for the directory under which the file was created
   411  		// 2. Create event for the file that was created
   412  		// Ignore 1 to avoid duplicate events.
   413  		if ignoreEvent || (stat.IsDir() && event.Op&fsnotify.Write == fsnotify.Write) {
   414  			ignoreEvent = true
   415  		}
   416  	}
   417  	return ignoreEvent
   418  }

   426  func removeDuplicates(input []string) []string {
   427  	valueMap := map[string]string{}
   428  	for _, str := range input {
   429  		valueMap[str] = str
   430  	}
   431  
   432  	result := []string{}
   433  	for str := range valueMap {
   434  		result = append(result, str)
   435  	}
   436  	return result
   437  }
