    26  // setUpF8AnalyticsComponentSrc sets up a mock analytics component source base for observing changes to source files.
    27  // Parameters:
    28  //	componentName: Name of the source directory
    29  //	requiredFilePaths: list of required sources, their description like whether regularfile/directory, parent directory path of source and desired modification type like update/create/delete/append
    30  // Returns:
    31  //	absolute base path of source code
    32  //	directory structure containing mappings from desired relative paths to their respective absolute path containing FileProperties.
    33  func setUpF8AnalyticsComponentSrc(componentName string, requiredFilePaths []testingutil.FileProperties) (string, map[string]testingutil.FileProperties, error) {
    34  
    35  	// retVal is mappings from desired relative paths to their respective absolute path containing FileProperties.
    36  	// This is required because ioutil#TempFile and ioutil#TempFolder creates paths with random numeric suffixes.
    37  	// So, to be able to refer to the file/folder at any later point in time the created paths returned by ioutil#TempFile or ioutil#TempFolder will need to be saved.
    38  	retVal := make(map[string]testingutil.FileProperties)
    39  	dirTreeMappings := make(map[string]string)
    40  	basePath := ""
    41  
    42  	// Create temporary directory for mock component source code
    43  	srcPath, err := testingutil.TempMkdir(basePath, componentName)
    44  	if err != nil {
    45  		return "", retVal, errors.Wrapf(err, "failed to create dir %s under %s", componentName, basePath)
    46  	}
    47  	dirTreeMappings[componentName] = srcPath
    48  
    49  	// For each of the passed(desired) files/folders under component source
    50  	for _, fileProperties := range requiredFilePaths {
    51  
    52  		// get relative path using file parent and file name passed
    53  		relativePath := filepath.Join(fileProperties.FileParent, fileProperties.FilePath)
    54  
    55  		// get its absolute path using the mappings preserved from previous creates
    56  		if realParentPath, ok := dirTreeMappings[fileProperties.FileParent]; ok {
    57  			// real path for the intended file operation is obtained from previously maintained directory tree mappings by joining parent path and file name
    58  			realPath := filepath.Join(realParentPath, fileProperties.FilePath)
    59  			// Preserve the new paths for further reference
    60  			fileProperties.FilePath = filepath.Base(realPath)
    61  			fileProperties.FileParent, _ = filepath.Rel(srcPath, filepath.Dir(realPath))
    62  			fileProperties.FileParent = filepath.FromSlash(fileProperties.FileParent)
    63  		}
    64  
    65  		// Perform mock operation as requested by the parameter
    66  		newPath, err := testingutil.SimulateFileModifications(srcPath, fileProperties)
    67  		dirTreeMappings[relativePath] = newPath
    68  		if err != nil {
    69  			return "", retVal, errors.Wrapf(err, "unable to setup test env")
    70  		}
    71  
    72  		fileProperties.FilePath = filepath.Base(newPath)
    73  		fileProperties.FileParent = filepath.Dir(newPath)
    74  		retVal[relativePath] = fileProperties
    75  	}
    76  
    77  	// Return base source path and directory tree mappings
    78  	return srcPath, retVal, nil
    79  }

   114  // Mocks the devFile push function that's called when odo watch pushes to a component
   115  func mockDevFilePush(parameters common.PushParameters, _ WatchParameters) error {
   116  	muLock.Lock()
   117  	defer muLock.Unlock()
   118  	if parameters.Show != mockPush.show || parameters.Debug != mockPush.isDebug || parameters.DebugPort != mockPush.debugPort {
   119  		fmt.Printf("some of the push parameters are different, wanted: %v, got: %v", mockPush, []string{
   120  			"show:" + strconv.FormatBool(parameters.Show),
   121  			"debug:" + strconv.FormatBool(parameters.Debug),
   122  			"debugPort:" + strconv.Itoa(parameters.DebugPort),
   123  		})
   124  		os.Exit(1)
   125  	}
   126  
   127  	if parameters.DevfileBuildCmd != mockPush.devfileBuildCmd || parameters.DevfileRunCmd != mockPush.devfileRunCmd || parameters.DevfileDebugCmd != mockPush.devfileDebugCmd {
   128  		fmt.Printf("some of the push parameters are different, wanted: %v, got: %v", mockPush, []string{
   129  			"devfileBuildCmd:" + parameters.DevfileBuildCmd,
   130  			"devfileRunCmd:" + parameters.DevfileRunCmd,
   131  			"devfileDebugCmd:" + parameters.DevfileDebugCmd,
   132  		})
   133  		os.Exit(1)
   134  	}
   135  
   136  	return commonChecks(parameters.Path, parameters.WatchFiles, parameters.WatchDeletedFiles, parameters.IgnoredFiles)

   139  // commonChecks is the common checker for both the push handlers
   140  func commonChecks(path string, files []string, delFiles []string, globExps []string) error {
   141  	sort.Strings(globExps)
   142  	mockPush.globExps = util.GetAbsGlobExps(path, mockPush.globExps)
   143  	sort.Strings(mockPush.globExps)
   144  	if !reflect.DeepEqual(globExps, mockPush.globExps) {
   145  		fmt.Printf("some of the push parameters are different, wanted: %v, got: %v", mockPush.globExps, globExps)
   146  		os.Exit(1)
   147  	}
   148  
   149  	for _, expChangedFile := range ExpectedChangedFiles {
   150  		found := false
   151  		// Verify every file in expected file changes to be actually observed to be changed
   152  		// If found exactly same or different, return from PushLocal and signal exit for watch so that the watch terminates gracefully
   153  		for _, gotChangedFile := range files {
   154  			wantedFileDetail := CompDirStructure[filepath.FromSlash(expChangedFile)]
   155  			if filepath.Join(wantedFileDetail.FileParent, wantedFileDetail.FilePath) == gotChangedFile {
   156  				found = true
   157  			}
   158  		}
   159  		if !found {
   160  			ExtChan <- true
   161  			fmt.Printf("received %+v which is not same as expected list %+v", files, strings.Join(ExpectedChangedFiles, ","))
   162  			os.Exit(1)
   163  		}
   164  	}
   165  
   166  	for _, deletedFile := range DeleteFiles {
   167  		found := false
   168  		// Verify every file in expected deleted file changes to be actually observed to be changed
   169  		// If found exactly same or different, return from PushLocal and signal exit for watch so that the watch terminates gracefully
   170  		for _, gotChangedFile := range delFiles {
   171  			wantedFileDetail := CompDirStructure[filepath.FromSlash(deletedFile)]
   172  			if filepath.Join(wantedFileDetail.FileParent, wantedFileDetail.FilePath) == filepath.Join(wantedFileDetail.FileParent, filepath.Base(gotChangedFile)) {
   173  				found = true
   174  			}
   175  		}
   176  		if !found {
   177  			ExtChan <- true
   178  			fmt.Printf("received deleted files: %+v which is not same as expected list %+v", delFiles, strings.Join(DeleteFiles, ","))
   179  			os.Exit(1)
   180  		}
   181  	}
   182  
   183  	ExtChan <- true
   184  	return nil
   185  }
