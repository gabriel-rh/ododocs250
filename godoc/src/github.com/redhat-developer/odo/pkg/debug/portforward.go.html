    40  // ready Chan is used to signal failure to the channel receiver

    42  	var pod *corev1.Pod
    43  	var conf *rest.Config
    44  	var err error
    45  
    46  	if f.kClient != nil && isDevfile {
    47  		conf, err = f.kClient.GetConfig().ClientConfig()
    48  		if err != nil {
    49  			return err
    50  		}
    51  
    52  		pod, err = f.kClient.GetOnePod(f.componentName, f.appName)
    53  		if err != nil {
    54  			return err
    55  		}
    56  	} else {
    57  		conf = f.kClient.GetClientConfig()
    58  	}
    59  
    60  	if pod.Status.Phase != corev1.PodRunning {
    61  		return fmt.Errorf("unable to forward port because pod is not running. Current status=%v", pod.Status.Phase)
    62  	}
    63  
    64  	transport, upgrader, err := spdy.RoundTripperFor(conf)
    65  	if err != nil {
    66  		return err
    67  	}
    68  
    69  	req := f.kClient.GeneratePortForwardReq(pod.Name)
    70  
    71  	dialer := spdy.NewDialer(upgrader, &http.Client{Transport: transport}, "POST", req.URL())
    72  	fw, err := portforward.New(dialer, []string{portPair}, stopChan, readyChan, f.Out, f.ErrOut)
    73  	if err != nil {
    74  		return err
    75  	}
    76  	log.Info("Started port forwarding at ports -", portPair)
