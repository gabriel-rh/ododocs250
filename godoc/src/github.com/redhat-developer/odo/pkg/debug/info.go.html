    34  func GetDebugInfoFilePath(componentName, appName string, projectName string) string {
    35  	tempDir := os.TempDir()
    36  	debugFileSuffix := "odo-debug.json"
    37  	var arr []string
    38  	if appName == "" {
    39  		arr = []string{projectName, componentName, debugFileSuffix}
    40  	} else {
    41  		arr = []string{projectName, appName, componentName, debugFileSuffix}
    42  	}
    43  	debugFileName := strings.Join(arr, "-")
    44  	return filepath.Join(tempDir, debugFileName)

    51  // createDebugInfoFile creates a file in the temp directory with information regarding the debugging session of a component
    52  func createDebugInfoFile(f *DefaultPortForwarder, portPair string, fs filesystem.Filesystem) error {
    53  	portPairs := strings.Split(portPair, ":")
    54  	if len(portPairs) != 2 {
    55  		return errors.New("port pair should be of the format localPort:RemotePort")
    56  	}
    57  
    58  	localPort, err := strconv.Atoi(portPairs[0])
    59  	if err != nil {
    60  		return errors.New("local port should be a int")
    61  	}
    62  	remotePort, err := strconv.Atoi(portPairs[1])
    63  	if err != nil {
    64  		return errors.New("remote port should be a int")
    65  	}
    66  
    67  	debugFile := Info{
    68  		TypeMeta: metav1.TypeMeta{
    69  			Kind:       "OdoDebugInfo",
    70  			APIVersion: "v1",
    71  		},
    72  		ObjectMeta: metav1.ObjectMeta{
    73  			Name:      f.componentName,
    74  			Namespace: f.projectName,
    75  		},
    76  		Spec: InfoSpec{
    77  			App:            f.appName,
    78  			DebugProcessID: os.Getpid(),
    79  			RemotePort:     remotePort,
    80  			LocalPort:      localPort,
    81  		},
    82  	}
    83  	odoDebugPathData, err := json.Marshal(debugFile)
    84  	if err != nil {
    85  		return errors.New("error marshalling json data")
    86  	}
    87  
    88  	// writes the data to the debug info file
    89  	file, err := fs.OpenFile(GetDebugInfoFilePath(f.componentName, f.appName, f.projectName), os.O_WRONLY|os.O_CREATE|os.O_TRUNC, 0666)
    90  	if err != nil {
    91  		return err
    92  	}
    93  	defer file.Close() // #nosec G307
    94  
    95  	_, err = file.Write(odoDebugPathData)
    96  	if err != nil {
    97  		return err
    98  	}
    99  	return nil
   100  }

   107  // getInfo gets information regarding the debugging session of the component
   108  // returns the OdoDebugFile from the debug info file
   109  // returns true if debugging is running else false
   110  func getInfo(f *DefaultPortForwarder, fs filesystem.Filesystem) (Info, bool) {
   111  	// gets the debug info file path and reads/unmarshalls it
   112  	debugInfoFilePath := GetDebugInfoFilePath(f.componentName, f.appName, f.projectName)
   113  	readFile, err := fs.ReadFile(debugInfoFilePath)
   114  	if err != nil {
   115  		klog.V(4).Infof("the debug %v is not present", debugInfoFilePath)
   116  		return Info{}, false
   117  	}
   118  
   119  	var info Info
   120  	err = json.Unmarshal(readFile, &info)
   121  	if err != nil {
   122  		klog.V(4).Infof("couldn't unmarshal the debug file %v", debugInfoFilePath)
   123  		return Info{}, false
   124  	}
   125  
   126  	// get the debug process id and send a signal 0 to check if it's alive or not
   127  	// according to https://golang.org/pkg/os/#FindProcess
   128  	// On Unix systems, FindProcess always succeeds and returns a Process for the given pid, regardless of whether the process exists.
   129  	// thus this step will pass on Unix systems and so for those systems and some others supporting signals
   130  	// we check if the process is alive or not by sending a signal 0 to the process
   131  	processInfo, err := os.FindProcess(info.Spec.DebugProcessID)
   132  	if err != nil || processInfo == nil {
   133  		klog.V(4).Infof("error getting the process info for pid %v", info.Spec.DebugProcessID)
   134  		return Info{}, false
   135  	}
   136  
   137  	// signal is not available on windows so we skip this step for windows
   138  	if runtime.GOOS != "windows" {
   139  		err = processInfo.Signal(syscall.Signal(0))
   140  		if err != nil {
   141  			klog.V(4).Infof("error sending signal 0 to pid %v, cause: %v", info.Spec.DebugProcessID, err)
   142  			return Info{}, false
   143  		}
   144  	}
   145  
   146  	// gets the debug local port and tries to listen on it
   147  	// if error doesn't occur the debug port was free and thus no debug process was using the port
   148  	addressLook := "localhost:" + strconv.Itoa(info.Spec.LocalPort)
   149  	listener, err := net.Listen("tcp", addressLook)
   150  	if err == nil {
   151  		klog.V(4).Infof("the debug port %v is free, thus debug is not running", info.Spec.LocalPort)
   152  		err = listener.Close()
   153  		if err != nil {
   154  			klog.V(4).Infof("error occurred while closing the listener, cause :%v", err)
   155  		}
   156  		return Info{}, false
   157  	}
   158  
   159  	return info, true
   160  }
