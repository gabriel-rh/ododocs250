   150  // WaitForDeploymentRollout waits for deployment to finish rollout. Returns the state of the deployment after rollout.
   151  func (c *Client) WaitForDeploymentRollout(deploymentName string) (*appsv1.Deployment, error) {
   152  	klog.V(3).Infof("Waiting for %s deployment rollout", deploymentName)
   153  	s := log.Spinner("Waiting for component to start")
   154  	defer s.End(false)
   155  
   156  	w, err := c.KubeClient.AppsV1().Deployments(c.Namespace).Watch(context.TODO(), metav1.ListOptions{FieldSelector: "metadata.name=" + deploymentName})
   157  	if err != nil {
   158  		return nil, errors.Wrapf(err, "unable to watch deployment")
   159  	}
   160  	defer w.Stop()
   161  
   162  	success := make(chan *appsv1.Deployment)
   163  	failure := make(chan error)
   164  
   165  	// Collect all the events in a separate go routine
   166  	failedEvents := make(map[string]corev1.Event)
   167  	quit := make(chan int)
   168  	go c.CollectEvents("", failedEvents, s, quit)
   169  
   170  	go func() {
   171  		defer close(success)
   172  		defer close(failure)
   173  
   174  		for {
   175  			val, ok := <-w.ResultChan()
   176  			if !ok {
   177  				failure <- errors.New("watch channel was closed")
   178  				return
   179  			}
   180  			//based on https://github.com/kubernetes/kubectl/blob/9a3954bf653c874c8af6f855f2c754a8e1a44b9e/pkg/polymorphichelpers/rollout_status.go#L66-L91
   181  			if deployment, ok := val.Object.(*appsv1.Deployment); ok {
   182  				for _, cond := range deployment.Status.Conditions {
   183  					// using this just for debugging message, so ignoring error on purpose
   184  					jsonCond, _ := json.Marshal(cond)
   185  					klog.V(3).Infof("Deployment Condition: %s", string(jsonCond))
   186  				}
   187  				if deployment.Generation <= deployment.Status.ObservedGeneration {
   188  					cond := getDeploymentCondition(deployment.Status, appsv1.DeploymentProgressing)
   189  					if cond != nil && cond.Reason == timedOutReason {
   190  						failure <- fmt.Errorf("deployment %q exceeded its progress deadline", deployment.Name)
   191  					} else if deployment.Spec.Replicas != nil && deployment.Status.UpdatedReplicas < *deployment.Spec.Replicas {
   192  						klog.V(3).Infof("Waiting for deployment %q rollout to finish: %d out of %d new replicas have been updated...\n", deployment.Name, deployment.Status.UpdatedReplicas, *deployment.Spec.Replicas)
   193  					} else if deployment.Status.Replicas > deployment.Status.UpdatedReplicas {
   194  						klog.V(3).Infof("Waiting for deployment %q rollout to finish: %d old replicas are pending termination...\n", deployment.Name, deployment.Status.Replicas-deployment.Status.UpdatedReplicas)
   195  					} else if deployment.Status.AvailableReplicas < deployment.Status.UpdatedReplicas {
   196  						klog.V(3).Infof("Waiting for deployment %q rollout to finish: %d of %d updated replicas are available...\n", deployment.Name, deployment.Status.AvailableReplicas, deployment.Status.UpdatedReplicas)
   197  					} else {
   198  						s.End(true)
   199  						klog.V(3).Infof("Deployment %q successfully rolled out\n", deployment.Name)
   200  						success <- deployment
   201  					}
   202  				}
   203  				klog.V(3).Infof("Waiting for deployment spec update to be observed...\n")
   204  
   205  			} else {
   206  				failure <- errors.New("unable to convert event object to Pod")
   207  			}
   208  		}
   209  	}()
   210  
   211  	select {
   212  	case val := <-success:
   213  		return val, nil
   214  	case err := <-failure:
   215  		return nil, err
   216  	case <-time.After(5 * time.Minute):
   217  		errorMessage := fmt.Sprintf("timeout while waiting for %s deployment roll out", deploymentName)
   218  		if len(failedEvents) != 0 {
   219  			tableString := getErrorMessageFromEvents(failedEvents)
   220  
   221  			errorMessage = errorMessage + fmt.Sprintf(`\nFor more information to help determine the cause of the error, re-run with '-v'.
   222  See below for a list of failed events that occured more than %d times during deployment:
   223  %s`, failedEventCount, tableString.String())
   224  
   225  			return nil, errors.Errorf(errorMessage)
   226  		}
   227  
   228  		return nil, errors.Errorf("timeout while waiting for %s deployment roll out", deploymentName)
   229  	}
   230  }

   255  // ApplyDeployment creates or updates a deployment based on the given deployment spec
   256  // It is using force:true to make sure that if someone changed one of the values that odo manages,
   257  // odo overrides it with the value it expects instead of failing due to conflict.
   258  func (c *Client) ApplyDeployment(deploy appsv1.Deployment) (*appsv1.Deployment, error) {
   259  	data, err := json.Marshal(deploy)
   260  	if err != nil {
   261  		return nil, errors.Wrapf(err, "unable to marshal deployment")
   262  	}
   263  	klog.V(5).Infoln("Applying Deployment via server-side apply:")
   264  	klog.V(5).Infoln(resourceAsJson(deploy))
   265  
   266  	err = c.removeDuplicateEnv(deploy.Name)
   267  	if err != nil {
   268  		return nil, err
   269  	}
   270  
   271  	deployment, err := c.KubeClient.AppsV1().Deployments(c.Namespace).Patch(context.TODO(), deploy.Name, types.ApplyPatchType, data, metav1.PatchOptions{FieldManager: FieldManager, Force: boolPtr(true)})
   272  	if err != nil {
   273  		return nil, errors.Wrapf(err, "unable to update Deployment %s", deploy.Name)
   274  	}
   275  	return deployment, nil
   276  }

   278  // removeDuplicateEnv removes duplicate environment variables from containers, due to a bug in Service Binding Operator:
   279  // https://github.com/redhat-developer/service-binding-operator/issues/983
   280  func (c *Client) removeDuplicateEnv(deploymentName string) error {
   281  	deployment, err := c.KubeClient.AppsV1().Deployments(c.Namespace).Get(context.Background(), deploymentName, metav1.GetOptions{})
   282  	if kerrors.IsNotFound(err) {
   283  		return nil
   284  	}
   285  	if err != nil {
   286  		return err
   287  	}
   288  	changes := false
   289  	containers := deployment.Spec.Template.Spec.Containers
   290  	for i := range containers {
   291  		found := map[string]bool{}
   292  		var newEnv []corev1.EnvVar
   293  		for _, env := range containers[i].Env {
   294  			if _, ok := found[env.Name]; !ok {
   295  				found[env.Name] = true
   296  				newEnv = append(newEnv, env)
   297  			} else {
   298  				changes = true
   299  			}
   300  		}
   301  		containers[i].Env = newEnv
   302  	}
   303  	if changes {
   304  		_, err = c.KubeClient.AppsV1().Deployments(c.Namespace).Update(context.Background(), deployment, metav1.UpdateOptions{})
   305  		if kerrors.IsNotFound(err) {
   306  			return nil
   307  		}
   308  		return err
   309  	}
   310  	return nil
   311  }

   397  // LinkSecret links a secret to the Deployment of a component
   398  func (c *Client) LinkSecret(secretName, componentName, applicationName string) error {
   399  
   400  	var deploymentPatchProvider = func(d *appsv1.Deployment) (string, error) {
   401  		if len(d.Spec.Template.Spec.Containers[0].EnvFrom) > 0 {
   402  			// we always add the link as the first value in the envFrom array. That way we don't need to know the existing value
   403  			return fmt.Sprintf(`[{ "op": "add", "path": "/spec/template/spec/containers/0/envFrom/0", "value": {"secretRef": {"name": "%s"}} }]`, secretName), nil
   404  		}
   405  
   406  		//in this case we need to add the full envFrom value
   407  		return fmt.Sprintf(`[{ "op": "add", "path": "/spec/template/spec/containers/0/envFrom", "value": [{"secretRef": {"name": "%s"}}] }]`, secretName), nil
   408  	}
   409  
   410  	return c.jsonPatchDeployment(componentlabels.GetSelector(componentName, applicationName), deploymentPatchProvider)
   411  }

   413  // UnlinkSecret unlinks a secret to the Deployment of a component
   414  func (c *Client) UnlinkSecret(secretName, componentName, applicationName string) error {
   415  	// Remove the Secret from the container
   416  	var deploymentPatchProvider = func(d *appsv1.Deployment) (string, error) {
   417  		indexForRemoval := -1
   418  		for i, env := range d.Spec.Template.Spec.Containers[0].EnvFrom {
   419  			if env.SecretRef.Name == secretName {
   420  				indexForRemoval = i
   421  				break
   422  			}
   423  		}
   424  
   425  		if indexForRemoval == -1 {
   426  			return "", fmt.Errorf("Deployment does not contain a link to %s", secretName)
   427  		}
   428  
   429  		return fmt.Sprintf(`[{"op": "remove", "path": "/spec/template/spec/containers/0/envFrom/%d"}]`, indexForRemoval), nil
   430  	}
   431  
   432  	return c.jsonPatchDeployment(componentlabels.GetSelector(componentName, applicationName), deploymentPatchProvider)
   433  }

   435  // jsonPatchDeployment will look up the appropriate Deployment, and execute the specified patch
   436  // the whole point of using patch is to avoid race conditions where we try to update
   437  // deployment while it's being simultaneously updated from another source (for example Kubernetes itself)
   438  // this will result in the triggering of a redeployment
   439  func (c *Client) jsonPatchDeployment(deploymentSelector string, deploymentPatchProvider deploymentPatchProvider) error {
   440  
   441  	deployment, err := c.GetOneDeploymentFromSelector(deploymentSelector)
   442  	if err != nil {
   443  		return errors.Wrapf(err, "unable to locate Deployment with selector %q", deploymentSelector)
   444  	}
   445  
   446  	if deploymentPatchProvider != nil {
   447  		patch, e := deploymentPatchProvider(deployment)
   448  		if e != nil {
   449  			return errors.Wrap(e, "Unable to create a patch for the Deployment")
   450  		}
   451  
   452  		// patch the Deployment with the secret
   453  		_, e = c.KubeClient.AppsV1().Deployments(c.Namespace).Patch(context.TODO(), deployment.Name, types.JSONPatchType, []byte(patch), metav1.PatchOptions{FieldManager: FieldManager})
   454  		if e != nil {
   455  			return errors.Wrapf(e, "Deployment not patched %s", deployment.Name)
   456  		}
   457  	} else {
   458  		return errors.Wrapf(err, "deploymentPatch was not properly set")
   459  	}
   460  
   461  	return nil
   462  }
