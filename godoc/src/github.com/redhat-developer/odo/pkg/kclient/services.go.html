    70  func (c *Client) GetOneServiceFromSelector(selector string) (*corev1.Service, error) {
    71  	services, err := c.ListServices(selector)
    72  	if err != nil {
    73  		return nil, fmt.Errorf("unable to get services for the selector %q : %w", selector, err)
    74  	}
    75  
    76  	num := len(services)
    77  	if num == 0 {
    78  		return nil, &ServiceNotFoundError{Selector: selector}
    79  	} else if num > 1 {
    80  		return nil, fmt.Errorf("multiple services exist for the selector: %v. Only one must be present", selector)
