    45  		{

    47  			ingressName:             "",

    51  		},
    52  		{
    53  			name:                    "Case: Valid extensions v1 beta1 ingress name",
    54  			ingressName:             "testIngress",
    55  			wantErr:                 false,
    56  			isNetworkingV1Supported: false,
    57  			ieExtensionV1Supported:  true,
    58  		},
    59  		{
    60  			name:                    "Case: Invalid extensions v1 beta1 ingress name",
    61  			ingressName:             "",
    62  			wantErr:                 true,
    63  			isNetworkingV1Supported: false,
    64  			ieExtensionV1Supported:  true,
    65  		},
    66  		{
    67  			name:                    "Case: fail if neither is supported",
    68  			ingressName:             "testIngress",
    69  			wantErr:                 true,
    70  			isNetworkingV1Supported: false,
    71  			ieExtensionV1Supported:  false,
    72  		},
    73  	}
    74  	for _, tt := range tests {
    75  		t.Run(tt.name, func(t *testing.T) {
    76  			// initialising the fakeclient
    77  			fkclient, fkclientset := FakeNewWithIngressSupports(tt.isNetworkingV1Supported, tt.ieExtensionV1Supported)
    78  			fkclient.Namespace = "default"
    79  
    80  			objectMeta := generator.GetObjectMeta(tt.ingressName, "default", nil, nil)
    81  			ingressParams := generator.IngressParams{
    82  				ObjectMeta:        objectMeta,
    83  				IngressSpecParams: generator.IngressSpecParams{ServiceName: tt.ingressName},
    84  			}
    85  			ingress := unions.NewKubernetesIngressFromParams(ingressParams)
    86  			createdIngress, err := fkclient.CreateIngress(*ingress)
    87  
    88  			// Checks for unexpected error cases
    89  			if !tt.wantErr == (err != nil) {
    90  				t.Errorf("fkclient.CreateIngress unexpected error %v, wantErr %v", err, tt.wantErr)
    91  			}
    92  
    93  			if err == nil {
    94  				if len(fkclientset.Kubernetes.Actions()) != 1 {
    95  					t.Errorf("expected 1 action, got: %v", fkclientset.Kubernetes.Actions())
    96  				} else {
    97  					if createdIngress.GetName() != tt.ingressName {
    98  						t.Errorf(ingressNameMatchError([]string{tt.ingressName}, []string{createdIngress.GetName()}))

   100  				}
   101  			}
