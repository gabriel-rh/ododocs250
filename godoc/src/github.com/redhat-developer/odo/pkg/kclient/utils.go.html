    16  // GetInputEnvVarsFromStrings generates corev1.EnvVar values from the array of string key=value pairs
    17  // envVars is the array containing the key=value pairs
    18  func GetInputEnvVarsFromStrings(envVars []string) ([]corev1.EnvVar, error) {
    19  	var inputEnvVars []corev1.EnvVar
    20  	var keys = make(map[string]int)
    21  	for _, env := range envVars {
    22  		splits := strings.SplitN(env, "=", 2)
    23  		if len(splits) < 2 {
    24  			return nil, errors.New("invalid syntax for env, please specify a VariableName=Value pair")
    25  		}
    26  		_, ok := keys[splits[0]]
    27  		if ok {
    28  			return nil, errors.Errorf("multiple values found for VariableName: %s", splits[0])
    29  		}
    30  
    31  		keys[splits[0]] = 1
    32  
    33  		inputEnvVars = append(inputEnvVars, corev1.EnvVar{
    34  			Name:  splits[0],
    35  			Value: splits[1],
    36  		})
    37  	}
    38  	return inputEnvVars, nil
    39  }

    41  // getErrorMessageFromEvents generates a error message from the given events
    42  func getErrorMessageFromEvents(failedEvents map[string]corev1.Event) strings.Builder {
    43  	// Create an output table
    44  	tableString := &strings.Builder{}
    45  	table := tablewriter.NewWriter(tableString)
    46  	table.SetAlignment(tablewriter.ALIGN_LEFT)
    47  	table.SetHeaderAlignment(tablewriter.ALIGN_LEFT)
    48  	table.SetCenterSeparator("")
    49  	table.SetColumnSeparator("")
    50  	table.SetRowSeparator("")
    51  
    52  	// Header
    53  	table.SetHeader([]string{"Name", "Count", "Reason", "Message"})
    54  
    55  	// List of events
    56  	for name, event := range failedEvents {
    57  		table.Append([]string{name, strconv.Itoa(int(event.Count)), event.Reason, event.Message})
    58  	}
    59  
    60  	// Here we render the table as well as a helpful error message
    61  	table.Render()
    62  
    63  	return *tableString
    64  }
