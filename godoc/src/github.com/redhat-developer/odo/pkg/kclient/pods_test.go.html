    33  func TestWaitAndGetPodWithEvents(t *testing.T) {
    34  	tests := []struct {
    35  		name                string
    36  		podName             string
    37  		status              corev1.PodPhase
    38  		wantEventWarning    bool
    39  		wantErr             bool
    40  		eventWarningMessage string
    41  	}{
    42  		{
    43  			name:             "Case 1: Pod running",
    44  			podName:          "ruby",
    45  			status:           corev1.PodRunning,
    46  			wantEventWarning: false,
    47  			wantErr:          false,
    48  		},
    49  		{
    50  			name:             "Case 2: Pod failed",
    51  			podName:          "ruby",
    52  			status:           corev1.PodFailed,
    53  			wantEventWarning: false,
    54  			wantErr:          true,
    55  		},
    56  		{
    57  			name:             "Case 3: Pod unknown",
    58  			podName:          "ruby",
    59  			status:           corev1.PodUnknown,
    60  			wantEventWarning: false,
    61  			wantErr:          true,
    62  		},
    63  	}
    64  
    65  	for _, tt := range tests {
    66  		t.Run(tt.name, func(t *testing.T) {
    67  
    68  			fakeClient, fakeClientSet := FakeNew()
    69  			fakePodWatch := watch.NewRaceFreeFake()
    70  
    71  			// Watch for Pods
    72  			fakePod := fakePodStatus(tt.status, tt.podName)
    73  			go func(pod *corev1.Pod) {
    74  				fakePodWatch.Modify(pod)
    75  			}(fakePod)
    76  
    77  			// Prepend watch reactor (beginning of the chain)
    78  			fakeClientSet.Kubernetes.PrependWatchReactor("pods", func(action ktesting.Action) (handled bool, ret watch.Interface, err error) {
    79  				return true, fakePodWatch, nil
    80  			})
    81  
    82  			podSelector := fmt.Sprintf("deploymentconfig=%s", tt.podName)
    83  
    84  			pod, err := fakeClient.WaitAndGetPodWithEvents(podSelector, corev1.PodRunning, "Waiting for component to start", preference.DefaultPushTimeout*time.Second)
    85  
    86  			if !tt.wantErr == (err != nil) {
    87  				t.Errorf("client.WaitAndGetPod(string) unexpected error %v, wantErr %v", err, tt.wantErr)
    88  				return
    89  			}
    90  
    91  			if err == nil {
    92  				if pod.Name != tt.podName {
    93  					t.Errorf("pod name is not matching to expected name, expected: %s, got %s", tt.podName, pod.Name)
    94  				}
    95  			}
    96  
    97  		})
    98  	}

   106  	fakePodWithDeletionTimeStamp.Labels["component"] = "nodejs"
   107  	currentTime := metav1.NewTime(time.Now())
   108  	fakePodWithDeletionTimeStamp.DeletionTimestamp = &currentTime
   109  
   110  	type args struct {
   111  		selector string
   112  	}

   115  		args         args

   117  		want         *corev1.Pod
   118  		wantErr      bool
   119  	}{
   120  		{
   121  			name: "valid number of pods",
   122  			args: args{selector: fmt.Sprintf("component=%s", "nodejs")},
   123  			returnedPods: &corev1.PodList{
   124  				Items: []corev1.Pod{
   125  					*fakePod,
   126  				},
   127  			},
   128  			want:    fakePod,
   129  			wantErr: false,
   130  		},
   131  		{
   132  			name: "zero pods",
   133  			args: args{selector: fmt.Sprintf("component=%s", "nodejs")},
   134  			returnedPods: &corev1.PodList{
   135  				Items: []corev1.Pod{},
   136  			},
   137  			want:    &corev1.Pod{},
   138  			wantErr: true,
   139  		},
   140  		{
   141  			name: "mutiple pods",
   142  			args: args{selector: fmt.Sprintf("component=%s", "nodejs")},
   143  			returnedPods: &corev1.PodList{
   144  				Items: []corev1.Pod{
   145  					*fakePod,
   146  					*fakePod,
   147  				},
   148  			},
   149  			want:    &corev1.Pod{},
   150  			wantErr: true,
   151  		},
   152  		{
   153  			name: "pod is in the deletion state",
   154  			args: args{selector: fmt.Sprintf("component=%s", "nodejs")},

   156  				Items: []corev1.Pod{
   157  					*fakePodWithDeletionTimeStamp,
   158  				},
   159  			},
   160  			want:    &corev1.Pod{},
   161  			wantErr: true,
   162  		},
   163  	}
   164  	for _, tt := range tests {
   165  		t.Run(tt.name, func(t *testing.T) {
   166  
   167  			fkclient, fkclientset := FakeNew()
   168  
   169  			fkclientset.Kubernetes.PrependReactor("list", "pods", func(action ktesting.Action) (handled bool, ret runtime.Object, err error) {
   170  				if action.(ktesting.ListAction).GetListRestrictions().Labels.String() != fmt.Sprintf("component=%s", "nodejs") {
   171  					t.Errorf("list called with different selector want:%s, got:%s", fmt.Sprintf("component=%s", "nodejs"), action.(ktesting.ListAction).GetListRestrictions().Labels.String())
   172  				}
   173  				return true, tt.returnedPods, nil
   174  			})
   175  
   176  			got, err := fkclient.GetOnePodFromSelector(tt.args.selector)
   177  			if (err != nil) != tt.wantErr {
   178  				t.Errorf("GetOnePodFromSelector() error = %v, wantErr %v", err, tt.wantErr)
   179  				return
   180  			} else if tt.wantErr && err != nil {
   181  				return
   182  			}
   183  			if !reflect.DeepEqual(got, tt.want) {
   184  				t.Errorf("GetOnePodFromSelector() got = %v, want %v", got, tt.want)
