    17  func TestCreateNewProject(t *testing.T) {
    18  	tests := []struct {
    19  		name     string
    20  		projName string
    21  		wait     bool
    22  		wantErr  bool
    23  	}{
    24  		{
    25  			name:     "Case 1: valid project name, not waiting",
    26  			projName: "testing",
    27  			wait:     false,
    28  			wantErr:  false,
    29  		},
    30  		{
    31  			name:     "Case 2: valid project name, waiting",
    32  			projName: "testing2",
    33  			wait:     true,
    34  			wantErr:  false,
    35  		},
    36  	}
    37  
    38  	for _, tt := range tests {
    39  		t.Run(tt.name, func(t *testing.T) {
    40  			fkclient, fkclientset := FakeNew()
    41  
    42  			fkclientset.ProjClientset.PrependReactor("create", "projectrequests", func(action ktesting.Action) (bool, runtime.Object, error) {
    43  				proj := projectv1.Project{
    44  					ObjectMeta: metav1.ObjectMeta{
    45  						Name: tt.projName,
    46  					},
    47  				}
    48  				return true, &proj, nil
    49  			})
    50  
    51  			if tt.wait {
    52  				fkWatch := watch.NewFake()
    53  				// Change the status
    54  				go func() {
    55  					fkWatch.Add(&projectv1.Project{
    56  						ObjectMeta: metav1.ObjectMeta{
    57  							Name: tt.projName,
    58  						},
    59  						Status: projectv1.ProjectStatus{Phase: "Active"},
    60  					})
    61  				}()
    62  				fkclientset.ProjClientset.PrependWatchReactor("projects", func(action ktesting.Action) (handled bool, ret watch.Interface, err error) {
    63  					if len(tt.projName) == 0 {
    64  						return true, nil, fmt.Errorf("error watching project")
    65  					}
    66  					return true, fkWatch, nil
    67  				})
    68  			}
    69  
    70  			err := fkclient.CreateNewProject(tt.projName, tt.wait)
    71  			if !tt.wantErr == (err != nil) {
    72  				t.Errorf("client.CreateNewProject(string) unexpected error %v, wantErr %v", err, tt.wantErr)
    73  			}
    74  
    75  			actions := fkclientset.ProjClientset.Actions()
    76  			actionsLen := len(actions)
    77  			if !tt.wait && actionsLen != 1 {
    78  				t.Errorf("expected 1 action in CreateNewProject got: %v", actions)
    79  			}
    80  			if tt.wait && actionsLen != 2 {
    81  				t.Errorf("expected 2 actions in CreateNewProject when waiting for project creation got: %v", actions)
    82  			}
    83  
    84  			if err == nil {
    85  				createdProj := actions[actionsLen-1].(ktesting.CreateAction).GetObject().(*projectv1.ProjectRequest)
    86  
    87  				if createdProj.Name != tt.projName {
    88  					t.Errorf("project name does not match the expected name, expected: %s, got: %s", tt.projName, createdProj.Name)
    89  				}
    90  
    91  				if tt.wait {
    92  					expectedFields := fields.OneTermEqualSelector("metadata.name", tt.projName)
    93  					gotFields := actions[0].(ktesting.WatchAction).GetWatchRestrictions().Fields
    94  
    95  					if !reflect.DeepEqual(expectedFields, gotFields) {
    96  						t.Errorf("Fields not matching: expected: %s, got %s", expectedFields, gotFields)
    97  					}
    98  				}
    99  			}
   100
