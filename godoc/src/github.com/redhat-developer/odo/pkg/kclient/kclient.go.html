    96  // NewForConfig creates a new client with the provided configuration or initializes the configuration if none is provided
    97  func NewForConfig(config clientcmd.ClientConfig) (client *Client, err error) {
    98  	if config == nil {
    99  		// initialize client-go clients
   100  		loadingRules := clientcmd.NewDefaultClientConfigLoadingRules()
   101  		configOverrides := &clientcmd.ConfigOverrides{}
   102  		config = clientcmd.NewNonInteractiveDeferredLoadingClientConfig(loadingRules, configOverrides)
   103  	}
   104  
   105  	client = new(Client)
   106  	client.KubeConfig = config
   107  
   108  	client.KubeClientConfig, err = client.KubeConfig.ClientConfig()
   109  	if err != nil {
   110  		return nil, errors.Wrapf(err, errorMsg)
   111  	}
   112  
   113  	client.KubeClient, err = kubernetes.NewForConfig(client.KubeClientConfig)
   114  	if err != nil {
   115  		return nil, err
   116  	}
   117  
   118  	client.Namespace, _, err = client.KubeConfig.Namespace()
   119  	if err != nil {
   120  		return nil, err
   121  	}
   122  
   123  	client.OperatorClient, err = operatorsclientset.NewForConfig(client.KubeClientConfig)
   124  	if err != nil {
   125  		return nil, err
   126  	}
   127  
   128  	client.DynamicClient, err = dynamic.NewForConfig(client.KubeClientConfig)
   129  	if err != nil {
   130  		return nil, err
   131  	}
   132  
   133  	client.appsClient, err = appsclientset.NewForConfig(client.KubeClientConfig)
   134  	if err != nil {
   135  		return nil, err
   136  	}
   137  
   138  	client.serviceCatalogClient, err = servicecatalogclienset.NewForConfig(client.KubeClientConfig)
   139  	if err != nil {
   140  		return nil, err
   141  	}
   142  
   143  	client.discoveryClient, err = discovery.NewDiscoveryClientForConfig(client.KubeClientConfig)
   144  	if err != nil {
   145  		return nil, err
   146  	}
   147  
   148  	client.checkIngressSupports = true
   149  
   150  	client.userClient, err = userclientset.NewForConfig(client.KubeClientConfig)
   151  	if err != nil {
   152  		return nil, err
   153  	}
   154  
   155  	client.projectClient, err = projectclientset.NewForConfig(client.KubeClientConfig)
   156  	if err != nil {
   157  		return nil, err
   158  	}
   159  
   160  	client.routeClient, err = routeclientset.NewForConfig(client.KubeClientConfig)
   161  	if err != nil {
   162  		return nil, err

   168  // Delete takes labels as a input and based on it, deletes respective resource
   169  func (c *Client) Delete(labels map[string]string, wait bool) error {
   170  
   171  	// convert labels to selector
   172  	selector := util.ConvertLabelsToSelector(labels)
   173  	klog.V(3).Infof("Selectors used for deletion: %s", selector)
   174  
   175  	var errorList []string
   176  	var deletionPolicy = metav1.DeletePropagationBackground
   177  
   178  	// for --wait flag, it deletes component dependents first and then delete component
   179  	if wait {
   180  		deletionPolicy = metav1.DeletePropagationForeground
   181  	}
   182  	// Delete Deployments
   183  	klog.V(3).Info("Deleting Deployments")
   184  	err := c.appsClient.Deployments(c.Namespace).DeleteCollection(context.TODO(), metav1.DeleteOptions{PropagationPolicy: &deletionPolicy}, metav1.ListOptions{LabelSelector: selector})
   185  	if err != nil {
   186  		errorList = append(errorList, "unable to delete deployments")
   187  	}
   188  
   189  	// for --wait it waits for component to be deleted
   190  	// TODO: Need to modify for `odo app delete`, currently wait flag is added only in `odo component delete`
   191  	//       so only one component gets passed in selector
   192  	if wait {
   193  		err = c.WaitForComponentDeletion(selector)
   194  		if err != nil {
   195  			errorList = append(errorList, err.Error())
   196  		}
   197  	}
   198  
   199  	// Error string
   200  	errString := strings.Join(errorList, ",")
   201  	if len(errString) != 0 {
   202  		return errors.New(errString)
   203  	}
   204  	return nil
   205  
   206  }

   208  // WaitForComponentDeletion waits for component to be deleted
   209  func (c *Client) WaitForComponentDeletion(selector string) error {
   210  
   211  	klog.V(3).Infof("Waiting for component to get deleted")
   212  
   213  	watcher, err := c.appsClient.Deployments(c.Namespace).Watch(context.TODO(), metav1.ListOptions{LabelSelector: selector})
   214  	if err != nil {
   215  		return err
   216  	}
   217  	defer watcher.Stop()
   218  	eventCh := watcher.ResultChan()
   219  
   220  	for {
   221  		select {
   222  		case event, ok := <-eventCh:
   223  			_, typeOk := event.Object.(*appsv1.Deployment)
   224  			if !ok || !typeOk {
   225  				return errors.New("Unable to watch deployments")
   226  			}
   227  			if event.Type == watch.Deleted {
   228  				klog.V(3).Infof("WaitForComponentDeletion, Event Received:Deleted")
   229  				return nil
   230  			} else if event.Type == watch.Error {
   231  				klog.V(3).Infof("WaitForComponentDeletion, Event Received:Deleted ")
   232  				return errors.New("Unable to watch deployments")
   233  			}
   234  		case <-time.After(waitForComponentDeletionTimeout):
   235  			klog.V(3).Infof("WaitForComponentDeletion, Timeout")
   236  			return errors.New("Time out waiting for component to get deleted")
   237  		}
   238  	}
   239  }

   255  func (c *Client) IsResourceSupported(apiGroup, apiVersion, resourceName string) (bool, error) {
   256  	klog.V(4).Infof("Checking if %q resource is supported", resourceName)
   257  
   258  	if c.supportedResources == nil {
   259  		c.supportedResources = make(map[string]bool, 7)
   260  	}
   261  	groupVersion := metav1.GroupVersion{Group: apiGroup, Version: apiVersion}.String()
   262  	resource := metav1.GroupVersionResource{Group: apiGroup, Version: apiVersion, Resource: resourceName}
   263  	groupVersionResource := resource.String()
   264  
   265  	supported, found := c.supportedResources[groupVersionResource]
   266  	if !found {
   267  		list, err := c.discoveryClient.ServerResourcesForGroupVersion(groupVersion)
   268  		if err != nil {
   269  			if kerrors.IsNotFound(err) {
   270  				supported = false
   271  			} else {
   272  				// don't record, just attempt again next time in case it's a transient error
   273  				return false, err
   274  			}
   275  		} else {
   276  			for _, resources := range list.APIResources {
   277  				if resources.Name == resourceName {
   278  					supported = true
   279  					break
   280  				}
   281  			}
   282  		}
   283  		c.supportedResources[groupVersionResource] = supported
   284  	}
   285  	return supported, nil
   286  }

   288  // IsSSASupported checks if Server Side Apply is supported by cluster
   289  // SSA was introduced in Kubernetes 1.16
   290  // If there is an error while parsing versions, it assumes that SSA is supported by cluster.
   291  // Most of clusters these days are 1.16 and up
   292  func (c *Client) IsSSASupported() bool {
   293  	// check if this was done before so we don't query cluster multiple times for the same info
   294  	if c.isSSASupported == nil {
   295  		versionWithSSA, err := semver.Make("1.16.0")
   296  		if err != nil {
   297  			klog.Warningf("unable to parse version %q", err)
   298  		}
   299  
   300  		kVersion, err := c.discoveryClient.ServerVersion()
   301  		if err != nil {
   302  			klog.Warningf("unable to get k8s server version %q", err)
   303  			return true
   304  		}
   305  		klog.V(4).Infof("Kubernetes version is %q", kVersion.String())
   306  
   307  		cleanupVersion := strings.TrimLeft(kVersion.String(), "v")
   308  		serverVersion, err := semver.Make(cleanupVersion)
   309  		if err != nil {
   310  			klog.Warningf("unable to parse k8s server version %q", err)
   311  			return true
   312  		}
   313  
   314  		isSSASupported := versionWithSSA.LE(serverVersion)
   315  		c.isSSASupported = &isSSASupported
   316  
   317  		klog.V(4).Infof("Cluster has support for SSA: %t", *c.isSSASupported)
   318  	}
   319  	return *c.isSSASupported
   320  
   321  }

   323  func (c *Client) checkIngressSupport() error {
   324  	var err error
   325  	if c.checkIngressSupports {
   326  		c.isNetworkingV1IngressSupported, err = c.IsResourceSupported("networking.k8s.io", "v1", "ingresses")
   327  		if err != nil {
   328  			return fmt.Errorf("failed to check networking v1 ingress support %w", err)
   329  		}
   330  		c.isExtensionV1Beta1IngressSupported, err = c.IsResourceSupported("extensions", "v1beta1", "ingresses")
   331  		if err != nil {
   332  			return fmt.Errorf("failed to check extensions v1beta1 ingress support %w", err)
   333  		}
   334  		c.checkIngressSupports = false
   335  	}
   336  	return nil
   337  }
