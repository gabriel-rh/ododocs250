    34  			}
    35  		} else {

    63  // DeleteProject deletes given project
    64  //
    65  // NOTE:
    66  // There is a very specific edge case that may happen during project deletion when deleting a project and then immediately creating another.
    67  // Unfortunately, despite the watch interface, we cannot safely determine if the project is 100% deleted. See this link:
    68  // https://stackoverflow.com/questions/48208001/deleted-openshift-online-pro-project-has-left-a-trace-so-cannot-create-project-o
    69  // Will Gordon (Engineer @ Red Hat) describes the issue:
    70  //
    71  // "Projects are deleted asynchronously after you send the delete command. So it's possible that the deletion just hasn't been reconciled yet. It should happen within a minute or so, so try again.
    72  // Also, please be aware that in a multitenant environment, like OpenShift Online, you are prevented from creating a project with the same name as any other project in the cluster, even if it's not your own. So if you can't create the project, it's possible that someone has already created a project with the same name."
    73  func (c *Client) DeleteProject(name string, wait bool) error {
    74  
    75  	// Instantiate watcher for our "wait" function
    76  	var watcher watch.Interface
    77  	var err error
    78  
    79  	// If --wait has been passed, we will wait for the project to fully be deleted
    80  	if wait {
    81  		watcher, err = c.projectClient.Projects().Watch(context.TODO(), metav1.ListOptions{
    82  			FieldSelector: fields.Set{"metadata.name": name}.AsSelector().String(),
    83  		})
    84  		if err != nil {
    85  			return errors.Wrapf(err, "unable to watch project")
    86  		}
    87  		defer watcher.Stop()
    88  	}
    89  
    90  	// Delete the project
    91  	err = c.projectClient.Projects().Delete(context.TODO(), name, metav1.DeleteOptions{})
    92  	if err != nil {
    93  		return errors.Wrap(err, "unable to delete project")
    94  	}
    95  
    96  	// If watcher has been created (wait was passed) we will create a go routine and actually **wait**
    97  	// until *EVERYTHING* is successfully deleted.
    98  	if watcher != nil {
    99  
   100  		// Project channel
   101  		// Watch error channel
   102  		projectChannel := make(chan *projectv1.Project)
   103  		watchErrorChannel := make(chan error)
   104  
   105  		// Create a go routine to run in the background
   106  		go func() {
   107  
   108  			for {
   109  
   110  				// If watch unexpected has been closed..
   111  				val, ok := <-watcher.ResultChan()
   112  				if !ok {
   113  					//return fmt.Errorf("received unexpected signal %+v on project watch channel", val)
   114  					watchErrorChannel <- errors.Errorf("watch channel was closed unexpectedly: %+v", val)
   115  					break
   116  				}
   117  
   118  				// So we depend on val.Type as val.Object.Status.Phase is just empty string and not a mapped value constant
   119  				if projectStatus, ok := val.Object.(*projectv1.Project); ok {
   120  
   121  					klog.V(3).Infof("Status of delete of project %s is '%s'", name, projectStatus.Status.Phase)
   122  
   123  					switch projectStatus.Status.Phase {
   124  					//projectStatus.Status.Phase can only be "Terminating" or "Active" or ""
   125  					case "":
   126  						if val.Type == watch.Deleted {
   127  							projectChannel <- projectStatus
   128  							break
   129  						}
   130  						if val.Type == watch.Error {
   131  							watchErrorChannel <- errors.Errorf("failed watching the deletion of project %s", name)
   132  							break
   133  						}
   134  					}
   135  
   136  				} else {
   137  					watchErrorChannel <- errors.New("unable to convert event object to Project")
   138  					break
   139  				}
   140  
   141  			}
   142  			close(projectChannel)
   143  			close(watchErrorChannel)
   144  		}()
   145  
   146  		select {
   147  		case val := <-projectChannel:
   148  			klog.V(3).Infof("Deletion information for project: %+v", val)
   149  			return nil
   150  		case err := <-watchErrorChannel:
   151  			return err
   152  		case <-time.After(waitForProjectDeletionTimeOut):
   153  			return errors.Errorf("waited %s but couldn't delete project %s in time", waitForProjectDeletionTimeOut, name)
   154  		}
   155  
   156  	}
   157  
   158  	// Return nil since we don't bother checking for the watcher..
   159  	return nil
   160  }

   162  // CreateNewProject creates project with given projectName
   163  func (c *Client) CreateNewProject(projectName string, wait bool) error {
   164  	// Instantiate watcher before requesting new project
   165  	// If watcher is created after the project it can lead to situation when the project is created before the watcher.
   166  	// When this happens, it gets stuck waiting for event that already happened.
   167  	var watcher watch.Interface
   168  	var err error
   169  	if wait {
   170  		watcher, err = c.projectClient.Projects().Watch(context.TODO(), metav1.ListOptions{
   171  			FieldSelector: fields.Set{"metadata.name": projectName}.AsSelector().String(),
   172  		})
   173  		if err != nil {
   174  			return errors.Wrapf(err, "unable to watch new project %s creation", projectName)
   175  		}
   176  		defer watcher.Stop()
   177  	}
   178  
   179  	projectRequest := &projectv1.ProjectRequest{
   180  		ObjectMeta: metav1.ObjectMeta{
   181  			Name: projectName,
   182  		},
   183  	}
   184  	_, err = c.projectClient.ProjectRequests().Create(context.TODO(), projectRequest, metav1.CreateOptions{FieldManager: FieldManager})
   185  	if err != nil {
   186  		return errors.Wrapf(err, "unable to create new project %s", projectName)
   187  	}
   188  
   189  	if watcher != nil {
   190  		for {
   191  			val, ok := <-watcher.ResultChan()
   192  			if !ok {
   193  				break
   194  			}
   195  			if prj, ok := val.Object.(*projectv1.Project); ok {
   196  				klog.V(3).Infof("Status of creation of project %s is %s", prj.Name, prj.Status.Phase)
   197  				switch prj.Status.Phase {
   198  				//prj.Status.Phase can only be "Terminating" or "Active" or ""
   199  				case corev1.NamespaceActive:
   200  					if val.Type == watch.Added {
   201  						klog.V(3).Infof("Project %s now exists", prj.Name)
   202  						return nil
   203  					}
   204  					if val.Type == watch.Error {
   205  						return fmt.Errorf("failed watching the deletion of project %s", prj.Name)
   206  					}
   207  				}
   208  			}
   209  		}
   210  	}
   211  
   212  	return nil
   213  }
