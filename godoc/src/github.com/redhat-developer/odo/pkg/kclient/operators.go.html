    89  func (c *Client) SearchClusterServiceVersionList(name string) (*olm.ClusterServiceVersionList, error) {
    90  	var result []olm.ClusterServiceVersion
    91  	csvs, err := c.ListClusterServiceVersions()
    92  	if err != nil {
    93  		return &olm.ClusterServiceVersionList{}, errors.Wrap(err, "unable to list services")
    94  	}
    95  
    96  	// do a partial search in all the services
    97  	for _, service := range csvs.Items {
    98  		if strings.Contains(service.ObjectMeta.Name, name) {
    99  			result = append(result, service)
   100  		} else {
   101  			for _, crd := range service.Spec.CustomResourceDefinitions.Owned {
   102  				if name == crd.Kind {
   103  					result = append(result, service)
   104  				}
   105  			}
   106  		}
   107  	}
   108  
   109  	return &olm.ClusterServiceVersionList{
   110  		TypeMeta: v1.TypeMeta{
   111  			Kind:       "List",
   112  			APIVersion: apiVersion,
   113  		},
   114  		Items: result,

   118  // GetCustomResource returns the CR matching the name
   119  func (c *Client) GetCustomResource(customResource string) (*olm.CRDDescription, error) {
   120  	// Get all csvs in the namespace
   121  	csvs, err := c.ListClusterServiceVersions()
   122  	if err != nil {
   123  		return &olm.CRDDescription{}, err
   124  	}
   125  
   126  	// iterate of csvs to find if CR of our interest is provided by any of those
   127  	for _, csv := range csvs.Items {
   128  		clusSerVer := csv
   129  		crs := c.GetCustomResourcesFromCSV(&clusSerVer)
   130  
   131  		for _, cr := range *crs {
   132  			if cr.Kind == customResource {
   133  				return &cr, nil
   134  			}
   135  		}
   136  	}
   137  
   138  	return &olm.CRDDescription{}, fmt.Errorf("could not find a Custom Resource named %q in the namespace", customResource)
   139  }

   168  // getResourceSpecDefinitionFromSwagger returns the OpenAPI v2 definition of the Kubernetes resource of a given group/version/kind, for a given swagger data
   169  func getResourceSpecDefinitionFromSwagger(data []byte, group, version, kind string) (*spec.Schema, error) {
   170  	schema := new(spec.Schema)
   171  	err := json.Unmarshal([]byte(data), schema)
   172  	if err != nil {
   173  		return nil, err
   174  	}
   175  
   176  	var crd spec.Schema
   177  	found := false
   178  loopDefinitions:
   179  	for _, definition := range schema.Definitions {
   180  		extensions := definition.Extensions
   181  		gvkI, ok := extensions["x-kubernetes-group-version-kind"]
   182  		if !ok {
   183  			continue
   184  		}
   185  		// The concrete type of this extension is expected to be an array of interface{}
   186  		// If not, we ignore it
   187  		gvkA, ok := gvkI.([]interface{})
   188  		if !ok {
   189  			continue
   190  		}
   191  
   192  		for i := range gvkA {
   193  			// The concrete type of each element is expected to be a map[string]interface{}
   194  			// If not, we ignore it
   195  			gvk, ok := gvkA[i].(map[string]interface{})
   196  			if !ok {
   197  				continue
   198  			}
   199  			gvkGroup := gvk["group"].(string)
   200  			gvkVersion := gvk["version"].(string)
   201  			gvkKind := gvk["kind"].(string)
   202  			if strings.HasSuffix(group, gvkGroup) && version == gvkVersion && kind == gvkKind {
   203  				crd = definition
   204  				found = true
   205  				break loopDefinitions
   206  			}
   207  		}
   208  
   209  	}
   210  	if !found {
   211  		return nil, errors.New("no definition found")
   212  	}
   213  
   214  	spec, ok := crd.Properties["spec"]
   215  	if ok {
   216  		return &spec, nil
   217  	}
   218  	return nil, nil
   219  }

   237  // toOpenAPISpec transforms Spec descriptors from a CRD description to an OpenAPI schema
   238  func toOpenAPISpec(repr *olm.CRDDescription) *spec.Schema {
   239  	if len(repr.SpecDescriptors) == 0 {
   240  		return nil
   241  	}
   242  	schema := new(spec.Schema).Typed("object", "")
   243  	schema.AdditionalProperties = &spec.SchemaOrBool{
   244  		Allows: false,
   245  	}
   246  	for _, param := range repr.SpecDescriptors {
   247  		addParam(schema, param)
   248  	}
   249  	return schema
   250  }

   252  // addParam adds a Spec Descriptor parameter to an OpenAPI schema
   253  func addParam(schema *spec.Schema, param olm.SpecDescriptor) {
   254  	parts := strings.SplitN(param.Path, ".", 2)
   255  	if len(parts) == 1 {
   256  		child := spec.StringProperty()
   257  		if len(param.XDescriptors) == 1 {
   258  			switch param.XDescriptors[0] {
   259  			case "urn:alm:descriptor:com.tectonic.ui:podCount":
   260  				child = spec.Int32Property()
   261  				// TODO(feloy) more cases, based on
   262  				// - https://github.com/openshift/console/blob/master/frontend/packages/operator-lifecycle-manager/src/components/descriptors/reference/reference.md
   263  				// - https://docs.google.com/document/d/17Tdmpu4R6pA5UC4LumyJ2EP6AcotMWM127Jy728hYCk
   264  			}
   265  		}
   266  		child = child.WithTitle(param.DisplayName).WithDescription(param.Description)
   267  		schema.SetProperty(parts[0], *child)
   268  	} else {
   269  		var child *spec.Schema
   270  		if _, ok := schema.Properties[parts[0]]; ok {
   271  			c := schema.Properties[parts[0]]
   272  			child = &c
   273  		} else {
   274  			child = new(spec.Schema).Typed("object", "")
   275  		}
   276  		param.Path = parts[1]
   277  		addParam(child, param)
   278  		schema.SetProperty(parts[0], *child)
   279  	}
   280  }
