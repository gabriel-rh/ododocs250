    25  // WaitAndGetPod block and waits until pod matching selector is in in Running state
    26  // desiredPhase cannot be PodFailed or PodUnknown
    27  func (c *Client) WaitAndGetPodWithEvents(selector string, desiredPhase corev1.PodPhase, waitMessage string, pushTimeout time.Duration) (*corev1.Pod, error) {
    28  
    29  	klog.V(3).Infof("Waiting for %s pod", selector)
    30  
    31  	var spinner *log.Status
    32  	defer func() {
    33  		if spinner != nil {
    34  			spinner.End(false)
    35  		}
    36  	}()
    37  
    38  	w, err := c.KubeClient.CoreV1().Pods(c.Namespace).Watch(context.TODO(), metav1.ListOptions{
    39  		LabelSelector: selector,
    40  	})
    41  	if err != nil {
    42  		return nil, errors.Wrapf(err, "unable to watch pod")
    43  	}
    44  	defer w.Stop()
    45  
    46  	// Here we are going to start a loop watching for the pod status
    47  	podChannel := make(chan *corev1.Pod)
    48  	watchErrorChannel := make(chan error)
    49  	failedEvents := make(map[string]corev1.Event)
    50  	go func() {
    51  	loop:
    52  		for {
    53  			val, ok := <-w.ResultChan()
    54  			if !ok {
    55  				watchErrorChannel <- errors.New("watch channel was closed")
    56  				break loop
    57  			}
    58  			if e, ok := val.Object.(*corev1.Pod); ok {
    59  				klog.V(3).Infof("Status of %s pod is %s", e.Name, e.Status.Phase)
    60  				for _, cond := range e.Status.Conditions {
    61  					// using this just for debugging message, so ignoring error on purpose
    62  					jsonCond, _ := json.Marshal(cond)
    63  					klog.V(3).Infof("Pod Conditions: %s", string(jsonCond))
    64  				}
    65  				for _, status := range e.Status.ContainerStatuses {
    66  					// using this just for debugging message, so ignoring error on purpose
    67  					jsonStatus, _ := json.Marshal(status)
    68  					klog.V(3).Infof("Container Status: %s", string(jsonStatus))
    69  				}
    70  				switch e.Status.Phase {
    71  				case desiredPhase:
    72  					klog.V(3).Infof("Pod %s is %v", e.Name, desiredPhase)
    73  					podChannel <- e
    74  					break loop
    75  				case corev1.PodFailed, corev1.PodUnknown:
    76  					watchErrorChannel <- errors.Errorf("pod %s status %s", e.Name, e.Status.Phase)
    77  					break loop
    78  				default:
    79  					// we start in a phase different from the desired one, let's wait
    80  					if spinner == nil {
    81  						spinner = log.Spinner(waitMessage)
    82  						// Collect all the events in a separate go routine
    83  						quit := make(chan int)
    84  						go c.CollectEvents(selector, failedEvents, spinner, quit)
    85  						defer close(quit)
    86  					}
    87  				}
    88  			} else {
    89  				watchErrorChannel <- errors.New("unable to convert event object to Pod")
    90  				break loop
    91  			}
    92  		}
    93  		close(podChannel)
    94  		close(watchErrorChannel)
    95  	}()
    96  
    97  	select {
    98  	case val := <-podChannel:
    99  		if spinner != nil {
   100  			spinner.End(true)
   101  		}
   102  		return val, nil
   103  	case err := <-watchErrorChannel:
   104  		return nil, err
   105  	case <-time.After(pushTimeout):
   106  
   107  		// Create a useful error if there are any failed events
   108  		errorMessage := fmt.Sprintf(`waited %s but couldn't find running pod matching selector: '%s'`, pushTimeout, selector)
   109  
   110  		if len(failedEvents) != 0 {
   111  
   112  			tableString := getErrorMessageFromEvents(failedEvents)
   113  
   114  			errorMessage = fmt.Sprintf(`waited %s but was unable to find a running pod matching selector: '%s'
   115  For more information to help determine the cause of the error, re-run with '-v'.
   116  See below for a list of failed events that occured more than %d times during deployment:
   117  %s`, pushTimeout, selector, failedEventCount, tableString.String())
   118  		}
   119  
   120  		return nil, errors.Errorf(errorMessage)
   121  	}
   122  }

   124  // ExecCMDInContainer execute command in the container of a pod, pass an empty string for containerName to execute in the first container of the pod
   125  func (c *Client) ExecCMDInContainer(containerName, podName string, cmd []string, stdout io.Writer, stderr io.Writer, stdin io.Reader, tty bool) error {
   126  	podExecOptions := corev1.PodExecOptions{
   127  		Command: cmd,
   128  		Stdin:   stdin != nil,
   129  		Stdout:  stdout != nil,
   130  		Stderr:  stderr != nil,
   131  		TTY:     tty,
   132  	}
   133  
   134  	// If a container name was passed in, set it in the exec options, otherwise leave it blank
   135  	if containerName != "" {
   136  		podExecOptions.Container = containerName
   137  	}
   138  
   139  	req := c.KubeClient.CoreV1().RESTClient().
   140  		Post().
   141  		Namespace(c.Namespace).
   142  		Resource("pods").
   143  		Name(podName).
   144  		SubResource("exec").
   145  		VersionedParams(&podExecOptions, scheme.ParameterCodec)
   146  
   147  	config, err := c.KubeConfig.ClientConfig()
   148  	if err != nil {
   149  		return errors.Wrapf(err, "unable to get Kubernetes client config")
   150  	}
   151  
   152  	// Connect to url (constructed from req) using SPDY (HTTP/2) protocol which allows bidirectional streams.
   153  	exec, err := remotecommand.NewSPDYExecutor(config, "POST", req.URL())
   154  	if err != nil {
   155  		return errors.Wrapf(err, "unable execute command via SPDY")
   156  	}
   157  	// initialize the transport of the standard shell streams
   158  	err = exec.Stream(remotecommand.StreamOptions{
   159  		Stdin:  stdin,
   160  		Stdout: stdout,
   161  		Stderr: stderr,
   162  		Tty:    tty,
   163  	})
   164  	if err != nil {
   165  		return errors.Wrapf(err, "error while streaming command")
   166  	}
   167  
   168  	return nil
   169  }

   172  func (c *Client) ExtractProjectToComponent(containerName, podName string, targetPath string, stdin io.Reader) error {
   173  	// cmdArr will run inside container
   174  	cmdArr := []string{"tar", "xf", "-", "-C", targetPath}
   175  	var stdout bytes.Buffer
   176  	var stderr bytes.Buffer
   177  	klog.V(3).Infof("Executing command %s", strings.Join(cmdArr, " "))

   179  	if err != nil {
   180  		log.Errorf("Command '%s' in container failed.\n", strings.Join(cmdArr, " "))
   181  		log.Errorf("stdout: %s\n", stdout.String())
   182  		log.Errorf("stderr: %s\n", stderr.String())
   183  		log.Errorf("err: %s\n", err.Error())
   184  		return err
   185  	}
   186  	return nil

   200  // GetOnePodFromSelector gets a pod from the selector
   201  func (c *Client) GetOnePodFromSelector(selector string) (*corev1.Pod, error) {
   202  	pods, err := c.KubeClient.CoreV1().Pods(c.Namespace).List(context.TODO(), metav1.ListOptions{
   203  		LabelSelector: selector,
   204  	})
   205  	if err != nil {
   206  		// Dont wrap error since we want to know if its a forbidden error
   207  		// if we wrap, we lose the err status reason and callers of this api rely on it
   208  		return nil, err
   209  	}
   210  	numPods := len(pods.Items)
   211  	if numPods == 0 {
   212  		return nil, &PodNotFoundError{Selector: selector}
   213  	} else if numPods > 1 {
   214  		return nil, fmt.Errorf("multiple Pods exist for the selector: %v. Only one must be present", selector)
   215  	}
   216  
   217  	// check if the pod is in the terminating state
   218  	if pods.Items[0].DeletionTimestamp != nil {
   219  		return nil, &PodNotFoundError{Selector: selector}
   220  	}
   221  
   222  	return &pods.Items[0], nil
   223  }

   225  // GetPodLogs prints the log from pod to stdout
   226  func (c *Client) GetPodLogs(podName, containerName string, followLog bool) (io.ReadCloser, error) {
   227  
   228  	// Set standard log options
   229  	podLogOptions := corev1.PodLogOptions{Follow: false, Container: containerName}
   230  
   231  	// If the log is being followed, set it to follow / don't wait
   232  	if followLog {
   233  		tailLines := int64(1)
   234  		podLogOptions = corev1.PodLogOptions{
   235  			Follow:    true,
   236  			Previous:  false,
   237  			TailLines: &tailLines,
   238  			Container: containerName,
   239  		}
   240  	}
   241  
   242  	// RESTClient call to kubernetes
   243  	rd, err := c.KubeClient.CoreV1().RESTClient().Get().
   244  		Namespace(c.Namespace).
   245  		Name(podName).
   246  		Resource("pods").
   247  		SubResource("log").
   248  		VersionedParams(&podLogOptions, scheme.ParameterCodec).
   249  		Stream(context.TODO())
   250  
   251  	return rd, err
   252  }
