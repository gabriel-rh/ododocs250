    29  	tests := []struct {
    30  		name                string
    31  		podName             string
    32  		eventWarningMessage string
    33  	}{
    34  		{
    35  			name:                "Case 1: Collect an arbitrary amount of events",
    36  			podName:             "ruby",
    37  			eventWarningMessage: "Fake event warning message",
    38  		},
    39  	}
    40  
    41  	for _, tt := range tests {
    42  		t.Run(tt.name, func(t *testing.T) {
    43  
    44  			// Create a fake client
    45  			fakeClient, fakeClientSet := FakeNew()
    46  			fakeEventWatch := watch.NewRaceFreeFake()
    47  			podSelector := fmt.Sprintf("deploymentconfig=%s", tt.podName)
    48  
    49  			// Create a fake event status / watch reactor for faking the events we are collecting
    50  			fakeEvent := fakeEventStatus(tt.podName, tt.eventWarningMessage, 10)
    51  			go func(event *corev1.Event) {
    52  				fakeEventWatch.Add(event)
    53  			}(fakeEvent)
    54  
    55  			fakeClientSet.Kubernetes.PrependWatchReactor("events", func(action ktesting.Action) (handled bool, ret watch.Interface, err error) {
    56  				return true, fakeEventWatch, nil
    57  			})
    58  
    59  			// Create a test spinner / variables / quit variable for the go channel
    60  			spinner := log.Spinner("Test spinner")
    61  			events := make(map[string]corev1.Event)
    62  			quit := make(chan int)
    63  			go fakeClient.CollectEvents(podSelector, events, spinner, quit)
    64  
    65  			// Sleep in order to make sure we actually collect some events
    66  			time.Sleep(2 * time.Second)
    67  			close(quit)
    68  
    69  			// We make sure to lock in order to prevent race conditions when retrieving the events (since they are a pointer
    70  			// by default since we pass in a map)
    71  			mu.Lock()
    72  			if len(events) == 0 {
    73  				t.Errorf("Expected events, got none")
    74  			}
    75  			mu.Unlock()
    76  
    77  			// Collect the first event in the map
    78  			var firstEvent corev1.Event
    79  			for _, val := range events {
    80  				firstEvent = val
    81  			}
    82  
    83  			if !strings.Contains(firstEvent.Reason, tt.eventWarningMessage) {
    84  				t.Errorf("expected warning message: '%s' in event message: '%+v'", tt.eventWarningMessage, firstEvent.Reason)
    85  			}
    86  
    87  		})
    88  	}
