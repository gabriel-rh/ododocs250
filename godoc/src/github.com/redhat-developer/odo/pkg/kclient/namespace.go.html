    42  	ns, err := c.KubeClient.CoreV1().Namespaces().Get(context.TODO(), name, metav1.GetOptions{})

    44  		istatus, ok := err.(kerrors.APIStatus)
    45  		if ok {
    46  			status := istatus.Status()

    48  				return nil, nil
    49  			}
    50  		} else {
    51  			return nil, err
    52  		}
    53  
    54  	}
    55  	return ns, err

    79  // DeleteNamespace deletes namespace
    80  // if wait=true , it will wait for deletion
    81  func (c *Client) DeleteNamespace(name string, wait bool) error {
    82  	var watcher watch.Interface
    83  	var err error
    84  	if wait {
    85  		watcher, err = c.KubeClient.CoreV1().Namespaces().Watch(context.TODO(), metav1.ListOptions{
    86  			FieldSelector: fields.Set{"metadata.name": name}.AsSelector().String(),
    87  		})
    88  		if err != nil {
    89  			return errors.Wrapf(err, "unable to watch namespace")
    90  		}
    91  		defer watcher.Stop()
    92  	}
    93  
    94  	err = c.KubeClient.CoreV1().Namespaces().Delete(context.TODO(), name, metav1.DeleteOptions{})
    95  	if err != nil {
    96  		return errors.Wrapf(err, "unable to delete Namespace %s", name)
    97  	}
    98  
    99  	if watcher != nil {
   100  		namespaceChannel := make(chan *corev1.Namespace)
   101  		watchErrorChannel := make(chan error)
   102  
   103  		go func() {
   104  			for {
   105  				val, ok := <-watcher.ResultChan()
   106  				if !ok {
   107  					watchErrorChannel <- errors.Errorf("watch channel was closed unexpectedly: %+v", val)
   108  					break
   109  				}
   110  				klog.V(3).Infof("Watch event.Type '%s'.", val.Type)
   111  
   112  				if namespaceStatus, ok := val.Object.(*corev1.Namespace); ok {
   113  					klog.V(3).Infof("Status of delete of namespace %s is '%s'.", name, namespaceStatus.Status.Phase)
   114  					if val.Type == watch.Deleted {
   115  						namespaceChannel <- namespaceStatus
   116  						break
   117  					}
   118  					if val.Type == watch.Error {
   119  						watchErrorChannel <- errors.Errorf("failed watching the deletion of namespace %s", name)
   120  						break
   121  					}
   122  
   123  				}
   124  
   125  			}
   126  			close(namespaceChannel)
   127  			close(watchErrorChannel)
   128  		}()
   129  
   130  		select {
   131  		case val := <-namespaceChannel:
   132  			klog.V(3).Infof("Namespace %s deleted", val.Name)
   133  			return nil
   134  		case err := <-watchErrorChannel:
   135  			return err
   136  		case <-time.After(waitForNamespaceDeletionTimeOut):
   137  			return errors.Errorf("waited %s but couldn't delete namespace %s in time", waitForNamespaceDeletionTimeOut, name)
   138  		}
   139  
   140  	}
   141  	return nil
   142  }

   170  // WaitForServiceAccountInNamespace waits for the given service account to be ready
   171  func (c *Client) WaitForServiceAccountInNamespace(namespace, serviceAccountName string) error {
   172  	if namespace == "" || serviceAccountName == "" {
   173  		return errors.New("namespace and serviceAccountName cannot be empty")
   174  	}
   175  	watcher, err := c.KubeClient.CoreV1().ServiceAccounts(namespace).Watch(context.TODO(), metav1.SingleObject(metav1.ObjectMeta{Name: serviceAccountName}))
   176  	if err != nil {
   177  		return err
   178  	}
   179  
   180  	timeout := time.After(getDefaultServiceAccTimeout)
   181  	if watcher != nil {
   182  		defer watcher.Stop()
   183  		for {
   184  			select {
   185  			case val, ok := <-watcher.ResultChan():
   186  				if !ok {
   187  					break
   188  				}
   189  				if serviceAccount, ok := val.Object.(*corev1.ServiceAccount); ok {
   190  					if serviceAccount.Name == serviceAccountName {
   191  						klog.V(3).Infof("Status of creation of service account %s is ready", serviceAccount)
   192  						return nil
   193  					}
   194  				}
   195  			case <-timeout:
   196  				return errors.New("Timed out waiting for service to be ready")
   197  			}
   198  		}
   199  	}
   200  	return nil
   201  }
