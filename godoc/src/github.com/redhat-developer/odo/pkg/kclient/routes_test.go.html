    17  // GenerateOwnerReference generates an ownerReference which can then be set as
    18  // owner for various OpenShift objects and ensure that when the owner object is
    19  // deleted from the cluster, all other objects are automatically removed by
    20  // OpenShift garbage collector
    21  func GenerateOwnerReference(deployment *appsV1.Deployment) metav1.OwnerReference {
    22  
    23  	ownerReference := metav1.OwnerReference{
    24  		APIVersion: "apps.openshift.io/v1",
    25  		Kind:       "Deployment",
    26  		Name:       deployment.Name,
    27  		UID:        deployment.UID,
    28  	}
    29  
    30  	return ownerReference
    31  }

    60  			name:       "Case : blog (urlName is different than service)",
    61  			urlName:    "example",
    62  			service:    "blog",
    63  			portNumber: intstr.FromInt(9100),
    64  			labels: map[string]string{
    65  				"SLA":                        "High",
    66  				"app.kubernetes.io/instance": "backend",

    68  			},

    72  
    73  		{
    74  			name:       "Case : secure url",

    79  				"SLA":                        "High",

    85  			secureURL:          true,
    86  		},
    87  
    88  		{
    89  			name:       "Case : specify a path",
    90  			urlName:    "example",
    91  			service:    "blog",
    92  			portNumber: intstr.FromInt(9100),
    93  			labels: map[string]string{
    94  				"SLA":                        "High",
    95  				"app.kubernetes.io/instance": "backend",
    96  				"app.kubernetes.io/name":     "golang",
    97  			},
    98  			wantErr:            false,
    99  			existingDeployment: testingutil.CreateFakeDeployment("blog"),
   100  			path:               "/testpath",
   101  		},
   102  	}
   103  	for _, tt := range tests {
   104  		t.Run(tt.name, func(t *testing.T) {
   105  			// initialising the fakeclient
   106  			fkclient, fkclientset := FakeNew()
   107  
   108  			ownerReferences := GenerateOwnerReference(tt.existingDeployment)
   109  
   110  			fkclientset.AppsClientset.PrependReactor("get", "deployments", func(action ktesting.Action) (bool, runtime.Object, error) {
   111  				return true, tt.existingDeployment, nil
   112  			})
   113  
   114  			createdRoute, err := fkclient.CreateRoute(tt.urlName, tt.service, tt.portNumber, tt.labels, tt.secureURL, tt.path, ownerReferences)
   115  
   116  			if tt.secureURL {
   117  				wantedTLSConfig := &routev1.TLSConfig{
   118  					Termination:                   routev1.TLSTerminationEdge,
   119  					InsecureEdgeTerminationPolicy: routev1.InsecureEdgeTerminationPolicyRedirect,
   120  				}
   121  				if !reflect.DeepEqual(createdRoute.Spec.TLS, wantedTLSConfig) {
   122  					t.Errorf("tls config is different, wanted %v, got %v", wantedTLSConfig, createdRoute.Spec.TLS)
   123  				}
   124  			} else {
   125  				if createdRoute.Spec.TLS != nil {
   126  					t.Errorf("tls config is set for a non secure url")
   127  				}
   128  				if tt.path == "" && createdRoute.Spec.Path != "/" {
   129  					t.Errorf("expect path: /, but got path: %v", createdRoute.Spec.Path)
   130  				} else if tt.path != "" && createdRoute.Spec.Path != tt.path {
   131  					t.Errorf("expect path: %v, but got path: %v", tt.path, createdRoute.Spec.Path)
   132  				}
   133  			}
   134  
   135  			// Checks for error in positive cases
   136  			if !tt.wantErr == (err != nil) {
   137  				t.Errorf(" client.CreateRoute(string, labels) unexpected error %v, wantErr %v", err, tt.wantErr)
   138  			}
   139  
   140  			// Check for validating actions performed
   141  			if len(fkclientset.RouteClientset.Actions()) != 1 {
   142  				t.Errorf("expected 1 action in CreateRoute got: %v", fkclientset.RouteClientset.Actions())
   143  			}
   144  			// Checks for return values in positive cases
   145  			if err == nil {
   146  				createdRoute := fkclientset.RouteClientset.Actions()[0].(ktesting.CreateAction).GetObject().(*routev1.Route)
   147  				// created route should be labeled with labels passed to CreateRoute
   148  				if !reflect.DeepEqual(createdRoute.Labels, tt.labels) {
   149  					t.Errorf("labels in created route is not matching expected labels, expected: %v, got: %v", tt.labels, createdRoute.Labels)
   150  				}
   151  				// route name and service that route is pointg to should match
   152  				if createdRoute.Spec.To.Name != tt.service {
   153  					t.Errorf("route is not matching to expected service name, expected: %s, got %s", tt.service, createdRoute)
   154  				}
   155  				if createdRoute.Name != tt.urlName {
   156  					t.Errorf("route name is not matching to expected route name, expected: %s, got %s", tt.urlName, createdRoute.Name)
   157  				}
   158  				if createdRoute.Spec.To.Name != tt.service {
   159  					t.Errorf("service name is not matching to expected service name, expected: %s, got %s", tt.service, createdRoute.Spec.To.Name)
   160  				}
   161  				if createdRoute.Spec.Port.TargetPort != tt.portNumber {
   162  					t.Errorf("port number is not matching to expected port number, expected: %v, got %v", tt.portNumber, createdRoute.Spec.Port.TargetPort)
   163  				}
   164  			}
   165  		})
   166  	}
