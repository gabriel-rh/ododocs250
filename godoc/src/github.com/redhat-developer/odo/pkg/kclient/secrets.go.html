    58  // GenerateSelfSignedCertificate creates a self-signed SSl certificate
    59  func GenerateSelfSignedCertificate(host string) (SelfSignedCertificate, error) {
    60  
    61  	privateKey, err := rsa.GenerateKey(rand.Reader, 2048)
    62  	if err != nil {
    63  		return SelfSignedCertificate{}, errors.Wrap(err, "unable to generate rsa key")
    64  	}
    65  	template := x509.Certificate{
    66  		SerialNumber: big.NewInt(time.Now().Unix()),
    67  		Subject: pkix.Name{
    68  			CommonName:   "Odo self-signed certificate",
    69  			Organization: []string{"Odo"},
    70  		},
    71  		NotBefore:             time.Now(),
    72  		NotAfter:              time.Now().Add(time.Hour * 24 * 365 * 10),
    73  		KeyUsage:              x509.KeyUsageCertSign | x509.KeyUsageKeyEncipherment | x509.KeyUsageDigitalSignature,
    74  		BasicConstraintsValid: true,
    75  		DNSNames:              []string{"*." + host},
    76  	}
    77  
    78  	certificateDerEncoding, err := x509.CreateCertificate(rand.Reader, &template, &template, &privateKey.PublicKey, privateKey)
    79  	if err != nil {
    80  		return SelfSignedCertificate{}, errors.Wrap(err, "unable to create certificate")
    81  	}
    82  	out := &bytes.Buffer{}
    83  	err = pem.Encode(out, &pem.Block{Type: "CERTIFICATE", Bytes: certificateDerEncoding})
    84  	if err != nil {
    85  		return SelfSignedCertificate{}, errors.Wrap(err, "unable to encode certificate")
    86  	}
    87  	certPemEncode := out.String()
    88  	certPemByteArr := []byte(certPemEncode)
    89  
    90  	tlsPrivKeyEncoding := x509.MarshalPKCS1PrivateKey(privateKey)
    91  	err = pem.Encode(out, &pem.Block{Type: "RSA PRIVATE KEY", Bytes: tlsPrivKeyEncoding})
    92  	if err != nil {
    93  		return SelfSignedCertificate{}, errors.Wrap(err, "unable to encode rsa private key")
    94  	}
    95  	keyPemEncode := out.String()
    96  	keyPemByteArr := []byte(keyPemEncode)
    97  
    98  	return SelfSignedCertificate{CertPem: certPemByteArr, KeyPem: keyPemByteArr}, nil
    99  }

   145  // CreateSecrets creates a secret for each port, containing the host and port of the component
   146  // This is done so other components can later inject the secret into the environment
   147  // and have the "coordinates" to communicate with this component
   148  func (c *Client) CreateSecrets(componentName string, commonObjectMeta metav1.ObjectMeta, svc *corev1.Service, ownerReference metav1.OwnerReference) error {
   149  	originalName := commonObjectMeta.Name
   150  	for _, svcPort := range svc.Spec.Ports {
   151  		portAsString := fmt.Sprintf("%v", svcPort.Port)
   152  
   153  		// we need to create multiple secrets, so each one has to contain the port in it's name
   154  		// so we change the name of each secret by adding the port number
   155  		commonObjectMeta.Name = fmt.Sprintf("%v-%v", originalName, portAsString)
   156  
   157  		// we also add the port as an annotation to the secret
   158  		// this comes in handy when we need to "query" for the appropriate secret
   159  		// of a component based on the port
   160  		commonObjectMeta.Annotations[ComponentPortAnnotationName] = portAsString
   161  
   162  		err := c.CreateSecret(
   163  			commonObjectMeta,
   164  			map[string]string{
   165  				secretKeyName(componentName, "host"): svc.Name,
   166  				secretKeyName(componentName, "port"): portAsString,
   167  			},
   168  			ownerReference)
   169  
   170  		if err != nil {
   171  			return errors.Wrapf(err, "unable to create Secret for %s", commonObjectMeta.Name)
   172  		}
   173  	}
   174  
   175  	// restore the original values of the fields we changed
   176  	commonObjectMeta.Name = originalName
   177  	delete(commonObjectMeta.Annotations, ComponentPortAnnotationName)
   178  
   179  	return nil
   180  }

   199  // WaitAndGetSecret blocks and waits until the secret is available
   200  func (c *Client) WaitAndGetSecret(name string, namespace string) (*corev1.Secret, error) {
   201  	klog.V(3).Infof("Waiting for secret %s to become available", name)
   202  
   203  	w, err := c.KubeClient.CoreV1().Secrets(namespace).Watch(context.TODO(), metav1.ListOptions{
   204  		FieldSelector: fields.Set{"metadata.name": name}.AsSelector().String(),
   205  	})
   206  	if err != nil {
   207  		return nil, errors.Wrapf(err, "unable to watch secret")
   208  	}
   209  	defer w.Stop()
   210  	for {
   211  		val, ok := <-w.ResultChan()
   212  		if !ok {
   213  			break
   214  		}
   215  		if e, ok := val.Object.(*corev1.Secret); ok {
   216  			klog.V(3).Infof("Secret %s now exists", e.Name)
   217  			return e, nil
   218  		}
   219  	}
   220  	return nil, errors.Errorf("unknown error while waiting for secret '%s'", name)
   221  }
