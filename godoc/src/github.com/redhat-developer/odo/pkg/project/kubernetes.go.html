    28  // Create a new project, either by creating a `project.openshift.io` resource if supported by the cluster
    29  // (which will trigger the creation of a namespace),
    30  // or by creating directly a `namespace` resource.
    31  // With the `wait` flag, the function will wait for the `default` service account
    32  // to be created in the namespace before returning
    33  func (o kubernetesClient) Create(projectName string, wait bool) error {
    34  	if projectName == "" {
    35  		return errors.Errorf("no project name given")
    36  	}
    37  
    38  	projectSupport, err := o.client.IsProjectSupported()
    39  	if err != nil {
    40  		return errors.Wrap(err, "unable to detect project support")
    41  	}
    42  
    43  	if projectSupport {
    44  		err = o.client.CreateNewProject(projectName, wait)
    45  
    46  	} else {
    47  		_, err = o.client.CreateNamespace(projectName)
    48  	}
    49  	if err != nil {
    50  		return errors.Wrap(err, "unable to create new project")
    51  	}
    52  
    53  	if wait {
    54  		err = o.client.WaitForServiceAccountInNamespace(projectName, "default")
    55  		if err != nil {
    56  			return errors.Wrap(err, "unable to wait for service account")
    57  		}
    58  	}
    59  	return nil
    60  }

    85  // List all the projects on the cluster and returns an error if any
    86  func (o kubernetesClient) List() (ProjectList, error) {
    87  	currentProject := o.client.GetCurrentNamespace()
    88  
    89  	projectSupport, err := o.client.IsProjectSupported()
    90  	if err != nil {
    91  		return ProjectList{}, errors.Wrap(err, "unable to detect project support")
    92  	}
    93  
    94  	var allProjects []string
    95  	if projectSupport {
    96  		allProjects, err = o.client.ListProjectNames()
    97  	} else {
    98  		allProjects, err = o.client.GetNamespaces()
    99  	}
   100  	if err != nil {
   101  		return ProjectList{}, errors.Wrap(err, "cannot get all the projects")
   102  	}
   103  
   104  	projects := make([]Project, len(allProjects))
   105  	for i, project := range allProjects {
   106  		isActive := project == currentProject
   107  		projects[i] = NewProject(project, isActive)
   108  	}
   109  
   110  	return NewProjectList(projects), nil
   111  }

   114  func (o kubernetesClient) Exists(projectName string) (bool, error) {
   115  	projectSupport, err := o.client.IsProjectSupported()
   116  	if err != nil {

   120  	if projectSupport {

   122  		if err != nil || project == nil {
   123  			return false, err
   124  		}
   125  	} else {
   126  		namespace, err := o.client.GetNamespace(projectName)
   127  		if err != nil || namespace == nil {
   128  			return false, err
   129  		}
   130  	}
   131  
   132  	return true, nil
