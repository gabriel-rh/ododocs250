    26  // GetDevfileRegistries gets devfile registries from preference file,
    27  // if registry name is specified return the specific registry, otherwise return all registries
    28  func GetDevfileRegistries(registryName string) ([]Registry, error) {
    29  	var devfileRegistries []Registry
    30  
    31  	cfg, err := preference.NewClient()
    32  	if err != nil {
    33  		return nil, err
    34  	}
    35  
    36  	hasName := len(registryName) != 0
    37  	if cfg.RegistryList() != nil {
    38  		registryList := *cfg.RegistryList()
    39  		// Loop backwards here to ensure the registry display order is correct (display latest newly added registry firstly)
    40  		for i := len(registryList) - 1; i >= 0; i-- {
    41  			registry := registryList[i]
    42  			if hasName {
    43  				if registryName == registry.Name {
    44  					reg := Registry{
    45  						Name:   registry.Name,
    46  						URL:    registry.URL,
    47  						Secure: registry.Secure,
    48  					}
    49  					devfileRegistries = append(devfileRegistries, reg)
    50  					return devfileRegistries, nil
    51  				}
    52  			} else {
    53  				reg := Registry{
    54  					Name:   registry.Name,
    55  					URL:    registry.URL,
    56  					Secure: registry.Secure,
    57  				}
    58  				devfileRegistries = append(devfileRegistries, reg)
    59  			}
    60  		}
    61  	} else {
    62  		return nil, nil
    63  	}
    64  
    65  	return devfileRegistries, nil
    66  }

    68  // convertURL converts GitHub regular URL to GitHub raw URL, do nothing if the URL is not GitHub URL
    69  // For example:
    70  // GitHub regular URL: https://github.com/elsony/devfile-registry/tree/johnmcollier-crw
    71  // GitHub raw URL: https://raw.githubusercontent.com/elsony/devfile-registry/johnmcollier-crw
    72  func convertURL(URL string) (string, error) {
    73  	url, err := url.Parse(URL)
    74  	if err != nil {
    75  		return "", err
    76  	}
    77  
    78  	if strings.Contains(url.Host, "github") && !strings.Contains(url.Host, "raw") {
    79  		// Convert path part of the URL
    80  		URLSlice := strings.Split(URL, "/")
    81  		if len(URLSlice) > 2 && URLSlice[len(URLSlice)-2] == "tree" {
    82  			// GitHub raw URL doesn't have "tree" structure in the URL, need to remove it
    83  			URL = strings.Replace(URL, "/tree", "", 1)
    84  		} else {
    85  			// Add "master" branch for GitHub raw URL by default if branch is not specified
    86  			URL = URL + "/master"
    87  		}
    88  
    89  		// Convert host part of the URL
    90  		if url.Host == "github.com" {
    91  			URL = strings.Replace(URL, "github.com", "raw.githubusercontent.com", 1)
    92  		} else {
    93  			URL = strings.Replace(URL, url.Host, "raw."+url.Host, 1)
    94  		}
    95  	}
    96  
    97  	return URL, nil
    98  }

   103  func getRegistryDevfiles(registry Registry) ([]DevfileComponentType, error) {
   104  	if !strings.Contains(registry.URL, "github") {
   105  		// OCI-based registry
   106  		devfileIndex, err := registryLibrary.GetRegistryIndex(registry.URL, segment.GetRegistryOptions(), indexSchema.StackDevfileType)
   107  		if err != nil {
   108  			return nil, err
   109  		}
   110  		return createRegistryDevfiles(registry, devfileIndex)
   111  	}
   112  	// Github-based registry
   113  	URL, err := convertURL(registry.URL)
   114  	if err != nil {
   115  		return nil, errors.Wrapf(err, "unable to convert URL %s", registry.URL)
   116  	}
   117  	registry.URL = URL
   118  	indexLink := registry.URL + indexPath
   119  	request := util.HTTPRequestParams{
   120  		URL: indexLink,
   121  	}
   122  
   123  	// TODO(feloy) Get from DI
   124  	cfg, err := preference.NewClient()
   125  	if err != nil {
   126  		return nil, err
   127  	}
   128  
   129  	secure := registryUtil.IsSecure(cfg, registry.Name)
   130  	if secure {
   131  		token, e := keyring.Get(fmt.Sprintf("%s%s", util.CredentialPrefix, registry.Name), registryUtil.RegistryUser)
   132  		if e != nil {
   133  			return nil, errors.Wrap(e, "unable to get secure registry credential from keyring")
   134  		}
   135  		request.Token = token
   136  	}
   137  
   138  	jsonBytes, err := util.HTTPGetRequest(request, cfg.GetRegistryCacheTime())
   139  	if err != nil {
   140  		return nil, errors.Wrapf(err, "unable to download the devfile index.json from %s", indexLink)
   141  	}
   142  
   143  	var devfileIndex []indexSchema.Schema
   144  	err = json.Unmarshal(jsonBytes, &devfileIndex)
   145  	if err != nil {
   146  		if err := util.CleanDefaultHTTPCacheDir(); err != nil {
   147  			log.Warning("Error while cleaning up cache dir.")
   148  		}
   149  		// we try once again
   150  		jsonBytes, err := util.HTTPGetRequest(request, cfg.GetRegistryCacheTime())
   151  		if err != nil {
   152  			return nil, errors.Wrapf(err, "unable to download the devfile index.json from %s", indexLink)
   153  		}
   154  
   155  		err = json.Unmarshal(jsonBytes, &devfileIndex)
   156  		if err != nil {
   157  			return nil, errors.Wrapf(err, "unable to unmarshal the devfile index.json from %s", indexLink)
   158  		}
   159  	}
   160  	return createRegistryDevfiles(registry, devfileIndex)
   161  }

   163  func createRegistryDevfiles(registry Registry, devfileIndex []indexSchema.Schema) ([]DevfileComponentType, error) {
   164  	registryDevfiles := make([]DevfileComponentType, 0, len(devfileIndex))
   165  	for _, devfileIndexEntry := range devfileIndex {
   166  		stackDevfile := DevfileComponentType{
   167  			Name:        devfileIndexEntry.Name,
   168  			DisplayName: devfileIndexEntry.DisplayName,
   169  			Description: devfileIndexEntry.Description,
   170  			Link:        devfileIndexEntry.Links["self"],
   171  			Registry:    registry,
   172  			Language:    devfileIndexEntry.Language,
   173  			Tags:        devfileIndexEntry.Tags,
   174  		}
   175  		registryDevfiles = append(registryDevfiles, stackDevfile)
   176  	}
   177  
   178  	return registryDevfiles, nil
   179  }

   181  // ListDevfileComponents lists all the available devfile components
   182  func ListDevfileComponents(registryName string) (DevfileComponentTypeList, error) {
   183  	catalogDevfileList := &DevfileComponentTypeList{}
   184  	var err error
   185  
   186  	// TODO: consider caching registry information for better performance since it should be fairly stable over time
   187  	// Get devfile registries
   188  	catalogDevfileList.DevfileRegistries, err = GetDevfileRegistries(registryName)
   189  	if err != nil {
   190  		return *catalogDevfileList, err
   191  	}
   192  	if catalogDevfileList.DevfileRegistries == nil {
   193  		return *catalogDevfileList, nil
   194  	}
   195  
   196  	// first retrieve the indices for each registry, concurrently
   197  	devfileIndicesMutex := &sync.Mutex{}
   198  	retrieveRegistryIndices := util.NewConcurrentTasks(len(catalogDevfileList.DevfileRegistries))
   199  
   200  	// The 2D slice index is the priority of the registry (highest priority has highest index)
   201  	// and the element is the devfile slice that belongs to the registry
   202  	registrySlice := make([][]DevfileComponentType, len(catalogDevfileList.DevfileRegistries))
   203  	for regPriority, reg := range catalogDevfileList.DevfileRegistries {
   204  		// Load the devfile registry index.json
   205  		registry := reg                 // Needed to prevent the lambda from capturing the value
   206  		registryPriority := regPriority // Needed to prevent the lambda from capturing the value
   207  		retrieveRegistryIndices.Add(util.ConcurrentTask{ToRun: func(errChannel chan error) {
   208  			registryDevfiles, err := getRegistryDevfiles(registry)
   209  			if err != nil {
   210  				log.Warningf("Registry %s is not set up properly with error: %v, please check the registry URL and credential (refer `odo registry update --help`)\n", registry.Name, err)
   211  				return
   212  			}
   213  
   214  			devfileIndicesMutex.Lock()
   215  			registrySlice[registryPriority] = registryDevfiles
   216  			devfileIndicesMutex.Unlock()
   217  		}})
   218  	}
   219  	if err := retrieveRegistryIndices.Run(); err != nil {
   220  		return *catalogDevfileList, err
   221  	}
   222  
   223  	for _, registryDevfiles := range registrySlice {
   224  		catalogDevfileList.Items = append(catalogDevfileList.Items, registryDevfiles...)
   225  	}
   226  
   227  	return *catalogDevfileList, nil
   228  }

   251  // ListOperatorServices fetches a list of Operators from the cluster and
   252  // returns only those Operators which are successfully installed on the cluster
   253  func ListOperatorServices(client kclient.ClientInterface) (*olm.ClusterServiceVersionList, error) {
   254  	var csvList olm.ClusterServiceVersionList
   255  
   256  	// first check for CSV support
   257  	csvSupport, err := client.IsCSVSupported()
   258  	if !csvSupport || err != nil {
   259  		return &csvList, err
   260  	}
   261  
   262  	allCsvs, err := client.ListClusterServiceVersions()
   263  	if err != nil {
   264  		return &csvList, err
   265  	}
   266  
   267  	// now let's filter only those csvs which are successfully installed
   268  	csvList.TypeMeta = allCsvs.TypeMeta
   269  	csvList.ListMeta = allCsvs.ListMeta
   270  	for _, csv := range allCsvs.Items {
   271  		if csv.Status.Phase == "Succeeded" {
   272  			csvList.Items = append(csvList.Items, csv)
   273  		}
   274  	}
   275  
   276  	return &csvList, nil
   277  }
