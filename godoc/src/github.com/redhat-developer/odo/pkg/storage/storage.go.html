    68  // Push creates and deletes the required persistent storages and returns the list of ephemeral storages
    69  // it compares the local storage against the storage on the cluster
    70  func Push(client Client, configProvider localConfigProvider.LocalConfigProvider) (ephemerals map[string]Storage, _ error) {
    71  	// list all the storage in the cluster
    72  	storageClusterList := StorageList{}
    73  
    74  	storageClusterList, err := client.ListFromCluster()
    75  	if err != nil {
    76  		return nil, err
    77  	}
    78  	storageClusterNames := make(map[string]Storage)
    79  	for _, storage := range storageClusterList.Items {
    80  		storageClusterNames[storage.Name] = storage
    81  	}
    82  
    83  	// list the persistent storages in the config
    84  	persistentConfigNames := make(map[string]Storage)
    85  	// list the ephemeral storages
    86  	ephemeralConfigNames := make(map[string]Storage)
    87  
    88  	localStorage, err := configProvider.ListStorage()
    89  	if err != nil {
    90  		return nil, err
    91  	}
    92  	for _, storage := range ConvertListLocalToMachine(localStorage).Items {
    93  		if storage.Spec.Ephemeral == nil || storage.Spec.Ephemeral != nil && !*storage.Spec.Ephemeral {
    94  			persistentConfigNames[storage.Name] = storage
    95  		} else {
    96  			ephemeralConfigNames[storage.Name] = storage
    97  		}
    98  	}
    99  
   100  	// find storage to delete
   101  	for storageName, storage := range storageClusterNames {
   102  		val, ok := persistentConfigNames[storageName]
   103  		if !ok {
   104  			// delete the pvc
   105  			err = client.Delete(storage.Name)
   106  			if err != nil {
   107  				return nil, err
   108  			}
   109  			log.Successf("Deleted storage %v from %v", storage.Name, configProvider.GetName())
   110  			continue
   111  		} else if storage.Name == val.Name {
   112  			if val.Spec.Size != storage.Spec.Size {
   113  				return nil, errors.Errorf("config mismatch for storage with the same name %s", storage.Name)
   114  			}
   115  		}
   116  	}
   117  
   118  	// find storage to create
   119  	for storageName, storage := range persistentConfigNames {
   120  		_, ok := storageClusterNames[storageName]
   121  		if ok {
   122  			continue
   123  		}
   124  		if e := client.Create(storage); e != nil {
   125  			return nil, e
   126  		}
   127  		log.Successf("Added storage %v to %v", storage.Name, configProvider.GetName())
   128  	}
   129  
   130  	return ephemeralConfigNames, nil
   131  }
