    65  // IsTerminal returns true if the writer w is a terminal
    66  // This function is modified if we are running within Windows..
    67  // as Golang's built-in "IsTerminal" command only works on UNIX-based systems:
    68  // https://github.com/golang/crypto/blob/master/ssh/terminal/util.go#L5
    69  func IsTerminal(w io.Writer) bool {
    70  	if runtime.GOOS == "windows" {
    71  		return true
    72  	} else if v, ok := (w).(*os.File); ok {
    73  		return term.IsTerminal(int(v.Fd()))
    74  	}
    75  	return false
    76  }

    84  // Updates the status and makes sure that if the previous status was longer, it
    85  // "clears" the rest of the message.
    86  func (s *Status) updateStatus() {
    87  	mu.Lock()
    88  	if s.warningStatus != "" {
    89  		yellow := color.New(color.FgYellow).SprintFunc()
    90  
    91  		// Determine the warning size, so that we can calculate its length and use that length as padding parameter
    92  		warningSubstring := fmt.Sprintf(" [%s %s]", yellow(getWarningString()), yellow(s.warningStatus))
    93  
    94  		// Combine suffix and spacing, then resize them
    95  		newSuffix := fmt.Sprintf(suffixSpacing+"%s", s.status)
    96  		newSuffix = truncateSuffixIfNeeded(newSuffix, s.writer, len(warningSubstring))
    97  
    98  		// Combine the warning and non-warning text (since we don't want to truncate the warning text)
    99  		s.spinner.SetSuffix(fmt.Sprintf("%s%s", newSuffix, warningSubstring))
   100  	} else {
   101  		newSuffix := fmt.Sprintf(suffixSpacing+"%s", s.status)
   102  		s.spinner.SetSuffix(truncateSuffixIfNeeded(newSuffix, s.writer, 0))
   103  	}
   104  	mu.Unlock()
   105  }

   107  // Start starts a new phase of the status, if attached to a terminal
   108  // there will be a loading spinner with this status
   109  func (s *Status) Start(status string, debug bool) {
   110  	s.End(true)
   111  
   112  	// set new status
   113  	isTerm := IsTerminal(s.writer)
   114  	s.status = status
   115  
   116  	// If we are in debug mode, don't spin!
   117  	// In under no circumstances do we output if we're using -o json.. to
   118  	// to avoid parsing errors.
   119  	if !IsJSON() {
   120  		if !isTerm || debug {
   121  			fmt.Fprintf(s.writer, prefixSpacing+getSpacingString()+suffixSpacing+"%s  ...\n", s.status)
   122  		} else {
   123  			s.spinner.SetPrefix(prefixSpacing)
   124  			newSuffix := fmt.Sprintf(suffixSpacing+"%s", s.status)
   125  			s.spinner.SetSuffix(truncateSuffixIfNeeded(newSuffix, s.writer, 0))
   126  			s.spinner.Start()
   127  		}
   128  	}
   129  }

   131  // truncateSuffixIfNeeded returns a representation of the 'suffix' parameter that fits within the terminal
   132  // (including the extra space occupied by the padding parameter).
   133  func truncateSuffixIfNeeded(suffix string, w io.Writer, padding int) string {
   134  
   135  	terminalWidth := getTerminalWidth(w)
   136  	if terminalWidth == nil {
   137  		return suffix
   138  	}
   139  
   140  	// Additional padding to account for animation widget on lefthand side, and to avoid getting too close to the righthand terminal edge
   141  	const additionalPadding = 10
   142  
   143  	maxWidth := *terminalWidth - padding - additionalPadding
   144  
   145  	// For VERY small terminals, or very large padding, just return the suffix unmodified
   146  	if maxWidth <= 20 {
   147  		return suffix
   148  	}
   149  
   150  	// If we are compliant, return the unmodified suffix...
   151  	if len(suffix) <= maxWidth {
   152  		return suffix
   153  	}
   154  
   155  	// Otherwise truncate down to the desired length and append '...'
   156  	abbrevSuffix := "..."
   157  	maxWidth -= len(abbrevSuffix) // maxWidth is necessarily >20 at this point
   158  
   159  	// len(suffix) is necessarily >= maxWidth at this point
   160  	suffix = suffix[:maxWidth] + abbrevSuffix
   161  
   162  	return suffix
   163  }

   165  func getTerminalWidth(w io.Writer) *int {
   166  
   167  	if runtime.GOOS != "windows" {
   168  
   169  		if v, ok := (w).(*os.File); ok {
   170  			w, _, err := term.GetSize(int(v.Fd()))
   171  			if err == nil {
   172  				return &w
   173  			}
   174  		}
   175  
   176  	}
   177  
   178  	return nil
   179  }

   181  // End completes the current status, ending any previous spinning and
   182  // marking the status as success or failure
   183  func (s *Status) End(success bool) {
   184  	if s.status == "" {
   185  		return
   186  	}
   187  
   188  	isTerm := IsTerminal(s.writer)
   189  	if isTerm {
   190  		s.spinner.Stop()
   191  		if !IsJSON() {
   192  			fmt.Fprint(s.writer, "\r")
   193  		}
   194  	}
   195  
   196  	if !IsJSON() {
   197  		if success {
   198  			// Clear the warning (unneeded now)
   199  			s.WarningStatus("")
   200  			green := color.New(color.FgGreen).SprintFunc()
   201  			fmt.Fprintf(s.writer, prefixSpacing+"%s"+suffixSpacing+"%s [%s]\n", green(getSuccessString()), s.status, s.spinner.TimeSpent())
   202  		} else {
   203  			red := color.New(color.FgRed).SprintFunc()
   204  			if s.warningStatus != "" {
   205  				fmt.Fprintf(s.writer, prefixSpacing+"%s"+suffixSpacing+"%s [%s] [%s]\n", red(getErrString()), s.status, s.spinner.TimeSpent(), s.warningStatus)
   206  			} else {
   207  				fmt.Fprintf(s.writer, prefixSpacing+"%s"+suffixSpacing+"%s [%s]\n", red(getErrString()), s.status, s.spinner.TimeSpent())
   208  			}
   209  		}
   210  	}
   211  
   212  	s.status = ""
   213  }

   367  // ExplicitSpinner creates a spinner that can or not spin based on the value of the preventSpinning parameter
   368  func ExplicitSpinner(status string, preventSpinning bool) *Status {
   369  	doNotSpin := true
   370  	if !preventSpinning {
   371  		doNotSpin = IsDebug()
   372  	}
   373  	s := NewStatus(GetStdout())
   374  	s.Start(status, doNotSpin)
   375  	return s
   376  }

   378  // IsJSON returns true if we are in machine output mode..
   379  // under NO circumstances should we output any logging.. as we are only outputting json
   380  func IsJSON() bool {
   381  
   382  	flag := pflag.Lookup("o")
   383  	if flag != nil && flag.Changed {
   384  		return strings.Contains(pflag.Lookup("o").Value.String(), "json")
   385  	}
   386  
   387  	return false
   388  }

   390  // IsDebug returns true if we are debugging (-v is set to anything but 0)
   391  func IsDebug() bool {
   392  
   393  	flag := pflag.Lookup("v")
   394  
   395  	if flag != nil {
   396  		return !strings.Contains(pflag.Lookup("v").Value.String(), "0")
   397  	}

   399  	return false
