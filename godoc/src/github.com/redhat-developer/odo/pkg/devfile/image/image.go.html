    67  // buildPushImage build an image using the provided backend
    68  // If push is true, also push the image to its registry
    69  func buildPushImage(backend Backend, image *devfile.ImageComponent, devfilePath string, push bool) error {
    70  	if image == nil {
    71  		return errors.New("image should not be nil")
    72  	}
    73  	err := backend.Build(image, devfilePath)
    74  	if err != nil {
    75  		return err
    76  	}
    77  	if push {
    78  		err = backend.Push(image.ImageName)
    79  		if err != nil {
    80  			return err
    81  		}
    82  	}
    83  	return nil
    84  }

    86  // selectBackend selects the container backend to use for building and pushing images

    95  	if _, err := lookPathCmd(podmanCmd); err == nil {

    98  
    99  	dockerCmd := os.Getenv("DOCKER_CMD")
   100  	if dockerCmd == "" {

   102  	}
   103  	if _, err := lookPathCmd(dockerCmd); err == nil {
   104  		return NewDockerCompatibleBackend(dockerCmd), nil
   105  	}
   106  	return nil, errors.New("no backend found")
