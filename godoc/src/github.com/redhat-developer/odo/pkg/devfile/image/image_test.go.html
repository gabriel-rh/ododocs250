    30  		},

    39  			name: "image and no push should call Build and not Push",

    43  				},
    44  			},
    45  			push:            false,
    46  			wantErr:         false,
    47  			wantBuildCalled: true,
    48  			wantPushCalled:  false,
    49  		},
    50  		{
    51  			name: "image and push should call Build and Push",
    52  			image: &devfile.ImageComponent{
    53  				Image: devfile.Image{
    54  					ImageName: "a name",
    55  				},
    56  			},
    57  			push:            true,
    58  			wantErr:         false,
    59  			wantBuildCalled: true,
    60  			wantPushCalled:  true,
    61  		},
    62  		{
    63  			name: "Build returns err",
    64  			image: &devfile.ImageComponent{
    65  				Image: devfile.Image{
    66  					ImageName: "a name",
    67  				},
    68  			},
    69  			push:            true,
    70  			BuildReturns:    errors.New(""),
    71  			PushReturns:     nil,
    72  			wantErr:         true,
    73  			wantBuildCalled: true,
    74  			wantPushCalled:  false,
    75  		},
    76  		{
    77  			name: "Push returns err",
    78  			image: &devfile.ImageComponent{
    79  				Image: devfile.Image{
    80  					ImageName: "a name",
    81  				},
    82  			},
    83  			push:            true,
    84  			BuildReturns:    nil,
    85  			PushReturns:     errors.New(""),
    86  			wantErr:         true,
    87  			wantBuildCalled: true,
    88  			wantPushCalled:  true,
    89  		},
    90  	}
    91  
    92  	for _, tt := range tests {

    98  			} else {
    99  				backend.EXPECT().Build(nil, tt.devfilePath).Times(0)
   100  			}
   101  			if tt.wantPushCalled {
   102  				backend.EXPECT().Push(tt.image.ImageName).Return(tt.PushReturns).Times(1)
   103  			} else {
   104  				backend.EXPECT().Push(nil).Times(0)
   105  			}
   106  			err := buildPushImage(backend, tt.image, "", tt.push)
   107  
   108  			if tt.wantErr != (err != nil) {
   109  				t.Errorf("%s: Error result wanted %v, got %v", tt.name, tt.wantErr, err != nil)
   110  			}
   111  			ctrl.Finish()

   145  			},
   146  			wantErr:  false,

   153  					return "podman", nil
   154  				}
   155  				return "", errors.New("")

   159  		},
   160  	}
   161
