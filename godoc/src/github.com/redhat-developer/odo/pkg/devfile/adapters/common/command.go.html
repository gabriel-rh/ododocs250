    22  // New returns a new command implementation based on the specified devfile command and the known commands
    23  func New(devfile devfilev1.Command, knowCommands map[string]devfilev1.Command, executor commandExecutor) (command, error) {
    24  	composite := devfile.Composite
    25  	if composite != nil {
    26  		cmds := composite.Commands
    27  		components := make([]command, 0, len(cmds))
    28  		for _, cmd := range cmds {
    29  			if devfileCommand, ok := knowCommands[strings.ToLower(cmd)]; ok {
    30  				c, err := New(devfileCommand, knowCommands, executor)
    31  				if err != nil {
    32  					return nil, errors.Wrapf(err, "couldn't create command %s", cmd)
    33  				}
    34  				components = append(components, c)
    35  			} else {
    36  				return nil, fmt.Errorf("composite command %q has command %v not found in devfile", cmd, devfile)
    37  			}
    38  		}
    39  		if util.SafeGetBool(composite.Parallel) {
    40  			return newParallelCompositeCommand(components...), nil
    41  		}
    42  		return newCompositeCommand(components...), nil
    43  	} else if devfile.Exec != nil {
    44  		return newExecCommand(devfile, executor)
    45  	} else {
    46  		return newApplyCommand(devfile, executor)
    47  	}
    48  }

    65  // getCommandFromDevfile iterates through the devfile commands and returns the command associated with the group
    66  func getCommandFromDevfile(data data.DevfileData, groupType devfilev1.CommandGroupKind) (supportedCommand devfilev1.Command, err error) {
    67  	commands, err := data.GetCommands(parsercommon.DevfileOptions{})
    68  	if err != nil {
    69  		return devfilev1.Command{}, err
    70  	}
    71  	var onlyCommand devfilev1.Command
    72  
    73  	for _, command := range commands {
    74  		cmdGroup := parsercommon.GetGroup(command)
    75  		if cmdGroup != nil && cmdGroup.Kind == groupType {
    76  			if util.SafeGetBool(cmdGroup.IsDefault) {
    77  				return command, nil
    78  			} else if reflect.DeepEqual(onlyCommand, devfilev1.Command{}) {
    79  				// return the only remaining command for the group if there is no default command
    80  				// NOTE: we return outside the for loop since the next iteration can have a default command
    81  				onlyCommand = command
    82  			}
    83  		}
    84  	}
    85  
    86  	// if default command is not found return the first command found for the matching type.
    87  	if !reflect.DeepEqual(onlyCommand, devfilev1.Command{}) {
    88  		return onlyCommand, nil
    89  	}
    90  
    91  	notFoundError := NoCommandForGroup{Group: groupType}
    92  	// if run command or test command is not found in devfile then it is an error
    93  	if groupType == devfilev1.RunCommandGroupKind || groupType == devfilev1.TestCommandGroupKind {
    94  		err = notFoundError
    95  	} else {
    96  		klog.V(2).Info(notFoundError)
    97  	}
    98  
    99  	return
   100  }

   102  // getCommandFromFlag iterates through the devfile commands and returns the command specified associated with the group
   103  func getCommandFromFlag(data data.DevfileData, groupType devfilev1.CommandGroupKind, commandName string) (supportedCommand devfilev1.Command, err error) {
   104  	commands, err := data.GetCommands(parsercommon.DevfileOptions{})
   105  	if err != nil {
   106  		return devfilev1.Command{}, err
   107  	}
   108  
   109  	for _, command := range commands {
   110  		if command.Id == commandName {
   111  
   112  			// Update Group only custom commands (specified by odo flags)
   113  			command = updateCommandGroupIfReqd(groupType, command)
   114  
   115  			// we have found the command with name, its groupType Should match to the flag
   116  			// e.g --build-command "mybuild"
   117  			// exec:
   118  			//   id: mybuild
   119  			//   group:
   120  			//     kind: build
   121  			cmdGroup := parsercommon.GetGroup(command)
   122  			if cmdGroup != nil && cmdGroup.Kind != groupType {
   123  				return command, fmt.Errorf("command group mismatched, command %s is of group %v in devfile.yaml", commandName, command.Exec.Group.Kind)
   124  			}
   125  
   126  			return command, nil
   127  		}
   128  	}
   129  
   130  	// if any command specified via flag is not found in devfile then it is an error.
   131  	err = fmt.Errorf("the command \"%v\" is not found in the devfile", commandName)
   132  
   133  	return
   134  }

   161  // ValidateAndGetPushDevfileCommands validates the build and the run command,
   162  // if provided through odo push or else checks the devfile for devBuild and devRun.
   163  // It returns the build and run commands if its validated successfully, error otherwise.
   164  func ValidateAndGetPushDevfileCommands(data data.DevfileData, devfileBuildCmd, devfileRunCmd string) (commandMap PushCommandsMap, err error) {
   165  	var emptyCommand devfilev1.Command
   166  	commandMap = NewPushCommandMap()
   167  
   168  	isBuildCommandValid, isRunCommandValid := false, false
   169  
   170  	buildCommand, buildCmdErr := GetBuildCommand(data, devfileBuildCmd)
   171  
   172  	isBuildCmdEmpty := reflect.DeepEqual(emptyCommand, buildCommand)
   173  	if isBuildCmdEmpty && buildCmdErr == nil {
   174  		// If there was no build command specified through odo push and no default build command in the devfile, default validate to true since the build command is optional
   175  		isBuildCommandValid = true
   176  		klog.V(2).Infof("No build command was provided")
   177  	} else if !reflect.DeepEqual(emptyCommand, buildCommand) && buildCmdErr == nil {
   178  		isBuildCommandValid = true
   179  		commandMap[devfilev1.BuildCommandGroupKind] = buildCommand
   180  		klog.V(2).Infof("Build command: %v", buildCommand.Id)
   181  	}
   182  
   183  	runCommand, runCmdErr := GetRunCommand(data, devfileRunCmd)
   184  	if runCmdErr == nil && !reflect.DeepEqual(emptyCommand, runCommand) {
   185  		isRunCommandValid = true
   186  		commandMap[devfilev1.RunCommandGroupKind] = runCommand
   187  		klog.V(2).Infof("Run command: %v", runCommand.Id)
   188  	}
   189  
   190  	// If either command had a problem, return an empty list of commands and an error
   191  	if !isBuildCommandValid || !isRunCommandValid {
   192  		commandErrors := ""
   193  
   194  		if buildCmdErr != nil {
   195  			commandErrors += fmt.Sprintf("\n%s", buildCmdErr.Error())
   196  		}
   197  		if runCmdErr != nil {
   198  			commandErrors += fmt.Sprintf("\n%s", runCmdErr.Error())
   199  		}
   200  		return commandMap, fmt.Errorf(commandErrors)
   201  	}
   202  
   203  	return commandMap, nil
   204  }

   206  // Need to update group on custom commands specified by odo flags
   207  func updateCommandGroupIfReqd(groupType devfilev1.CommandGroupKind, command devfilev1.Command) devfilev1.Command {
   208  	// Update Group only for exec commands
   209  	// Update Group only when Group is not nil, devfile v2 might contain group for custom commands.
   210  	if command.Exec != nil && command.Exec.Group == nil {
   211  		command.Exec.Group = &devfilev1.CommandGroup{Kind: groupType}
   212  		return command
   213  	}
   214  	return command
   215  }
