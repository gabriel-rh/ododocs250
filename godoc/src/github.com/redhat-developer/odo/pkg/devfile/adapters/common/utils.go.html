   179  // GetCommandsFromEvent returns the list of commands from the event name.
   180  // If the event is a composite command, it returns the sub-commands from the tree
   181  func GetCommandsFromEvent(commandsMap map[string]devfilev1.Command, eventName string) []string {
   182  	var commands []string
   183  
   184  	if command, ok := commandsMap[eventName]; ok {
   185  		if command.Composite != nil {
   186  			klog.V(4).Infof("%s is a composite command", command.Id)
   187  			for _, compositeSubCmd := range command.Composite.Commands {
   188  				klog.V(4).Infof("checking if sub-command %s is either an exec or a composite command ", compositeSubCmd)
   189  				subCommands := GetCommandsFromEvent(commandsMap, strings.ToLower(compositeSubCmd))
   190  				commands = append(commands, subCommands...)
   191  			}
   192  		} else {
   193  			klog.V(4).Infof("%s is an exec command", command.Id)
   194  			commands = append(commands, command.Id)
   195  		}
   196  	}
   197  
   198  	return commands
   199  }

   213  func GetSyncFilesFromAttributes(commandsMap PushCommandsMap) map[string]string {
   214  	syncMap := make(map[string]string)
   215  	if value, ok := commandsMap[devfilev1.RunCommandGroupKind]; ok {
   216  		for key, value := range value.Attributes.Strings(nil) {
   217  			if strings.HasPrefix(key, "dev.odo.push.path:") {

   219  				syncMap[filepath.Clean(localValue)] = filepath.ToSlash(filepath.Clean(value))

   221  		}
   222  	}
   223  	return syncMap

   232  func removeDevfileURIContents(devfile devfileParser.DevfileObj, componentContext string, fs devfilefs.Filesystem) error {
   233  	components, err := devfile.Data.GetComponents(parsercommon.DevfileOptions{})
   234  	if err != nil {
   235  		return err
   236  	}
   237  	for _, component := range components {
   238  		var uri string
   239  		if component.Kubernetes != nil && component.Kubernetes.Uri != "" {
   240  			uri = component.Kubernetes.Uri
   241  		}
   242  
   243  		if component.Openshift != nil && component.Openshift.Uri != "" {
   244  			uri = component.Openshift.Uri
   245  		}
   246  
   247  		if uri == "" {
   248  			continue
   249  		}
   250  
   251  		parsedURL, err := url.Parse(uri)
   252  		if err != nil {
   253  			continue
   254  		}
   255  		if len(parsedURL.Host) != 0 && len(parsedURL.Scheme) != 0 {
   256  			continue
   257  		}
   258  
   259  		completePath := filepath.Join(componentContext, uri)
   260  		err = fs.Remove(completePath)
   261  		if err != nil {
   262  			return err
   263  		}
   264  	}
   265  	return nil
   266  }
