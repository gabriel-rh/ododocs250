    67  // getLatestContainerStatus returns a KubernetesDeploymentStatus for the given component; this function blocks until it is available
    68  func getLatestContainerStatus(adapter *Adapter) *KubernetesDeploymentStatus {
    69  
    70  	// Keep trying to acquire the ReplicaSet and DeploymentSet of the component, so that we can reliably find its pods
    71  	for {
    72  		containerStatus, err := adapter.getDeploymentStatus()
    73  		if err == nil {
    74  
    75  			if containerStatus.DeploymentUID == "" || containerStatus.ReplicaSetUID == "" {
    76  				adapter.Logger().ReportError(fmt.Errorf("unable to retrieve component deployment and replica set, trying again in a few moments"), machineoutput.TimestampNow())
    77  				time.Sleep(KubernetesResourceFailureInterval)
    78  				continue
    79  			}
    80  
    81  			return containerStatus
    82  		}
    83  
    84  		adapter.Logger().ReportError(errors.Wrapf(err, "unable to retrieve component deployment and replica set, trying again in a few moments"), machineoutput.TimestampNow())
    85  		time.Sleep(KubernetesResourceFailureInterval)
    86  	}
    87  
    88  }

    90  func (pw *podWatcher) startWatchThread(adapter *Adapter) {
    91  
    92  	// Kick off the goroutine then return execution
    93  	go func() {
    94  
    95  		watchAttempts := 1
    96  
    97  		var w watch.Interface = nil
    98  		for {
    99  
   100  			klog.V(4).Infof("Attempting to acquire watch, attempt #%d", watchAttempts)
   101  
   102  			var err error
   103  			w, err = adapter.Client.GetClient().CoreV1().Pods(adapter.Client.GetCurrentNamespace()).Watch(context.TODO(), metav1.ListOptions{})
   104  
   105  			if err != nil || w == nil {
   106  
   107  				if err != nil {
   108  					adapter.Logger().ReportError(err, machineoutput.TimestampNow())
   109  				}
   110  
   111  				klog.V(4).Infof("Unable to establish watch, trying again in a few moments seconds. Error was:  %v", err)
   112  
   113  				time.Sleep(KubernetesResourceFailureInterval)
   114  				watchAttempts++
   115  			} else {
   116  				// Success!
   117  				break
   118  			}
   119  		}
   120  
   121  		klog.V(4).Infof("Watch is successfully established.")
   122  
   123  		kubeContainerStatus := getLatestContainerStatus(adapter)
   124  
   125  		// After the watch is established, provide the reconciler with a list of all the current pods in the namespace (not just delta), so that
   126  		// old pods may be deleted from the reconciler (eg those pods that were deleted in the namespace while the watch was dead).
   127  		// (This prevents a race condition where pods deleted during a watch outage might be missed forever).
   128  		pw.statusReconcilerChan <- statusReconcilerChannelEntry{
   129  			pods:                   kubeContainerStatus.Pods,
   130  			isCompleteListOfPods:   true,
   131  			isDeleteEventFromWatch: false,
   132  			err:                    nil,
   133  		}
   134  
   135  		// We have succesfully established the watch, so kick off the watch event listener
   136  		go pw.watchEventListener(w, kubeContainerStatus.ReplicaSetUID)
   137  
   138  	}()
   139  
   140  }

   142  // This function runs in a goroutine for each watch. This goroutine exits if the watch dies (for example due to network disconnect),
   143  // at which point the watch acquisition process begins again.
   144  func (pw *podWatcher) watchEventListener(w watch.Interface, replicaSetUID types.UID) {
   145  	for {
   146  
   147  		// Retrieve watch event
   148  		entry := <-w.ResultChan()
   149  
   150  		// Restart the watch acquisition process on death, then exit
   151  		if entry.Object == nil && entry.Type == "" {
   152  			klog.V(4).Infof("Watch has died; initiating re-establish.")
   153  			pw.statusReconcilerChan <- statusReconcilerChannelEntry{
   154  				isWatchThreadRestarted: true,
   155  			}
   156  			pw.startWatchThread(pw.adapter)
   157  			return
   158  		}
   159  
   160  		// We only care about watch events that are related to Pods
   161  		if pod, ok := entry.Object.(*corev1.Pod); ok && pod != nil {
   162  
   163  			// Look for pods that are owned by the replicaset of our deployment
   164  			ownerRefMatches := false
   165  			for _, ownerRef := range pod.OwnerReferences {
   166  				if ownerRef.UID == replicaSetUID {
   167  					ownerRefMatches = true
   168  					break
   169  				}
   170  			}
   171  			if !ownerRefMatches {
   172  				continue
   173  			}
   174  
   175  			// We located the component pod, so now pass it to our status reconciler to report to the console (if required)
   176  			pw.statusReconcilerChan <- statusReconcilerChannelEntry{
   177  				pods:                   []*corev1.Pod{pod},
   178  				err:                    nil,
   179  				isCompleteListOfPods:   false, // only a delta
   180  				isDeleteEventFromWatch: entry.Type == watch.Deleted,
   181  			}
   182  
   183  		}
   184  	}
   185  }

   187  // createStatusReconciler kicks off a goroutine which receives messages containing updates to odo-managed k8s Pod resources.
   188  // For each message received, this function must determine if that resources has changed (in a way that we care about), and
   189  // if so, report that as a change event.
   190  func createStatusReconciler(adapter *Adapter) chan statusReconcilerChannelEntry {
   191  
   192  	senderChannel := make(chan statusReconcilerChannelEntry)
   193  
   194  	go func() {
   195  
   196  		// This map is the single source of truth re: what odo expects the cluster namespace to look like; when
   197  		// new events are received that contain pod data that differs from this, the user should be informed of the delta
   198  		// (and this 'truth' should be updated.)
   199  		//
   200  		// Map key is pod UID
   201  		mostRecentPodStatus := map[string]*KubernetesPodStatus{}
   202  
   203  		for {
   204  
   205  			entry := <-senderChannel
   206  
   207  			if entry.isWatchThreadRestarted {
   208  				// On network disconnect, clear the status map
   209  				mostRecentPodStatus = map[string]*KubernetesPodStatus{}
   210  			}
   211  
   212  			if entry.err != nil {
   213  				adapter.Logger().ReportError(entry.err, machineoutput.TimestampNow())
   214  				klog.V(4).Infof("Error received on status reconciler channel %v", entry.err)
   215  				continue
   216  			}
   217  
   218  			if entry.pods == nil {
   219  				continue
   220  			}
   221  
   222  			// Map key is pod UID (we don't use the map value)
   223  			entryPodUIDs := map[string]string{}
   224  			for _, pod := range entry.pods {
   225  				entryPodUIDs[string(pod.UID)] = string(pod.UID)
   226  			}
   227  
   228  			changeDetected := false
   229  
   230  			// This section of the algorithm only works if the entry was from a podlist (which contains the full list
   231  			// of all pods that exist in the namespace), rather than the watch (which contains only one pod in
   232  			// the namespace.)
   233  			if entry.isCompleteListOfPods {
   234  				// Detect if there exists a UID in mostRecentPodStatus that is not in entry; if so, one or more previous
   235  				// pods have disappeared, so set changeDetected to true.
   236  				for mostRecentPodUID := range mostRecentPodStatus {
   237  					if _, exists := entryPodUIDs[mostRecentPodUID]; !exists {
   238  						klog.V(4).Infof("Status change detected: Could not find previous pod %s in most recent pod status", mostRecentPodUID)
   239  						delete(mostRecentPodStatus, mostRecentPodUID)
   240  						changeDetected = true
   241  					}
   242  				}
   243  			}
   244  
   245  			if !changeDetected {
   246  
   247  				// For each pod we received a status for, determine if it is a change, and if so, update mostRecentPodStatus
   248  				for _, pod := range entry.pods {
   249  					podVal := CreateKubernetesPodStatusFromPod(*pod)
   250  
   251  					if entry.isDeleteEventFromWatch {
   252  						delete(mostRecentPodStatus, string(pod.UID))
   253  						klog.V(4).Infof("Removing deleted pod %s", pod.UID)
   254  						changeDetected = true
   255  						continue
   256  					}
   257  
   258  					// If a pod exists in the new pod status, that we have not seen before, then a change is detected.
   259  					prevValue, exists := mostRecentPodStatus[string(pod.UID)]
   260  					if !exists {
   261  						mostRecentPodStatus[string(pod.UID)] = &podVal
   262  						klog.V(4).Infof("Adding new pod to most recent pod status %s", pod.UID)
   263  						changeDetected = true
   264  
   265  					} else {
   266  						// If the pod exists in both the old and new status, then do a deep comparison
   267  						areEqual := areEqual(&podVal, prevValue)
   268  						if areEqual != "" {
   269  							mostRecentPodStatus[string(pod.UID)] = &podVal
   270  							klog.V(4).Infof("Pod value %s has changed:  %s", pod.UID, areEqual)
   271  							changeDetected = true
   272  						}
   273  					}
   274  				}
   275  			}
   276  
   277  			// On change: output all pods (our full knowledge of the odo-managed components in the namespace) as a single JSON event
   278  			if changeDetected {
   279  
   280  				podStatuses := []machineoutput.KubernetesPodStatusEntry{}
   281  
   282  				for _, v := range mostRecentPodStatus {
   283  
   284  					startTime := ""
   285  					if v.StartTime != nil {
   286  						startTime = machineoutput.FormatTime(*v.StartTime)
   287  					}
   288  
   289  					podStatuses = append(podStatuses, machineoutput.KubernetesPodStatusEntry{
   290  						Name:           v.Name,
   291  						Containers:     v.Containers,
   292  						InitContainers: v.InitContainers,
   293  						Labels:         v.Labels,
   294  						Phase:          v.Phase,
   295  						UID:            v.UID,
   296  						StartTime:      startTime,
   297  					})
   298  				}
   299  
   300  				adapter.Logger().KubernetesPodStatus(podStatuses, machineoutput.TimestampNow())
   301  			}
   302  		}
   303  	}()
   304  
   305  	return senderChannel
   306  }

   308  // areEqual compares two KubernetesPodStatus and returns a non-empty string if the two are not equivalent.
   309  // Note: returned strings are for logging/debug purposes only.
   310  func areEqual(one *KubernetesPodStatus, two *KubernetesPodStatus) string {
   311  
   312  	if one.UID != two.UID {
   313  		return fmt.Sprintf("UIDs differ %s %s", one.UID, two.UID)
   314  	}
   315  
   316  	if one.Name != two.Name {
   317  		return fmt.Sprintf("Names differ %s %s", one.Name, two.Name)
   318  	}
   319  
   320  	if !reflect.DeepEqual(one.StartTime, two.StartTime) {
   321  		return fmt.Sprintf("Start times differ %v %v", one.StartTime, two.StartTime)
   322  	}
   323  
   324  	if one.Phase != two.Phase {
   325  		return fmt.Sprintf("Pod phase differs %s %s", one.Phase, two.Phase)
   326  	}
   327  
   328  	if !reflect.DeepEqual(one.Labels, two.Labels) {
   329  		return fmt.Sprintf("Labels differ %v %v", one.Labels, two.Labels)
   330  	}
   331  
   332  	initContainerComparison := compareCoreContainerStatusList(one.InitContainers, two.InitContainers)
   333  	if initContainerComparison != "" {
   334  		return fmt.Sprintf("Init containers differ: %s", initContainerComparison)
   335  	}
   336  
   337  	containerComparison := compareCoreContainerStatusList(one.Containers, two.Containers)
   338  	if containerComparison != "" {
   339  		return fmt.Sprintf("Containers differ %s", containerComparison)
   340  	}
   341  
   342  	return ""
   343  }

   345  // compareCoreContainerStatusList compares two ContainerStatus arrays and returns a non-empty string if the two are not equivalent.
   346  // Note: returned strings are for logging/debug purposes only.
   347  func compareCoreContainerStatusList(oneParam []corev1.ContainerStatus, twoParam []corev1.ContainerStatus) string {
   348  
   349  	// One-way list compare, using container name to identify individual entries
   350  	compareFunc := func(paramA []corev1.ContainerStatus, paramB []corev1.ContainerStatus) string {
   351  
   352  		// key: container name
   353  		oneMap := map[string]*corev1.ContainerStatus{}
   354  
   355  		// Populate oneMap
   356  		for index, one := range paramA {
   357  			oneMap[one.Name] = &paramA[index]
   358  		}
   359  
   360  		// Iterate through paramB and compare with the corresponding container name in paramA
   361  		for index, two := range paramB {
   362  
   363  			oneEntry, exists := oneMap[two.Name]
   364  
   365  			// If an entry is present in two but not one
   366  			if !exists || oneEntry == nil {
   367  				return fmt.Sprintf("Container with id %s was present in one state but not the other", two.Name)
   368  			}
   369  
   370  			comparison := areCoreContainerStatusesEqual(oneEntry, &paramB[index])
   371  
   372  			if comparison != "" {
   373  				return comparison
   374  			}
   375  		}
   376  
   377  		return ""
   378  	}
   379  
   380  	// Since compareFunc is unidirectional, we do it twice
   381  	result := compareFunc(oneParam, twoParam)
   382  	if result != "" {
   383  		return result
   384  	}
   385  
   386  	result = compareFunc(twoParam, oneParam)
   387  	return result
   388  
   389  }

   391  // areCoreContainerStatusesEqual compares two ContainerStatus and returns a non-empty string if the two are not equivalent.
   392  // Note: returned strings are for logging/debug purposes only.
   393  func areCoreContainerStatusesEqual(one *corev1.ContainerStatus, two *corev1.ContainerStatus) string {
   394  
   395  	if one.Name != two.Name {
   396  		return fmt.Sprintf("Core status names differ [%s] [%s]", one.Name, two.Name)
   397  	}
   398  
   399  	if one.ContainerID != two.ContainerID {
   400  		return fmt.Sprintf("Core status container IDs differ: [%s] [%s]", one.ContainerID, two.ContainerID)
   401  	}
   402  
   403  	compareStates := compareCoreContainerState(one.State, two.State)
   404  	if compareStates != "" {
   405  		return fmt.Sprintf("Core status states differ %s", compareStates)
   406  	}
   407  
   408  	return ""
   409  }

   411  // compareCoreContainerState compares two ContainerState and returns a non-empty string if the two are not equivalent.
   412  // Note: returned strings are for logging/debug purposes only.
   413  func compareCoreContainerState(oneParam corev1.ContainerState, twoParam corev1.ContainerState) string {
   414  
   415  	// At present, we only compare the state, and not the state contents, so convert the state to a string and
   416  	// discard the other information.
   417  	toString := func(one corev1.ContainerState) string {
   418  		if one.Running != nil {
   419  			return "Running"
   420  		}
   421  
   422  		if one.Terminated != nil {
   423  			return "Terminated"
   424  		}
   425  
   426  		if one.Waiting != nil {
   427  			return "Waiting"
   428  		}
   429  
   430  		return ""
   431  	}
   432  
   433  	oneParamState := toString(oneParam)
   434  	twoParamState := toString(twoParam)
   435  
   436  	if oneParamState != twoParamState {
   437  		return "Core container states different: " + oneParamState + " " + twoParamState
   438  	}
   439  
   440  	return ""
   441
