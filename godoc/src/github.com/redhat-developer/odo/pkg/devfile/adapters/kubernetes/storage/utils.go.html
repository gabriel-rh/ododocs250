    31  // GetVolumeInfos returns the PVC name attached to the `odo-projects` directory and a map of other PVCs
    32  func GetVolumeInfos(pvcs []corev1.PersistentVolumeClaim) (odoSourcePVCName string, infos map[string]VolumeInfo, _ error) {
    33  	infos = make(map[string]VolumeInfo)
    34  	for _, pvc := range pvcs {
    35  		// check if the pvc is in the terminating state
    36  		if pvc.DeletionTimestamp != nil {
    37  			continue
    38  		}
    39  
    40  		generatedVolumeName, e := generateVolumeNameFromPVC(pvc.Name)
    41  		if e != nil {
    42  			return "", nil, errors.Wrapf(e, "Unable to generate volume name from pvc name")
    43  		}
    44  
    45  		if pvc.Labels[storagelabels.StorageLabel] == storagepkg.OdoSourceVolume {
    46  			odoSourcePVCName = pvc.Name
    47  			continue
    48  		}
    49  
    50  		infos[pvc.Labels[storagelabels.StorageLabel]] = VolumeInfo{
    51  			PVCName:    pvc.Name,
    52  			VolumeName: generatedVolumeName,
    53  		}
    54  	}
    55  	return odoSourcePVCName, infos, nil
    56  }

    58  // GetPersistentVolumesAndVolumeMounts gets the PVC volumes and updates the containers with the volume mounts.
    59  // volumeNameToVolInfo is a map of the devfile volume name to the volume info containing the pvc name and the volume name.
    60  // To be moved to devfile/library.
    61  func GetPersistentVolumesAndVolumeMounts(devfileObj devfileParser.DevfileObj, containers []corev1.Container, initContainers []corev1.Container, volumeNameToVolInfo map[string]VolumeInfo, options parsercommon.DevfileOptions) ([]corev1.Volume, error) {
    62  
    63  	containerComponents, err := devfileObj.Data.GetDevfileContainerComponents(options)
    64  	if err != nil {
    65  		return nil, err
    66  	}
    67  
    68  	var pvcVols []corev1.Volume
    69  	for volName, volInfo := range volumeNameToVolInfo {
    70  		pvcVols = append(pvcVols, getPVC(volInfo.VolumeName, volInfo.PVCName))
    71  
    72  		// containerNameToMountPaths is a map of the Devfile container name to their Devfile Volume Mount Paths for a given Volume Name
    73  		containerNameToMountPaths := make(map[string][]string)
    74  		for _, containerComp := range containerComponents {
    75  			for _, volumeMount := range containerComp.Container.VolumeMounts {
    76  				if volName == volumeMount.Name {
    77  					containerNameToMountPaths[containerComp.Name] = append(containerNameToMountPaths[containerComp.Name], envinfo.GetVolumeMountPath(volumeMount))
    78  				}
    79  			}
    80  		}
    81  
    82  		addVolumeMountToContainers(containers, initContainers, volInfo.VolumeName, containerNameToMountPaths)
    83  	}
    84  	return pvcVols, nil
    85  }

   128  // getEmptyDir gets an emptyDir type volume with the given volume name and size.
   129  // size should be parseable as a Kubernetes `Quantity` or an error will be returned
   130  func getEmptyDir(volumeName string, size string) (corev1.Volume, error) {
   131  
   132  	emptyDir := &corev1.EmptyDirVolumeSource{}
   133  	qty, err := resource.ParseQuantity(size)
   134  	if err != nil {
   135  		return corev1.Volume{}, err
   136  	}
   137  	emptyDir.SizeLimit = &qty
   138  	return corev1.Volume{
   139  		Name: volumeName,
   140  		VolumeSource: corev1.VolumeSource{
   141  			EmptyDir: emptyDir,
   142  		},
   143  	}, nil
   144  }

   147  // containerNameToMountPaths is a map of a container name to an array of its Mount Paths.
   148  // To be moved to devfile/library.
   149  func addVolumeMountToContainers(containers []corev1.Container, initContainers []corev1.Container, volumeName string, containerNameToMountPaths map[string][]string) {
   150  
   151  	for containerName, mountPaths := range containerNameToMountPaths {
   152  		for i := range containers {
   153  			if containers[i].Name == containerName {
   154  				for _, mountPath := range mountPaths {
   155  					containers[i].VolumeMounts = append(containers[i].VolumeMounts, corev1.VolumeMount{
   156  						Name:      volumeName,
   157  						MountPath: mountPath,
   158  						SubPath:   "",
   159  					},
   160  					)
   161  				}
   162  			}
   163  		}
   164  		for i := range initContainers {
   165  			if strings.HasPrefix(initContainers[i].Name, containerName) {
   166  				for _, mountPath := range mountPaths {
   167  					initContainers[i].VolumeMounts = append(initContainers[i].VolumeMounts, corev1.VolumeMount{
   168  						Name:      volumeName,
   169  						MountPath: mountPath,
   170  						SubPath:   "",
   171  					},
   172  					)
   173  				}
   174  			}
   175  		}
   176  	}

   188  // HandleEphemeralStorage creates or deletes the ephemeral volume based on the preference setting
   189  func HandleEphemeralStorage(client kclient.ClientInterface, storageClient storage.Client, componentName string, isEphemeral bool) error {
   190  	selector := fmt.Sprintf("%v=%s,%s=%s", componentlabels.ComponentLabel, componentName, storagelabels.SourcePVCLabel, storage.OdoSourceVolume)
   191  
   192  	pvcs, err := client.ListPVCs(selector)
   193  	if err != nil && !kerrors.IsNotFound(err) {
   194  		return err
   195  	}
   196  
   197  	if !isEphemeral {
   198  		if len(pvcs) == 0 {
   199  			err := storageClient.Create(storage.Storage{
   200  				ObjectMeta: metav1.ObjectMeta{
   201  					Name: storage.OdoSourceVolume,
   202  				},
   203  				Spec: storage.StorageSpec{
   204  					Size: storage.OdoSourceVolumeSize,
   205  				},
   206  			})
   207  
   208  			if err != nil {
   209  				return err
   210  			}
   211  		} else if len(pvcs) > 1 {
   212  			return fmt.Errorf("number of source volumes shouldn't be greater than 1")
   213  		}
   214  	} else {
   215  		if len(pvcs) > 0 {
   216  			for _, pvc := range pvcs {
   217  				err := client.DeletePVC(pvc.Name)
   218  				if err != nil {
   219  					return err
   220  				}
   221  			}
   222  		}
   223  	}
   224  	return nil
   225  }
