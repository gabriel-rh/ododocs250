    40  // Find the pod for the component and convert to KubernetesDeploymentStatus
    41  func (a Adapter) getDeploymentStatus() (*KubernetesDeploymentStatus, error) {
    42  
    43  	// 1) Retrieve the deployment
    44  	deployment, err := a.Client.GetOneDeployment(a.ComponentName, a.AppName)
    45  	if err != nil {
    46  		klog.V(4).Infof("Unable to retrieve deployment %s in %s ", a.ComponentName, a.Client.GetCurrentNamespace())
    47  		return nil, err
    48  	}
    49  
    50  	if deployment == nil {
    51  		return nil, errors.New("deployment status from Kubernetes API was nil")
    52  	}
    53  
    54  	deploymentUID := deployment.UID
    55  
    56  	// 2) Retrieve the replica set that is owned by the deployment; if multiple, go with one with largest generation
    57  	replicaSetList, err := a.Client.GetClient().AppsV1().ReplicaSets(a.Client.GetCurrentNamespace()).List(context.TODO(), metav1.ListOptions{})
    58  	if err != nil {
    59  		return nil, err
    60  	}
    61  
    62  	matchingReplicaSets := []v1.ReplicaSet{}
    63  	sort.Slice(replicaSetList.Items, func(i, j int) bool {
    64  		iGen := replicaSetList.Items[i].Generation
    65  		jGen := replicaSetList.Items[j].Generation
    66  
    67  		// Sort descending by generation
    68  		return iGen > jGen
    69  	})
    70  
    71  	// Locate the first matching replica, after above sort
    72  outer:
    73  	for _, replicaSet := range replicaSetList.Items {
    74  		for _, ownerRef := range replicaSet.OwnerReferences {
    75  			if ownerRef.UID == deploymentUID {
    76  				matchingReplicaSets = append(matchingReplicaSets, replicaSet)
    77  				break outer
    78  			}
    79  		}
    80  	}
    81  
    82  	if len(matchingReplicaSets) == 0 {
    83  		return nil, errors.New("no replica sets found")
    84  	}
    85  
    86  	replicaSetUID := matchingReplicaSets[0].UID
    87  
    88  	// 3) Retrieves the pods that are owned by the ReplicaSet and return
    89  	podList, err := a.Client.GetClient().CoreV1().Pods(a.Client.GetCurrentNamespace()).List(context.TODO(), metav1.ListOptions{})
    90  	if err != nil {
    91  		return nil, err
    92  	}
    93  
    94  	matchingPods := []*corev1.Pod{}
    95  	for i, podItem := range podList.Items {
    96  		for _, ownerRef := range podItem.OwnerReferences {
    97  
    98  			if string(ownerRef.UID) == string(replicaSetUID) {
    99  				matchingPods = append(matchingPods, &podList.Items[i])
   100  			}
   101  		}
   102  	}
   103  	result := KubernetesDeploymentStatus{}
   104  	result.Pods = append(result.Pods, matchingPods...)
   105  
   106  	result.DeploymentUID = deploymentUID
   107  	result.ReplicaSetUID = replicaSetUID
   108  
   109  	return &result, nil
   110  
   111  }

   174  // createSupervisordStatusReconciler contains the status reconciler implementation.
   175  // The reconciler receives (is sent) channel messages that contains the 'supervisord ctl status' values for each odo-managed container,
   176  // with the result reported to the console.
   177  func createSupervisordStatusReconciler(loggingClient machineoutput.MachineEventLoggingClient) chan supervisordStatusEvent {
   178  
   179  	senderChannel := make(chan supervisordStatusEvent)
   180  
   181  	go func() {
   182  		// Map key: 'podUID:containerName' (within pod) -> list of statuses from 'supervisord ctl status'
   183  		lastContainerStatus := map[string][]supervisordStatus{}
   184  
   185  		for {
   186  
   187  			event := <-senderChannel
   188  
   189  			key := event.podUID + ":" + event.containerName
   190  
   191  			previousStatus, hasLastContainerStatus := lastContainerStatus[key]
   192  			lastContainerStatus[key] = event.status
   193  
   194  			reportChange := false
   195  
   196  			if hasLastContainerStatus {
   197  				// If we saw a status for this container previously...
   198  				if !supervisordStatusesEqual(previousStatus, event.status) {
   199  					reportChange = true
   200  				} else {
   201  					reportChange = false
   202  				}
   203  
   204  			} else {
   205  				// No status from the container previously...
   206  				reportChange = true
   207  			}
   208  
   209  			entries := []machineoutput.SupervisordStatusEntry{}
   210  
   211  			for _, status := range event.status {
   212  				entries = append(entries, machineoutput.SupervisordStatusEntry{
   213  					Program: status.program,
   214  					Status:  status.status,
   215  				})
   216  			}
   217  
   218  			loggingClient.SupervisordStatus(entries, machineoutput.TimestampNow())
   219  
   220  			if reportChange {
   221  				klog.V(4).Infof("Ccontainer %v status has changed - is: %v", event.containerName, event.status)
   222  			}
   223  
   224  		}
   225  
   226  	}()
   227  
   228  	return senderChannel
   229  }

   231  // querySupervisordStatusFromContainers locates the correct component's pod, and for each container within the pod queries the supervisord ctl status.
   232  // The status results are sent to the reconciler.
   233  func (sw *supervisordStatusWatcher) querySupervisordStatusFromContainers(a Adapter) {
   234  
   235  	status, err := a.getDeploymentStatus()
   236  	if err != nil {
   237  		a.Logger().ReportError(errors.Wrap(err, "unable to retrieve container status"), machineoutput.TimestampNow())
   238  		return
   239  	}
   240  
   241  	if status == nil {
   242  		return
   243  	}
   244  
   245  	// Given a list of odo-managed pods, we want to find the newest; if there are multiple with the same age, then find the most
   246  	// alive by container status.
   247  	var podPhaseSortOrder = map[corev1.PodPhase]int{
   248  		corev1.PodFailed:    0,
   249  		corev1.PodSucceeded: 1,
   250  		corev1.PodUnknown:   2,
   251  		corev1.PodPending:   3,
   252  		corev1.PodRunning:   4,
   253  	}
   254  	sort.Slice(status.Pods, func(i, j int) bool {
   255  
   256  		iPod := status.Pods[i]
   257  		jPod := status.Pods[j]
   258  
   259  		iTime := iPod.CreationTimestamp.Time
   260  		jTime := jPod.CreationTimestamp.Time
   261  
   262  		if !jTime.Equal(iTime) {
   263  			// Sort descending by creation timestamp
   264  			return jTime.After(iTime)
   265  		}
   266  
   267  		// Next, sort descending to find the pod with most successful pod phase:
   268  		// PodRunning > PodPending > PodUnknown > PodSucceeded > PodFailed
   269  		return podPhaseSortOrder[jPod.Status.Phase] > podPhaseSortOrder[iPod.Status.Phase]
   270  	})
   271  
   272  	if len(status.Pods) < 1 {
   273  		return
   274  	}
   275  
   276  	// Retrieve the first pod, which post-sort should be the most recent and most alive
   277  	pod := status.Pods[0]
   278  
   279  	debugCommand, err := common.GetDebugCommand(a.Devfile.Data, a.devfileDebugCmd)
   280  	if err != nil {
   281  		a.Logger().ReportError(errors.Wrap(err, "unable to retrieve debug command"), machineoutput.TimestampNow())
   282  		return
   283  	}
   284  
   285  	runCommand, err := common.GetRunCommand(a.Devfile.Data, a.devfileRunCmd)
   286  	if err != nil {
   287  		a.Logger().ReportError(errors.Wrap(err, "unable to retrieve run command"), machineoutput.TimestampNow())
   288  		return
   289  	}
   290  
   291  	// For each of the containers, retrieve the status of the tasks and send that status back to the status reconciler
   292  	for _, container := range pod.Status.ContainerStatuses {
   293  
   294  		if (runCommand.Exec != nil && container.Name == runCommand.Exec.Component) || (debugCommand.Exec != nil && container.Name == debugCommand.Exec.Component) {
   295  			status := getSupervisordStatusInContainer(pod.Name, container.Name, a)
   296  
   297  			sw.statusReconcilerChannel <- supervisordStatusEvent{
   298  				containerName: container.Name,
   299  				status:        status,
   300  				podUID:        string(pod.UID),
   301  			}
   302  		}
   303  	}
   304  }

   306  // supervisordStatusesEqual is a simple comparison of []supervisord that ignores slice element order
   307  func supervisordStatusesEqual(one []supervisordStatus, two []supervisordStatus) bool {
   308  	if len(one) != len(two) {
   309  		return false
   310  	}
   311  
   312  	for _, oneVal := range one {
   313  
   314  		match := false
   315  		for _, twoVal := range two {
   316  
   317  			if reflect.DeepEqual(oneVal, twoVal) {
   318  				match = true
   319  			}
   320  		}
   321  		if !match {
   322  			return false
   323  		}
   324  	}
   325  	return true
   326  }

   328  // getSupervisordStatusInContainer executes 'supervisord ctl status' within the pod and container, parses the output,
   329  // and returns the status for the container
   330  func getSupervisordStatusInContainer(podName string, containerName string, a Adapter) []supervisordStatus {
   331  
   332  	command := []string{common.SupervisordBinaryPath, common.SupervisordCtlSubCommand, "status"}
   333  	compInfo := common.ComponentInfo{
   334  		ContainerName: containerName,
   335  		PodName:       podName,
   336  	}
   337  
   338  	stdoutWriter, stdoutOutputChannel := common.CreateConsoleOutputWriterAndChannel()
   339  	stderrWriter, stderrOutputChannel := common.CreateConsoleOutputWriterAndChannel()
   340  
   341  	err := common.ExecuteCommand(&a, compInfo, command, false, stdoutWriter, stderrWriter)
   342  
   343  	// Close the writer and wait the console output
   344  	stdoutWriter.Close()
   345  	consoleResult := <-stdoutOutputChannel
   346  
   347  	stderrWriter.Close()
   348  	consoleStderrResult := <-stderrOutputChannel
   349  
   350  	if err != nil {
   351  		a.Logger().ReportError(errors.Wrapf(err, "unable to execute command on %s within container %s, %v, output: %v %v", podName, containerName, err, consoleResult, consoleStderrResult), machineoutput.TimestampNow())
   352  		return nil
   353  	}
   354  
   355  	result := []supervisordStatus{}
   356  
   357  	for _, line := range consoleResult {
   358  
   359  		fields := strings.Fields(line)
   360  		if len(fields) < 2 {
   361  			continue
   362  		}
   363  
   364  		result = append(result, supervisordStatus{program: fields[0], status: fields[1]})
   365  	}
   366  
   367  	return result
   368  }
