   411  // debugSprintAll returns a list of all machine readable JSON events that have been output thus far
   412  func (lfo *logFuncOutput) debugSprintAll() string {
   413  
   414  	lfo.listMutex.Lock()
   415  	defer lfo.listMutex.Unlock()
   416  
   417  	result := ""
   418  
   419  	for _, entry := range lfo.jsonList {
   420  		jsonVal, err := json.Marshal(entry)
   421  		if err != nil {
   422  			lfo.errorOccurred = err
   423  			return fmt.Sprint(err)
   424  		}
   425  		result += string(jsonVal)
   426  	}

   429  }

   439  // Any machine readable JSON events that are output by odo are passed to this function, and this function

   441  func (lfo *logFuncOutput) logFunc(wrapper machineoutput.MachineEventWrapper) {
   442  
   443  	lfo.listMutex.Lock()
   444  	defer lfo.listMutex.Unlock()
   445  
   446  	json, err := wrapper.GetEntry()
   447  	if err != nil {
   448  		lfo.errorOccurred = err
   449  		return
   450  	}
   451  
   452  	machineoutput.OutputSuccessUnindented(wrapper)
   453  
   454  	lfo.jsonList = append(lfo.jsonList, json)
   455  }
