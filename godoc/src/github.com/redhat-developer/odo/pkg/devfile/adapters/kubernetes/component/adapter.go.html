    50  // getPod lazily records and retrieves the pod associated with the component associated with this adapter. If refresh parameter
    51  // is true, then the pod is refreshed from the cluster regardless of its current local state
    52  func (a *Adapter) getPod(refresh bool) (*corev1.Pod, error) {
    53  	if refresh || a.pod == nil {
    54  		podSelector := fmt.Sprintf("component=%s", a.ComponentName)
    55  
    56  		// Wait for Pod to be in running state otherwise we can't sync data to it.
    57  		pod, err := a.Client.WaitAndGetPodWithEvents(podSelector, corev1.PodRunning, "Waiting for component to start", time.Duration(a.prefClient.GetPushTimeout())*time.Second)
    58  		if err != nil {
    59  			return nil, errors.Wrapf(err, "error while waiting for pod %s", podSelector)
    60  		}
    61  		a.pod = pod
    62  	}
    63  	return a.pod, nil
    64  }

   108  // Push updates the component if a matching component exists or creates one if it doesn't exist
   109  // Once the component has started, it will sync the source code to it.
   110  func (a Adapter) Push(parameters common.PushParameters) (err error) {
   111  
   112  	a.deployment, err = a.Client.GetOneDeployment(a.ComponentName, a.AppName)
   113  	if err != nil {
   114  		if _, ok := err.(*kclient.DeploymentNotFoundError); !ok {
   115  			return errors.Wrapf(err, "unable to determine if component %s exists", a.ComponentName)
   116  		}
   117  	}
   118  
   119  	componentExists := false
   120  	if a.deployment != nil {
   121  		componentExists = true
   122  	}
   123  
   124  	a.devfileBuildCmd = parameters.DevfileBuildCmd
   125  	a.devfileRunCmd = parameters.DevfileRunCmd
   126  	a.devfileDebugCmd = parameters.DevfileDebugCmd
   127  	a.devfileDebugPort = parameters.DebugPort
   128  
   129  	podChanged := false
   130  	var podName string
   131  
   132  	// If the component already exists, retrieve the pod's name before it's potentially updated
   133  	if componentExists {
   134  		pod, podErr := a.getPod(true)
   135  		if podErr != nil {
   136  			return errors.Wrapf(podErr, "unable to get pod for component %s", a.ComponentName)
   137  		}
   138  		podName = pod.GetName()
   139  	}
   140  
   141  	// Validate the devfile build and run commands
   142  	log.Info("\nValidation")
   143  	s := log.Spinner("Validating the devfile")
   144  	err = util.ValidateK8sResourceName("component name", a.ComponentName)
   145  	if err != nil {
   146  		return err
   147  	}
   148  
   149  	err = util.ValidateK8sResourceName("component namespace", parameters.EnvSpecificInfo.GetNamespace())
   150  	if err != nil {
   151  		return err
   152  	}
   153  
   154  	pushDevfileCommands, err := common.ValidateAndGetPushDevfileCommands(a.Devfile.Data, a.devfileBuildCmd, a.devfileRunCmd)
   155  	if err != nil {
   156  		s.End(false)
   157  		return errors.Wrap(err, "failed to validate devfile build and run commands")
   158  	}
   159  	s.End(true)
   160  
   161  	labels := componentlabels.GetLabels(a.ComponentName, a.AppName, true)
   162  
   163  	previousMode := parameters.EnvSpecificInfo.GetRunMode()
   164  	currentMode := envinfo.Run
   165  
   166  	if parameters.Debug {
   167  		pushDevfileDebugCommands, e := common.ValidateAndGetDebugDevfileCommands(a.Devfile.Data, a.devfileDebugCmd)
   168  		if e != nil {
   169  			return fmt.Errorf("debug command is not valid")
   170  		}
   171  		pushDevfileCommands[devfilev1.DebugCommandGroupKind] = pushDevfileDebugCommands
   172  		currentMode = envinfo.Debug
   173  	}
   174  
   175  	if currentMode != previousMode {
   176  		parameters.RunModeChanged = true
   177  	}
   178  
   179  	// fetch the "kubernetes inlined components" to create them on cluster
   180  	// from odo standpoint, these components contain yaml manifest of an odo service or an odo link
   181  	k8sComponents, err := devfile.GetKubernetesComponentsToPush(a.Devfile)
   182  	if err != nil {
   183  		return errors.Wrap(err, "error while trying to fetch service(s) from devfile")
   184  	}
   185  
   186  	log.Infof("\nCreating Services for component %s", a.ComponentName)
   187  
   188  	// validate if the GVRs represented by Kubernetes inlined components are supported by the underlying cluster
   189  	err = service.ValidateResourcesExist(a.Client, k8sComponents, a.Context)
   190  	if err != nil {
   191  		return err
   192  	}
   193  
   194  	// create the Kubernetes objects from the manifest and delete the ones not in the devfile
   195  	err = service.PushKubernetesResources(a.Client, k8sComponents, labels, a.Context)
   196  	if err != nil {
   197  		return errors.Wrap(err, "failed to create service(s) associated with the component")
   198  	}
   199  
   200  	log.Infof("\nCreating Kubernetes resources for component %s", a.ComponentName)
   201  
   202  	isMainStorageEphemeral := a.prefClient.GetEphemeralSourceVolume()
   203  	err = a.createOrUpdateComponent(componentExists, parameters.EnvSpecificInfo, isMainStorageEphemeral)
   204  	if err != nil {
   205  		return errors.Wrap(err, "unable to create or update component")
   206  	}
   207  
   208  	a.deployment, err = a.Client.WaitForDeploymentRollout(a.deployment.Name)
   209  	if err != nil {
   210  		return errors.Wrap(err, "error while waiting for deployment rollout")
   211  	}
   212  
   213  	// Wait for Pod to be in running state otherwise we can't sync data or exec commands to it.
   214  	pod, err := a.getPod(true)
   215  	if err != nil {
   216  		return errors.Wrapf(err, "unable to get pod for component %s", a.ComponentName)
   217  	}
   218  
   219  	// list the latest state of the PVCs
   220  	pvcs, err := a.Client.ListPVCs(fmt.Sprintf("%v=%v", "component", a.ComponentName))
   221  	if err != nil {
   222  		return err
   223  	}
   224  
   225  	ownerReference := generator.GetOwnerReference(a.deployment)
   226  	// update the owner reference of the PVCs with the deployment
   227  	for i := range pvcs {
   228  		if pvcs[i].OwnerReferences != nil || pvcs[i].DeletionTimestamp != nil {
   229  			continue
   230  		}
   231  		err = a.Client.UpdateStorageOwnerReference(&pvcs[i], ownerReference)
   232  		if err != nil {
   233  			return err
   234  		}
   235  	}
   236  
   237  	err = service.UpdateServicesWithOwnerReferences(a.Client, k8sComponents, ownerReference, a.Context)
   238  	if err != nil {
   239  		return err
   240  	}
   241  
   242  	// create the Kubernetes objects from the manifest and delete the ones not in the devfile
   243  	needRestart, err := service.PushLinks(a.Client, k8sComponents, labels, a.deployment, a.Context)
   244  	if err != nil {
   245  		return errors.Wrap(err, "failed to create service(s) associated with the component")
   246  	}
   247  
   248  	if needRestart {
   249  		s := log.Spinner("Restarting the component")
   250  		defer s.End(false)
   251  		err = a.Client.WaitForPodDeletion(pod.Name)
   252  		if err != nil {
   253  			return err
   254  		}
   255  		s.End(true)
   256  	}
   257  
   258  	a.deployment, err = a.Client.WaitForDeploymentRollout(a.deployment.Name)
   259  	if err != nil {
   260  		return errors.Wrap(err, "error while waiting for deployment rollout")
   261  	}
   262  
   263  	// Wait for Pod to be in running state otherwise we can't sync data or exec commands to it.
   264  	pod, err = a.getPod(true)
   265  	if err != nil {
   266  		return errors.Wrapf(err, "unable to get pod for component %s", a.ComponentName)
   267  	}
   268  
   269  	parameters.EnvSpecificInfo.SetDevfileObj(a.Devfile)
   270  	err = component.ApplyConfig(a.Client, parameters.EnvSpecificInfo)
   271  	if err != nil {
   272  		return errors.Wrapf(err, "Failed to update config to component deployed.")
   273  	}
   274  
   275  	// Compare the name of the pod with the one before the rollout. If they differ, it means there's a new pod and a force push is required
   276  	if componentExists && podName != pod.GetName() {
   277  		podChanged = true
   278  	}
   279  
   280  	// Find at least one pod with the source volume mounted, error out if none can be found
   281  	containerName, syncFolder, err := getFirstContainerWithSourceVolume(pod.Spec.Containers)
   282  	if err != nil {
   283  		return errors.Wrapf(err, "error while retrieving container from pod %s with a mounted project volume", podName)
   284  	}
   285  
   286  	log.Infof("\nSyncing to component %s", a.ComponentName)
   287  	// Get a sync adapter. Check if project files have changed and sync accordingly
   288  	syncAdapter := sync.New(a.AdapterContext, &a)
   289  	compInfo := common.ComponentInfo{
   290  		ContainerName: containerName,
   291  		PodName:       pod.GetName(),
   292  		SyncFolder:    syncFolder,
   293  	}
   294  	syncParams := common.SyncParameters{
   295  		PushParams:      parameters,
   296  		CompInfo:        compInfo,
   297  		ComponentExists: componentExists,
   298  		PodChanged:      podChanged,
   299  		Files:           common.GetSyncFilesFromAttributes(pushDevfileCommands),
   300  	}
   301  
   302  	execRequired, err := syncAdapter.SyncFiles(syncParams)
   303  	if err != nil {
   304  		return errors.Wrapf(err, "Failed to sync to component with name %s", a.ComponentName)
   305  	}
   306  
   307  	// PostStart events from the devfile will only be executed when the component
   308  	// didn't previously exist
   309  	postStartEvents := a.Devfile.Data.GetEvents().PostStart
   310  	if !componentExists && len(postStartEvents) > 0 {
   311  		err = a.ExecDevfileEvent(postStartEvents, common.PostStart, parameters.Show)
   312  		if err != nil {
   313  			return err
   314  
   315  		}
   316  	}
   317  
   318  	runCommand := pushDevfileCommands[devfilev1.RunCommandGroupKind]
   319  	if parameters.Debug {
   320  		runCommand = pushDevfileCommands[devfilev1.DebugCommandGroupKind]
   321  	}
   322  	running, err := a.GetSupervisordCommandStatus(runCommand)
   323  	if err != nil {
   324  		return err
   325  	}
   326  
   327  	if !running || execRequired || parameters.RunModeChanged {
   328  		log.Infof("\nExecuting devfile commands for component %s", a.ComponentName)
   329  		err = a.ExecDevfile(pushDevfileCommands, componentExists, parameters)
   330  		if err != nil {
   331  			return err
   332  		}
   333  
   334  		// wait for a second
   335  		wait := time.After(supervisorDStatusWaitTimeInterval * time.Second)
   336  		<-wait
   337  
   338  		err := a.CheckSupervisordCommandStatus(runCommand)
   339  		if err != nil {
   340  			return err
   341  		}
   342  	} else {
   343  		// no file was modified/added/deleted/renamed, thus return without syncing files
   344  		log.Success("No file changes detected, skipping build. Use the '-f' flag to force the build.")
   345  	}
   346  
   347  	return nil
   348  }

   350  // GetSupervisordCommandStatus returns true if the command is running
   351  // based on `supervisord ctl` output and returns an error if
   352  // the command is not known by supervisord
   353  func (a Adapter) GetSupervisordCommandStatus(command devfilev1.Command) (bool, error) {
   354  	statusInContainer := getSupervisordStatusInContainer(a.pod.Name, command.Exec.Component, a)
   355  
   356  	supervisordProgramName := "devrun"
   357  
   358  	// if the command is a debug one, we check against `debugrun`
   359  	if command.Exec.Group.Kind == devfilev1.DebugCommandGroupKind {
   360  		supervisordProgramName = "debugrun"
   361  	}
   362  
   363  	for _, status := range statusInContainer {
   364  		if strings.EqualFold(status.program, supervisordProgramName) {
   365  			return strings.EqualFold(status.status, "running"), nil
   366  		}
   367  	}
   368  	return false, fmt.Errorf("the supervisord program %s not found", supervisordProgramName)
   369  }

   371  // CheckSupervisordCommandStatus checks if the command is running based on supervisord status output.
   372  // if the command is not in a running state, we fetch the last 20 lines of the component's log and display it
   373  func (a Adapter) CheckSupervisordCommandStatus(command devfilev1.Command) error {
   374  
   375  	running, err := a.GetSupervisordCommandStatus(command)
   376  	if err != nil {
   377  		return err
   378  	}
   379  
   380  	if !running {
   381  		numberOfLines := 20
   382  		log.Warningf("devfile command %q exited with error status within %d sec", command.Id, supervisorDStatusWaitTimeInterval)
   383  		log.Infof("Last %d lines of the component's log:", numberOfLines)
   384  
   385  		rd, err := a.Log(false, command)
   386  		if err != nil {
   387  			return err
   388  		}
   389  
   390  		err = util.DisplayLog(false, rd, os.Stderr, a.ComponentName, numberOfLines)
   391  		if err != nil {
   392  			return err
   393  		}
   394  
   395  		log.Info("To get the full log output, please run 'odo log'")
   396  	}
   397  	return nil
   398  }

   428  func (a *Adapter) createOrUpdateComponent(componentExists bool, ei envinfo.EnvSpecificInfo, isMainStorageEphemeral bool) (err error) {
   429  	ei.SetDevfileObj(a.Devfile)
   430  
   431  	storageClient := storagepkg.NewClient(storagepkg.ClientOptions{
   432  		Client:              a.Client,
   433  		LocalConfigProvider: &ei,
   434  	})
   435  
   436  	// handle the ephemeral storage
   437  	err = storage.HandleEphemeralStorage(a.Client, storageClient, a.ComponentName, isMainStorageEphemeral)
   438  	if err != nil {
   439  		return err
   440  	}
   441  
   442  	// From devfile info, create PVCs and return ephemeral storages
   443  	ephemerals, err := storagepkg.Push(storageClient, &ei)
   444  	if err != nil {
   445  		return err
   446  	}
   447  
   448  	componentName := a.ComponentName
   449  	var componentType string
   450  	// We insert the component type in deployment annotations because its value might be in a namespaced/versioned format,
   451  	// since labels do not support such formats, we extract the component type from these formats before assigning its value to the corresponding label.
   452  	// This annotated value will later be used when listing the components; we do this to list/describe and stay inline with the component type value set in the devfile.
   453  	annotatedComponentType := component.GetComponentTypeFromDevfileMetadata(a.AdapterContext.Devfile.Data.GetMetadata())
   454  	if annotatedComponentType != component.NotAvailable {
   455  		componentType = strings.TrimSuffix(util.ExtractComponentType(componentType), "-")
   456  	}
   457  
   458  	labels := componentlabels.GetLabels(componentName, a.AppName, true)
   459  	labels["component"] = componentName
   460  	labels[componentlabels.ComponentTypeLabel] = componentType
   461  
   462  	containers, err := generator.GetContainers(a.Devfile, parsercommon.DevfileOptions{})
   463  	if err != nil {
   464  		return err
   465  	}
   466  	if len(containers) == 0 {
   467  		return fmt.Errorf("no valid components found in the devfile")
   468  	}
   469  
   470  	// Add the project volume before generating init containers
   471  	utils.AddOdoProjectVolume(&containers)
   472  
   473  	containers, err = utils.UpdateContainersWithSupervisord(a.Devfile, containers, a.devfileRunCmd, a.devfileDebugCmd, a.devfileDebugPort)
   474  	if err != nil {
   475  		return err
   476  	}
   477  
   478  	initContainers, err := generator.GetInitContainers(a.Devfile)
   479  	if err != nil {
   480  		return err
   481  	}
   482  
   483  	initContainers = append(initContainers, kclient.GetBootstrapSupervisordInitContainer())
   484  
   485  	// list all the pvcs for the component
   486  	pvcs, err := a.Client.ListPVCs(fmt.Sprintf("%v=%v", "component", a.ComponentName))
   487  	if err != nil {
   488  		return err
   489  	}
   490  
   491  	odoSourcePVCName, volumeNameToVolInfo, err := storage.GetVolumeInfos(pvcs)
   492  	if err != nil {
   493  		return err
   494  	}
   495  
   496  	var allVolumes []corev1.Volume
   497  
   498  	// Get PVC volumes and Volume Mounts
   499  	pvcVolumes, err := storage.GetPersistentVolumesAndVolumeMounts(a.Devfile, containers, initContainers, volumeNameToVolInfo, parsercommon.DevfileOptions{})
   500  	if err != nil {
   501  		return err
   502  	}
   503  	allVolumes = append(allVolumes, pvcVolumes...)
   504  
   505  	ephemeralVolumes, err := storage.GetEphemeralVolumesAndVolumeMounts(a.Devfile, containers, initContainers, ephemerals, parsercommon.DevfileOptions{})
   506  	if err != nil {
   507  		return err
   508  	}
   509  	allVolumes = append(allVolumes, ephemeralVolumes...)
   510  
   511  	odoMandatoryVolumes := utils.GetOdoContainerVolumes(odoSourcePVCName)
   512  	allVolumes = append(allVolumes, odoMandatoryVolumes...)
   513  
   514  	selectorLabels := map[string]string{
   515  		"component": componentName,
   516  	}
   517  
   518  	deploymentObjectMeta, err := a.generateDeploymentObjectMeta(labels)
   519  	if err != nil {
   520  		return err
   521  	}
   522  
   523  	deployParams := generator.DeploymentParams{
   524  		TypeMeta:          generator.GetTypeMeta(kclient.DeploymentKind, kclient.DeploymentAPIVersion),
   525  		ObjectMeta:        deploymentObjectMeta,
   526  		InitContainers:    initContainers,
   527  		Containers:        containers,
   528  		Volumes:           allVolumes,
   529  		PodSelectorLabels: selectorLabels,
   530  	}
   531  
   532  	deployment, err := generator.GetDeployment(a.Devfile, deployParams)
   533  	if err != nil {
   534  		return err
   535  	}
   536  
   537  	if deployment.Annotations == nil {
   538  		deployment.Annotations = make(map[string]string)
   539  	}
   540  
   541  	// Add annotation for component type; this will later be used while listing/describing a component
   542  	deployment.Annotations[componentlabels.ComponentTypeAnnotation] = annotatedComponentType
   543  
   544  	if vcsUri := util.GetGitOriginPath(a.Context); vcsUri != "" {
   545  		deployment.Annotations["app.openshift.io/vcs-uri"] = vcsUri
   546  	}
   547  
   548  	// add the annotations to the service for linking
   549  	serviceAnnotations := make(map[string]string)
   550  	serviceAnnotations["service.binding/backend_ip"] = "path={.spec.clusterIP}"
   551  	serviceAnnotations["service.binding/backend_port"] = "path={.spec.ports},elementType=sliceOfMaps,sourceKey=name,sourceValue=port"
   552  
   553  	serviceName, err := util.NamespaceKubernetesObjectWithTrim(componentName, a.AppName)
   554  	if err != nil {
   555  		return err
   556  	}
   557  	serviceObjectMeta := generator.GetObjectMeta(serviceName, a.Client.GetCurrentNamespace(), labels, serviceAnnotations)
   558  	serviceParams := generator.ServiceParams{
   559  		ObjectMeta:     serviceObjectMeta,
   560  		SelectorLabels: selectorLabels,
   561  	}
   562  	svc, err := generator.GetService(a.Devfile, serviceParams, parsercommon.DevfileOptions{})
   563  
   564  	if err != nil {
   565  		return err
   566  	}
   567  	klog.V(2).Infof("Creating deployment %v", deployment.Spec.Template.GetName())
   568  	klog.V(2).Infof("The component name is %v", componentName)
   569  	if componentExists {
   570  		// If the component already exists, get the resource version of the deploy before updating
   571  		klog.V(2).Info("The component already exists, attempting to update it")
   572  		if a.Client.IsSSASupported() {
   573  			a.deployment, err = a.Client.ApplyDeployment(*deployment)
   574  		} else {
   575  			a.deployment, err = a.Client.UpdateDeployment(*deployment)
   576  		}
   577  		if err != nil {
   578  			return err
   579  		}
   580  		klog.V(2).Infof("Successfully updated component %v", componentName)
   581  		e := a.createOrUpdateServiceForComponent(svc, componentName)
   582  		if e != nil {
   583  			return e
   584  		}
   585  	} else {
   586  		if a.Client.IsSSASupported() {
   587  			a.deployment, err = a.Client.ApplyDeployment(*deployment)
   588  		} else {
   589  			a.deployment, err = a.Client.CreateDeployment(*deployment)
   590  		}
   591  
   592  		if err != nil {
   593  			return err
   594  		}
   595  		klog.V(2).Infof("Successfully created component %v", componentName)
   596  		ownerReference := generator.GetOwnerReference(a.deployment)
   597  		svc.OwnerReferences = append(svc.OwnerReferences, ownerReference)
   598  		if len(svc.Spec.Ports) > 0 {
   599  			_, err = a.Client.CreateService(*svc)
   600  			if err != nil {
   601  				return err
   602  			}
   603  			klog.V(2).Infof("Successfully created Service for component %s", componentName)
   604  		}
   605  
   606  	}
   607  
   608  	return nil

   611  func (a *Adapter) createOrUpdateServiceForComponent(svc *corev1.Service, componentName string) error {
   612  	oldSvc, err := a.Client.GetOneService(a.ComponentName, a.AppName)
   613  	ownerReference := generator.GetOwnerReference(a.deployment)
   614  	svc.OwnerReferences = append(svc.OwnerReferences, ownerReference)
   615  	if err != nil {
   616  		// no old service was found, create a new one
   617  		if len(svc.Spec.Ports) > 0 {
   618  			_, err = a.Client.CreateService(*svc)
   619  			if err != nil {
   620  				return err
   621  			}
   622  			klog.V(2).Infof("Successfully created Service for component %s", componentName)
   623  		}
   624  		return nil
   625  	}
   626  	if len(svc.Spec.Ports) > 0 {
   627  		svc.Spec.ClusterIP = oldSvc.Spec.ClusterIP
   628  		svc.ResourceVersion = oldSvc.GetResourceVersion()
   629  		_, err = a.Client.UpdateService(*svc)
   630  		if err != nil {
   631  			return err
   632  		}
   633  		klog.V(2).Infof("Successfully update Service for component %s", componentName)
   634  		return nil
   635  	}
   636  	// delete the old existing service if the component currently doesn't expose any ports
   637  	return a.Client.DeleteService(oldSvc.Name)

   671  // Delete deletes the component
   672  func (a Adapter) Delete(labels map[string]string, show bool, wait bool) error {
   673  	if labels == nil {
   674  		return fmt.Errorf("cannot delete with labels being nil")
   675  	}
   676  	log.Infof("\nGathering information for component %s", a.ComponentName)
   677  	podSpinner := log.Spinner("Checking status for component")
   678  	defer podSpinner.End(false)
   679  
   680  	pod, err := a.Client.GetOnePod(a.ComponentName, a.AppName)
   681  	if kerrors.IsForbidden(err) {
   682  		klog.V(2).Infof("Resource for %s forbidden", a.ComponentName)
   683  		// log the error if it failed to determine if the component exists due to insufficient RBACs
   684  		podSpinner.End(false)
   685  		log.Warningf("%v", err)
   686  		return nil
   687  	} else if e, ok := err.(*kclient.PodNotFoundError); ok {
   688  		podSpinner.End(false)
   689  		log.Warningf("%v", e)
   690  		return nil
   691  	} else if err != nil {
   692  		return errors.Wrapf(err, "unable to determine if component %s exists", a.ComponentName)
   693  	}
   694  
   695  	podSpinner.End(true)
   696  
   697  	// if there are preStop events, execute them before deleting the deployment
   698  	preStopEvents := a.Devfile.Data.GetEvents().PreStop
   699  	if len(preStopEvents) > 0 {
   700  		if pod.Status.Phase != corev1.PodRunning {
   701  			return fmt.Errorf("unable to execute preStop events, pod for component %s is not running", a.ComponentName)
   702  		}
   703  
   704  		err = a.ExecDevfileEvent(preStopEvents, common.PreStop, show)
   705  		if err != nil {
   706  			return err
   707  		}
   708  	}
   709  
   710  	log.Infof("\nDeleting component %s", a.ComponentName)
   711  	spinner := log.Spinner("Deleting Kubernetes resources for component")
   712  	defer spinner.End(false)
   713  
   714  	err = a.Client.Delete(labels, wait)
   715  	if err != nil {
   716  		return err
   717  	}
   718  
   719  	spinner.End(true)
   720  	log.Successf("Successfully deleted component")
   721  	return nil
   722  }
