    25  // newExecCommand creates a new execCommand instance, adapting the devfile-defined command to run in the target component's
    26  // container, modifying it to add environment variables or adapting the path as needed.
    27  func newExecCommand(command devfilev1.Command, executor commandExecutor) (command, error) {
    28  	exe := command.Exec
    29  
    30  	// deal with environment variables
    31  	var cmdLine string
    32  	setEnvVariable := util.GetCommandStringFromEnvs(exe.Env)
    33  
    34  	if setEnvVariable == "" {
    35  		cmdLine = exe.CommandLine
    36  	} else {
    37  		cmdLine = setEnvVariable + " && " + exe.CommandLine
    38  	}
    39  
    40  	// Change to the workdir and execute the command
    41  	var cmd []string
    42  	if exe.WorkingDir != "" {
    43  		// since we are using /bin/sh -c, the command needs to be within a single double quote instance, for example "cd /tmp && pwd"
    44  		cmd = []string{ShellExecutable, "-c", "cd " + exe.WorkingDir + " && " + cmdLine}
    45  	} else {
    46  		cmd = []string{ShellExecutable, "-c", cmdLine}
    47  	}
    48  
    49  	return newOverriddenExecCommand(command, executor, cmd)
    50  }

    52  // newOverriddenExecCommand creates a new execCommand albeit overriding the command specified in the devfile with the specified one
    53  // returning a pointer to the newly created instance so that clients can further modify it if needed.
    54  // Note that the specified command will be run as-is in the target component's container so needs to be set accordingly as
    55  // opposed to the implementation provided by newExecCommand which will take the devfile's command definition and adapt it to
    56  // run in the container.
    57  func newOverriddenExecCommand(command devfilev1.Command, executor commandExecutor, cmd []string) (*execCommand, error) {
    58  	// create the component info associated with the command
    59  	info, err := executor.ComponentInfo(command)
    60  	if err != nil {
    61  		return nil, err
    62  	}
    63  
    64  	originalCmd := command.Exec.CommandLine
    65  	return &execCommand{
    66  		info:        info,
    67  		adapter:     executor,
    68  		cmd:         cmd,
    69  		id:          command.Id,
    70  		component:   command.Exec.Component,
    71  		originalCmd: originalCmd,
    72  		group:       convertGroupKindToString(command.Exec),
    73  		msg:         fmt.Sprintf("Executing %s command %q", command.Id, originalCmd),
    74  	}, nil
    75  }

    77  func (s execCommand) Execute(show bool) error {
    78  	var spinner *log.Status
    79  	showSpinner := len(s.msg) > 0
    80  	if showSpinner {
    81  		spinner = log.ExplicitSpinner(s.msg, show)
    82  		defer spinner.End(false)
    83  	}
    84  
    85  	// Emit DevFileCommandExecutionBegin JSON event (if machine output logging is enabled)
    86  	logger := s.adapter.Logger()
    87  	logger.DevFileCommandExecutionBegin(s.id, s.component, s.originalCmd, s.group, machineoutput.TimestampNow())
    88  
    89  	// Capture container text and log to the screen as JSON events (machine output only)
    90  	stdoutWriter, stdoutChannel, stderrWriter, stderrChannel := logger.CreateContainerOutputWriter()
    91  
    92  	err := ExecuteCommand(s.adapter, s.info, s.cmd, show, stdoutWriter, stderrWriter)
    93  
    94  	// Close the writers and wait for an acknowledgement that the reader loop has exited (to ensure we get ALL container output)
    95  	closeWriterAndWaitForAck(stdoutWriter, stdoutChannel, stderrWriter, stderrChannel)
    96  
    97  	// Emit close event
    98  	logger.DevFileCommandExecutionComplete(s.id, s.component, s.originalCmd, s.group, machineoutput.TimestampNow(), err)
    99  
   100  	if err != nil {
   101  		return errors.Wrapf(err, "unable to execute the run command")
   102  	}
   103  
   104  	if showSpinner {
   105  		spinner.End(true)
   106  	}
   107  
   108  	return nil
   109  }
