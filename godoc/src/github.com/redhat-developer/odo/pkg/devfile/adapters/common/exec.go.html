    20  // ExecuteCommand executes the given command in the pod's container
    21  func ExecuteCommand(client ExecClient, compInfo ComponentInfo, command []string, show bool, consoleOutputStdout *io.PipeWriter, consoleOutputStderr *io.PipeWriter) (err error) {
    22  	stdoutReader, stdoutWriter := io.Pipe()
    23  	stderrReader, stderrWriter := io.Pipe()
    24  
    25  	var cmdOutput string
    26  
    27  	klog.V(2).Infof("Executing command %v for pod: %v in container: %v", command, compInfo.PodName, compInfo.ContainerName)
    28  
    29  	// Read stdout and stderr, store their output in cmdOutput, and also pass output to consoleOutput Writers (if non-nil)
    30  	stdoutCompleteChannel := startReaderGoroutine(stdoutReader, show, &cmdOutput, consoleOutputStdout)
    31  	stderrCompleteChannel := startReaderGoroutine(stderrReader, show, &cmdOutput, consoleOutputStderr)
    32  
    33  	err = client.ExecCMDInContainer(compInfo, command, stdoutWriter, stderrWriter, nil, false)
    34  
    35  	// Block until we have received all the container output from each stream
    36  	_ = stdoutWriter.Close()
    37  	<-stdoutCompleteChannel
    38  	_ = stderrWriter.Close()
    39  	<-stderrCompleteChannel
    40  
    41  	if err != nil {
    42  		// It is safe to read from cmdOutput here, as the goroutines are guaranteed to have terminated at this point.
    43  		klog.V(2).Infof("ExecuteCommand returned an an err: %v. for command '%v'. output: %v", err, command, cmdOutput)
    44  
    45  		return errors.Wrapf(err, "unable to exec command %v: \n%v", command, cmdOutput)
    46  	}
    47  
    48  	return
    49  }

    51  // This goroutine will automatically pipe the output from the writer (passed into ExecCMDInContainer) to
    52  // the loggers.
    53  // The returned channel will contain a single nil entry once the reader has closed.
    54  func startReaderGoroutine(reader io.Reader, show bool, cmdOutput *string, consoleOutput *io.PipeWriter) chan interface{} {
    55  
    56  	result := make(chan interface{})
    57  
    58  	go func() {
    59  		scanner := bufio.NewScanner(reader)
    60  		for scanner.Scan() {
    61  			line := scanner.Text()
    62  
    63  			if log.IsDebug() || show {
    64  				_, err := fmt.Fprintln(os.Stdout, line)
    65  				if err != nil {
    66  					log.Errorf("Unable to print to stdout: %s", err.Error())
    67  				}
    68  			}
    69  
    70  			*cmdOutput += fmt.Sprintln(line)
    71  
    72  			if consoleOutput != nil {
    73  				_, err := consoleOutput.Write([]byte(line + "\n"))
    74  				if err != nil {
    75  					log.Errorf("Error occurred on writing string to consoleOutput writer: %s", err.Error())
    76  				}
    77  			}
    78  		}
    79  		result <- nil
    80  	}()
    81  
    82  	return result
    83

    86  // CreateConsoleOutputWriterAndChannel is a utility function that returns a pipeWriter and a channel;
    87  // any strings written to that PipeWriter will be output to the channel (as lines) when the
    88  // writer closes. This is used to retrieve the stdout/stderr output from the container exec commands.
    89  //
    90  // The io.PipeWriter can be passed to ExecuteCommand(...) above, in order to receive the full
    91  // stderr/stdout output from the process.
    92  // See calling functions of CreateConsoleOutputWriterAndChannel for examples of usage.
    93  func CreateConsoleOutputWriterAndChannel() (*io.PipeWriter, chan []string) {
    94  	reader, writer := io.Pipe()
    95  
    96  	closeChannel := make(chan []string)
    97  
    98  	go func() {
    99  
   100  		consoleContents := []string{}
   101  
   102  		bufReader := bufio.NewReader(reader)
   103  		for {
   104  			line, _, err := bufReader.ReadLine()
   105  			if err != nil {
   106  				if err != io.EOF {
   107  					klog.V(2).Infof("Unexpected error on reading container output reader: %v", err)
   108  				}
   109  
   110  				break
   111  			}
   112  
   113  			consoleContents = append(consoleContents, string(line))
   114  		}
   115  
   116  		// Output the final console contents to the channel
   117  		closeChannel <- consoleContents
   118  	}()
   119  
   120  	return writer, closeChannel
   121  }
