   105  // ExecDevFile executes all the commands from the devfile in order: init and build - which are both optional, and a compulsory run.
   106  // Init only runs once when the component is created.
   107  func (a GenericAdapter) ExecDevfile(commandsMap PushCommandsMap, componentExists bool, params PushParameters) (err error) {
   108  	// Need to get mapping of all commands in the devfile since the composite command may reference any exec or composite command in the devfile
   109  	devfileCommands, err := a.Devfile.Data.GetCommands(common.DevfileOptions{})
   110  	if err != nil {
   111  		return err
   112  	}
   113  
   114  	devfileCommandMap := GetCommandsMap(devfileCommands)
   115  
   116  	// If nothing has been passed, then the devfile is missing the required run command
   117  	if len(commandsMap) == 0 {
   118  		return errors.New("error executing devfile commands - there should be at least 1 command")
   119  	}
   120  
   121  	commands := make([]command, 0, 7)
   122  
   123  	// Get Build Command
   124  	commands, err = a.addToComposite(commandsMap, devfilev1.BuildCommandGroupKind, devfileCommandMap, commands)
   125  	if err != nil {
   126  		return err
   127  	}
   128  
   129  	group := devfilev1.RunCommandGroupKind
   130  	defaultCmd := string(DefaultDevfileRunCommand)
   131  
   132  	if params.Debug {
   133  		group = devfilev1.DebugCommandGroupKind
   134  		defaultCmd = string(DefaultDevfileDebugCommand)
   135  
   136  	}
   137  
   138  	if command, ok := commandsMap[group]; ok {
   139  		// if the component doesn't exist, initialize the supervisor if needed
   140  		if !componentExists {
   141  			if cmd, err := newSupervisorInitCommand(command, a); cmd != nil {
   142  				if err != nil {
   143  					return err
   144  				}
   145  				commands = append(commands, cmd)
   146  			}
   147  		}
   148  
   149  		restart := IsRestartRequired(util.SafeGetBool(command.Exec.HotReloadCapable), params.RunModeChanged)
   150  
   151  		// if we need to restart, issue supervisor command to stop all running commands first
   152  		// we do not need to restart Hot reload capable commands
   153  		if componentExists {
   154  			if restart {
   155  				klog.V(2).Infof("supervisord stop command to restart or start other command")
   156  				if cmd, err := newSupervisorStopCommand(command, a); cmd != nil {
   157  					if err != nil {
   158  						return err
   159  					}
   160  					commands = append(commands, cmd)
   161  				}
   162  			} else {
   163  				klog.V(2).Infof("command is hot reload capable, not restarting %s", defaultCmd)
   164  			}
   165  		}
   166  
   167  		// with restart false, executing only supervisord start command, if the command is already running, supvervisord will not restart it.
   168  		// if the command is failed or not running supervisord would start it.
   169  		if cmd, err := newSupervisorStartCommand(command, defaultCmd, a, restart); cmd != nil {
   170  			if err != nil {
   171  				return err
   172  			}
   173  			commands = append(commands, cmd)
   174  		}
   175  
   176  		c := newCompositeCommand(commands...)
   177  		return c.Execute(params.Show)
   178  	}
   179  
   180  	return nil
   181  }

   195  // ExecDevfileEvent receives a Devfile Event (PostStart, PreStop etc.) and loops through them
   196  // Each Devfile Command associated with the given event is retrieved, and executed in the container specified
   197  // in the command
   198  func (a GenericAdapter) ExecDevfileEvent(events []string, eventType DevfileEventType, show bool) error {
   199  	if len(events) > 0 {
   200  		log.Infof("\nExecuting %s event commands for component %s", string(eventType), a.ComponentName)
   201  		commands, err := a.Devfile.Data.GetCommands(common.DevfileOptions{})
   202  		if err != nil {
   203  			return err
   204  		}
   205  
   206  		commandMap := GetCommandsMap(commands)
   207  		for _, commandName := range events {
   208  			// Convert commandName to lower because GetCommands converts Command.Exec.Id's to lower
   209  			command, ok := commandMap[strings.ToLower(commandName)]
   210  			if !ok {
   211  				return errors.New("unable to find devfile command " + commandName)
   212  			}
   213  
   214  			c, err := New(command, commandMap, a)
   215  			if err != nil {
   216  				return err
   217  			}
   218  			// Execute command in container
   219  			err = c.Execute(show)
   220  			if err != nil {
   221  				return errors.Wrapf(err, "unable to execute devfile command %s", commandName)
   222  			}
   223  		}
   224  	}
   225  	return nil
   226  }
