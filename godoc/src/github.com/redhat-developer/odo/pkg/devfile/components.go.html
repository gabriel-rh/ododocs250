    22  // GetKubernetesComponentsToPush returns the list of Kubernetes components to push,
    23  // by getting the list of Kubernetes components and removing the ones
    24  // referenced from a command in the devfile
    25  func GetKubernetesComponentsToPush(devfileObj parser.DevfileObj) ([]devfilev1.Component, error) {
    26  	k8sComponents, err := devfileObj.Data.GetComponents(parsercommon.DevfileOptions{
    27  		ComponentOptions: parsercommon.ComponentOptions{ComponentType: devfilev1.KubernetesComponentType},
    28  	})
    29  	if err != nil {
    30  		return nil, err
    31  	}
    32  
    33  	componentsMap := map[string]devfilev1.Component{}
    34  	for _, component := range k8sComponents {
    35  		componentsMap[component.Name] = component
    36  	}
    37  
    38  	commands, err := devfileObj.Data.GetCommands(parsercommon.DevfileOptions{})
    39  	if err != nil {
    40  		return nil, err
    41  	}
    42  
    43  	for _, command := range commands {
    44  		componentName := ""
    45  		if command.Exec != nil {
    46  			componentName = command.Exec.Component
    47  		} else if command.Apply != nil {
    48  			componentName = command.Apply.Component
    49  		}
    50  		if componentName == "" {
    51  			continue
    52  		}
    53  		delete(componentsMap, componentName)
    54  	}
    55  
    56  	k8sComponents = make([]devfilev1.Component, len(componentsMap))
    57  	i := 0
    58  	for _, v := range componentsMap {
    59  		k8sComponents[i] = v
    60  		i++
    61  	}
    62  
    63  	return k8sComponents, err
    64  }

   107  // addKubernetesComponent adds the crd information to a separate file and adds the uri information to a devfile component
   108  func addKubernetesComponent(crd, name, componentContext string, devfileObj parser.DevfileObj, fs devfilefs.Filesystem) error {
   109  	filePath := filepath.Join(componentContext, consts.UriFolder, filePrefix+name+".yaml")
   110  	if _, err := fs.Stat(filepath.Join(componentContext, consts.UriFolder)); os.IsNotExist(err) {
   111  		err = fs.MkdirAll(filepath.Join(componentContext, consts.UriFolder), os.ModePerm)
   112  		if err != nil {
   113  			return err
   114  		}
   115  	}
   116  
   117  	if _, err := fs.Stat(filePath); !os.IsNotExist(err) {
   118  		return fmt.Errorf("the file %q already exists", filePath)
   119  	}
   120  
   121  	err := fs.WriteFile(filePath, []byte(crd), 0755)
   122  	if err != nil {
   123  		return err
   124  	}
   125  
   126  	err = devfileObj.Data.AddComponents([]devfilev1.Component{{
   127  		Name: name,
   128  		ComponentUnion: devfilev1.ComponentUnion{
   129  			Kubernetes: &devfilev1.KubernetesComponent{
   130  				K8sLikeComponent: devfilev1.K8sLikeComponent{
   131  					BaseComponent: devfilev1.BaseComponent{},
   132  					K8sLikeComponentLocation: devfilev1.K8sLikeComponentLocation{
   133  						Uri: filepath.Join(consts.UriFolder, filePrefix+name+".yaml"),
   134  					},
   135  				},
   136  			},
   137  		},
   138  	}})
   139  	if err != nil {
   140  		return err
   141  	}
   142  
   143  	return devfileObj.WriteYamlDevfile()
   144  }

   151  // deleteKubernetesComponentFromDevfile deletes an inlined Kubernetes component from devfile, if one exists
   152  func deleteKubernetesComponentFromDevfile(name string, devfileObj parser.DevfileObj, componentContext string, fs devfilefs.Filesystem) error {
   153  	components, err := devfileObj.Data.GetComponents(common.DevfileOptions{})
   154  	if err != nil {
   155  		return err
   156  	}
   157  
   158  	found := false
   159  	for _, c := range components {
   160  		if c.Name == name {
   161  			err = devfileObj.Data.DeleteComponent(c.Name)
   162  			if err != nil {
   163  				return err
   164  			}
   165  
   166  			if c.Kubernetes.Uri != "" {
   167  				parsedURL, err := url.Parse(c.Kubernetes.Uri)
   168  				if err != nil {
   169  					return err
   170  				}
   171  				if len(parsedURL.Host) == 0 || len(parsedURL.Scheme) == 0 {
   172  					err := fs.Remove(filepath.Join(componentContext, c.Kubernetes.Uri))
   173  					if err != nil {
   174  						return err
   175  					}
   176  				}
   177  			}
   178  			found = true
   179  			break
   180  		}
   181  	}
   182  
   183  	if !found {
   184  		return fmt.Errorf("could not find the service %q in devfile", name)
   185  	}
   186  
   187  	return devfileObj.WriteYamlDevfile()
   188  }
