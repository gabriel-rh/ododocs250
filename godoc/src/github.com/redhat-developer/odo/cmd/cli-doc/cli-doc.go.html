   137  func commandPrinter(command *cobra.Command, level int) string {
   138  	var finalCommand string
   139  	// add indentation
   140  	for i := 0; i < level; i++ {
   141  		finalCommand = finalCommand + "    "
   142  	}
   143  	finalCommand = finalCommand +
   144  		command.Name() +
   145  		" " +
   146  		flattenFlags(getFlags(command.NonInheritedFlags())) +
   147  		": " +
   148  		command.Short +
   149  		"\n"
   150  	for _, subcommand := range command.Commands() {
   151  		finalCommand = finalCommand + commandPrinter(subcommand, level+1)
   152  	}
   153  	return finalCommand
   154  }

   168  		ValidArgs: []string{"help", "reference", "structure"},
   169  
   170  		Run: func(command *cobra.Command, args []string) {
   171  			if len(args) == 0 {
   172  				fmt.Print(command.Usage())
   173  			} else {
   174  				switch args[0] {
   175  				case "reference":
   176  					fmt.Print(referencePrinter(cli.NewCmdOdo(cli.OdoRecommendedName, cli.OdoRecommendedName), 0))

   178  					fmt.Print(commandPrinter(cli.NewCmdOdo(cli.OdoRecommendedName, cli.OdoRecommendedName), 0))
   179  				default:
   180  					fmt.Print(command.Usage())
   181  				}
   182  			}
   183  		},
   184  	}
   185
