    21  func main() {
    22  	// create the complete command
    23  	klog.InitFlags(nil)
    24  
    25  	root := cli.NewCmdOdo(cli.OdoRecommendedName, cli.OdoRecommendedName)
    26  	rootCmp := createCompletion(root)
    27  	cmp := complete.New("odo", rootCmp)
    28  
    29  	// AddFlags adds the completion flags to the program flags, specifying custom names
    30  	cmp.CLI.InstallName = "complete"
    31  	cmp.CLI.UninstallName = "uncomplete"
    32  	cmp.AddFlags(nil)
    33  
    34  	// add the completion flags to the root command, though they won't appear in completions
    35  	root.Flags().AddGoFlagSet(flag.CommandLine)
    36  	// override usage so that flag.Parse uses root command's usage instead of default one when invoked with -h
    37  	flag.Usage = func() {
    38  		_ = root.Help()
    39  	}
    40  
    41  	// parse the flags but hack around to avoid exiting with error code 2 on help
    42  	flag.CommandLine.Init(os.Args[0], flag.ContinueOnError)
    43  	args := os.Args[1:]
    44  	if err := flag.CommandLine.Parse(args); err != nil {
    45  		if err == flag.ErrHelp {
    46  			os.Exit(0)
    47  		}
    48  	}
    49  
    50  	// run the completion, in case that the completion was invoked
    51  	// and ran as a completion script or handled a flag that passed
    52  	// as argument, the Run method will return true,
    53  	// in that case, our program have nothing to do and should return.
    54  	if cmp.Complete() {
    55  		return
    56  	}
    57  
    58  	cfg, err := preference.NewClient()
    59  	if err != nil {
    60  		util.LogErrorAndExit(err, "")
    61  	}
    62  
    63  	// Call commands
    64  	// checking the value of updatenotification in config
    65  	// before proceeding with fetching the latest version
    66  	if cfg.GetUpdateNotification() {
    67  		updateInfo := make(chan string)
    68  		go version.GetLatestReleaseInfo(updateInfo)
    69  
    70  		util.LogErrorAndExit(root.ExecuteContext(segment.NewContext(context.Background())), "")
    71  		select {
    72  		case message := <-updateInfo:
    73  			log.Italic(message)
    74  		default:
    75  			klog.V(4).Info("Could not get the latest release information in time. Never mind, exiting gracefully :)")
    76  		}
    77  	} else {
    78  		util.LogErrorAndExit(root.ExecuteContext(segment.NewContext(context.Background())), "")
    79  	}
    80  }

    82  func createCompletion(root *cobra.Command) complete.Command {
    83  	rootCmp := complete.Command{}
    84  	rootCmp.Flags = make(complete.Flags)
    85  	addFlags := func(flag *pflag.Flag) {
    86  		if flag.Hidden {
    87  			return
    88  		}
    89  		var handler complete.Predictor
    90  		handler, ok := completion.GetCommandFlagHandler(root, flag.Name)
    91  		if !ok {
    92  			handler = complete.PredictAnything
    93  		}
    94  
    95  		if len(flag.Shorthand) > 0 {
    96  			rootCmp.Flags["-"+flag.Shorthand] = handler
    97  		}
    98  
    99  		rootCmp.Flags["--"+flag.Name] = handler
   100  	}
   101  	root.LocalFlags().VisitAll(addFlags)
   102  	root.InheritedFlags().VisitAll(addFlags)
   103  	if root.HasAvailableSubCommands() {
   104  		rootCmp.Sub = make(complete.Commands)
   105  		for _, c := range root.Commands() {
   106  			if !c.Hidden {
   107  				rootCmp.Sub[c.Name()] = createCompletion(c)
   108  			}
   109  		}
   110  	}
   111  
   112  	var handler complete.Predictor
   113  	handler, ok := completion.GetCommandHandler(root)
   114  	if !ok {
   115  		handler = complete.PredictNothing
   116  	}
   117  	rootCmp.Args = handler
   118  
   119  	return rootCmp
   120  }
